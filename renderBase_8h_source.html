151struct Arcball {
152 double angle;
153 triple axis;
154
155 Arcball(double x0, double y0, double x, double y)
156 {
157 triple v0 = norm(x0, y0);
158 triple v1 = norm(x, y);
159 double Dot = dot(v0, v1);
160 angle = Dot > 1.0 ? 0.0 : Dot < -1.0 ? M_PI
161 : acos(Dot);
162 static triple Z(0.0, 0.0, 1.0);
163 axis = unit(cross(v0, v1), Z);
164 }
165
166 triple norm(double x, double y)
167 {
168 double norm = hypot(x, y);
169 if (norm > 1.0) {
170 double denom = 1.0 / norm;
171 x *= denom;
172 y *= denom;
173 }
174 return triple(x, y, sqrt(max(1.0 - x * x - y * y, 0.0)));
175 }
