147void ifile::ignoreComment()
148{
149 if(comment == 0) return;
150 int c=stream->peek();
151 bool eol=c == '\n';
152 if(csvmode && eol) {nullfield=true; return;}
153 if(csvmode && c == ',') nullfield=true;
154 for(;;) {
155 while(isspace(c=stream->peek())) {
156 stream->ignore();
157 whitespace += (char) c;
158 }
159 if(c == comment) {
160 whitespace="";
161 while((c=stream->peek()) != '\n' && c != EOF)
162 stream->ignore();
163 if(c == '\n')
164 stream->ignore();
165 } else {if(c != EOF && eol) stream->unget(); return;}
166 }

169void ifile::Read(double& val) {
170 char c;
171 std::string str;
172 bool neg;
173
174 while(isspace(c=stream->peek()))
175 stream->ignore();
176 neg=stream->peek() == '-';
177 // Try parsing the input as a number.
178 if(*stream >> val)
179 return;
180
181 clear();
182
183 switch(stream->peek()) {
184 case 'I': case 'i': // inf
185 case 'N': case 'n': // NaN
186 for(Int i=0; i < 3 && stream->good(); i++)
187 str += stream->get();
188 break;
189 default:
190 stream->setstate(std::ios_base::failbit);
191 return;
192 }
193
194 if(strcasecmp(str.c_str(),"inf") == 0)
195 val=std::numeric_limits < double > ::infinity();
196 else if(strcasecmp(str.c_str(),"nan") == 0)
197 val=std::numeric_limits < double > ::quiet_NaN();
198 else {
199 for(auto it=str.rbegin(); it != str.rend(); ++it)
200 stream->putback(*it);
201 stream->setstate(std::ios_base::failbit);
202 return;
203 }
204
205 if(neg)
206 val=-val;
207}

210{

222bool ifile::nexteol()
223{
224 int c;

227 return true;
228 }
229
230 while(isspace(c=stream->peek())) {

232 nullfield=true;
233 return false;
234 }
235 stream->ignore();
236 if(c == '\n') {
237 while(isspace(c=stream->peek())) {
238 if(c == '\n') {nullfield=true; return true;}

240 stream->ignore();

242 }
243 }
244 return true;
245 }

247 }
248 return false;

251void ifile::csv()
252{
253 comma=false;
254 nullfield=false;
255 if(!csvmode || stream->eof()) return;
256 std::ios::iostate rdstate=stream->rdstate();
257 if(stream->fail()) stream->clear();
258 int c=stream->peek();
259 if(c == ',') stream->ignore();
260 else if(c == '\n') {
261 stream->ignore();
262 if(linemode && stream->peek() != EOF) stream->unget();
263 } else stream->clear(rdstate);
264 if(c == ',') comma=true;

267void ifile::Read(string& val)
268{
269 string s;
270 if(wordmode) {
271 whitespace="";
272 while(isspace(stream->peek())) stream->ignore();
273 }
274 if(csvmode || wordmode) {
275 bool quote=false;
276 while(stream->good()) {
277 int c=stream->peek();
278 if(c == '"') {quote=!quote; stream->ignore(); continue;}
279 if(!quote) {
280 if(comment && c == comment) {
281 while((c=stream->peek()) != '\n' && c != EOF)
282 stream->ignore();
283 if(wordmode && !linemode)
284 while(isspace(stream->peek())) stream->ignore();
285 if(stream->peek() == '"') {quote=!quote; stream->ignore(); continue;}
286 if(s.empty() && c == '\n') {
287 stream->ignore();
288 continue;
289 }
290 }
291 if(csvmode && (c == ',' || c == '\n'))
292 break;
293 if(wordmode && isspace(c)) {
294 if(!linemode) while(isspace(stream->peek())) stream->ignore();
295 break;
296 }
297 }
298 s += (char) stream->get();
299 }
300 } else
301 getline(*stream,s);
302
303 if(comment) {
304 size_t p=0;
305 while((p=s.find(comment,p)) < string::npos) {
306 if(p+1 < s.length() && s[p+1] == comment) {
307 s.erase(p,1);
308 ++p;
309 } else {
310 s.erase(p);
311 break;
312 }
313 }
314 }
315 size_t n=s.length();
316 if(n > 0) {
317 size_t pos=n-1;
318 if(s[pos] == '\r') s.erase(pos,1);
319 }
320 val=whitespace+s;
321}
