594void importv3d(string name)
595{
596 if(name == stripextension(name)) name += ".v3d";
597 v3dfile xf=v3dfile(name);
598 xf.process();
599 xf.setCameraInfo();
600
601 for(int c=0; c < xf.surfaces.length; ++c) {
602 triple center=c > 0 ? xf.centers[c-1] : O;
603 render r=render(interaction(c == 0 ? Embedded : Billboard,center=center));
604 surface[][] S=xf.surfaces[c];
605 for(int m=0; m < S.length; ++m)
606 if(S.initialized(m))
607 draw(S[m],xf.materials[m],r);
608 }
609
610 for(int c=0; c < xf.paths3.length; ++c) {
611 triple center=c > 0 ? xf.centers[c-1] : O;
612 render r=render(interaction(c == 0 ? Embedded : Billboard,center=center));
613 path3[][] G=xf.paths3[c];
614 for(int m=0; m < G.length; ++m)
615 if(G.initialized(m)) {
616 material material=material(xf.materials[m]);
617 material.p[0] += thin();
618 draw(G[m],material,currentlight,r);
619 }
620 }
621
622 for(int c=0;c < xf.triangles.length; ++c) {
623 triple center=c > 0 ? xf.centers[c-1] : O;
624 render r=render(interaction(c == 0 ? Embedded : Billboard,center=center));
625 triangleGroup[][] groups=xf.triangles[c];
626 for(int m=0; m < groups.length; ++m) {
627 if(groups.initialized(m)) {
628 material material=xf.materials[m];
629 triangleGroup[] triangleGroups=groups[m];
630 for(triangleGroup g : triangleGroups) {
631 if(g.colors.length > 0)
632 draw(g.positions,g.positionIndices,g.normals,g.normalIndices,
633 g.colors,g.colorIndices,r);
634 else
635 draw(g.positions,g.positionIndices,g.normals,g.normalIndices,
636 material,r);
637 }
638 }
639 }
640 }
641
642 for(int m=0; m < xf.pixels.length; ++m) {
643 if(xf.pixels.initialized(m)) {
644 material material=xf.materials[m];
645 pixel[] pixels=xf.pixels[m];
646 for(pixel P : pixels)
647 pixel(P.point,material.p[0],P.width);
648 }
649 }
