1// A module for reading simple obj files with groups.
2// Authors: Jens Schwaiger and John Bowman
3//
4// Here simple means that :
5//
6// 1) all vertex statements should come before the face statements;
7//
8// 2) face informations with respect to texture and/or normal vectors are
9// ignored;
10//
11// 3) face statements only contain positive numbers(no relative positions).
12//
13// The reading process only takes into account lines starting with "v" or
14// "f" or "g"(group).

23 path3[][] read(string datafile, bool verbose=false) {

30 void Face(int[] vertnr, int groupnr) {
31 guide3 gh;
32 for(int i=0; i < vertnr.length; ++i)
33 gh=gh--vert[vertnr[i]-1];
34 gh=gh--cycle;
35 g[groupnr].push(gh);
36 }
37 if(verbose) write("Reading data from "+datafile+".");
38 int groupnr;
39 while(true) {
40 string[] str=in;
41 if(str.length == 0) break;
42 str=sequence(new string(int i) {return split(str[i],"/")[0];},str.length);
43 if(str[0] == "g" && str.length > 1) {
44 int tst=find(G == str[1]);
45 if(tst == -1) {
46 G.push(str[1]);
47 groupnr=G.length-1;
48 g[groupnr]=new path3[] ;
49 }
50 if(tst > -1) groupnr=tst;
51 }
52 if(str[0] == "v") Vertex((real) str[1],(real) str[2],(real) str[3]);
53 if(str[0] == "f") {
54 int[] vertnr;
55 for(int i=1; i < str.length; ++i) vertnr[i-1]=(int) str[i];
56 Face(vertnr,groupnr);
57 }
58 if(eof(in)) break;
59 }
60 close(in);
61 if(verbose) {
62 write("Number of groups: ",G.length);
63 write("Groups and their names:");
64 write(G);
65 write("Reading done.");
66 write("Number of faces contained in the groups: ");
67 for(int j=0; j < G.length; ++j)
68 write(G[j],": ",(string) g[j].length);
69 }
70 return g;
