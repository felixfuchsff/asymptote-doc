294// Search a sorted ordered array a of n elements for key, returning the index i
295// if a[i] <= key < a[i+1], -1 if key is less than all elements of a, or
296// n-1 if key is greater than or equal to the last element of a.
297template<class T>
298void searchArray(vm::stack *s)
299{
300 T key=pop<T>(s);
301 array *a=pop<array*>(s);
302 size_t size= a->size();
303 if(size == 0 || key < read<T>(a,0)) {s->push(-1); return;}
304 size_t u=size-1;
305 if(key >= read<T>(a,u)) {s->push((Int) u); return;}
306 size_t l=0;
307
308 while (l < u) {
309 size_t i=(l+u)/2;
310 if(key < read<T>(a,i)) u=i;
311 else if(key < read<T>(a,i+1)) {s->push((Int) i); return;}
312 else l=i+1;
313 }
314 s->push(0);
315}
