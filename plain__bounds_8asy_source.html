18// Adds another sizing restriction to the coordinates, but only if it is
19// maximal, that is, if under some scaling, this coordinate could be the
20// largest.
21private void addIfMaximal(coord[] coords, real user, real truesize) {
22 // TODO: Test promoting coordinates for efficiency.
23
24 for (coord c : coords)
25 if (user <= c.user && truesize <= c.truesize)
26 // Not maximal.
27 return;
28
29 // The coordinate is not dominated by any existing extreme, so it is
30 // maximal and will be added, but first remove any coords it now dominates.
31 int i=0;
32 while (i < coords.length) {
33 coord c=coords[i];
34 if (c.user <= user && c.truesize <= truesize)
35 coords.delete(i);
36 else
37 ++i;
38 }
39
40 // Add the coordinate to the extremes.
41 coords.push(coord.build(user, truesize));
42}

592 pair min(transform t) {

617 // Returns the transform for turning user-space pairs into true-space pairs.
618 transform scaling(real xsize, real ysize,
619 real xunitsize, real yunitsize,
620 bool keepAspect, bool warn) {
621 if(xsize == 0 && xunitsize == 0 && ysize == 0 && yunitsize == 0)
622 return identity();
623
624 // Get the extremal coordinates.
625 extremes e=extremes();
626
627 real sx;
628 if(xunitsize == 0) {
629 if(xsize != 0) sx=calculateScaling("x",e.left,e.right,xsize,warn);
630 } else sx=xunitsize;
631
632 /* Possible alternative code :
633 real sx=xunitsize != 0 ? xunitsize :
634 xsize != 0 ? calculateScaling("x", Coords.x, xsize, warn) :
635 0; */
636
637 real sy;
638 if(yunitsize == 0) {
639 if(ysize != 0) sy=calculateScaling("y",e.bottom,e.top,ysize,warn);
