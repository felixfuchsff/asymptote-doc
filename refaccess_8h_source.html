44template <class T>
45void pointerRead(vm::stack *s) {

50template <class T>
51void pointerWrite(vm::stack *s) {

58template <class T>
59void refAccess<T>::encode(action act, position, coder &e)
60{
61 // You may be able to use typeid(T).name() to get a better label.
62 REGISTER_BLTIN((bltin) pointerRead<T>, "pointerRead");
63 REGISTER_BLTIN((bltin) pointerWrite<T>, "pointerWrite");
64
65 e.encode(vm::inst::constpush, (vm::item)ref);
66
67 switch (act) {
68 case READ:
69 e.encode(vm::inst::builtin, (bltin) pointerRead<T>);
70 break;
71 case WRITE:
72 e.encode(vm::inst::builtin, (bltin) pointerWrite<T>);
73 break;
74 case CALL:
75 e.encode(vm::inst::builtin, (bltin) pointerRead<T>);
76 e.encode(vm::inst::popcall);
77 break;
78 };
79}
