88void storeExp(coenv &e, types::ty *t, exp *expr) {
89 assert(t->kind != ty_error);
90 assert(t->kind != ty_void);
91 assert(t->kind != ty_overloaded);
92
93 expr->transAsType(e, t);
94
95 // Store the value in a new variable of the proper type.
96 varEntry *v = makeVarEntry(expr->getPos(), e, 0, t);
97 e.e.addVar(symbol::trans("operator answer"), v);
98 v->getLocation()->encode(WRITE, expr->getPos(), e.c);
99 e.c.encodePop();
100}

110void tryToWriteExp(coenv &e, exp *expr)
111{
112 position pos=expr->getPos();
113 types::ty *t=expr->cgetType(e);
114
115 if(!t) return;
116
117 // If the original expression is bad, just print the errors.
118 // If it is a function which returns void, just call the function.
119 if (t->kind == ty_error || t->kind == ty_void) {
120 baseExpTrans(e, expr);
121 return;
122 }
123
124 exp *callee=new nameExp(pos, symbol::trans("write"));
125 exp *call=new callExp(pos, callee, expr);
126
127 types::ty *ct=call->getType(e);
128 if (ct->kind == ty_error || ct->kind == ty_overloaded) {
129 if (t->kind == ty_overloaded) {
130 // Translate the expr in order to print the ambiguity error first.
131 expr->trans(e);
132 em.sync(true);
133 assert(em.errors());
134
135 // Then, write out all of the types.
136 tryToWriteTypeOfExp(t, expr);
137 }
138 else {
139 // Write the type of the expression and, since it is unique, assign it to
140 // 'operator answer' even though its value isn't printed.
141 tryToWriteTypeOfExp(t, expr);
142 storeExp(e, t, expr);
143 }
144 }
145 else if (t->kind == ty_overloaded) {
146 // If the exp is overloaded, but the act of writing makes it
147 // unambiguous, add a suffix to the output to warn the user of this.
148 exp *suffix=new nameExp(pos,
149 symbol::trans("overloadedMessage"));
150 exp *callWithSuffix=new callExp(pos,
151 callee, expr, suffix);
152
153 if (callWithSuffix->getType(e)->kind != ty_error)
154 baseExpTrans(e, callWithSuffix);
155 else
156 baseExpTrans(e, call);
157 }
158 else {
159 // Interactive writing can proceed normally.
160 storeAndWriteExp(e, t, expr);
161 }
162}

211void transLoopBody(coenv &e, stm *body) {
212 // The semantics of the language are defined so that any variable declared
213 // inside a loop are new variables for each iteration of the loop. For
214 // instance, the code
215 //
216 // int f();
217 // for (int i = 0; i < 10; ++i) {
218 // int j=10*i;
219 // if (i == 5)
220 // f = new int() { return j; };
221 // }
222 // write(f());
223 //
224 // will write 50. This is implemented by allocating a new frame for each
225 // iteration. However, this can have a big performance hit, so we first
226 // translate the code without the frame, check if it needed the closure, and
227 // rewrite the code if necessary.
228
229 label start = e.c.defNewLabel();
230
231 // Encode a no-op, in case we need to jump over the default implementation
232 // to a special case.
233 e.c.encode(inst::nop);
234
235 body->markTrans(e);
236
237 // Don't re-translate if there were errors.
238 if (em.errors())
239 return;
240
241 if (e.c.usesClosureSinceLabel(start)){
242 // Jump over the old section.
243 label end = e.c.defNewLabel();
244 e.c.encodePatch(start, end);
245
246 // Let coder know that break and continue need to pop the frame.
247 e.c.loopPushesFrame();
248
249 e.c.encodePushFrame();
250 body->markTrans(e);
251 e.c.encodePopFrame();
252 }
253}

318void forStm::trans(coenv &e)
319{
320 // Any vardec in the initializer needs its own scope.
321 e.e.beginScope();
322 if (init)
323 init->markTrans(e);
324
325 label ctarget = e.c.fwdLabel();
326 label end = e.c.fwdLabel();
327 e.c.pushLoop(ctarget, end);
328
329 label start = e.c.defNewLabel();
330 if(test) {
331 test->transConditionalJump(e, false, end);
332 }
333
334 transLoopBody(e,body);
335
336 e.c.defLabel(ctarget);
337
338 if (update)

342 e.c.defLabel(end);

344 e.c.popLoop();

415extendedForStm::LoopType extendedForStm::transObjectDec(symbol a, coenv &e) {
416 // Get the start type. Handle type inference as a special case.
417 types::ty *t;
418 { // Suppress errors while probing the declared iteration variable type.
419 auto modeGuard = em.modeGuard(ErrorMode::SUPPRESS);
420 t = start->trans(e);
421 }
422 if (t->kind == types::ty_inferred) {
423 // First ensure the array expression is an unambiguous array.
424
425 types::ty *atOriginal = set->cgetType(e);
426 types::ty *at = atOriginal->signatureless();
427 if (at && at->kind == ty_array) {
428 // var a=set;
429 tyEntryTy tet(pos, at);
430 decid dec1(pos, new decidstart(pos, a), set);
431 vardec(pos, &tet, &dec1).trans(e);
432 return LoopType::ARRAY;
433 }
434 em.error(set->getPos());
435 if (atOriginal->isOverloaded()) {
436 em << "cannot resolve type for iteration";
437 } else {
438 em << "cannot iterate over expression of type '" << *atOriginal << "'";
439 }
440
441 // On failure, don't bother trying to translate the loop.
442 return LoopType::TY_ERROR;
443
444 }
445 // start[] a=set;
446 arrayTy at(pos, start, new dimensions(pos));
447 decid dec1(pos, new decidstart(pos, a), set);
448 vardec(pos, &at, &dec1).trans(e);
449 return LoopType::ARRAY;
450}

452void extendedForStm::trans(coenv &e) {
453 // Translate into the syntax:
454 //
455 // start[] a = set;
456 // for (int i=0; i < a.length; ++i) {
457 // start var=a[i];
458 // body
459 // }
460
461 position pos=getPos();
462
463 // Use gensyms for the variable names so as not to pollute the namespace.
464 symbol a=symbol::gensym("a");
465 symbol i=symbol::gensym("i");
466
467 if (set->cgetType(e)->isError()) {
468 // Translate the object for the error message.
469 set->trans(e);
470 // On failure, don't bother trying to translate the loop.
471 return;
472 }
473 // Is `set.operator iter()` a valid expression?
474 exp* iterExp=
475 new callExp(pos, new fieldExp(pos, set, symbol::opTrans("iter")));
476 LoopType loopType;
477 if (!iterExp->cgetType(e)->isError()) {
478 loopType = LoopType::ITERABLE;
479 } else {
480 loopType = transObjectDec(a, e);
481 }
482 // On failure, don't bother trying to translate the loop.
483 if (loopType == LoopType::TY_ERROR)
484 return;
485 exp *varInitExp = nullptr;
486 if (loopType == LoopType::ITERABLE) {
487 // start var=i.get();
488 varInitExp = new callExp(
489 pos,
490 new fieldExp(pos, new nameExp(pos, i), symbol::literalTrans("get"))
491 );
492 } else {
493 // start var=a[i];
494 varInitExp = new subscriptExp(pos, new nameExp(pos, a),
495 new nameExp(pos, i));
496 }
497 // { start var = <varInitExp>; body }
498 block b(pos);
499 decid dec2(pos, new decidstart(pos, var), varInitExp);
500 b.add(new vardec(pos, start, &dec2));
501 b.add(body);
502
503 if (loopType == LoopType::ARRAY) {
504 // for (int i=0; i < a.length; ++i)
505 // <block>
506 forStm(pos,
507 forArrayInit(pos, i),
508 forArrayTest(pos, i, a),
509 forArrayUpdate(pos, i),
510 new blockStm(pos, &b)).trans(e);
511 } else {
512 // for (var i=set.operator iter(); i.valid(); i.advance())
513 // <block>
514 forStm(pos,
515 forIterInit(pos, i, iterExp),
516 forIterTest(pos, i),
517 forIterUpdate(pos, i),
518 new blockStm(pos, &b)).trans(e);
519 }
520}

561void returnStm::trans(coenv &e)
562{
563 types::ty *t = e.c.getReturnType();
564
565 if (t->kind == ty_void) {
566 if (value) {
567 em.error(getPos());
568 em << "function cannot return a value";
569 }
570 if (e.c.isRecord())
571 e.c.encode(inst::pushclosure);
572 }
573 else {
574 if (value) {
575 value->transToType(e, t);
576 }
577 else {
578 em.error(getPos());
579 em << "function must return a value";
580 }
581 }
582
583 // NOTE: Currently, a return statement in a module definition will end
584 // the initializer. Should this be allowed?
585 e.c.encode(inst::ret);
586}
