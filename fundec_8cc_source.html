74void formals::addToSignature(signature& sig,
75 coenv &e, bool encodeDefVal)
76{
77 for (list<formal *>::iterator p = fields.begin(); p != fields.end(); ++p) {
78 formal& f=**p;
79 types::formal tf = f.trans(e, encodeDefVal);
80
81 if (f.isKeywordOnly())
82 sig.addKeywordOnly(tf);
83 else
84 sig.add(tf);
85 }
86
87 if (rest) {
88 if (rest->getDefaultValue()) {
89 em.error(rest->getPos());
90 em << "rest parameters cannot have default values";
91 }
92 sig.addRest(rest->trans(e, encodeDefVal));
93 }
94}

159void transDefault(coenv &e, position pos, varEntry *v, varinit *init) {
160 // This roughly translates into the statement
161 // if (isDefault(x))
162 // x=init;
163 // where x is the variable in v and isDefault is a function that tests
164 // whether x is the default argument token.
165
166 v->encode(READ, pos, e.c);
167
168 label end = e.c.fwdLabel();
169 e.c.useLabel(inst::jump_if_not_default, end);
170
171 init->transToType(e, v->getType());
172 v->encode(WRITE, pos, e.c);
173 e.c.encodePop();
174
175 e.c.defLabel(end);
176}

178void formal::transAsVar(coenv &e, Int index) {
179 symbol name = getName();
180 if (name) {
181 trans::access *a = e.c.accessFormal(index);
182 assert(a);
183
184 types::ty *t;
185 { // Suppress error messages because they will already be reported
186 // when the formals are translated to yield the type earlier.
187 auto modeGuard = em.modeGuard(ErrorMode::SUPPRESS);
188 t = getType(e);
189 }
190 varEntry *v = new varEntry(t, a, 0, getPos());
191
192 // Translate the default argument before adding the formal to the
193 // environment, consistent with the initializers of variables.
194 if (defval)
195 transDefault(e, getPos(), v, defval);
196
197 e.e.addVar(name, v);
198 }
199}

260void fundef::baseTrans(coenv &e, types::function *ft)
261{
262 string name = id ? string(id) : string("<anonymous function>");
263
264 // Create a new function environment.
265 coder fc = e.c.newFunction(getPos(), name, ft);
266 coenv fe(fc,e.e);
267
268 // Translate the function.
269 fe.e.beginScope();
270 params->trans(fe);
271
272 body->trans(fe);
273
274 types::ty *rt = ft->result;
275 if (rt->kind != ty_void &&
276 rt->kind != ty_error &&
277 !body->returns()) {
278 em.error(body->getPos());
279 em << "function must return a value";
280 }
281
282 fe.e.endScope();
283
284 // Put an instance of the new function on the stack.
285 vm::lambda *l = fe.c.close();
286 e.c.encode(inst::pushclosure);
287 e.c.encode(inst::makefunc, l);
288}

290types::ty *fundef::trans(coenv &e) {
291 // We must addOps for the return type and formals. ex:
292 //
293 // new guide[] (guide f(int)) {
294 // return sequence(f, 10);
295 // };
296 function *ft=transTypeAndAddOps(e, (record *)0);
297 assert(ft);
