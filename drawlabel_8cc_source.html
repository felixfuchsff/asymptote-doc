29// Reads one of the dimensions from the pipe.
30void texdim(iopipestream& tex, double& dest, const string command,
31 const string name)
32{
33 string start(">dim(");
34 string stop(")dim");
35 string expect("pt"+stop+"\n\n*");
36
37 // ask the tex engine for dimension
38 tex << "\\immediate\\write16{" << start << "\\the\\" << command << "\\ASYbox"
39 << stop << "}\n";
40 // keep reading output until ')dim\n\n*' is read
41 tex.wait(expect.c_str());
42 string buffer = tex.getbuffer();
43
44 size_t dim1=buffer.find(start);
45 size_t dim2=buffer.find("pt" + stop);
46 string cannotread="Cannot read label "+name;
47 if (dim1 != string::npos && dim2 != string::npos) {
48 string n=buffer.substr(dim1+start.size(),dim2-dim1-start.size());
49 try {
50 dest=lexical::cast<double>(n,true)*tex2ps;
51 } catch(lexical::bad_cast&) {
52 camp::reportError(cannotread);
53 }
54 } else {
55 camp::reportError(cannotread);
56 }
57}

93void drawLabel::getbounds(iopipestream& tex, const string& texengine)
94{
95 if(havebounds) return;
96 havebounds=true;
97
98 setpen(tex,texengine,pentype);
99 texbounds(width,height,depth,tex,label);
100
101 if(width == 0.0 && height == 0.0 && depth == 0.0 && !size.empty())
102 texbounds(width,height,depth,tex,size);
103
104 enabled=true;
105
106 Align=inverse(T)*align;
107 double scale0=max(fabs(Align.getx()),fabs(Align.gety()));
108 if(scale0) Align *= 0.5/scale0;
109 Align -= pair(0.5,0.5);
110 double Depth=(pentype.Baseline() == NOBASEALIGN) ? depth :
111 -depth*Align.gety();
112 texAlign=Align;
113 const double vertical=height+depth;
114 if(Depth > 0) texAlign += pair(0.0,Depth/vertical);
115 Align.scale(width,vertical);
116 Align += pair(0.0,Depth-depth);
117 Align=T*Align;
118}

120void drawLabel::bounds(bbox& b, iopipestream& tex, boxvector& labelbounds,
121 bboxlist&)
122{
123 string texengine=getSetting<string>("tex");
124 if(texengine == "none") {b += position; return;}
125
126 getbounds(tex,texengine);
127
128 // alignment point
129 pair p=position+Align;
130 const double vertical=height+depth;
131 const double fuzz=pentype.size()*0.1+0.3;
132 pair A=p+T*pair(-fuzz,-fuzz);
133 pair B=p+T*pair(-fuzz,vertical+fuzz);
134 pair C=p+T*pair(width+fuzz,vertical+fuzz);
135 pair D=p+T*pair(width+fuzz,-fuzz);
136
137 if(pentype.Overwrite() != ALLOW && label != "") {
138 size_t n=labelbounds.size();
139 box Box=box(A,B,C,D);
140 for(size_t i=0; i < n; i++) {
141 if(labelbounds[i].intersect(Box)) {
142 switch(pentype.Overwrite()) {
143 case SUPPRESS:
144 labelwarning("suppressed");
145 case SUPPRESSQUIET:
146 suppress=true;
147 return;
148 case MOVE:
149 labelwarning("moved");
150 default:
151 break;
152 }
153
154 pair Align=(align == pair(0,0)) ? unit(pair(urand(),urand())) :
155 unit(align);
156 double s=0.1*pentype.size();
157 double dx=0, dy=0;
158 if(Align.getx() > 0.1) dx=labelbounds[i].xmax()-Box.xmin()+s;
159 if(Align.getx() < -0.1) dx=labelbounds[i].xmin()-Box.xmax()-s;
160 if(Align.gety() > 0.1) dy=labelbounds[i].ymax()-Box.ymin()+s;
161 if(Align.gety() < -0.1) dy=labelbounds[i].ymin()-Box.ymax()-s;
162 pair offset=pair(dx,dy);
163 position += offset;
164 A += offset;
165 B += offset;
166 C += offset;
167 D += offset;
168 Box=box(A,B,C,D);
169 i=0;
170 }
171 }
172 labelbounds.resize(n+1);
173 labelbounds[n]=Box;
174 }
175
176 Box=bbox();
177 Box += A;
178 Box += B;
179 Box += C;
180 Box += D;
181
182 b += Box;
183}

206void drawLabelPath::bounds(bbox& b, iopipestream& tex, boxvector&, bboxlist&)
207{
208 string texengine=getSetting<string>("tex");
209 if(texengine == "none") {b += position; return;}
210
211 getbounds(tex,texengine);
212 double L=p.arclength();
213
214 double s1,s2;
215 if(justify == "l") {
216 s1=0.0;
217 s2=width;
218 } else if(justify == "r") {
219 s1=L-width;
220 s2=L;
221 } else {
222 double s=0.5*L;
223 double h=0.5*width;
224 s1=s-h;
225 s2=s+h;
226 }
227
228 double Sx=shift.getx();
229 double Sy=shift.gety();
230 s1 += Sx;
231 s2 += Sx;
232
233 if(width > L || (!p.cyclic() && (s1 < 0 || s2 > L))) {
234 ostringstream buf;
235 buf << "Cannot fit label \"" << label << "\" to path";
236 reportError(buf);
237 }
238
239 path q=p.subpath(p.arctime(s1),p.arctime(s2));
240
241 b += q.bounds(Sy,Sy+height);
242 Box=b;
243}

246{
247 double Hoffset=getSetting<bool>("inlinetex") ? Box.right : Box.left;
248 Box=Box.shift(pair(-Hoffset,-Box.bottom));
