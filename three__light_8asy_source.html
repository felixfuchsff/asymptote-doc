93 triple[] position=light.position;
94 if(invisible((pen) m)) return invisible;
95 if(position.length == 0) return m.diffuse();
96 normal=unit(transpose(inverse(shiftless(T)))*normal);
97 if(settings.twosided) normal *= sgn(normal.z);
98 real s=m.shininess*128;
99 real[] Diffuse=rgba(m.diffuse());
100 real[] Specular=rgba(m.specular());
101 real[] p=rgba(m.emissive());
102 real[] diffuse={0,0,0,0};
103 real[] specular={0,0,0,0};
104 for(int i=0; i < position.length; ++i) {
105 triple L=position[i];
106 real dotproduct=abs(dot(normal,L));
107 diffuse += dotproduct*light.diffuse[i];
108 dotproduct=abs(dot(normal,unit(L+Z)));
109 // Phong-Blinn model of specular reflection
110 specular += dotproduct^s*light.specular[i];
111 }
112 p += diffuse*Diffuse;
113 // Apply specularfactor to partially compensate non-pixel-based rendering.
114 p += specular*Specular*light.specularfactor;
115 return rgb(p[0],p[1],p[2])+opacity(opacity(m.diffuse()));
116}
