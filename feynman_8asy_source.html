123// internal function for overpainting
124private void do_overpaint(picture pic, path p, pen bgpen,
125 real halfwidth, real vertexangle)
126{
127 real tanvertexangle = tan(vertexangle*pi/180);
128 if(tanvertexangle != 0) {
129 real t1 = arctime(p, halfwidth/tanvertexangle+halfwidth);
130 real t2 = arctime(p, arclength(p)-halfwidth/tanvertexangle-halfwidth);
131 draw(pic, subpath(p, t1, t2),
132 bgpen+linewidth(2*halfwidth));
133 }
134}

136// returns the path of a gluon line along path p, with amplitude amp and width
137// width (distance between two loops). If width is negative, the width is
138// set to amp*gluonratio
139path gluon(path p, real amp = gluonamplitude, real width=-1)
140{
141 if(width < 0) width = abs(gluonratio*amp);
142
143 real pathlen = arclength(p);
144 int ncurls = floor(pathlen/width);
145 real firstlen = (pathlen - width*(ncurls-1))/2;
146 real firstt = arctime(p, firstlen);
147 pair firstv = dir(p, firstt);
148 guide g = point(p, 0)..{firstv}( point(p, firstt)
149 +amp*unit(rotate(90)*firstv));
150
151 real t1;
152 pair v1;
153 real t2;
154 pair v2;
155 pathlen -= firstlen;
156 for(real len = firstlen+width/2; len < pathlen; len += width) {
157 t1 = arctime(p, len);
158 v1 = dir(p, t1);
159 t2 = arctime(p, len + width/2);
160 v2 = dir(p, t2);
161
162 g=g..{-v1}(point(p, t1)+amp*unit(rotate(-90)*v1))
163 ..{+v2}(point(p, t2)+amp*unit(rotate(+90)*v2));
164 }
165 g = g..point(p, size(p));
166 return g;
167}
168
169// returns the path of a photon line along path p, with amplitude amp and width
170// width (distance between two crests). If width is negative, the width is
171// set to amp*photonratio
172path photon(path p, real amp = photonamplitude, real width=-1)
173{
174 if(width < 0)
175 width = abs(photonratio*amp)/2;
176 else
177 width = width/2;
178
179 real pathlen = arclength(p);
180 int ncurls = floor(pathlen/width);
181 real firstlen = (pathlen - width*ncurls)/2;
182 real firstt = arctime(p, firstlen+width);
183 guide g = point(p, 0){unit(point(p, firstt)-point(p, 0))};
184
185 real t;
186 pair v;
187 pathlen -= firstlen;
188 for(real len = firstlen+width; len < pathlen; len += width) {
189 t = arctime(p, len);
190 v = dir(p, t);
191
192 g=g..{v}(point(p, t)+amp*unit(rotate(90)*v));
193 amp = -amp;
194 }
195 g = g..{unit(point(p, size(p))-point(p, t))}point(p, size(p));
196 return g;
197}

199// returns the path of a momentum arrow along path p, with length length,
200// an offset offset from the path p and at position position. position will
201// usually be one of the predefined pairs left or right. Making adjust
202// nonzero shifts the momentum arrow along the path.
203path momArrowPath(path p,
204 align align,
205 position pos,
206 real offset = momarrowoffset,
207 real length = momarrowlength)
208{
209 real pathlen = arclength(p);
210
211 real t1, t2;
212 if(pos.relative) {
213 t1 = arctime(p, (pathlen-length)*pos.position.x);
214 t2 = arctime(p, (pathlen-length)*pos.position.x+length);
215 } else {
216 t1 = arctime(p, (pathlen-length)/2 + pos.position.x);
217 t2 = arctime(p, (pathlen+length)/2+ pos.position.x);
218 }
219
220 pair v1 = dir(p, t1);
221 pair v2 = dir(p, t2);
222
223 pair p1, p2;
224 if(align.relative) {
225 p1 = point(p, t1) + offset*abs(align.dir)
226 *unit(rotate(degrees(align.dir)-90)*v1);
227 p2 = point(p, t2) + offset*abs(align.dir)
228 *unit(rotate(degrees(align.dir)-90)*v2);
229 } else {
230 p1 = point(p, t1) + offset*align.dir;
231 p2 = point(p, t2) + offset*align.dir;
232 }
233
234 return p1{v1}..{v2}p2;
235}

243// width (distance between loops) and with pen fgpen. If erasebg is true,
244// bgpen is used to erase the background behind the line and at a margin
245// margin around it. The background is not erased at a certain distance to

247// default parameter minvertexangle). For negative values of width, the width
248// is set to gluonratio*amp.

594// texshipout("filename", pic) creates two files: filename.eps holding the
595// picture pic and filename.tex holding some LaTeX code that includes the
596// picture from filename.eps and shifts it vertically in such a way that the
597// point (0,0) lies on the baseline.
598void texshipout(string stem,
599 picture pic = currentpicture,
600 bool xalign = YAlign)
601{
602 file tf = output(stem + ".tex");
603 pair min=pic.min();
604 real depth = min.y;
605 real xoffset = min.x;
606 if(xalign) {
607 write(tf, "\makebox[0pt][l]{\kern");
608 write(tf, xoffset);
609 write(tf, "bp\relax");
610 }
611 write(tf, "\raisebox{");
612 write(tf, depth);
613 write(tf, "bp}{"+includegraphicscommand+"{");
614 write(tf, stem);
615 string suffix="."+nativeformat();
616 if(appendsuffix)
617 write(tf, suffix);
618 write(tf, "}}");
619 if(xalign)
620 write(tf, "}");
621 close(tf);
622 shipout(stem+suffix, pic);
623}
