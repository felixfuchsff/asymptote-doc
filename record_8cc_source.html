50mem::pair<ty*, ty*> computeKVTypes(
51 trans::venv& ve, const position& pos
52)
53{
54 mem::pair<ty*, ty*> errorPair(primError(), primError());
55
56 // TODO: Make the lookup more efficient. (See DEFSYMBOL in camp.l.)
57 const symbol SYM_BRACKETS= symbol::trans("[]");
58 const symbol SYM_BRACKETS_ASSIGN= symbol::trans("[=]");
59 ty* getTy= ve.getType(SYM_BRACKETS);
60 ty* setTy= ve.getType(SYM_BRACKETS_ASSIGN);
61 if (getTy == nullptr) {
62 if (setTy != nullptr) {
63 em.error(pos);
64 em << "operator[=] defined without operator[]";
65 }
66 return errorPair;
67 }
68
69 // Find the keytype and valuetype based on operator[].
70 if (getTy->isOverloaded()) {
71 em.error(pos);
72 em << "multiple operator[] definitions in one struct";
73 return errorPair;
74 }
75 if (getTy->kind != ty_function) {
76 em.error(pos);
77 em << "operator[] is not a function";
78 return errorPair;
79 }
80 types::function* get= static_cast<types::function*>(getTy);
81 types::ty* valTy= get->getResult();
82 signature* getSig= get->getSignature();
83 // TODO: Can we get the position of the definition of operator[] rather than
84 // the end of the struct?
85 if (getSig->hasRest() || getSig->getNumFormals() != 1) {
86 em.error(pos);
87 em << "operator[] must have exactly one parameter";
88 return errorPair;
89 }
90 ty* keyTy= getSig->getFormal(0).t;
91
92 if (setTy != nullptr) {
93 // Find the keytype and valuetype based on operator[=].
94 if (setTy->isOverloaded()) {
95 em.error(pos);
96 em << "multiple operator[=] definitions in one struct";
97 return errorPair;
98 }
99 if (setTy->kind != ty_function) {
100 em.error(pos);
101 em << "operator[=] is not a function";
102 return errorPair;
103 }
104 types::function* set= static_cast<types::function*>(setTy);
105 types::ty* setResult= set->getResult();
106 if (setResult->kind != ty_void) {
107 em.error(pos);
108 em << "operator[=] must return void";
109 return errorPair;
110 }
111 signature* setSig= set->getSignature();
112 if (setSig->hasRest() || setSig->getNumFormals() != 2) {
113 em.error(pos);
114 em << "operator[=] must have exactly two parameters";
115 return errorPair;
116 }
117 ty* setKeyTy= setSig->getFormal(0).t;
118 ty* setValTy= setSig->getFormal(1).t;
119
120 // Check that they agree.
121 if (!keyTy->equiv(setKeyTy) || !setKeyTy->equiv(keyTy)) {
122 em.error(pos);
123 em << "first parameter of operator[] and operator[=] must match";
124 return errorPair;
125 }
126 if (!valTy->equiv(setValTy) || !setValTy->equiv(valTy)) {
127 em.error(pos);
128 em << "return type of operator[] and second parameter of operator[=] "
129 "must match";
130 return errorPair;
131 }
132 }
133
134 return mem::pair<ty*, ty*>(keyTy, valTy);
135}
