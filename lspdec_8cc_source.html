165DEC_CREATE_SYM_MAP_FUNCTION_DEF(idpair)
166{
167 if (valid) {
168 string fullSrc(settings::locateFile(src, true));
169 if (not AsymptoteLsp::isVirtualFile((std::string) (fullSrc.c_str()))) {
170 if (not fullSrc.empty()) {
171 symContext->addEmptyExtRef((std::string) (fullSrc.c_str()));
172 }
173
174 // add (dest, source) to reference map.
175 auto s= symContext->extRefs.fileIdPair.emplace(
176 dest, (std::string) fullSrc.c_str()
177 );
178 auto it= std::get<0>(s);
179 auto success= std::get<1>(s);
180 if (not success) {
181 it->second= (std::string) (fullSrc.c_str());
182 }
183
184 symContext->extRefs.addAccessVal(static_cast<std::string>(dest));
185 }
186 }
