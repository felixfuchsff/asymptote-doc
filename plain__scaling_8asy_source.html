197 Solve the two-variable linear programming problem using the simplex method.
198 This problem is specialized in that the second variable, "b", does not have
199 a non-negativity condition, and the first variable, "a", is the quantity
200 being maximized.
201*/

206
207 void addMinCoord(coord c) {
208 // (a*user + b) + truesize >= 0:
209 p.addRestriction(c.user,1,c.truesize);
210 }
211 void addMaxCoord(coord c) {
212 // (a*user + b) + truesize <= size:
213 p.addRestriction(-c.user,-1,size-c.truesize);
214 }
215

217 addMinCoord(m[i]);
218 for (int i=0; i < M.length; ++i)
219 addMaxCoord(M[i]);

230 if(!warn) return 1;
