180/**
181 * Use key to look up an entry in the MSWindows registry,
182 * @param baseRegLocation base location for a key
183 * @param key Key to look up, respecting wild cards. Note that wildcards
184 * only support single-level glob. Recursive globs are not supported.
185 * @param value Value to look up
186 * @remark Wildcards can only be in keys, not in the final value
187 * @return Entry value, or nullopt if not found
188 */
189optional<string>
190getEntry(HKEY const& baseRegLocation, string const& key, string const& value)
191{
192 string path= key;
193 if (key.find('\\') == 0) {
194 path= path.substr(1);// strip the prefix separator
195 }
196
197 size_t const star= path.find('*');
198 if (star == string::npos) {
199 // absolute path, can return right away
200 DWORD dataSize= 0;
201 if (RegGetValueA(
202 baseRegLocation, path.c_str(), value.c_str(), RRF_RT_REG_SZ,
203 nullptr, nullptr, &dataSize
204 ) != ERROR_SUCCESS) {
205 return nullopt;
206 }
207
208 mem::vector<BYTE> outputBuffer(dataSize);
209
210 if (RegGetValueA(
211 baseRegLocation, path.c_str(), value.c_str(), RRF_RT_REG_SZ,
212 nullptr, outputBuffer.data(), &dataSize
213 ) != ERROR_SUCCESS) {
214 return nullopt;
215 }
216
217 return make_optional<string>(
218 reinterpret_cast<char const*>(outputBuffer.data())
219 );
220 }
221
222 // has a glob, search until we find one
223 string const prefix= path.substr(0, star);
224 string const pathSuffix= path.substr(star + 1);
225
226 // open the key in prefix
227
228 camp::w32::RegKeyWrapper directoryWithPrefix;
229 if (RegOpenKeyExA(
230 baseRegLocation, prefix.c_str(), 0, KEY_READ,
231 directoryWithPrefix.put()
232 ) != ERROR_SUCCESS) {
233 return nullopt;// prefix path does not exist, or some other error
234 }
235
236 DWORD numSubKeys= 0;
237 DWORD longestSubkeySize= 0;
238
239 // querying # of subkeys + their longest path length
240 if (RegQueryInfoKeyA(
241 directoryWithPrefix.getKey(), nullptr, nullptr, nullptr,
242 &numSubKeys, &longestSubkeySize, nullptr, nullptr, nullptr,
243 nullptr, nullptr, nullptr
244 ) != ERROR_SUCCESS) {
245 return nullopt;
246 }
247
248 mem::vector<CHAR> subkeyBuffer(longestSubkeySize + 1);
249
250 for (DWORD i= 0; i < numSubKeys; ++i) {
251 DWORD cchValue= longestSubkeySize + 1;
252
253 // get subkey's name
254 if (RegEnumKeyExA(
255 directoryWithPrefix.getKey(), i, subkeyBuffer.data(), &cchValue,
256 nullptr, nullptr, nullptr, nullptr
257 ) != ERROR_SUCCESS) {
258 continue;
259 }
260
261 // open the subkey
262 camp::w32::RegKeyWrapper searchKey;
263 if (RegOpenKeyExA(
264 directoryWithPrefix.getKey(), subkeyBuffer.data(), 0, KEY_READ,
265 searchKey.put()
266 ) != ERROR_SUCCESS) {
267 continue;
268 }
269
270 // do a recursive search starting at the opened key
271 if (auto retResult= getEntry(searchKey.getKey(), pathSuffix, value);
272 retResult.has_value()) {
273 return retResult;
274 }
275 }
276 return nullopt;
277}

279// Use key to look up an entry in the MSWindows registry, respecting wild cards
280string getEntry(const string& key, const string& value)
281{
282 for (HKEY const keyToSearch : {HKEY_LOCAL_MACHINE, HKEY_CURRENT_USER}) {
283 camp::w32::RegKeyWrapper baseRegKey;
284 if (RegOpenKeyExA(keyToSearch, "SOFTWARE", 0, KEY_READ, baseRegKey.put()) !=
285 ERROR_SUCCESS) {
286 baseRegKey.release();
287 continue;
288 }
289 optional<string> entry= getEntry(baseRegKey.getKey(), key, value);
290
291 if (entry.has_value()) {
292 return entry.value();
293 }
294 }
295
296 return "";

409struct option : public gc {
410 string name;
411 char code; // Command line option, i.e. 'V' for -V.
412 bool argument; // If it takes an argument on the command line. This is set
413 // based on whether argname is empty.
414 string argname; // The argument name for printing the description.
415 string desc; // One line description of what the option does.
416 bool cmdlineonly; // If it is only available on the command line.
417 string Default; // A string containing an optional default value.
418
419 option(string name, char code, string argname, string desc,
420 bool cmdlineonly=false, string Default="")
421 : name(name), code(code), argument(!argname.empty()), argname(argname),
422 desc(desc), cmdlineonly(cmdlineonly), Default(Default) {}
423
424 virtual ~option() {}
425
426 // Builds this option's contribution to the optstring argument of get_opt().
427 virtual string optstring() {
428 if (code) {
429 string base;
430 base.push_back(code);
431 if(argument) base.push_back(':');
432 return base;
433 }
434 else return "";
435 }
436
437 // Sets the contribution to the longopt array.
438 virtual void longopt(c_option &o) {
439 o.name=name.c_str();
440 o.has_arg=argument ? 1 : 0;
441 o.flag=0;
442 o.val=0;
443 }
444
445 // Add to the dictionaries of options.
446 virtual void add() {
447 optionsMap[name]=this;
448 if (code)
449 codeMap[code]=this;
450 }
451
452 // Set the option from the command-line argument. Return true if the option
453 // was correctly parsed.
454 virtual bool getOption() = 0;
455
456 void error(string msg) {
457 cerr << endl << argv0 << ": ";
458 if (code)
459 cerr << "-" << code << " ";
460 cerr << "(-" << name << ") " << msg << endl;
461 }
462
463 // The "-f,-outformat format" part of the option.
464 virtual string describeStart() {
465 ostringstream ss;
466 if (code)
467 ss << "-" << code << ",";
468 ss << "-" << name;
469 if (argument)
470 ss << " " << argname;
471 return ss.str();
472 }
473
474 // Outputs description of the command for the -help option.
475 virtual void describe(char option) {
476 // Don't show the option if it has no description.
477 if(!hide() && ((option == 'h') ^ env())) {
478 const unsigned WIDTH=22;
479 string start=describeStart();
480 cerr << std::left << std::setw(WIDTH) << start;
481 if (start.size() >= WIDTH) {
482 cerr << endl;
483 cerr << std::left << std::setw(WIDTH) << "";
484 }
485 cerr << " " << desc;
486 if(cmdlineonly) cerr << "; command-line only";
487 if(Default != "") {
488 if(!desc.empty()) cerr << " ";
489 cerr << Default;
490 }
491 cerr << endl;
492 }
493 }
494
495 virtual void reset() {
496 }
497
498 virtual bool env() {return false;}
499 virtual bool hide() {return false;}

1516{
1517 bool syntax=false;
1518 optind=0;
1519
1520 string optstring=build_optstring();
1521 //cerr << "optstring: " << optstring << endl;
1522 c_option *longopts=build_longopts();
1523 int long_index = 0;
1524
1525 errno=0;
1526 for(;;) {
1527 int c = getopt_long_only(argc,argv,
1528 optstring.c_str(), longopts, &long_index);
1529 if (c == -1)
1530 break;
1531
1532 if (c == 0) {
1533 const char *name=longopts[long_index].name;
1534 //cerr << "long option: " << name << endl;
1535 if (!optionsMap[name]->getOption())
1536 syntax=true;
1537 }
1538 else if (codeMap.find((char)c) != codeMap.end()) {
1539 //cerr << "char option: " << (char)c << endl;
1540 if (!codeMap[(char)c]->getOption())
1541 syntax=true;
1542 }
1543 else {
1544 syntax=true;
1545 }
1546
1547 if (showVersion) {
1548 // Don't exit yet — continue parsing remaining options
1549 }
1550
1551 errno=0;
1552 }
1553
1554 if (showVersion) {
1555 // Don't exit yet — continue parsing remaining options, then exit
1556 // from setOptions() after setPath() has been called so that the
1557 // renderer can locate its shared libraries.
1558 }
1559
1560 if (syntax)
1561 reportSyntax();

1994string lookup(const string& symbol)
1995{
1996 string s;
1997 mem::vector<string> cmd;
1998 string kpsewhich="kpsewhich";
1999 string fullname=stripFile(argv0)+kpsewhich;
2000 std::ifstream exists(fullname.c_str());
2001 if(!exists) fullname=kpsewhich;
2002 cmd.push_back(fullname);
2003 cmd.push_back("--var-value="+symbol);
2004 iopipestream pipe(cmd);
2005 pipe >> s;
2006 size_t n=s.find('\r');
2007 if(n != string::npos)
2008 s.erase(n,1);
2009 n=s.find('\n');
2010 if(n != string::npos)
2011 s.erase(n,1);
2012 return s;
2013}
