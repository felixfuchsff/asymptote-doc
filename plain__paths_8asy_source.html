250// return a cyclic path enclosing a region bounded by a list of two or more
251// consecutively intersecting paths
252path buildcycle(... path[] p)
253{
254 int n=p.length;
255 if(n < 2) return nullpath;
256 real[] ta=new real[n];
257 real[] tb=new real[n];
258 if(n == 2) {
259 real[][] t=intersections(p[0],p[1]);
260 if(t.length < 2)
261 return nullpath;
262 int k=t.length-1;
263 ta[0]=t[0][0]; tb[0]=t[k][0];
264 ta[1]=t[k][1]; tb[1]=t[0][1];
265 } else {
266 int j=n-1;
267 for(int i=0; i < n; ++i) {
268 real[][] t=intersections(p[i],p[j]);
269 if(t.length == 0)
270 return nullpath;
271 ta[i]=t[0][0]; tb[j]=t[0][1];
272 j=i;
273 }
274 }
275
276 pair c;
277 for(int i=0; i < n ; ++i)
278 c += point(p[i],ta[i]);
279 c /= n;
280
281 path G;
282 for(int i=0; i < n ; ++i) {
283 real Ta=ta[i];
284 real Tb=tb[i];
285 if(cyclic(p[i])) {
286 int L=length(p[i]);
287 real t=Tb-L;
288 if(abs(c-point(p[i],0.5(Ta+t))) <
289 abs(c-point(p[i],0.5(Ta+Tb)))) Tb=t;
290 while(Tb < Ta) Tb += L;
291 }
292 G=G&subpath(p[i],Ta,Tb);
293 }
294 return G&cycle;
295}

308// Return an arbitrary point strictly inside a cyclic path p according to
309// the specified fill rule.
310pair inside(path p, pen fillrule=currentpen)
311{
312 if(!cyclic(p)) abort("path is not cyclic");
313 int n=length(p);
314 for(int i=0; i < n; ++i) {
315 pair z=point(p,i);
316 pair dir=dir(p,i);
317 if(dir == 0) continue;
318 real[] T=intersections(p,z,z+I*dir);
319 // Check midpoints of line segments formed between the
320 // corresponding intersection points and z.
321 for(int j=0; j < T.length; ++j) {
322 if(T[j] != i) {
323 pair w=point(p,T[j]);
324 pair m=0.5*(z+w);
325 if(interior(windingnumber(p,m),fillrule)) return m;
326 }
327 }
328 }
329 // cannot find an interior point: path is degenerate
330 return point(p,0);
331}
