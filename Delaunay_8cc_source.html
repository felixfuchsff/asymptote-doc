33///////////////////////////////////////////////////////////////////////////////
34// Triangulate():
35// Triangulation subroutine
36// Takes as input NV vertices in array pxyz
37// Returned is a list of ntri triangular faces in the array v
38// These triangles are arranged in a consistent clockwise order.
39// The triangle array v should be allocated to 4 * nv
40// The vertex array pxyz must be big enough to hold 3 additional points.
41// By default, the array pxyz is automatically presorted and postsorted.
42///////////////////////////////////////////////////////////////////////////////
43
44Int Triangulate(Int nv, XYZ pxyz[], ITRIANGLE v[], Int &ntri,
45 bool presort, bool postsort)
46{
47 Int emax = 200;
48
49 if(presort) qsort(pxyz,nv,sizeof(XYZ),XYZCompare);
50 else postsort=false;
51
52/* Allocate memory for the completeness list, flag for each triangle */
53 Int trimax = 4 * nv;
54 Int *complete = new Int[trimax];
55/* Allocate memory for the edge list */
56 IEDGE *edges = new IEDGE[emax];
57/*
58 Find the maximum and minimum vertex bounds.
59 This is to allow calculation of the bounding triangle
60*/
61 double xmin = pxyz[0].p[0];
62 double ymin = pxyz[0].p[1];
63 double xmax = xmin;
64 double ymax = ymin;
65 for(Int i = 1; i < nv; i++) {
66 XYZ *pxyzi=pxyz+i;
67 double x=pxyzi->p[0];
68 double y=pxyzi->p[1];
69 if (x < xmin) xmin = x;
70 if (x > xmax) xmax = x;
71 if (y < ymin) ymin = y;
72 if (y > ymax) ymax = y;
73 }
74 double dx = xmax - xmin;
75 double dy = ymax - ymin;
76/*
77 Set up the supertriangle.
78 This is a triangle which encompasses all the sample points.
79 The supertriangle coordinates are added to the end of the
80 vertex list. The supertriangle is the first triangle in
81 the triangle list.
82*/
83 static const double margin=0.01;
84 double xmargin=margin*dx;
85 double ymargin=margin*dy;
86 pxyz[nv+0].p[0] = xmin-xmargin;
87 pxyz[nv+0].p[1] = ymin-ymargin;
88 pxyz[nv+1].p[0] = xmin-xmargin;
89 pxyz[nv+1].p[1] = ymax+ymargin+dx;
90 pxyz[nv+2].p[0] = xmax+xmargin+dy;
91 pxyz[nv+2].p[1] = ymin-ymargin;
92 v->p1 = nv;
93 v->p2 = nv+1;
94 v->p3 = nv+2;
95 complete[0] = false;
96 ntri = 1;
97/*
98 Include each point one at a time into the existing mesh
99*/
100 for(Int i = 0; i < nv; i++) {
101 Int nedge = 0;
102 double *d=pxyz[i].p;
103/*
104 Set up the edge buffer.
105 If the point d lies inside the circumcircle then the
106 three edges of that triangle are added to the edge buffer
107 and that triangle is removed.
108*/
109 for(Int j = 0; j < ntri; j++) {
110 if(complete[j])
111 continue;
112 ITRIANGLE *vj=v+j;
113
114 double *a=pxyz[vj->p1].p;
115 double *b=pxyz[vj->p2].p;
116 double *c=pxyz[vj->p3].p;
117
118 if(incircle(a,b,c,d) <= 0.0) { // Point d is inside or on circumcircle
119/* Check that we haven't exceeded the edge list size */
120 if(nedge + 3 >= emax) {
121 emax += 100;
122 IEDGE *p_EdgeTemp = new IEDGE[emax];
123 for (Int i = 0; i < nedge; i++) {
124 p_EdgeTemp[i] = edges[i];
125 }
126 delete[] edges;
127 edges = p_EdgeTemp;
128 }
129 ITRIANGLE *vj=v+j;
130 Int p1=vj->p1;
131 Int p2=vj->p2;
132 Int p3=vj->p3;
133 edges[nedge].p1 = p1;
134 edges[nedge].p2 = p2;
135 edges[++nedge].p1 = p2;
136 edges[nedge].p2 = p3;
137 edges[++nedge].p1 = p3;
138 edges[nedge].p2 = p1;
139 ++nedge;
140 ntri--;
141 v[j] = v[ntri];
142 complete[j] = complete[ntri];
143 j--;
144 } else {
145 double A=hypot2(a);
146 double B=hypot2(b);
147 double C=hypot2(c);
148 double a0=orient2d(a,b,c);
149 // Is d[0] > xc+r for circumcircle abc of radius r about (xc,yc)?
150 if(d[0]*a0 < 0.5*orient2d(A,a[1],B,b[1],C,c[1]))
151 complete[j]=
152 incircle(a[0]*a0,a[1]*a0,b[0]*a0,b[1]*a0,c[0]*a0,c[1]*a0,
153 d[0]*a0,0.5*orient2d(a[0],A,b[0],B,c[0],C)) > 0.0;
154 }
155 }
156/*
157 Tag multiple edges
158 Note: if all triangles are specified anticlockwise then all
159 interior edges are opposite pointing in direction.
160*/
161 for(Int j = 0; j < nedge - 1; j++) {
162 for(Int k = j + 1; k < nedge; k++) {
163 if((edges[j].p1 == edges[k].p2) && (edges[j].p2 == edges[k].p1)) {
164 edges[j].p1 = -1;
165 edges[j].p2 = -1;
166 edges[k].p1 = -1;
167 edges[k].p2 = -1;
168 }
169 }
170 }
171/*
172 Form new triangles for the current point
173 Skipping over any tagged edges.
174 All edges are arranged in clockwise order.
175*/
176 for(Int j = 0; j < nedge; j++) {
177 if(edges[j].p1 < 0 || edges[j].p2 < 0)
178 continue;
179 v[ntri].p1 = edges[j].p1;
180 v[ntri].p2 = edges[j].p2;
181 v[ntri].p3 = i;
182 complete[ntri] = false;
183 ntri++;
184 assert(ntri < trimax);
185 }
186 }
187/*
188 Remove triangles with supertriangle vertices
189 These are triangles which have a vertex number greater than nv
190*/
191 for(Int i = 0; i < ntri; i++) {
192 ITRIANGLE *vi=v+i;
193 if(vi->p1 >= nv || vi->p2 >= nv || vi->p3 >= nv) {
194 ntri--;
195 *vi = v[ntri];
196 i--;
197 }
198 }
199 delete[] edges;
200 delete[] complete;
201
202 if(postsort) {
203 for(Int i = 0; i < ntri; i++) {
204 ITRIANGLE *vi=v+i;
205 vi->p1=pxyz[vi->p1].i;
206 vi->p2=pxyz[vi->p2].i;
207 vi->p3=pxyz[vi->p3].i;
208 }
209 }
210
211 return 0;
212}
