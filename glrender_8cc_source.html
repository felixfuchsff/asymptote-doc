76camp::GLTexture3<float,GL_FLOAT> fromEXR3(
77 mem::vector<string> const& EXRFiles, camp::GLTexturesFmt const& fmt, GLint const& textureNumber)
78{
79 // 3d reflectance textures
80 std::vector<float> data;
81 size_t count=EXRFiles.size();
82 int wi=0, ht=0;
83
84 for(string const& EXRFile : EXRFiles) {
85 IEXRFile fil3(EXRFile);
86 std::tie(wi,ht)=fil3.size();
87 size_t imSize=4*wi*ht;
88 std::copy(fil3.getData(),fil3.getData()+imSize,std::back_inserter(data));
89 }
90
91 return camp::GLTexture3<float,GL_FLOAT> {
92 data.data(),
93 std::tuple<int,int,int>(wi,ht,static_cast<int>(count)),textureNumber,
94 fmt
95 };
96}

136{
137 string sum1=locateFile(SHADERS+"sum1.glsl");
138 string sum2=locateFile(SHADERS+"sum2.glsl");
139 string sum3=locateFile(SHADERS+"sum3.glsl");
140
141 if(sum1.empty() || sum2.empty() || sum3.empty())
142 noShaders();
143
144 std::vector<ShaderfileModePair> shaders(1);
145 std::vector<std::string> shaderParams;
146
147 shaders[0]=ShaderfileModePair(sum1.c_str(),GL_COMPUTE_SHADER);

149 s << "LOCALSIZE " << localSize << "u" << endl;

159// glGetIntegeri_v(GL_MAX_COMPUTE_WORK_GROUP_COUNT,0,&maxgroups);
160// maxgroups=min(1024,maxgroups/(GLint) (localSize*blockSize));
161 sum1Shader=rc;
162
163 shaders[0]=ShaderfileModePair(sum2.c_str(),GL_COMPUTE_SHADER);
164 sum2Shader=compileAndLinkShader(shaders,shaderParams,true,false,
165 true);
166
167 shaders[0]=ShaderfileModePair(sum3.c_str(),GL_COMPUTE_SHADER);
168 sum3Shader=compileAndLinkShader(shaders,shaderParams,true,false,

227void AsyGLRender::initShaders()
228{
229 Nlights = nlights == 0 ? 0 : std::max(Nlights, nlights);
230 nmaterials = materials.size();
231
232 string zero=locateFile(SHADERS+"zero.glsl");
233 string compress=locateFile(SHADERS+"compress.glsl");
234 string vertex=locateFile(SHADERS+"vertex.glsl");
235 string count=locateFile(SHADERS+"count.glsl");
236 string fragment=locateFile(SHADERS+"fragment.glsl");
237 string screen=locateFile(SHADERS+"screen.glsl");
238
239 if(zero.empty() || compress.empty() || vertex.empty() || fragment.empty() ||
240 screen.empty() || count.empty())
241 noShaders();
242
243 // Only try compute shaders if GPUindexing is explicitly enabled
244 if(GPUindexing) {
245 initComputeShaders();
246 }
247
248 std::vector<ShaderfileModePair> shaders(2);
249 std::vector<std::string> shaderParams;
250
251 if(ibl) {
252 shaderParams.push_back("USE_IBL");
253 initIBL();
254 }
255
256 shaders[0]=ShaderfileModePair(vertex.c_str(),GL_VERTEX_SHADER);
257
258#ifdef HAVE_SSBO
259 if(GPUindexing)
260 shaderParams.push_back("GPUINDEXING");
261 if(GPUcompress)
262 shaderParams.push_back("GPUCOMPRESS");
263 shaders[1]=ShaderfileModePair(count.c_str(),GL_FRAGMENT_SHADER);
264 countShader=compileAndLinkShader(shaders,shaderParams,
265 true,false,false,true);
266 if(countShader)
267 shaderParams.push_back("HAVE_SSBO");
268#else
269 countShader=0;
270#endif
271
272 ssbo=countShader;
273 if(!ssbo) {
274 glEnable(GL_BLEND);
275 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
276 }
277
278 interlock=ssbo && getSetting<bool>("GPUinterlock");
279
280 if(isNVIDIA30xx((const char*)glGetString(GL_RENDERER)))
281 interlock = false;
282
283 if(!ssbo && settings::verbose > 2)
284 cout << "No SSBO support; order-independent transparency unavailable"
285 << endl;
286
287 shaders[1]=ShaderfileModePair(fragment.c_str(),GL_FRAGMENT_SHADER);
288 shaderParams.push_back("MATERIAL");
289 if(orthographic)
290 shaderParams.push_back("ORTHOGRAPHIC");
291
292 ostringstream lights,materials,opaque;
293 lights << "Nlights " << Nlights;
294 shaderParams.push_back(lights.str().c_str());
295 materials << "Nmaterials " << nmaterials;
296 shaderParams.push_back(materials.str().c_str());
297
298 shaderParams.push_back("WIDTH");
299 pixelShader=compileAndLinkShader(shaders,shaderParams,ssbo);
300 shaderParams.pop_back();
301
302 shaderParams.push_back("NORMAL");
303 if(interlock) shaderParams.push_back("HAVE_INTERLOCK");
304 materialShader[0]=compileAndLinkShader(shaders,shaderParams,
305 ssbo,interlock,false,true);
306 if(interlock && !materialShader[0]) {
307 shaderParams.pop_back();
308 interlock=false;
309 materialShader[0]=compileAndLinkShader(shaders,shaderParams,ssbo);
310 if(settings::verbose > 2)
311 cout << "No fragment shader interlock support" << endl;
312 }
313
314 shaderParams.push_back("OPAQUE");
315 materialShader[1]=compileAndLinkShader(shaders,shaderParams,ssbo);
316 shaderParams.pop_back();
317
318 shaderParams.push_back("COLOR");
319 colorShader[0]=compileAndLinkShader(shaders,shaderParams,ssbo,
320 interlock);
321 shaderParams.push_back("OPAQUE");
322 colorShader[1]=compileAndLinkShader(shaders,shaderParams,ssbo);
323 shaderParams.pop_back();
324
325 shaderParams.push_back("GENERAL");
326 if(mode != DRAWMODE_NORMAL)
327 shaderParams.push_back("WIREFRAME");
328 generalShader[0]=compileAndLinkShader(shaders,shaderParams,ssbo,
329 interlock);
330 shaderParams.push_back("OPAQUE");
331 generalShader[1]=compileAndLinkShader(shaders,shaderParams,ssbo);
332 shaderParams.pop_back();
333
334 shaderParams.push_back("TRANSPARENT");
335 transparentShader=compileAndLinkShader(shaders,shaderParams,ssbo,
336 interlock);
337 shaderParams.clear();
338
339 if(ssbo) {
340 if(GPUindexing)
341 shaderParams.push_back("GPUINDEXING");
342 shaders[0]=ShaderfileModePair(screen.c_str(),GL_VERTEX_SHADER);
343 shaders[1]=ShaderfileModePair(compress.c_str(),GL_FRAGMENT_SHADER);
344 compressShader=compileAndLinkShader(shaders,shaderParams,ssbo);
345 if(GPUindexing)
346 shaderParams.pop_back();
347 else {
348 shaders[1]=ShaderfileModePair(zero.c_str(),GL_FRAGMENT_SHADER);
349 zeroShader=compileAndLinkShader(shaders,shaderParams,ssbo);
350 }
351 maxSize=1;
352 initBlendShader();
353 }
354 lastshader=-1;
355
356 if(vao == 0)
357 setBuffers();
358}

373{

405void AsyGLRender::drawFrame()
406{
407 if((nlights == 0 && Nlights > 0) || nlights > Nlights ||
408 materials.size() > nmaterials) {
409 deleteShaders();
410 initShaders();
411 }
412
413 // Apply srgb setting each frame so changes take effect dynamically
414 if(getSetting<bool>("srgb"))
415 glEnable(GL_FRAMEBUFFER_SRGB);
416 else
417 glDisable(GL_FRAMEBUFFER_SRGB);
418
419 // Set viewport before clearing (in case it wasn't set)
420 // Skip during export - trBeginTile handles viewport for tiling
421 if(!exporting)
422 glViewport(0, 0, Width, Height);
423
424 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
425
426 // Use member variables from AsyGLRender (following Vulkan pattern)
427 if(xmin >= xmax || ymin >= ymax || Zmin >= Zmax) return;
428
429 drawBuffers();
430
431 if(queueExport) {
432 queueExport=false;
433 Export();
434 }
435}

443void AsyGLRender::Export(int)
444{
445 size_t ndata=3*fullWidth*fullHeight;
446 if(ndata == 0) return;
447 glReadBuffer(GL_BACK_LEFT);
448 glPixelStorei(GL_PACK_ALIGNMENT,1);
449 glFinish();
450 exporting=true;
451
452 try {
453 unsigned char *data=new unsigned char[ndata];
454 if(data) {
455 TRcontext *tr=trNew();
456 int width=ceilquotient(fullWidth,
457 ceilquotient(fullWidth,std::min(maxTileWidth,Width)));
458 int height=ceilquotient(fullHeight,
459 ceilquotient(fullHeight,
460 std::min(maxTileHeight,Height)));
461 if(settings::verbose > 1)
462 cout << "Exporting " << Prefix << " as " << fullWidth << "x"
463 << fullHeight << " image" << " using tiles of size "
464 << width << "x" << height << endl;
465
466 unsigned border=std::min(std::min(1,(width-1)/2),(height-1)/2);
467 trTileSize(tr,width,height,border);
468 trImageSize(tr,fullWidth,fullHeight);
469 trImageBuffer(tr,GL_RGB,GL_UNSIGNED_BYTE,data);
470
471 setDimensions(fullWidth,fullHeight,X/Width*fullWidth,Y/Width*fullWidth);
472
473 size_t count=0;
474 if(haveScene) {
475 (orthographic ? trOrtho : trFrustum)(tr,xmin,xmax,ymin,ymax,-Zmax,-Zmin);
476 do {
477 trBeginTile(tr);
478 remesh=true;
479 redraw=true;
480 prepareScene();
481 drawFrame();
482 lastshader=-1;
483 ++count;
484 } while (trEndTile(tr));
485 } else {// clear screen and return
486 redraw=true;
487 prepareScene();
488 drawFrame();
489 }
490
491 if(settings::verbose > 1)
492 cout << count << " tile" << (count != 1 ? "s" : "") << " drawn" << endl;
493 trDelete(tr);
494
495 picture pic;
496 drawRawImage *Image=NULL;
497 if(haveScene) {
498 double w=oWidth;
499 double h=oHeight;
500 double Aspect=((double) fullWidth)/fullHeight;
501 if(w > h*Aspect) w=(int) (h*Aspect+0.5);
502 else h=(int) (w/Aspect+0.5);
503 // Render an antialiased image.
504
505 Image=new drawRawImage(data,fullWidth,fullHeight,
506 transform(0.0,0.0,w,0.0,0.0,h),
507 antialias);
508 pic.append(Image);
509 }
510
511 pic.shipout(NULL,Prefix,Format,false,ViewExport);
512 if(Image)
513 delete Image;
514 delete[] data;
515 }
516 } catch(handled_error const&) {
517 } catch(std::bad_alloc&) {
518 outOfMemory();
519 }
520 // Restore viewport and redraw full scene so back buffer has correct content
521 // (matches Vulkan pattern which renders export to separate framebuffer)
522 glViewport(0, 0, Width, Height);
523 setProjection();
524 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
525 remesh=true;
526 redraw=true;
527 prepareScene();
528 drawBuffers();
529
530
531#ifdef HAVE_PTHREAD
532 if(threads && readyAfterExport) {
533 readyAfterExport=false;
534 threadMgr.endwait(threadMgr.readySignal,threadMgr.readyLock);
535 }
536#endif
537
538 exporting=false;
539 initSSBO=true;
540}

542// Return the greatest power of 2 less than or equal to n.
543inline unsigned int floorpow2(unsigned int n)
544{
545 n |= n >> 1;
546 n |= n >> 2;
547 n |= n >> 4;
548 n |= n >> 8;
549 n |= n >> 16;
550 return n-(n >> 1);
551}

558void AsyGLRender::cycleMode()
559{
560 // Call base class to handle mode cycling and ibl
561 AsyRender::cycleMode();
562
563 if(ssbo)
564 initSSBO=true;
565
566 switch(mode) {
567 case DRAWMODE_NORMAL: // regular
568 nlights=nlights0; // Restore original number of lights
569 lastshader=-1;
570 glPolygonMode(GL_FRONT_AND_BACK,GL_FILL);

577 Nlights=1; // Force shader recompilation
578 break;
579 }

661void AsyGLRender::resizeFragmentBuffer()
662{
663 if(GPUindexing) {
664 glMemoryBarrier(GL_BUFFER_UPDATE_BARRIER_BIT);
665
666 glBindBuffer(GL_SHADER_STORAGE_BUFFER,feedbackBuffer);
667 GLuint *feedback=(GLuint *) glMapBuffer(GL_SHADER_STORAGE_BUFFER,GL_READ_ONLY);
668
669 GLuint maxDepth=feedback[0];
670 if(maxDepth > maxSize)
671 resizeBlendShader(maxDepth);
672
673 fragments=feedback[1];
674 glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
675 }
676
677 if(fragments > maxFragments) {
678 // Initialize the alpha buffer
679 maxFragments=11*fragments/10;
680 glBindBuffer(GL_SHADER_STORAGE_BUFFER,fragmentBuffer);
681 glBufferData(GL_SHADER_STORAGE_BUFFER,maxFragments*sizeof(vec4),
682 NULL,GL_DYNAMIC_DRAW);
683 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,4,fragmentBuffer);
684
685
686 glBindBuffer(GL_SHADER_STORAGE_BUFFER,depthBuffer);
687 glBufferData(GL_SHADER_STORAGE_BUFFER,maxFragments*sizeof(GLfloat),
688 NULL,GL_DYNAMIC_DRAW);
689 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,5,depthBuffer);
690
691 glBindBuffer(GL_SHADER_STORAGE_BUFFER,feedbackBuffer);
692 }

695void AsyGLRender::refreshBuffers()
696{
697 GLuint zero=0;
698 pixels=(Width+1)*(Height+1);
699
700 if(initSSBO) {
701 processors=1;
702
703 GLuint Pixels;
704 if(GPUindexing) {
705 GLuint G=ceilquotient(pixels,groupSize);
706 Pixels=groupSize*G;
707
708 GLuint globalSize=localSize*ceilquotient(G,localSize);
709 glBindBuffer(GL_SHADER_STORAGE_BUFFER,globalSumBuffer);
710 glBufferData(GL_SHADER_STORAGE_BUFFER,globalSize*sizeof(GLuint),NULL,
711 GL_DYNAMIC_READ);
712 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32UI,GL_RED_INTEGER,
713 GL_UNSIGNED_INT,&zero);
714 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,3,globalSumBuffer);
715 } else Pixels=pixels;
716
717 glBindBuffer(GL_SHADER_STORAGE_BUFFER,offsetBuffer);
718 glBufferData(GL_SHADER_STORAGE_BUFFER,(Pixels+2)*sizeof(GLuint),
719 NULL,GL_DYNAMIC_DRAW);
720 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32UI,GL_RED_INTEGER,
721 GL_UNSIGNED_INT,&zero);
722 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,0,offsetBuffer);
723
724 glBindBuffer(GL_SHADER_STORAGE_BUFFER,countBuffer);
725 glBufferData(GL_SHADER_STORAGE_BUFFER,(Pixels+2)*sizeof(GLuint),
726 NULL,GL_DYNAMIC_DRAW);
727 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,2,countBuffer);
728
729 if(GPUcompress) {
730 GLuint one=1;
731 glBindBuffer(GL_ATOMIC_COUNTER_BUFFER,elementsBuffer);
732 glBufferData(GL_ATOMIC_COUNTER_BUFFER,sizeof(GLuint),&one,
733 GL_DYNAMIC_DRAW);
734 glBindBufferBase(GL_ATOMIC_COUNTER_BUFFER,0,elementsBuffer);
735
736 glBindBuffer(GL_SHADER_STORAGE_BUFFER,indexBuffer);
737 glBufferData(GL_SHADER_STORAGE_BUFFER,pixels*sizeof(GLuint),
738 NULL,GL_DYNAMIC_DRAW);
739 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,1,indexBuffer);
740 }
741 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32UI,GL_RED_INTEGER,
742 GL_UNSIGNED_INT,&zero); // Clear count or index buffer
743
744 glBindBuffer(GL_SHADER_STORAGE_BUFFER,opaqueBuffer);
745 glBufferData(GL_SHADER_STORAGE_BUFFER,pixels*sizeof(vec4),NULL,
746 GL_DYNAMIC_DRAW);
747 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,6,opaqueBuffer);
748
749 glBindBuffer(GL_SHADER_STORAGE_BUFFER,opaqueDepthBuffer);
750 glBufferData(GL_SHADER_STORAGE_BUFFER,
751 sizeof(GLuint)+pixels*sizeof(GLfloat),NULL,
752 GL_DYNAMIC_DRAW);
753 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,7,opaqueDepthBuffer);
754 const GLfloat zerof=0.0;
755 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32F,GL_RED,GL_FLOAT,&zerof);
756
757 glBindBuffer(GL_SHADER_STORAGE_BUFFER,feedbackBuffer);
758 glBufferData(GL_SHADER_STORAGE_BUFFER,2*sizeof(GLuint),NULL,
759 GL_DYNAMIC_DRAW);
760 glBindBufferBase(GL_SHADER_STORAGE_BUFFER,8,feedbackBuffer);
761
762 glBindBuffer(GL_SHADER_STORAGE_BUFFER,feedbackBuffer);
763 initSSBO=false;
764 }
765
766 // Determine the fragment offsets
767
768 if(exporting && GPUindexing && !GPUcompress) {
769 glBindBuffer(GL_SHADER_STORAGE_BUFFER,countBuffer);
770 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32UI,GL_RED_INTEGER,
771 GL_UNSIGNED_INT,&zero);
772 glBindBuffer(GL_SHADER_STORAGE_BUFFER,feedbackBuffer);
773 }
774
775 if(!interlock) {
776 drawBuffer(lineData,countShader,false,1);
777 drawBuffer(materialData,countShader,false,4);
778 drawBuffer(colorData,countShader,true,4);
779 drawBuffer(triangleData,countShader,true,4);
780 }
781
782 glDepthMask(GL_FALSE); // Don't write to depth buffer
783 glDisable(GL_MULTISAMPLE);
784 drawBuffer(transparentData,countShader,true,4);
785 glEnable(GL_MULTISAMPLE);
786 glDepthMask(GL_TRUE); // Write to depth buffer
787
788 if(GPUcompress) {
789 compressCount();
790 GLuint *p=(GLuint *) glMapBuffer(GL_ATOMIC_COUNTER_BUFFER,GL_READ_WRITE);
791 elements=GPUindexing ? p[0] : p[0]-1;
792 p[0]=1;
793 glUnmapBuffer(GL_ATOMIC_COUNTER_BUFFER);
794 if(elements == 0) return;
795 } else
796 elements=pixels;
797
798 if(GPUindexing) {
799 // Ensure fragment shader writes to countBuffer are visible to compute shaders.
800 glMemoryBarrier(GL_FRAMEBUFFER_BARRIER_BIT | GL_SHADER_STORAGE_BARRIER_BIT);
801
802 g=ceilquotient(elements,groupSize);
803 elements=groupSize*g;
804
805 if(settings::verbose > timePartialSumVerbosity) {
806 static bool first=true;
807 if(first) {
808 partialSums();
809 first=false;
810 }
811 unsigned int N=10000;
812 stopWatch Timer;
813 for(unsigned int i=0; i < N; ++i)
814 partialSums();
815 glFinish();
816 double T=Timer.seconds()/N;
817 cout << "elements=" << elements << endl;
818 cout << "Tmin (ms)=" << T*1e3 << endl;
819 cout << "Megapixels/second=" << elements/T/1e6 << endl;
820 }
821
822 partialSums(true);
823 } else {
824 size_t size=elements*sizeof(GLuint);
825
826 // Compute partial sums on the CPU
827 glBindBuffer(GL_SHADER_STORAGE_BUFFER,countBuffer);
828 GLuint *p=(GLuint *) glMapBufferRange(GL_SHADER_STORAGE_BUFFER,
829 0,size+sizeof(GLuint),
830 GL_MAP_READ_BIT);
831 GLuint maxDepth=p[0];
832 GLuint *count=p+1;
833
834 glBindBuffer(GL_SHADER_STORAGE_BUFFER,offsetBuffer);
835 GLuint *offset=(GLuint *) glMapBufferRange(GL_SHADER_STORAGE_BUFFER,
836 sizeof(GLuint),size,
837 GL_MAP_WRITE_BIT);
838
839 size_t Offset=offset[0]=count[0];
840 for(size_t i=1; i < elements; ++i)
841 offset[i]=Offset += count[i];
842 fragments=Offset;
843
844 glBindBuffer(GL_SHADER_STORAGE_BUFFER,offsetBuffer);
845 glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
846
847 glBindBuffer(GL_SHADER_STORAGE_BUFFER,countBuffer);
848 glUnmapBuffer(GL_SHADER_STORAGE_BUFFER);
849
850 if(exporting) {
851 glBindBuffer(GL_SHADER_STORAGE_BUFFER,countBuffer);
852 glClearBufferData(GL_SHADER_STORAGE_BUFFER,GL_R32UI,GL_RED_INTEGER,
853 GL_UNSIGNED_INT,&zero);
854 } else
855 clearCount();
856
857 if(maxDepth > maxSize)
858 resizeBlendShader(maxDepth);
859 }
860 lastshader=-1;
861}

863void AsyGLRender::setUniformsOpenGL(GLint shader)
864{
865 bool normal=shader != pixelShader;
866
867 if(shader != lastshader) {
868 glUseProgram(shader);
869
870 // Cache uniform locations when shader changes
871 projViewLoc = glGetUniformLocation(shader,"projViewMat");
872 viewMatLoc = glGetUniformLocation(shader,"viewMat");
873 normMatLoc = glGetUniformLocation(shader,"normMat");
874
875 if(normal)
876 glUniform1ui(glGetUniformLocation(shader,"width"),Width);
877 }
878
879 glUniformMatrix4fv(projViewLoc,1,GL_FALSE,
880 value_ptr(mat4(projViewMat)));
881
882 glUniformMatrix4fv(viewMatLoc,1,GL_FALSE,
883 value_ptr(mat4(viewMat)));
884 if(normal)
885 glUniformMatrix3fv(normMatLoc,1,GL_FALSE,
886 value_ptr(mat3(normMat)));
887
888 if(shader == countShader) {
889 lastshader=shader;
890 return;
891 }
892
893 if(shader != lastshader) {
894 lastshader=shader;
895 glUniform1ui(glGetUniformLocation(shader,"nlights"),nlights);
896
897 for(size_t i=0; i < nlights; ++i) {
898 triple Lighti=Lights[i];
899 double *Diffusei= LightsDiffuse+4*i;
900 glUniform3f(glGetUniformLocation(shader,
901 getLightIndex(i,"direction").c_str()),
902 (GLfloat) Lighti.getx(),(GLfloat) Lighti.gety(),
903 (GLfloat) Lighti.getz());
904
905 glUniform3f(glGetUniformLocation(shader,
906 getLightIndex(i,"color").c_str()),
907 (GLfloat) Diffusei[0],(GLfloat) Diffusei[1],
908 (GLfloat) Diffusei[2]);
909 }
910
911 if(settings::getSetting<bool>("ibl")) {
912 iblbrdfTex.setUniform(glGetUniformLocation(shader, "reflBRDFSampler"));
913 irradianceTex.setUniform(glGetUniformLocation(shader, "diffuseSampler"));
914 reflTexturesTex.setUniform(glGetUniformLocation(shader, "reflImgSampler"));
915 }
916 }
917
918 // Bind global materials buffer
919 GLuint binding=0;
920 GLuint blockindex=glGetUniformBlockIndex(shader,"MaterialBuffer");
921 if(blockindex != GL_INVALID_INDEX) {
922 glUniformBlockBinding(shader,blockindex,binding);
923 bool copy=shouldUpdateBuffers;
924 registerBuffer(materials, materialsBuffer, copy, GL_UNIFORM_BUFFER);
925 shouldUpdateBuffers=false;
926 glBindBufferBase(GL_UNIFORM_BUFFER, binding, materialsBuffer);
927 }
928}

930void AsyGLRender::drawBuffer(VertexBuffer& data, GLint shader, bool color, unsigned int drawType) // drawType: 0=GL_POINTS, 1=GL_LINES, 4=GL_TRIANGLES
931{
932 if(data.indices.empty()) return;
933
934 // Check for OpenGL errors before drawing
935
936 if(settings::verbose > 3) {
937 GLenum err = glGetError();
938 if(err != GL_NO_ERROR) {
939 cerr << "drawBuffer: OpenGL error at start: " << err << endl;
940 }
941 }
942 bool normal=shader != pixelShader;
943
944 bool copy = (remesh || data.renderCount < 1) && !copied;
945
946 // Get persistent GL buffer handles for this VertexBuffer instance
947 auto& glBuf = glBuffers[&data];
948
949 // Upload vertex data using persistent buffer
950 if(color) {
951 registerBuffer(data.colorVertices, glBuf.vertexBuffer, copy, GL_ARRAY_BUFFER);
952 } else if(normal) {
953 registerBuffer(data.materialVertices, glBuf.vertexBuffer, copy, GL_ARRAY_BUFFER);
954 } else {
955 registerBuffer(data.pointVertices, glBuf.vertexBuffer, copy, GL_ARRAY_BUFFER);
956 }
957
958 // Upload index data using persistent buffer
959 registerBuffer(data.indices, glBuf.indexBuffer, copy, GL_ELEMENT_ARRAY_BUFFER);
960
961 setUniformsOpenGL(shader);
962
963 // Position attribute (3 floats)
964 if(color) {
965 glVertexAttribPointer(positionAttrib, 3, GL_FLOAT, GL_FALSE, sizeof(ColorVertex),
966 (void *) offsetof(ColorVertex, position));
967 } else if(normal) {
968 glVertexAttribPointer(positionAttrib, 3, GL_FLOAT, GL_FALSE, sizeof(MaterialVertex),
969 (void *) offsetof(MaterialVertex, position));
970 } else {
971 glVertexAttribPointer(positionAttrib, 3, GL_FLOAT, GL_FALSE, sizeof(PointVertex),
972 (void *) offsetof(PointVertex, position));
973 }
974 glEnableVertexAttribArray(positionAttrib);
975
976 if(normal && nlights > 0) {
977 // Normal attribute (3 floats)
978 glVertexAttribPointer(normalAttrib, 3, GL_FLOAT, GL_FALSE,
979 color ? sizeof(ColorVertex) : sizeof(MaterialVertex),
980 (void *) offsetof(MaterialVertex, normal));
981 glEnableVertexAttribArray(normalAttrib);
982 } else if(!normal) {
983 // Width attribute for points (1 float)
984 glVertexAttribPointer(widthAttrib, 1, GL_FLOAT, GL_FALSE, sizeof(PointVertex),
985 (void *) offsetof(PointVertex, width));
986 glEnableVertexAttribArray(widthAttrib);
987 }
988
989 // Material index attribute (1 int)
990 if(color) {
991 glVertexAttribIPointer(materialAttrib, 1, GL_INT, sizeof(ColorVertex),
992 (void *) offsetof(ColorVertex, material));
993 } else if(normal) {
994 glVertexAttribIPointer(materialAttrib, 1, GL_INT, sizeof(MaterialVertex),
995 (void *) offsetof(MaterialVertex, material));
996 } else {
997 glVertexAttribIPointer(materialAttrib, 1, GL_INT, sizeof(PointVertex),
998 (void *) offsetof(PointVertex, material));
999 }
1000 glEnableVertexAttribArray(materialAttrib);
1001
1002 if(color) {
1003 // Color attribute (4 floats)
1004 glVertexAttribPointer(colorAttrib, 4, GL_FLOAT, GL_FALSE, sizeof(ColorVertex),
1005 (void *) offsetof(ColorVertex, color));
1006 glEnableVertexAttribArray(colorAttrib);
1007 }
1008
1009 fpu_trap(false); // Work around FE_INVALID
1010 glDrawElements(drawType, data.indices.size(), GL_UNSIGNED_INT, (void *) 0);
1011 fpu_trap(settings::trap());
1012
1013 // Disable attribute arrays but keep VAO bound for next draw call
1014 glDisableVertexAttribArray(positionAttrib);
1015 if(normal && nlights > 0)
1016 glDisableVertexAttribArray(normalAttrib);
1017 if(!normal)
1018 glDisableVertexAttribArray(widthAttrib);
1019 glDisableVertexAttribArray(materialAttrib);
1020 if(color)
1021 glDisableVertexAttribArray(colorAttrib);
1022
1023 glBindBuffer(GL_UNIFORM_BUFFER, 0);
1024 glBindBuffer(GL_ARRAY_BUFFER, 0);
1025 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
1026}

1132AsyGLRender::~AsyGLRender()
1133{
1134 if (this->View) {
1135 ::glfwDestroyWindow(getRenderWindow());
1136 glfwWindow = nullptr;
1137 }
1138
1139 // Cleanup OpenGL resources
1140 glDeleteProgram(pixelShader);
1141 for(int i=0; i<2; ++i) {
1142 glDeleteProgram(materialShader[i]);
1143 glDeleteProgram(colorShader[i]);
1144 glDeleteProgram(generalShader[i]);
1145 }
1146 glDeleteProgram(countShader);
1147 glDeleteProgram(transparentShader);
1148 glDeleteProgram(blendShader);
1149 glDeleteProgram(zeroShader);
1150 glDeleteProgram(compressShader);
1151 glDeleteProgram(sum1Shader);
1152 glDeleteProgram(sum2Shader);
1153 glDeleteProgram(sum3Shader);
1154
1155 if(vao) glDeleteVertexArrays(1, &vao);
1156 if(offsetBuffer) glDeleteBuffers(1, &offsetBuffer);
1157 if(indexBuffer) glDeleteBuffers(1, &indexBuffer);
1158 if(elementsBuffer) glDeleteBuffers(1, &elementsBuffer);
1159 if(countBuffer) glDeleteBuffers(1, &countBuffer);
1160 if(globalSumBuffer) glDeleteBuffers(1, &globalSumBuffer);
1161 if(fragmentBuffer) glDeleteBuffers(1, &fragmentBuffer);
1162 if(depthBuffer) glDeleteBuffers(1, &depthBuffer);
1163 if(opaqueBuffer) glDeleteBuffers(1, &opaqueBuffer);
1164 if(opaqueDepthBuffer) glDeleteBuffers(1, &opaqueDepthBuffer);
1165 if(feedbackBuffer) glDeleteBuffers(1, &feedbackBuffer);
1166
1167 // Cleanup persistent vertex/index buffers from VertexBuffer instances
1168 for (auto& [vb, glBuf] : glBuffers) {
1169 if (glBuf.vertexBuffer) glDeleteBuffers(1, &glBuf.vertexBuffer);
1170 if (glBuf.indexBuffer) glDeleteBuffers(1, &glBuf.indexBuffer);
1171 }
1172 glBuffers.clear();
1173}

1175void AsyGLRender::render(RenderFunctionArgs const& args)
1176{
1177#if !defined(_WIN32)
1178 setenv("XMODIFIERS","",true);
1179#endif
1180 lastshader = -1;
1181
1182 copyRenderArgs(args);
1183
1184 nlights0 = nlights; // Save original for mode restoration
1185
1186 pair maxtile=getSetting<pair>("maxtile");
1187 maxTileWidth=(int) maxtile.getx();
1188 maxTileHeight=(int) maxtile.gety();
1189 if(maxTileWidth <= 0) maxTileWidth=1024;
1190 if(maxTileHeight <= 0) maxTileHeight=768;
1191
1192#ifdef HAVE_PTHREAD
1193 static bool initializedView=false;
1194#endif
1195
1196 if(!initialized)
1197 Fitscreen=1;
1198
1199 if(!(initialized && interact::interactive)) {
1200 antialias=settings::getSetting<Int>("antialias") > 1;
1201
1202 Aspect = args.width/args.height;
1203
1204 initDisplay(args.width, args.height);
1205
1206 // Force a hard viewport limit to work around direct rendering bugs.
1207 // Alternatively, one can use -glOptions=-indirect (with a performance
1208 // penalty).
1209 {
1210 pair maxViewport = settings::getSetting<pair>("maxviewport");
1211 int maxWidth = maxViewport.getx() > 0 ? (int)ceil(maxViewport.getx()) : screenWidth;
1212 int maxHeight = maxViewport.gety() > 0 ? (int)ceil(maxViewport.gety()) : screenHeight;
1213 if(maxWidth <= 0) maxWidth = max(maxHeight, 2);
1214 if(maxHeight <= 0) maxHeight = max(maxWidth, 2);
1215
1216 if(screenWidth <= 0) screenWidth=maxWidth;
1217 else screenWidth=min(screenWidth,maxWidth);
1218 if(screenHeight <= 0) screenHeight=maxHeight;
1219 else screenHeight=min(screenHeight,maxHeight);
1220 }
1221 }
1222
1223 havewindow = View && threads;
1224
1225 maxFragments = 0;
1226 clearMaterials();
1227 shouldUpdateBuffers=true;
1228 initialized=true;
1229
1230#ifdef HAVE_PTHREAD
1231 if(threads && initializedView) {
1232 if(View) {
1233 // Called from asymain thread, main thread handles rendering
1234 hideWindow=false;
1235 threadMgr.messageQueue.enqueue(RendererMessage::updateRenderer);
1236 } else readyAfterExport=queueExport=true;
1237 return;
1238 }
1239#endif
1240
1241// Create GLFW window BEFORE OpenGL initialization
1242 if(!glfwWindow) {
1243 // For non-View rendering, hide the window during creation to prevent flash
1244 if(!View)
1245 glfwWindowHint(GLFW_VISIBLE, 0);
1246
1247 GLFWwindow* newWindow = glfwCreateRenderWindow(Width, Height, title.empty() ? Prefix.c_str() : title.c_str(), this);
1248 if(newWindow == nullptr) {
1249 cerr << "Failed to create GLFW window" << endl;
1250 exit(-1);
1251 }
1252 glfwWindow = newWindow;
1253
1254 // Make context current before GLEW initialization (matching reference pattern)
1255 glfwMakeContextCurrent(newWindow);
1256
1257 // Initialize GLEW immediately after context creation (matching reference pattern)
1258 glewExperimental = GL_TRUE;
1259 GLenum glewErr = glewInit();
1260 if(glewErr != GLEW_OK) {
1261 const char *glVer = (const char *)glGetString(GL_VERSION);
1262 if(glVer == NULL) {
1263 cerr << "GLEW initialization error: " << glewGetErrorString(glewErr) << endl;
1264 exit(-1);
1265 }
1266 }
1267
1268 // Set swap interval based on vsync setting (0 = no vsync, 1 = vsync)
1269 glfwRendererSwapInterval(getSetting<bool>("vsync") ? 1 : 0);
1270
1271 const char *GLSL_VERSION=(const char *)glGetString(GL_SHADING_LANGUAGE_VERSION);
1272 if(GLSL_VERSION)
1273 GLSLversion=(int) (100*atof(GLSL_VERSION)+0.5);
1274 if(settings::verbose > 2)
1275 cout << "GLSL version " << GLSL_VERSION << " (GLSLversion=" << GLSLversion << ")" << endl;
1276
1277 // Check multisampling
1278 {
1279 int samples = 0;
1280 glGetIntegerv(GL_SAMPLES, &samples);
1281 if(settings::verbose > 1 && samples > 1)
1282 cout << "Multisampling enabled with sample width " << samples << endl;
1283 }
1284 }
1285
1286#if defined(HAVE_COMPUTE_SHADER)
1287 GPUindexing=getSetting<bool>("GPUindexing");
1288 GPUcompress=getSetting<bool>("GPUcompress");
1289#else
1290 GPUindexing=false;
1291 GPUcompress=false;
1292#endif
1293
1294 // Initialize GPU compute parameters
1295 if(GPUindexing) {
1296 localSize = settings::getSetting<Int>("GPUlocalSize");
1297 checkpow2(localSize,"GPUlocalSize");
1298 blockSize = settings::getSetting<Int>("GPUblockSize");
1299 checkpow2(blockSize,"GPUblockSize");
1300 groupSize = localSize * blockSize;
1301 }
1302
1303 glClearColor(Background[0], Background[1],
1304 Background[2], Background[3]);
1305
1306 if(View) {
1307 if(!getSetting<bool>("fitscreen"))
1308 Fitscreen=0;
1309 firstFit=true;
1310 fitscreen();
1311#ifdef HAVE_PTHREAD
1312 initializedView = true;
1313#endif
1314 }
1315
1316 glEnable(GL_DEPTH_TEST);
1317
1318 mode = DRAWMODE_WIREFRAME;
1319 cycleMode();
1320
1321 ViewExport = View;
1322
1323
1324 havewindow = initialized && threads;
1325
1326 // Enter main loop or export
1327 if(View) {
1328 // Process pending resize events from fitscreen() before entering the main loop,
1329 // so the framebuffer has reached its target size before the first frame renders.
1330 glfwPollEvents();
1331 mainLoop();
1332 } else {
1333 update();
1334 display();
1335 if(threads) {
1336 exportHandler(0);
1337#ifdef HAVE_PTHREAD
1338 if(!pthread_equal(pthread_self(), threadMgr.mainthread))
1339 threadMgr.endwait(threadMgr.readySignal, threadMgr.readyLock);
1340#endif
1341 } else {
1342 exportHandler(0);
1343 quit();
1344 }
1345 }
1346}
