246// Return the transformation corresponding to moving the camera from the target
247// (looking in the negative z direction) to the point 'eye' (looking at target,
248// orienting the camera so that direction 'up' points upwards.
249// Since, in actuality, we are transforming the points instead of the camera,
250// we calculate the inverse matrix.
251// Based on the gluLookAt implementation in the OpenGL manual.
252transform3 look(triple eye, triple up=Z, triple target=O)
253{
254 triple f=unit(target-eye);
255 if(f == O)
256 f=-Z; // The eye is already at the origin: look down.
257
258 triple s=cross(f,up);
259
260 // If the eye is pointing either directly up or down, there is no
261 // preferred "up" direction. Pick one arbitrarily.
262 s=s != O ? unit(s) : perp(f);
263
264 triple u=cross(s,f);
265
266 transform3 M={{ s.x, s.y, s.z, 0},
267 { u.x, u.y, u.z, 0},
268 {-f.x, -f.y, -f.z, 0},
269 { 0, 0, 0, 1}};
270
271 return M*shift(-eye);
272}

437// Map pair z to a triple by inverting the projection P onto the
438// plane perpendicular to normal and passing through point.
439triple invert(pair z, triple normal, triple point,
440 projection P=currentprojection)
441{
442 transform3 t=P.t;
443 real[][] A={{t[0][0]-z.x*t[3][0],t[0][1]-z.x*t[3][1],t[0][2]-z.x*t[3][2]},
444 {t[1][0]-z.y*t[3][0],t[1][1]-z.y*t[3][1],t[1][2]-z.y*t[3][2]},
445 {normal.x,normal.y,normal.z}};

447 real[] x=solve(A,b,warn=false);
448 return x.length > 0 ? (x[0],x[1],x[2]) : P.camera;
449}

830 // (cf. The MetaFont Book, page 131),
831 // as described in John C. Bowman and A. Hammerlindl,
832 // TUGBOAT: The Communications of the TeX Users Group 29:2 (2008).
833
834 void operator init(triple v0, triple v1, triple d0, triple d1, real tout,
835 real tin, bool atLeast) {
836 triple v=v1-v0;
837 triple u=unit(v);
838 real L=length(v);
839 d0=unit(d0);
840 d1=unit(d1);
841 real theta=acos1(dot(d0,u));
842 real phi=acos1(dot(d1,u));
843 if(dot(cross(d0,v),cross(v,d1)) < 0) phi=-phi;
844 c0=v0+d0*L*relativedistance(theta,phi,tout,atLeast);
845 c1=v1-d1*L*relativedistance(phi,theta,tin,atLeast);
846 }
847}

869// 3D extension of John Hobby's angle formula (The MetaFont Book, page 131).
870// Notational differences: here psi[i] is the turning angle at z[i+1],
871// beta[i] is the tension for segment i, and in[i] is the incoming
872// direction for segment i (where segment i begins at node i).
873
874real[] theta(triple[] v, real[] alpha, real[] beta,
875 triple dir0, triple dirn, real g0, real gn, triple reference)
876{
877 real[] a,b,c,f,l,psi;
878 int n=alpha.length;
879 bool cyclic=v.cyclic;
880 for(int i=0; i < n; ++i)
881 l[i]=1/length(v[i+1]-v[i]);
882 int i0,in;
883 if(cyclic) {i0=0; in=n;}
884 else {i0=1; in=n-1;}
885 for(int i=0; i < in; ++i)
886 psi[i]=angle(v[i+1]-v[i],v[i+2]-v[i+1],reference);
887 if(cyclic) {
888 l.cyclic=true;
889 psi.cyclic=true;
890 } else {
891 psi[n-1]=0;
892 if(dir0 == O) {
893 real a0=alpha[0];
894 real b0=beta[0];
895 real chi=g0*(b0/a0)^2;
896 a[0]=0;
897 b[0]=3a0-a0/b0+chi;
898 real C=chi*(3a0-1)+a0/b0;
899 c[0]=C;
900 f[0]=-C*psi[0];
901 } else {
902 a[0]=c[0]=0;
903 b[0]=1;
904 f[0]=angle(v[1]-v[0],dir0,reference);
905 }
906 if(dirn == O) {
907 real an=alpha[n-1];
908 real bn=beta[n-1];
909 real chi=gn*(an/bn)^2;
910 a[n]=chi*(3bn-1)+bn/an;
911 b[n]=3bn-bn/an+chi;
912 c[n]=f[n]=0;
913 } else {
914 a[n]=c[n]=0;
915 b[n]=1;
916 f[n]=angle(v[n]-v[n-1],dirn,reference);
917 }
918 }
919
920 for(int i=i0; i < n; ++i) {
921 real in=beta[i-1]^2*l[i-1];
922 real A=in/alpha[i-1];
923 a[i]=A;
924 real B=3*in-A;
925 real out=alpha[i]^2*l[i];
926 real C=out/beta[i];
927 b[i]=B+3*out-C;
928 c[i]=C;
929 f[i]=-B*psi[i-1]-C*psi[i];
930 }
931
932 return tridiagonal(a,b,c,f);
933}

935triple reference(triple[] v, int n, triple d0, triple d1)
936{
937 triple[] V=sequence(new triple(int i) {
938 return cross(v[i+1]-v[i],v[i+2]-v[i+1]);
939 },n-1);
940 if(n > 0) {
941 V.push(cross(d0,v[1]-v[0]));
942 V.push(cross(v[n]-v[n-1],d1));
943 }
944
945 triple max=V[0];
946 real M=abs(max);
947 for(int i=1; i < V.length; ++i) {
948 triple vi=V[i];
949 real a=abs(vi);
950 if(a > M) {
951 M=a;
952 max=vi;
953 }
954 }
955
956 triple reference;
957 for(int i=0; i < V.length; ++i) {
958 triple u=unit(V[i]);
959 reference += dot(u,max) < 0 ? -u : u;
960 }
961
962 return reference;
963}

965// Fill in missing directions for n cyclic nodes.
966void aim(flatguide3 g, int N)
967{
968 bool cyclic=true;
969 int start=0, end=0;
970
971 // If the cycle contains one or more direction specifiers, break the loop.
972 for(int k=0; k < N; ++k)
973 if(g.solved(k)) {cyclic=false; end=k; break;}
974 for(int k=N-1; k >= 0; --k)
975 if(g.solved(k)) {cyclic=false; start=k; break;}
976 while(start < N && g.control[start].active) ++start;
977
978 int n=N-(start-end);
979 if(n <= 1 || (cyclic && n <= 2)) return;
980
981 triple[] v=new triple[cyclic ? n : n+1];
982 real[] alpha=new real[n];
983 real[] beta=new real[n];
984 for(int k=0; k < n; ++k) {
985 int K=(start+k) % N;
986 v[k]=g.nodes[K];
987 alpha[k]=g.Tension[K].out;
988 beta[k]=g.Tension[K].in;
989 }
990 if(cyclic) {
991 v.cyclic=true;
992 alpha.cyclic=true;
993 beta.cyclic=true;
994 } else v[n]=g.nodes[(start+n) % N];
995 int final=(end-1) % N;
996
997 triple d0=g.out[start].dir;
998 triple d1=g.in[final].dir;
999
1000 triple reference=reference(v,n,d0,d1);
1001
1002 real[] theta=theta(v,alpha,beta,d0,d1,g.out[start].gamma,g.in[final].gamma,
1003 reference);
1004
1005 v.cyclic=true;
1006 theta.cyclic=true;
1007
1008 for(int k=1; k < (cyclic ? n+1 : n); ++k) {
1009 triple w=dir(theta[k],v[k]-v[k-1],v[k+1]-v[k],reference);
1010 g.in[(start+k-1) % N].init(w);
1011 g.out[(start+k) % N].init(w);
1012 }
1013
1014 if(g.out[start].dir == O)
1015 g.out[start].init(dir(theta[0],v[0]-g.nodes[(start-1) % N],v[1]-v[0],
1016 reference));
1017 if(g.in[final].dir == O)
1018 g.in[final].init(dir(theta[n],v[n-1]-v[n-2],v[n]-v[n-1],reference));
1019}

1021// Fill in missing directions for the sequence of nodes i...n.
1022void aim(flatguide3 g, int i, int n)
1023{
1024 int j=n-i;
1025 if(j > 1 || g.out[i].dir != O || g.in[i].dir != O) {
1026 triple[] v=new triple[j+1];
1027 real[] alpha=new real[j];
1028 real[] beta=new real[j];
1029 for(int k=0; k < j; ++k) {
1030 v[k]=g.nodes[i+k];
1031 alpha[k]=g.Tension[i+k].out;
1032 beta[k]=g.Tension[i+k].in;
1033 }
1034 v[j]=g.nodes[n];
1035
1036 triple d0=g.out[i].dir;
1037 triple d1=g.in[n-1].dir;
1038
1039 triple reference=reference(v,j,d0,d1);
1040
1041 real[] theta=theta(v,alpha,beta,d0,d1,g.out[i].gamma,g.in[n-1].gamma,
1042 reference);
1043
1044 for(int k=1; k < j; ++k) {
1045 triple w=dir(theta[k],v[k]-v[k-1],v[k+1]-v[k],reference);
1046 g.in[i+k-1].init(w);
1047 g.out[i+k].init(w);
1048 }
1049 if(g.out[i].dir == O) {
1050 triple w=dir(theta[0],g.in[i].dir,v[1]-v[0],reference);
1051 if(i > 0) g.in[i-1].init(w);
1052 g.out[i].init(w);
1053 }
1054 if(g.in[n-1].dir == O) {
1055 triple w=dir(theta[j],g.out[n-1].dir,v[j]-v[j-1],reference);
1056 g.in[n-1].init(w);
1057 g.out[n].init(w);
1058 }
1059 }

1247path3 solve(flatguide3 g)
1248{
1249 int n=g.nodes.length-1;
1250
1251 // If duplicate points occur consecutively, add dummy controls (if absent).
1252 for(int i=0; i < n; ++i) {
1253 if(g.nodes[i] == g.nodes[i+1] && !g.control[i].active)
1254 g.control[i]=control(g.nodes[i],g.nodes[i],straight=true);
1255 }
1256
1257 // Fill in empty direction specifiers inherited from explicit control points.
1258 for(int i=0; i < n; ++i) {
1259 if(g.control[i].active) {
1260 g.out[i].init(g.control[i].post-g.nodes[i]);
1261 g.in[i].init(g.nodes[i+1]-g.control[i].pre);
1262 }
1263 }
1264
1265 // Propagate directions across nodes.
1266 for(int i=0; i < n; ++i) {
1267 int next=g.cyclic[i+1] ? 0 : i+1;
1268 if(g.out[next].active())
1269 g.in[i].default(g.out[next]);
1270 if(g.in[i].active()) {
1271 g.out[next].default(g.in[i]);
1272 g.out[i+1].default(g.in[i]);
1273 }
1274 }
1275
1276 // Compute missing 3D directions.
1277 // First, resolve cycles
1278 int i=find(g.cyclic);
1279 if(i > 0) {
1280 aim(g,i);
1281 // All other cycles can now be reduced to sequences.
1282 triple v=g.out[0].dir;
1283 for(int j=i; j <= n; ++j) {
1284 if(g.cyclic[j]) {
1285 g.in[j-1].default(v);
1286 g.out[j].default(v);
1287 if(g.nodes[j-1] == g.nodes[j] && !g.control[j-1].active)
1288 g.control[j-1]=control(g.nodes[j-1],g.nodes[j-1]);
1289 }
1290 }
1291 }
1292
1293 // Next, resolve sequences.
1294 int i=0;
1295 int start=0;
1296 while(i < n) {
1297 // Look for a missing outgoing direction.
1298 while(i <= n && g.solved(i)) {start=i; ++i;}
1299 if(i > n) break;
1300 // Look for the end of the sequence.
1301 while(i < n && !g.solved(i)) ++i;
1302
1303 while(start < i && g.control[start].active) ++start;
1304
1305 if(start < i)
1306 aim(g,start,i);
1307 }
1308
1309 // Compute missing 3D control points.
1310 for(int i=0; i < n; ++i) {
1311 int next=g.cyclic[i+1] ? 0 : i+1;
1312 if(!g.control[i].active) {
1313 control c;
1314 if((g.out[i].Curl && g.in[i].Curl) ||
1315 (g.out[i].dir == O && g.in[i].dir == O)) {
1316 // fill in straight control points for path3 functions
1317 triple delta=(g.nodes[i+1]-g.nodes[i])/3;
1318 c=control(g.nodes[i]+delta,g.nodes[i+1]-delta,straight=true);
1319 } else {
1320 Controls C=Controls(g.nodes[i],g.nodes[next],g.out[i].dir,g.in[i].dir,
1321 g.Tension[i].out,g.Tension[i].in,
1322 g.Tension[i].atLeast);
1323 c=control(C.c0,C.c1);
1324 }
1325 g.control[i]=c;
1326 }
1327 }
1328
1329 // Convert to Knuth's format (control points stored with nodes)
1330 int n=g.nodes.length;
1331 bool cyclic;
1332 if(n > 0) {
1333 cyclic=g.cyclic[n-1];
1334 if(cyclic) --n;
1335 }
1336 triple[] pre=new triple[n];
1337 triple[] point=new triple[n];
1338 triple[] post=new triple[n];
1339 bool[] straight=new bool[n];
1340 if(n > 0) {
1341 for(int i=0; i < n-1; ++i) {
1342 point[i]=g.nodes[i];
1343 post[i]=g.control[i].post;
1344 pre[i+1]=g.control[i].pre;
1345 straight[i]=g.control[i].straight;
1346 }
1347 point[n-1]=g.nodes[n-1];
1348 if(cyclic) {
1349 pre[0]=g.control[n-1].pre;
1350 post[n-1]=g.control[n-1].post;
1351 straight[n-1]=g.control[n-1].straight;
1352 } else {
1353 pre[0]=point[0];
1354 post[n-1]=point[n-1];
1355 straight[n-1]=false;
1356 }
1357 }
1358
1359 return path3(pre,point,post,straight,cyclic);
1360}

1392path project(path3 p, projection P=currentprojection,
1393 int ninterpolate=P.ninterpolate)
1394{
1395 guide g;
1396
1397 int last=length(p);
1398 if(last < 0) return g;
1399
1400 transform3 t=P.t;
1401
1402 if(ninterpolate == 1 || piecewisestraight(p)) {
1403 g=project(point(p,0),t);
1404 // Construct the path.
1405 int stop=cyclic(p) ? last-1 : last;
1406 for(int i=0; i < stop; ++i) {
1407 if(straight(p,i))
1408 g=g--project(point(p,i+1),t);
1409 else {
1410 g=g..controls project(postcontrol(p,i),t) and
1411 project(precontrol(p,i+1),t)..project(point(p,i+1),t);
1412 }
1413 }
1414 } else return nurb(p,P);
1415
1416 if(cyclic(p))

1462// Return a unit normal vector to a planar path p (or O if the path is
1463// nonplanar).
1464triple normal(path3 p)
1465{
1466 triple normal;
1467 real fuzz=sqrtEpsilon*abs(max(p)-min(p));
1468 real absnormal;
1469 real theta;
1470
1471 bool Cross(triple a, triple b) {
1472 if(abs(a) >= fuzz && abs(b) >= fuzz) {
1473 triple n=cross(unit(a),unit(b));
1474 real absn=abs(n);
1475 if(absn < sqrtEpsilon) return false;
1476 n=unit(n);
1477 if(absnormal > 0 &&
1478 abs(normal-n) > sqrtEpsilon && abs(normal+n) > sqrtEpsilon)
1479 return true;
1480 else {
1481 int sign=dot(n,normal) >= 0 ? 1 : -1;
1482 theta += sign*asin1(absn);
1483 if(absn > absnormal) {
1484 absnormal=absn;
1485 normal=n;
1486 theta=sign*theta;
1487 }
1488 }
1489 }
1490 return false;
1491 }
1492
1493 int L=length(p);
1494 if(L <= 0) return O;
1495
1496 triple zi=point(p,0);
1497 triple v0=zi-precontrol(p,0);
1498 for(int i=0; i < L; ++i) {
1499 triple c0=postcontrol(p,i);
1500 triple c1=precontrol(p,i+1);
1501 triple zp=point(p,i+1);
1502 triple v1=c0-zi;
1503 triple v2=c1-c0;
1504 triple v3=zp-c1;
1505 if(Cross(v0,v1) || Cross(v1,v2) || Cross(v2,v3)) return O;
1506 v0=v3;
1507 zi=zp;
1508 }
1509 return theta >= 0 ? normal : -normal;
1510}
