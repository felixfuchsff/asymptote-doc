81restoreThunk buildRestoreThunk()
82{
83 // Call the save functions in reverse order, storing their restore thunks.
84 restoreThunk[] thunks={};
85 for(int i=saveFunctions.length-1; i >= 0; --i)
86 thunks.push(saveFunctions[i]());
87
88 return new void() {
89 // Call the restore thunks in an order matching the saves.
90 for(int i=thunks.length-1; i >= 0; --i)
91 thunks[i]();
92 };
93}

95// Add the default save function.
96addSaveFunction(new restoreThunk() {

102 defaultpen(defaultpen);
103 currentpen=p;

106 restore=r;

255void asy(string format, bool overwrite=false ... string[] s)
256{
257 for(string f : s) {
258 f=stripsuffix(f);
259 string suffix="."+format;
260 string fsuffix=stripdirectory(f+suffix);
261 if(overwrite || error(input(outdirectory()+fsuffix,check=false))) {
262 string outformat=settings.outformat;
263 bool interactiveView=settings.interactiveView;
264 bool batchView=settings.batchView;
265 settings.outformat=format;

270 eval("import \""+f+"\" as dummy");
271 rename(stripsuffix(outname)+suffix,fsuffix);
272 settings.outformat=outformat;
273 settings.interactiveView=interactiveView;

296cputime cputime()
297{
298 static processtime last;
299 real[] a=_cputime();
300 cputime cputime;
301 real clock=a[4];
302 cputime.parent.user=a[0]; // Includes system time
303 cputime.parent.system=0;
304 cputime.parent.clock=clock;
305 cputime.child.user=a[2];
306 cputime.child.system=a[3];
307 cputime.child.clock=0;
308 real user=cputime.parent.user+cputime.child.user;
309 real system=cputime.parent.system+cputime.child.system;
310 cputime.change.user=user-last.user;
311 cputime.change.system=system-last.system;
312 cputime.change.clock=clock-last.clock;
313 last.user=user;
314 last.system=system;
315 last.clock=clock;
316 return cputime;
317}
