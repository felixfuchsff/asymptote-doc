272ImpDatum *ImpDatum::getField(const char *name)
273{
274 coenv &e = coenvInOngoingProcess();
275 symbol id = symbol::trans(name);
276
277 absyntax::exp *ex = getFieldExp(id);
278
279 types::ty *t = ex->getType(e);
280 if (t->isError())
281 return datumError("no field of that name");
282
283 if (t->isOverloaded())

395ImpDatum *callDatum(ImpDatum *callee, ImpArguments *args)
396{
397 coenv &e = coenvInOngoingProcess();
398
399 callExp callex(nullPos, callee->getExp(), args->getArgs());
400
401 types::ty *t = callex.getType(e);
402 if (t->isError()) {
403 // Run for errors.
404 runExp(&callex);
405 em.sync(true);
406 return datumError("invalid call");
407 }

411 if (t->kind == types::ty_void) {
