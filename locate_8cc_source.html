60// Find the appropriate file, first looking in the local directory, then the
61// directory given in settings, and finally the global system directory.
62string locateFile(string id, bool full, string suffix)
63{
64 if(id.empty()) return "";
65 file_list_t filenames = mungeFileName(id,suffix);
66 for (auto const& leaf : filenames) {
67 if (leaf[0] == '/') { // FIXME: Add windows path check
68 string file = leaf;
69 if (fs::exists(file))
70 return file;
71 } else {
72 for (auto const& dir : searchPath) {
73 string file = join(dir,leaf,full);
74 if (fs::exists(file))
75 return file;
76 }
77 }
78 }
79 return string();
