115template<class T>
116void readArray(vm::stack *s, Int nx=-1, Int ny=-1, Int nz=-1)
117{
118 camp::file *f = pop<camp::file*>(s);
119 vm::array *c=new vm::array(0);
120 if(f->isOpen()) {
121 if(nx != -1 && f->Nx() != -1) nx=f->Nx();
122 if(nx == -2) {f->read(nx); f->Nx(-1); if(nx == 0) {s->push(c); return;}}
123 if(ny != -1 && f->Ny() != -1) ny=f->Ny();
124 if(ny == -2) {f->read(ny); f->Ny(-1); if(ny == 0) {s->push(c); return;}}
125 if(nz != -1 && f->Nz() != -1) nz=f->Nz();
126 if(nz == -2) {f->read(nz); f->Nz(-1); if(nz == 0) {s->push(c); return;}}
127 T v;
128 if(nx >= 0) {
129 for(Int i=0; i < Limit(nx); i++) {
130 if(ny >= 0) {
131 vm::array *ci=new vm::array(0);
132 for(Int j=0; j < Limit(ny); j++) {
133 if(nz >= 0) {
134 vm::array *cij=new vm::array(0);
135 bool break2=false;
136 for(Int k=0; k < Limit(nz); k++) {
137 f->read(v);
138 if(f->error()) {
139 if(nx && ny && nz) reportEof(f,(i*ny+j)*nz+k);
140 s->push(c);
141 return;
142 }
143 if(k == 0) {
144 if(j == 0) c->push(ci);
145 ci->push(cij);
146 }
147 cij->push(v);
148 if(f->LineMode() && f->nexteol()) {
149 if(f->nexteol() && !f->CSVMode()) break2=true;
150 break;
151 }
152 }
153 if(break2) break;
154 } else {
155 f->read(v);
156 if(f->error()) {
157 if(nx && ny) reportEof(f,i*ny+j);
158 s->push(c);
159 return;
160 }
161 if(j == 0) c->push(ci);
162 ci->push(v);
163 if(f->LineMode() && f->nexteol()) break;
164 }
165 }
166 } else {
167 f->read(v);
168 if(f->error()) {
169 if(nx) reportEof(f,i);
170 s->push(c);
171 return;
172 }
173 c->push(v);
174 if(f->LineMode() && f->nexteol()) break;
175 }
176 }
177 } else {
178 for(;;) {
179 f->read(v);
180 if(f->error()) break;
181 c->push(v);
182 if(f->LineMode() && f->nexteol()) break;
183 }
184 }
185 if(interact::interactive) f->purgeStandard(v);
186 }
187 s->push(c);
188}
