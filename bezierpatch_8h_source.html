45 triple normal(triple left3, triple left2, triple left1, triple middle,
46 triple right1, triple right2, triple right3) {
47 triple lp=3.0*(left1-middle);
48 triple rp=3.0*(right1-middle);
49
50 triple n=cross(rp,lp);
51 if(abs2(n) > epsilon)
52 return n;
53
54 triple lpp=bezierPP(middle,left1,left2);
55 triple rpp=bezierPP(middle,right1,right2);
56
57 n=cross(rpp,lp)+cross(rp,lpp);

88 // Determine the flatness of a Bezier patch.
89 pair Distance(const triple *p) {
90 triple p0=p[0];
91 triple p3=p[3];
92 triple p12=p[12];
93 triple p15=p[15];
94
95 // Check the horizontal flatness.
96 double h=Flatness(p0,p12,p3,p15);
97 // Check straightness of the horizontal edges and interior control curves.
98 h=std::max(h,Straightness(p0,p[4],p[8],p12));
99 h=std::max(h,Straightness(p[1],p[5],p[9],p[13]));
100 h=std::max(h,Straightness(p[2],p[6],p[10],p[14]));
101 h=std::max(h,Straightness(p3,p[7],p[11],p15));
102
103 // Check the vertical flatness.
104 double v=Flatness(p0,p3,p12,p15);
105 // Check straightness of the vertical edges and interior control curves.
106 v=std::max(v,Straightness(p0,p[1],p[2],p3));
107 v=std::max(v,Straightness(p[4],p[5],p[6],p[7]));
108 v=std::max(v,Straightness(p[8],p[9],p[10],p[11]));
109 v=std::max(v,Straightness(p12,p[13],p[14],p15));
110
111 return pair(h,v);
112 }

198 // Check how far the internal point is from the centroid of the vertices.
