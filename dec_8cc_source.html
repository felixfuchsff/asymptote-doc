35bool usableInTemplate(ty *t) {
36 assert(t);
37 if (t->primitive()) return true;
38 assert(t->kind != ty_null);
39 assert(t->kind != ty_overloaded);
40
41 if (t->kind == ty_record) {
42 record* r= dynamic_cast<record*>(t);
43 assert(r);
44 assert(r->getLevel());
45 if (!r->getLevel()->getParent()) return false;// r is actually a module
46 if (!r->getLevel()->getParent()->getParent()) {
47 return true;// r is a top-level record, or all nestings are static
48 }
49 return false; // r is nested non-statically
50 }
51 if (t->kind == ty_function) {
52 function* f= dynamic_cast<function*>(t);
53 assert(f);
54 // Check the types of the result and all the parameters.
55 if (!usableInTemplate(f->result)) return false;
56 const signature& sig= *f->getSignature();
57 for (const types::formal& f : sig.formals) {
58 if (!usableInTemplate(f.t)) return false;
59 }
60 if (sig.hasRest() && !usableInTemplate(sig.getRest().t)) return false;
61 return true;
62 }
63 if (t->kind == ty_array) {
64 array* a= dynamic_cast<array*>(t);
65 assert(a);
66 return usableInTemplate(a->celltype);
67 }
68 // We should have already handled all the cases.
69 assert(false);
70 return false;
71}

89void addNameOps(coenv &e, record *r, record *qt, varEntry *qv, position pos) {
90 for (auto au : qt->autounravelRegistry.getAutoUnravels()) {
91 symbol auName = au.first;
92 varEntry *v = au.second;
93 if (!v->checkPerm(READ, e.c)) {
94 em.error(pos);
95 em << "cannot access '" << auName << "' in current scope";
96 continue;
97 }
98 if (r && e.c.getPermission() != PUBLIC) {
99 // Add an additional restriction to v based on c.getPermission().
100 v = new varEntry(*v, e.c.getPermission(), r);
101 }
102 varEntry *qqv = qualifyVarEntry(qv, v);
103 auto enter= [&](trans::venv& ve) {
104 // Add op only if it does not already exist.
105 if (!ve.lookByType(auName, qqv->getType())) {
106 ve.enter(auName, qqv);
107 }
108 };
109 if (r) {
110 enter(r->e.ve);
111 }
112 enter(e.e.ve);
113 }

329record *block::transAsFile(genv& ge, symbol id)
330{
331 // Create the new module.
332 record *r = new record(id, new frame(id,0,0));
333
334 // Create coder and environment to translate the module.
335 // File-level modules have dynamic fields by default.
336 coder c(getPos(), r, 0);
337 env e(ge);
338 coenv ce(c, e);
339
340 if (settings::getSetting<bool>("autoplain")) {
341 autoplainRunnable()->transAsField(ce, r);
342 }
343 // If the file starts with a template declaration and it was accessed
344 // or imported without template arguments, further translation is
345 // likely futile.
346 if (getTypedefDec() != nullptr) {
347 if(!settings::getSetting<bool>("listvariables")) {
348 em.error(getPos());
349 em << "templated module access requires template parameters";
350 em.sync();
351 }
352 return r;
353 }
354
355 // Translate the abstract syntax.
356 transAsField(ce, r);
357 ce.c.closeRecord();
358
359 em.sync();
360 if (em.errors()) return nullptr;
361
362 return r;
363}

716types::ty *inferType(position pos, coenv &e, varinit *init)
717{
718 if (!init) {

982 formals *args)

987 if (!computedArgs) {

991 record *imp=e.e.getTemplatedModule(moduleName,computedArgs);
992 if (!imp) {

1001 // order, meaning that if more than one parameter has an error, the first

1111void recordInitializer(coenv &e, symbol id, record *r, position here)
1112{
1113 // This is almost equivalent to the code
1114 // autounravel A operator init() { return new A; }
1115 // where A is the name of the record. The "almost" is because the code below
1116 // does not add the operator init to the environment, since in practice it
1117 // would be added to the outer environment, not the record's environment.
1118 // Since it is always added *after* any code in the record, we lose nothing
1119 // by not adding it to the environment.
1120 // Additionally, the "autounravel" is made low-priority (will not override
1121 // user-defined operator init) which is possible only for built-in functions.
1122 formals formals(here);
1123 simpleName recordName(here, id);
1124 nameTy result(here, &recordName);
1125 newRecordExp exp(here, &result);
1126 returnStm stm(here, &exp);
1127 fundef fun(here, &result, &formals, &stm);
1128 assert(r);
1129 {
1130 e.c.pushModifier(AUTOUNRAVEL);
1131 function *ft = fun.transType(e);
1132 assert(ft);
1133
1134 symbol initSym=symbol::opTrans("init");
1135 varinit *init=fun.makeVarInit(ft);
1136
1137 assert(ft->kind != types::ty_inferred);
1138
1139 varEntry *v=makeVarEntry(here, e, r, ft);
1140 r->e.addVar(initSym, v);
1141 r->autounravelRegistry.registerAutoUnravel(
1142 initSym, v, trans::AutounravelPriority::OFFER
1143 );
1144 initializeVar(here, e, v, init);
1145 e.c.popModifier();
1146 }

1150 symbol name = arg->dest;
1151 ty *t = arg->t;
1152 if (name != paramSym) {
1153 em.error(arg->pos);

1163 return true;

1165

1167 record *r = dynamic_cast<record *>(t);
1168 assert(r);

1170 // Build a varEntry v for the parent level and encode the bytecode
1171 // to pop the parent off the stack into v. The varEntry needs a type,
1172 // but the only thing we use from the type is the level, so make a
1173 // new fake record.

1178 oss << "<fake parent of " << name << " holding ";
1179 print(oss, r->getLevel()->getParent());
1180 oss << ">";

1187 symbol::literalTrans(fakeParentName), r->getLevel()->getParent()
1188 );

1197
1198 // Add any autounravel fields.
1199 addNameOps(e, module, r, v, getPos());
