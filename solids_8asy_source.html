8real[] tangent(path p, path q, bool side)
9{
10 static real fuzz=1.0e-5;
11
12 if((cyclic(p) && inside(p,point(q,0)) ||
13 cyclic(q) && inside(q,point(p,0))) &&
14 intersect(p,q,fuzz).length == 0) return new real[];

16 for(int i=0; i < 100; ++i) {
17 real ta=side ? mintimes(p)[1] : maxtimes(p)[1];
18 real tb=side ? mintimes(q)[1] : maxtimes(q)[1];
19 pair a=point(p,ta);
20 pair b=point(q,tb);
21 real angle=angle(b-a,warn=false);
22 if(abs(angle) <= sqrtEpsilon || abs(abs(0.5*angle)-pi) <= sqrtEpsilon)

36// Return the projection of a generalized cylinder of height h constructed
37// from area base in the XY plane and aligned with axis.
38path[] cylinder(path3 base, real h, triple axis=Z, projection P)
39{
40 base=rotate(-colatitude(axis),cross(axis,Z))*base;
41 path3 top=shift(h*axis)*base;
42 path Base=project(base,P);
43 path Top=project(top,P);
44 real[] t1=tangent(Base,Top,true);
45 real[] t2=tangent(Base,Top,false);
46 path p=subpath(Base,t1[0]/P.ninterpolate,t2[0]/P.ninterpolate);
47 path q=subpath(Base,t2[0]/P.ninterpolate,t1[0]/P.ninterpolate);
48 return Base^^Top^^line(Base,Top,t1)^^line(Base,Top,t2);
