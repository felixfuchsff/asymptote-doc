53 // Sort the points in the pair array z according to whether they lie on the
54 // left or right side of the line L in the direction dir passing through P.
55 // Points exactly on L are considered to be on the right side.
56 // Also push any points of intersection of L with the path operator --(... z)
57 // onto each of the arrays left and right.
58 void operator init(pair dir, pair P ... pair[] z) {
59 pair lastz;
60 pair invdir=dir != 0 ? 1/dir : 0;
61 bool left,last;
62 for(int i=0; i < z.length; ++i) {
63 left=(invdir*z[i]).y > (invdir*P).y;
64 if(i > 0 && last != left) {
65 pair w=extension(P,P+dir,lastz,z[i]);
66 this.left.push(w);
67 this.right.push(w);
68 }
69 if(left) this.left.push(z[i]);
70 else this.right.push(z[i]);
71 last=left;
72 lastz=z[i];
73 }
74 }

82splitface split(face a, face cut, projection P)
83{
84 splitface S;
85
86 void nointersection() {
87 if(abs(dot(a.point-P.camera,a.normal)) >=
88 abs(dot(cut.point-P.camera,cut.normal))) {
89 S.back=a;
90 S.front=null;
91 } else {
92 S.back=null;
93 S.front=a;
94 }
95 }
96
97 if(P.infinity) {
98 P=P.copy();
99 static real factor=1/sqrtEpsilon;
100 P.camera *= factor*max(abs(a.min),abs(a.max),
101 abs(cut.min),abs(cut.max));
102 }
103
104 if((abs(a.normal-cut.normal) < epsilon ||
105 abs(a.normal+cut.normal) < epsilon)) {
106 nointersection();
107 return S;
108 }
109
110 line L=intersection(a,cut);
111
112 if(dot(P.camera-L.point,P.camera-P.target) < 0) {
113 nointersection();
114 return S;
115 }
116
117 pair point=a.t*project(L.point,P);
118 pair dir=a.t*project(L.point+L.dir,P)-point;
119 pair invdir=dir != 0 ? 1/dir : 0;
120 triple apoint=L.point+cross(L.dir,a.normal);
121 bool left=(invdir*(a.t*project(apoint,P))).y >= (invdir*point).y;
122
123 real t=intersect(apoint,P.camera,cut.normal,cut.point);
124 bool rightfront=left ^ (t <= 0 || t >= 1);
125
126 face back=a, front=a.copy();
127 pair max=max(a.fit);
128 pair min=min(a.fit);
129 half h=half(dir,point,max,(min.x,max.y),min,(max.x,min.y),max);
130 if(h.right.length == 0) {
131 if(rightfront) front=null;
132 else back=null;
133 } else if(h.left.length == 0) {
134 if(rightfront) back=null;
135 else front=null;
136 }
137 if(front != null)
138 clip(front.fit,operator --(... rightfront ? h.right : h.left)--cycle,
139 zerowinding);
140 if(back != null)
141 clip(back.fit,operator --(... rightfront ? h.left : h.right)--cycle,
142 zerowinding);
143 S.back=back;
144 S.front=front;
145 return S;
146}
