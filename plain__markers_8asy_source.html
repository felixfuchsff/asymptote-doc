15// Return a unit n-point cyclic cross, with optional inner radius r and
16// end rounding.
17path cross(int n, bool round=true, real r=0)
18{
19 assert(n > 1);
20 real r=min(r,1);
21 real theta=pi/n;
22 real s=sin(theta);
23 real c=cos(theta);
24 pair z=(c,s);
25 transform mirror=reflect(0,z);
26 pair p1=(r,0);
27 path elementary;
28 if(round) {
29 pair e1=p1+z*max(1-r*(s+c),0);
30 elementary=p1--e1..(c,s)..mirror*e1--mirror*p1;
31 } else {
32 pair p2=p1+z*(max(sqrt(1-(r*s)^2)-r*c),0);
33 elementary=p1--p2--mirror*p2--mirror*p1;
34 }
35
36 guide g;
37 real step=360/n;
38 for(int i=0; i < n; ++i)
39 g=g--rotate(i*step-90)*elementary;
40
41 return g--cycle;
42}

55// On picture pic, add n copies of frame f to path g, evenly spaced in
56// arclength.
57// If rotated=true, the frame will be rotated by the angle of the tangent
58// to the path at the points where the frame will be added.
59// If centered is true, center the frames within n evenly spaced arclength
60// intervals.
61markroutine markuniform(bool centered=false, int n, bool rotated=false) {
62 return new void(picture pic=currentpicture, frame f, path g) {
63 if(n <= 0) return;
64 void add(real x) {
65 real t=reltime(g,x);
66 add(pic,rotated ? rotate(degrees(dir(g,t)))*f : f,point(g,t));
67 }
68 if(centered) {
69 real width=1/n;
70 for(int i=0; i < n; ++i) add((i+0.5)*width);
71 } else {
72 if(n == 1) add(0.5);
73 else {
74 real width=1/(n-1);
75 for(int i=0; i < n; ++i)
76 add(i*width);
77 }
78 }
79 };
80}

122marker markthin(path g, pen p=currentpen,

187picture legend(Legend[] Legend, int perline=1, real linelength,
188 real hskip, real vskip, real maxwidth=0, real maxheight=0,
189 bool hstretch=false, bool vstretch=false)
190{
191 if(maxwidth <= 0) hstretch=false;
192 if(maxheight <= 0) vstretch=false;
193 if(Legend.length <= 1) vstretch=hstretch=false;
194
195 picture inset;
196 size(inset,0,0,IgnoreAspect);
197
198 if(Legend.length == 0)
199 return inset;
200
201 // Check for legend entries with lines:
202 bool bLineEntriesAvailable=false;
203 for(int i=0; i < Legend.length; ++i) {
204 if(Legend[i].p != invisible) {
205 bLineEntriesAvailable=true;
206 break;
207 }
208 }
209
210 real markersize=0;
211 for(int i=0; i < Legend.length; ++i)
212 markersize=max(markersize,size(Legend[i].mark).x);
213
214 // If no legend has a line, set the line length to zero
215 if(!bLineEntriesAvailable)
216 linelength=0;
217
218 linelength=max(linelength,markersize*(linelength == 0 ? 1 : 2));
219
220 // Get the maximum dimensions per legend entry;
221 // calculate line length for a one-line legend
222 real heightPerEntry=0;
223 real widthPerEntry=0;
224 real totalwidth=0;
225 for(int i=0; i < Legend.length; ++i) {
226 picture pic=legenditem(Legend[i],linelength);
227 pair lambda=size(pic);
228 heightPerEntry=max(heightPerEntry,lambda.y);
229 widthPerEntry=max(widthPerEntry,lambda.x);
230 if(Legend[i].p != invisible)
231 totalwidth += lambda.x;
232 else {
233 // Legend entries without leading line need less space in one-line legends
234 picture pic=legenditem(Legend[i],0);
235 totalwidth += size(pic).x;
236 }
237 }
238 // Does everything fit into one line?
239 if(((perline < 1) || (perline >= Legend.length)) &&
240 (maxwidth >= totalwidth+(totalwidth/Legend.length)*
241 (Legend.length-1)*(hskip-1))) {
242 // One-line legend
243 real currPosX=0;
244 real itemDistance;
245 if(hstretch)
246 itemDistance=(maxwidth-totalwidth)/(Legend.length-1);
247 else
248 itemDistance=(totalwidth/Legend.length)*(hskip-1);
249 for(int i=0; i < Legend.length; ++i) {
250 picture pic=legenditem(Legend[i],
251 Legend[i].p == invisible ? 0 : linelength);
252 add(inset,pic,(currPosX,0));
253 currPosX += size(pic).x+itemDistance;
254 }
255 } else {
256 // multiline legend
257 if(maxwidth > 0) {
258 int maxperline=floor(maxwidth/(widthPerEntry*hskip));
259 if((perline < 1) || (perline > maxperline))
260 perline=maxperline;
261 }
262 if(perline < 1) // This means: maxwidth < widthPerEntry
263 perline=1;
264
265 if(perline <= 1) hstretch=false;
266 if(hstretch) hskip=(maxwidth/widthPerEntry-perline)/(perline-1)+1;
267 if(vstretch) {
268 int rows=ceil(Legend.length/perline);
269 vskip=(maxheight/heightPerEntry-rows)/(rows-1)+1;
270 }
271
272 if(hstretch && (perline == 1)) {
273 Draw(inset,(0,0)--(maxwidth,0),invisible());
274 for(int i=0; i < Legend.length; ++i)
275 add(inset,legenditem(Legend[i],linelength),
276 (0.5*(maxwidth-widthPerEntry),
277 -quotient(i,perline)*heightPerEntry*vskip));
278 } else
279 for(int i=0; i < Legend.length; ++i)
280 add(inset,legenditem(Legend[i],linelength),
281 ((i%perline)*widthPerEntry*hskip,
282 -quotient(i,perline)*heightPerEntry*vskip));
283 }
284
285 return inset;
286}
