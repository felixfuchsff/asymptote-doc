732 SymbolArgContainer<std::string> fromDeclCreateTraverse(std::unordered_set<std::string> const& symbols)
733 {
734 // base
735 SymbolArgContainer<std::string> base=defaultCreateTraverse(symbols);
736
737 for (auto const& sym : symbols)
738 {
739 auto aliasSearch = extRefs.fromAccessVals.find(sym);
740 if (aliasSearch != extRefs.fromAccessVals.end())
741 {
742 // there's an alias to dest -> [src, ctx].
743 auto const& src=std::get<0>(aliasSearch->second);
744 auto const& fileName=std::get<1>(aliasSearch->second);
745
746 auto baseTrav = base.find(fileName);
747 if (baseTrav == base.end())
748 {
749 base.emplace(fileName, std::unordered_set<std::string> { src });
750 }
751 else
752 {
753 baseTrav->second.emplace(src);
754 }
755 }
756 }
757
758 return base;
759 }
