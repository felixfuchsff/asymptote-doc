46// Solve for the real roots of the quadratic equation ax^2+bx+c=0.
47quadraticroots::quadraticroots(double a, double b, double c)
48{
49 // Remove roots at numerical infinity.
50 if(fabs(a) <= Fuzz2*fabs(b)+Fuzz4*fabs(c)) {
51 if(fabs(b) > Fuzz2*fabs(c)) {
52 distinct=quadraticroots::ONE;
53 roots=1;
54 t1=-c/b;
55 } else if(c == 0.0) {
56 distinct=quadraticroots::MANY;
57 roots=1;
58 t1=0.0;
59 } else {
60 distinct=quadraticroots::NONE;
61 roots=0;
62 }
63 } else {
64 double factor=0.5*b/a;
65 double denom=b*factor;
66 if(fabs(denom) <= Fuzz2*fabs(c)) {
67 double x=-c/a;
68 if(x >= 0.0) {
69 distinct=quadraticroots::TWO;
70 roots=2;
71 t2=sqrt(x);
72 t1=-t2;
73 } else {
74 distinct=quadraticroots::NONE;
75 roots=0;
76 }
77 } else {
78 double x=-2.0*c/denom;
79 if(x > -1.0) {
80 distinct=quadraticroots::TWO;
81 roots=2;
82 double r2=factor*sqrt1pxm1(x);
83 double r1=-r2-2.0*factor;
84 if(r1 <= r2) {
85 t1=r1;
86 t2=r2;
87 } else {
88 t1=r2;
89 t2=r1;
90 }
91 } else if(x == -1.0) {
92 distinct=quadraticroots::ONE;
93 roots=2;
94 t1=t2=-factor;
95 } else {
96 distinct=quadraticroots::NONE;
97 roots=0;
98 }
99 }
100 }
101}

103// Solve for the complex roots of the quadratic equation ax^2+bx+c=0.
104Quadraticroots::Quadraticroots(pair a, pair b, pair c)
105{
106 if(a == 0.0) {
107 if(b != 0.0) {
108 roots=1;
109 z1=-c/b;
110 } else if(c == 0.0) {
111 roots=1;
112 z1=0.0;
113 } else
114 roots=0;
115 } else {
116 roots=2;
117 pair factor=0.5*b/a;
118 pair denom=b*factor;
119 if(denom == 0.0) {
120 z1=Sqrt(-c/a);
121 z2=-z1;
122 } else {
123 z1=factor*sqrt1pxm1(-2.0*c/denom);
124 z2=-z1-2.0*factor;
125 }
126 }

134// Accurate computation of cbrt(sqrt(1+x)+1)-cbrt(sqrt(1+x)-1).
135inline double cbrtsqrt1pxm(double x)

137 double s=sqrt1pxm1(x);
138 return 2.0/(cbrt(x+2.0*(sqrt(1.0+x)+1.0))+cbrt(x)+cbrt(s*s));

141// Taylor series of cos((atan(1.0/w)+pi)/3.0).
142static inline double costhetapi3(double w)
143{
144 static const double c1=1.0/3.0;
145 static const double c3=-19.0/162.0;
146 static const double c5=425.0/5832.0;
147 static const double c7=-16829.0/314928.0;
148 double w2=w*w;
149 double w3=w2*w;
150 double w5=w3*w2;
151 return c1*w+c3*w3+c5*w5+c7*w5*w2;
152}

154// Solve for the real roots of the cubic equation ax^3+bx^2+cx+d=0.
155cubicroots::cubicroots(double a, double b, double c, double d)
156{
157 static const double ninth=1.0/9.0;
158 static const double fiftyfourth=1.0/54.0;
159
160 // Remove roots at numerical infinity.
161 if(fabs(a) <= Fuzz2*(fabs(b)+fabs(c)*Fuzz2+fabs(d)*Fuzz4)) {
162 quadraticroots q(b,c,d);
163 roots=q.roots;
164 if(q.roots >= 1) t1=q.t1;
165 if(q.roots == 2) t2=q.t2;
166 return;
167 }
168
169 // Detect roots at numerical zero.
170 if(fabs(d) <= Fuzz2*(fabs(c)+fabs(b)*Fuzz2+fabs(a)*Fuzz4)) {
171 quadraticroots q(a,b,c);
172 roots=q.roots+1;
173 t1=0;
174 if(q.roots >= 1) t2=q.t1;
175 if(q.roots == 2) t3=q.t2;
176 return;
177 }
178
179 b /= a;
180 c /= a;
181 d /= a;
182
183 double b2=b*b;
184 double Q=3.0*c-b2;
185 if(fabs(Q) < Fuzz2*(3.0*fabs(c)+fabs(b2)))
186 Q=0.0;
187
188 double R=(3.0*Q+b2)*b-27.0*d;
189 if(fabs(R) < Fuzz2*((3.0*fabs(Q)+fabs(b2))*fabs(b)+27.0*fabs(d)))
190 R=0.0;
191
192 Q *= ninth;
193 R *= fiftyfourth;
194
195 double Q3=Q*Q*Q;
196 double R2=R*R;
197 double D=Q3+R2;
198 double mthirdb=-b*third;
199
200 if(D > 0.0) {
201 roots=1;
202 t1=mthirdb;
203 if(R2 != 0.0) t1 += cbrt(R)*cbrtsqrt1pxm(Q3/R2);
204 } else {
205 roots=3;
206 double v=0.0,theta;
207 if(R2 > 0.0) {
208 v=sqrt(-D/R2);
209 theta=atan(v);
210 } else theta=0.5*PI;
211 double factor=2.0*sqrt(-Q)*(R >= 0 ? 1 : -1);
212
213 t1=mthirdb+factor*cos(third*theta);
214 t2=mthirdb-factor*cos(third*(theta-PI));
215 t3=mthirdb;
216 if(R2 > 0.0)
217 t3 -= factor*((v < 100.0) ? cos(third*(theta+PI)) : costhetapi3(1.0/v));
218 }
219}

223 checkEmpty(n);

251 bcd = one_t*bc + t*cd,

376inline void splitCubic(solvedKnot sn[], double t, const solvedKnot& left_,
377 const solvedKnot& right_)
378{
379 solvedKnot &left=(sn[0]=left_), &mid=sn[1], &right=(sn[2]=right_);
380 if(left.straight) {
381 mid.point=split(t,left.point,right.point);
382 pair deltaL=third*(mid.point-left.point);
383 left.post=left.point+deltaL;
384 mid.pre=mid.point-deltaL;
385 pair deltaR=third*(right.point-mid.point);
386 mid.post=mid.point+deltaR;
387 right.pre=right.point-deltaR;
388 mid.straight=true;
389 } else {
390 pair x=split(t,left.post,right.pre); // m1
391 left.post=split(t,left.point,left.post); // m0
392 right.pre=split(t,right.pre,right.point); // m2
393 mid.pre=split(t,left.post,x); // m3
394 mid.post=split(t,x,right.pre); // m4
395 mid.point=split(t,mid.pre,mid.post); // m5
396 }
397}

399path path::subpath(double a, double b) const
400{
401 if(empty()) return path();
402
403 if (a > b) {
404 const path &rp = reverse();
405 Int len=length();
406 return rp.subpath(len-a, len-b);
407 }
408
409 solvedKnot aL, aR, bL, bR;
410 if (!cycles) {
411 if (a < 0) {
412 a = 0;
413 if (b < 0)
414 b = 0;
415 }
416 if (b > n-1) {
417 b = n-1;
418 if (a > b)
419 a = b;
420 }
421 aL = nodes[(Int)floor(a)];
422 aR = nodes[(Int)ceil(a)];
423 bL = nodes[(Int)floor(b)];
424 bR = nodes[(Int)ceil(b)];
425 } else {
426 if(run::validInt(a) && run::validInt(b)) {
427 aL = nodes[imod((Int) floor(a),n)];
428 aR = nodes[imod((Int) ceil(a),n)];
429 bL = nodes[imod((Int) floor(b),n)];
430 bR = nodes[imod((Int) ceil(b),n)];
431 } else reportError("invalid path index");
432 }
433
434 if (a == b) return path(point(a));
435
436 solvedKnot sn[3];
437 path p = subpath(Ceil(a), Floor(b));
438 if (a > floor(a)) {
439 if (b < ceil(a)) {
440 splitCubic(sn,a-floor(a),aL,aR);
441 splitCubic(sn,(b-a)/(ceil(b)-a),sn[1],sn[2]);
442 return path(sn[0],sn[1]);
443 }
444 splitCubic(sn,a-floor(a),aL,aR);
445 p=concat(path(sn[1],sn[2]),p);
446 }
447 if (ceil(b) > b) {
448 splitCubic(sn,b-floor(b),bL,bR);
449 p=concat(p,path(sn[0],sn[1]));
450 }
451 return p;
452}

463// Calculate the coefficients of a Bezier derivative divided by 3.
464static inline void derivative(pair& a, pair& b, pair& c,
465 const pair& z0, const pair& c0,
466 const pair& c1, const pair& z1)
467{
468 a=z1-z0+3.0*(c0-c1);
469 b=2.0*(z0+c1)-4.0*c0;
470 c=c0-z0;
471}

589// Calculates arclength of a cubic Bezier curve using adaptive Simpson
590// integration.
591double arcLength(const pair& z0, const pair& c0, const pair& c1,
592 const pair& z1)
593{
594 double integral;
595 derivative(a,b,c,z0,c0,c1,z1);
596
597 if(!simpson(integral,ds,0.0,1.0,DBL_EPSILON,1.0))
598 reportError("nesting capacity exceeded in computing arclength");
599 return integral;
600}

602double path::cubiclength(Int i, double goal) const
603{
604 const pair& z0=point(i);
605 const pair& z1=point(i+1);
606 double L;
607 if(straight(i)) {
608 L=(z1-z0).length();
609 if(goal < 0 || L == 0 || goal >= L) return L;
610 return -goal/L;
611 }
612
613 double integral=arcLength(z0,postcontrol(i),precontrol(i+1),z1);
614
615 L=3.0*integral;
616 if(goal < 0 || goal >= L) return L;
617
618 double t=goal/L;
619 goal *= third;
620 static double dxmin=sqrt(DBL_EPSILON);
621 if(!unsimpson(goal,ds,0.0,t,100.0*DBL_EPSILON,integral,1.0,dxmin))
622 reportError("nesting capacity exceeded in computing arctime");
623 return -t;
624}

628 if (cached_length != -1) return cached_length;

630 double L=0.0;

639double path::arctime(double goal) const
640{
641 if (cycles) {
642 if (goal == 0 || cached_length == 0) return 0;
643 if (goal < 0) {
644 const path &rp = this->reverse();
645 double result = -rp.arctime(-goal);
646 return result;
647 }
648 if (cached_length > 0 && goal >= cached_length) {
649 Int loops = (Int)(goal / cached_length);
650 goal -= loops*cached_length;
651 return loops*n+arctime(goal);
652 }
653 } else {
654 if (goal <= 0)
655 return 0;
656 if (cached_length > 0 && goal >= cached_length)
657 return n-1;
658 }
659
660 double l,L=0;
661 for (Int i = 0; i < n-1; i++) {
662 l = cubiclength(i,goal);
663 if (l < 0)
664 return (-l+i);
665 else {
666 L += l;
667 goal -= l;
668 if (goal <= 0)
669 return i+1;
670 }
671 }
672 if (cycles) {
673 l = cubiclength(n-1,goal);
674 if (l < 0)
675 return -l+n-1;
676 if (cached_length > 0 && cached_length != L+l) {
677 reportError("arclength != length.\n"
678 "path::arclength(double) must have broken semantics.\n"
679 "Please report this error.");
680 }
681 cached_length = L += l;
682 goal -= l;
683 return arctime(goal)+n;
684 }
685 else {
686 cached_length = L;
687 return length();
688 }
689}

693// {{{ Direction Time Calculation
694// Algorithm Stolen from Knuth's MetaFont
695inline double cubicDir(const solvedKnot& left, const solvedKnot& right,
696 const pair& rot)
697{
698 pair a,b,c;
699 derivative(a,b,c,left.point,left.post,right.pre,right.point);
700 a *= rot; b *= rot; c *= rot;
701
702 quadraticroots ret(a.gety(),b.gety(),c.gety());
703 switch(ret.distinct) {
704 case quadraticroots::MANY:
705 case quadraticroots::ONE:
706 {
707 if(goodroot(a.getx(),b.getx(),c.getx(),ret.t1)) return ret.t1;
708 } break;
709
710 case quadraticroots::TWO:
711 {
712 if(goodroot(a.getx(),b.getx(),c.getx(),ret.t1)) return ret.t1;
713 if(goodroot(a.getx(),b.getx(),c.getx(),ret.t2)) return ret.t2;
714 } break;
715
716 case quadraticroots::NONE:
717 break;
718 }
719
720 return -1;
721}

725double path::directiontime(const pair& dir) const {
726 if (dir == pair(0,0)) return 0;
727 pair rot = pair(1,0)/unit(dir);
728
729 double t; double pre,post;
730 for (Int i = 0; i < n-1+cycles; ) {
731 t = cubicDir(this->nodes[i],(cycles && i==n-1) ? nodes[0]:nodes[i+1],rot);
732 if (t >= 0) return i+t;
733 i++;
734 if (cycles || i != n-1) {
735 pair Pre = (point(i)-precontrol(i))*rot;
736 pair Post = (postcontrol(i)-point(i))*rot;
737 static pair zero(0.0,0.0);
738 if(Pre != zero && Post != zero) {
739 pre = angle(Pre);
740 post = angle(Post);
741 if ((pre <= 0 && post >= 0 && pre >= post - PI) ||
742 (pre >= 0 && post <= 0 && pre <= post + PI))
743 return i;
744 }
745 }
746 }
747
748 return -1;
749}

766 double x)

775// Return all intersection times of path g with the pair z.
776void intersections(std::vector<double>& T, const path& g, const pair& z,
777 double fuzz)
778{
779 double fuzz2=fuzz*fuzz;
780 Int n=g.length();
781 bool cycles=g.cyclic();
782 for(Int i=0; i < n; ++i) {
783 // Check both directions to circumvent degeneracy.
784 std::vector<double> r;
785 roots(r,g.point(i).getx(),g.postcontrol(i).getx(),
786 g.precontrol(i+1).getx(),g.point(i+1).getx(),z.getx());
787 roots(r,g.point(i).gety(),g.postcontrol(i).gety(),
788 g.precontrol(i+1).gety(),g.point(i+1).gety(),z.gety());
789
790 size_t m=r.size();
791 for(size_t j=0 ; j < m; ++j) {
792 double t=r[j];
793 if(t >= -Fuzz2 && t <= 1.0+Fuzz2) {
794 double s=i+t;
795 if((g.point(s)-z).abs2() <= fuzz2) {
796 if(cycles && s >= n-Fuzz2) s=0;
797 T.push_back(s);
798 }
799 }
800 }
801 }
802}

804inline bool online(const pair& p, const pair& q, const pair& z, double fuzz)
805{
806 double norm=max(max(p.abs2(),q.abs2()),z.abs2());
807 if(p == q) return (z-p).abs2() <= fuzz*fuzz*norm;
808 pair v=q-p;
809 double cross=(z.getx()-p.getx())*v.gety()-v.getx()*(z.gety()-p.gety());
810 return cross*cross <= fuzz*fuzz*v.abs2()*norm;
811}

813// Return all intersection times of path g with the (infinite)
814// line through p and q; if there are an infinite number of intersection points,
815// the returned list is guaranteed to include the endpoint times of
816// the intersection if endpoints=true.
817void lineintersections(std::vector<double>& T, const path& g,
818 const pair& p, const pair& q, double fuzz,
819 bool endpoints=false)
820{
821 Int n=g.length();
822 if(n == 0) {
823 if(online(p,q,g.point((Int) 0),fuzz)) T.push_back(0.0);
824 return;
825 }
826 bool cycles=g.cyclic();
827 double dx=q.getx()-p.getx();
828 double dy=q.gety()-p.gety();
829 double det=p.gety()*q.getx()-p.getx()*q.gety();
830 double norm=max(p.abs2(),q.abs2());
831 for(Int i=0; i < n; ++i) {
832 pair z0=g.point(i);
833 pair c0=g.postcontrol(i);
834 pair c1=g.precontrol(i+1);
835 pair z1=g.point(i+1);
836 pair t3=z1-z0+3.0*(c0-c1);
837 pair t2=3.0*(z0+c1)-6.0*c0;
838 pair t1=3.0*(c0-z0);
839 double a=dy*t3.getx()-dx*t3.gety();
840 double b=dy*t2.getx()-dx*t2.gety();
841 double c=dy*t1.getx()-dx*t1.gety();
842 double d=dy*z0.getx()-dx*z0.gety()+det;
843 std::vector<double> r;
844 if(max(max(max(a*a,b*b),c*c),d*d) >
845 Fuzz4*max(norm,max(z0.abs2(),max(z1.abs2(),max(c0.abs2(),c1.abs2())))))
846 roots(r,a,b,c,d);
847 else r.push_back(0.0);
848 if(endpoints) {
849 path h=g.subpath(i,i+1);
850 intersections(r,h,p,fuzz);
851 intersections(r,h,q,fuzz);
852 if(online(p,q,z0,fuzz)) r.push_back(0.0);
853 if(online(p,q,z1,fuzz)) r.push_back(1.0);
854 }
855 size_t m=r.size();
856 for(size_t j=0 ; j < m; ++j) {
857 double t=r[j];
858 if(t >= -Fuzz2 && t <= 1.0+Fuzz2) {
859 double s=i+t;
860 if(cycles && s >= n-Fuzz2) s=0;
861 if(online(p,q,g.point(s),fuzz))
862 T.push_back(s);
863 }
864 }
865 }
866}

868// An optimized implementation of intersections(g,p--q);
869// if there are an infinite number of intersection points, the returned list is
870// only guaranteed to include the endpoint times of the intersection.
871void intersections(std::vector<double>& S, std::vector<double>& T,
872 const path& g, const pair& p, const pair& q, double fuzz)
873{
874 double length2=(q-p).abs2();
875 if(length2 == 0.0) {
876 std::vector<double> S1;
877 intersections(S1,g,p,fuzz);
878 size_t n=S1.size();
879 for(size_t i=0; i < n; ++i) {
880 S.push_back(S1[i]);
881 T.push_back(0.0);
882 }
883 } else {
884 pair factor=(q-p)/length2;
885 std::vector<double> S1;
886 lineintersections(S1,g,p,q,fuzz,true);
887 size_t n=S1.size();
888 for(size_t i=0; i < n; ++i) {
889 double s=S1[i];
890 double t=dot(g.point(s)-p,factor);
891 if(t >= -Fuzz2 && t <= 1.0+Fuzz2) {
892 S.push_back(s);
893 T.push_back(t);
894 }
895 }
896 }
897}

961bool intersections(double &s, double &t, std::vector<double>& S,
962 std::vector<double>& T, path& p, path& q,
963 double fuzz, bool single, bool exact, unsigned depth)
964{
965 if(errorstream::interrupt) throw interrupted();
966
967 double fuzz2=max(fuzzFactor*fuzz*fuzz,Fuzz2);
968
969 Int lp=p.length();
970 if(((lp == 1 && p.straight(0)) || lp == 0) && exact) {
971 std::vector<double> T1,S1;
972 intersections(T1,S1,q,p.point((Int) 0),p.point(lp),fuzz);
973 add(s,t,S,T,S1,T1,p,fuzz2,single);
974 return S1.size() > 0;
975 }
976
977 Int lq=q.length();
978 if(((lq == 1 && q.straight(0)) || lq == 0) && exact) {
979 std::vector<double> S1,T1;
980 intersections(S1,T1,p,q.point((Int) 0),q.point(lq),fuzz);
981 add(s,t,S,T,S1,T1,p,fuzz2,single);
982 return S1.size() > 0;
983 }
984
985 pair maxp=p.max();
986 pair minp=p.min();
987 pair maxq=q.max();
988 pair minq=q.min();
989
990 if(maxp.getx()+fuzz >= minq.getx() &&
991 maxp.gety()+fuzz >= minq.gety() &&
992 maxq.getx()+fuzz >= minp.getx() &&
993 maxq.gety()+fuzz >= minp.gety()) {
994 // Overlapping bounding boxes
995
996 --depth;
997// fuzz *= 2;
998// fuzz2=max(fuzzFactor*fuzz*fuzz,Fuzz2);
999
1000 if((maxp-minp).length()+(maxq-minq).length() <= fuzz || depth == 0) {
1001 if(single) {
1002 s=0.5;
1003 t=0.5;
1004 } else {
1005 S.push_back(0.5);
1006 T.push_back(0.5);
1007 }
1008 return true;
1009 }
1010
1011 path p1,p2;
1012 double pscale,poffset;
1013 std::vector<double> S1,T1;
1014
1015 if(lp <= 1) {
1016 if(lp == 1) p.halve(p1,p2);
1017 if(lp == 0 || p1 == p || p2 == p) {
1018 intersections(T1,S1,q,p.point((Int) 0),p.point((Int) 0),fuzz);
1019 add(s,t,S,T,S1,T1,p,fuzz2,single);
1020 return S1.size() > 0;
1021 }
1022 pscale=poffset=0.5;
1023 } else {
1024 Int tp=lp/2;
1025 p1=p.subpath(0,tp);
1026 p2=p.subpath(tp,lp);
1027 poffset=tp;
1028 pscale=1.0;
1029 }
1030
1031 path q1,q2;
1032 double qscale,qoffset;
1033
1034 if(lq <= 1) {
1035 if(lq == 1) q.halve(q1,q2);
1036 if(lq == 0 || q1 == q || q2 == q) {
1037 intersections(S1,T1,p,q.point((Int) 0),q.point((Int) 0),fuzz);
1038 add(s,t,S,T,S1,T1,p,fuzz2,single);
1039 return S1.size() > 0;
1040 }
1041 qscale=qoffset=0.5;
1042 } else {
1043 Int tq=lq/2;
1044 q1=q.subpath(0,tq);
1045 q2=q.subpath(tq,lq);
1046 qoffset=tq;
1047 qscale=1.0;
1048 }
1049
1050 bool Short=lp == 1 && lq == 1;
1051
1052 static size_t maxcount=9;
1053 size_t count=0;
1054
1055 if(intersections(s,t,S1,T1,p1,q1,fuzz,single,exact,depth)) {
1056 add(s,t,S,T,S1,T1,pscale,qscale,0.0,0.0,p,fuzz2,single);
1057 if(single || depth <= mindepth)
1058 return true;
1059 count += S1.size();
1060 if(Short && count > maxcount) return true;
1061 }
1062
1063 S1.clear();
1064 T1.clear();
1065 if(intersections(s,t,S1,T1,p1,q2,fuzz,single,exact,depth)) {
1066 add(s,t,S,T,S1,T1,pscale,qscale,0.0,qoffset,p,fuzz2,single);
1067 if(single || depth <= mindepth)
1068 return true;
1069 count += S1.size();
1070 if(Short && count > maxcount) return true;
1071 }
1072
1073 S1.clear();
1074 T1.clear();
1075 if(intersections(s,t,S1,T1,p2,q1,fuzz,single,exact,depth)) {
1076 add(s,t,S,T,S1,T1,pscale,qscale,poffset,0.0,p,fuzz2,single);
1077 if(single || depth <= mindepth)
1078 return true;
1079 count += S1.size();
1080 if(Short && count > maxcount) return true;
1081 }
1082
1083 S1.clear();
1084 T1.clear();
1085 if(intersections(s,t,S1,T1,p2,q2,fuzz,single,exact,depth)) {
1086 add(s,t,S,T,S1,T1,pscale,qscale,poffset,qoffset,p,fuzz2,single);
1087 if(single || depth <= mindepth)
1088 return true;
1089 count += S1.size();
1090 if(Short && count > maxcount) return true;
1091 }
1092
1093 return S.size() > 0;
1094 }
1095 return false;
1096}

1151// Interface to orient2d predicate optimized for pairs.
1152double orient2d(const pair& a, const pair& b, const pair& c)
1153{
1154 double detleft, detright, det;
1155 double detsum, errbound;
1156 double orient;
1157
1158 FPU_ROUND_DOUBLE;
1159
1160 detleft = (a.getx() - c.getx()) * (b.gety() - c.gety());
1161 detright = (a.gety() - c.gety()) * (b.getx() - c.getx());
1162 det = detleft - detright;
1163
1164 if (detleft > 0.0) {
1165 if (detright <= 0.0) {
1166 FPU_RESTORE;
1167 return det;
1168 } else {
1169 detsum = detleft + detright;
1170 }
1171 } else if (detleft < 0.0) {
1172 if (detright >= 0.0) {
1173 FPU_RESTORE;
1174 return det;
1175 } else {
1176 detsum = -detleft - detright;
1177 }
1178 } else {
1179 FPU_RESTORE;
1180 return det;
1181 }
1182
1183 errbound = ccwerrboundA * detsum;
1184 if ((det >= errbound) || (-det >= errbound)) {
1185 FPU_RESTORE;
1186 return det;
1187 }
1188
1189 double pa[]={a.getx(),a.gety()};
1190 double pb[]={b.getx(),b.gety()};
1191 double pc[]={c.getx(),c.gety()};
1192
1193 orient = orient2dadapt(pa, pb, pc, detsum);
1194 FPU_RESTORE;
1195 return orient;
1196}

1198// Returns true iff the point z lies in or on the bounding box
1199// of a,b,c, and d.
1200bool insidebbox(const pair& a, const pair& b, const pair& c, const pair& d,
1201 const pair& z)

1203 bbox B(a);
1204 B.addnonempty(b);
1205 B.addnonempty(c);
1206 B.addnonempty(d);

1208 && z.gety() <= B.top;

1216// Return true if point z is on z0--z1; otherwise compute contribution to
1217// winding number.
1218bool checkstraight(const pair& z0, const pair& z1, const pair& z, Int& count)
1219{
1220 if(z0.gety() <= z.gety() && z.gety() <= z1.gety()) {
1221 double side=orient2d(z0,z1,z);
1222 if(side == 0.0 && inrange(z0.getx(),z1.getx(),z.getx()))
1223 return true;
1224 if(z.gety() < z1.gety() && side > 0) ++count;
1225 } else if(z1.gety() <= z.gety() && z.gety() <= z0.gety()) {
1226 double side=orient2d(z0,z1,z);
1227 if(side == 0.0 && inrange(z0.getx(),z1.getx(),z.getx()))
1228 return true;
1229 if(z.gety() < z0.gety() && side < 0) --count;
1230 }
1231 return false;
1232}

1234// returns true if point is on curve; otherwise compute contribution to
1235// winding number.
1236bool checkcurve(const pair& z0, const pair& c0, const pair& c1,
1237 const pair& z1, const pair& z, Int& count, unsigned depth)
1238{
1239 if(depth == 0) return true;
1240 --depth;
1241
1242 if(insidebbox(z0,c0,c1,z1,z)) {
1243 const pair m0=0.5*(z0+c0);
1244 const pair m1=0.5*(c0+c1);
1245 const pair m2=0.5*(c1+z1);
1246 const pair m3=0.5*(m0+m1);
1247 const pair m4=0.5*(m1+m2);
1248 const pair m5=0.5*(m3+m4);
1249 if(checkcurve(z0,m0,m3,m5,z,count,depth) ||
1250 checkcurve(m5,m4,m2,z1,z,count,depth)) return true;
1251 } else
1252 if(checkstraight(z0,z1,z,count)) return true;
1253 return false;
1254}

1256// Return the winding number of the region bounded by the (cyclic) path
1257// relative to the point z, or the largest odd integer if the point lies on
1258// the path.
1259Int path::windingnumber(const pair& z) const
1260{
1261 static const Int undefined=Int_MAX % 2 ? Int_MAX : Int_MAX-1;
1262
1263 if(!cycles)
1264 reportError("path is not cyclic");
1265
1266 bbox b=bounds();
1267
1268 if(z.getx() < b.left || z.getx() > b.right ||
1269 z.gety() < b.bottom || z.gety() > b.top) return 0;
1270
1271 Int count=0;
1272 for(Int i=0; i < n; ++i)
1273 if(straight(i)) {
1274 if(checkstraight(point(i),point(i+1),z,count))
1275 return undefined;
1276 } else
1277 if(checkcurve(point(i),postcontrol(i),precontrol(i+1),point(i+1),z,count,
1278 maxdepth)) return undefined;
1279 return count;
1280}

1312path nurb(pair z0, pair z1, pair z2, pair z3,
1313 double w0, double w1, double w2, double w3, Int m)
1314{
1315 mem::vector<solvedKnot> nodes(m+1);
1316
1317 if(m < 1) reportError("invalid sampling interval");
1318
1319 double step=1.0/m;
1320 for(Int i=0; i <= m; ++i) {
1321 double t=i*step;
1322 double t2=t*t;
1323 double onemt=1.0-t;
1324 double onemt2=onemt*onemt;
1325 double W0=w0*onemt2*onemt;
1326 double W1=w1*3.0*t*onemt2;
1327 double W2=w2*3.0*t2*onemt;
1328 double W3=w3*t2*t;
1329 nodes[i].point=(W0*z0+W1*z1+W2*z2+W3*z3)/(W0+W1+W2+W3);
