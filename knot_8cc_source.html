54// Computes the relative distance of a control point given the angles.
55// The name is somewhat misleading as the velocity (with respect to the
56// variable that parameterizes the path) is relative to the distance
57// between the knots and even then, is actually three times this.
58// The routine is based on the velocity function in Section 131 of the MetaPost
59// code, but differs in that it automatically accounts for the tension and
60// bounding with tension atleast.
61double velocity(double theta, double phi, tension t)
62{
63 static const double VELOCITY_BOUND = 4.0;
64 static const double a = sqrt(2.0);
65 static const double b = 1.0/16.0;
66 static const double c = 1.5*(sqrt(5.0)-1.0);
67 static const double d = 1.5*(3.0-sqrt(5.0));
68
69 double st = sin(theta), ct = cos(theta),
70 sf = sin(phi), cf = cos(phi);
71
72 double denom = t.val * (3.0 + c*ct + d*cf);
73
74 double r = denom != 0.0 ? (2.0 + a*(st - b*sf)*(sf - b*st)*(ct-cf)) / denom
75 : VELOCITY_BOUND;
76
77 //cerr << " velocity(" << theta << "," << phi <<")= " << r << endl;
78 if (r > VELOCITY_BOUND)
79 r = VELOCITY_BOUND;
80
81 // Apply boundedness condition for tension atleast cases.
82 if (t.atleast)
83 {
84 double sine = sin(theta + phi);
85 if ((st >= 0.0 && sf >= 0.0 && sine > 0.0) ||
86 (st <= 0.0 && sf <= 0.0 && sine < 0.0))
87 {
88 double rmax = sf / sine;
89 if (r > rmax)
90 r = rmax;
91 }
92 }
93
94 return r;
95}

137eqn dirSpec::eqnOut(Int j, knotlist& l, cvector<double>&, cvector<double>&)
138{
139 // When choosing the control points, the path will come out the first knot
140 // going straight to the next knot rotated by the angle theta.
141 // Therefore, the angle theta we want is the difference between the
142 // specified heading given and the heading to the next knot.
143 double theta=reduceAngle(given-niceAngle(l[j+1].z-l[j].z));
144
145 // Give a simple linear equation to ensure this theta is picked.
146 return eqn(0.0,1.0,0.0,theta);
147}

149eqn dirSpec::eqnIn(Int j, knotlist& l, cvector<double>&, cvector<double>&)
150{
151 double theta=reduceAngle(given-niceAngle(l[j].z-l[j-1].z));
152 return eqn(0.0,1.0,0.0,theta);
153}

155eqn curlSpec::eqnOut(Int j, knotlist& l, cvector<double>&,
156 cvector<double>& psi)
157{
158 double alpha=l[j].alpha();
159 double beta=l[j+1].beta();
160
161 double chi=alpha*alpha*gamma/(beta*beta);
162
163 double C=alpha*chi+3-beta;
164 double D=(3.0-alpha)*chi+beta;
165
166 return eqn(0.0,C,D,-D*psi[j+1]);
167}

169eqn curlSpec::eqnIn(Int j, knotlist& l, cvector<double>&, cvector<double>&)
170{
171 double alpha=l[j-1].alpha();
172 double beta=l[j].beta();
173
174 double chi=beta*beta*gamma/(alpha*alpha);
175
176 double A=(3-beta)*chi+alpha;
177 double B=beta*chi+3-alpha;
178
179 return eqn(A,B,0.0,0.0);
180}

238struct eqnprop : public knotprop<eqn> {
239 cvector<double>& d;
240 cvector<double>& psi;
241
242 eqnprop(knotlist &l, cvector<double>& d, cvector<double>& psi)
243 : knotprop<eqn>(l), d(d), psi(psi) {}
244
245 eqn solo(Int) {
246 assert(False);
247 return eqn(0.0,1.0,0.0,0.0);
248 }
249
250 eqn start(Int j) {
251 // Defer to the specifier, as it knows the specifics.
252 return dynamic_cast<endSpec *>(l[j].out)->eqnOut(j,l,d,psi);
253 }
254
255 eqn end(Int j) {
256 return dynamic_cast<endSpec *>(l[j].in)->eqnIn(j,l,d,psi);
257 }
258
259 eqn mid(Int j) {
260 double lastAlpha = l[j-1].alpha();
261 double thisAlpha = l[j].alpha();
262 double thisBeta = l[j].beta();
263 double nextBeta = l[j+1].beta();
264
265 // Values based on the linear approximation of the curvature coming
266 // into the knot with respect to theta[j-1] and theta[j].
267 double inFactor = 1.0/(thisBeta*thisBeta*d[j-1]);
268 double A = lastAlpha*inFactor;
269 double B = (3.0 - lastAlpha)*inFactor;
270
271 // Values based on the linear approximation of the curvature going out of
272 // the knot with respect to theta[j] and theta[j+1].
273 double outFactor = 1.0/(thisAlpha*thisAlpha*d[j]);
274 double C = (3.0 - nextBeta)*outFactor;
275 double D = nextBeta*outFactor;
276
277 return eqn(A,B+C,D,-B*psi[j]-D*psi[j+1]);
278 }
279};

281// If the system of equations is homogeneous (ie. we are solving for x in
282// Ax=0), there is no need to solve for theta; we can just use zeros for the
283// thetas. In fact, our general solving method may not work in this case.
284// A common example of this is
285//
286// a{curl 1}..{curl 1}b
287//
288// which arises when solving a one-length path a..b or in a larger path a
289// section a--b.
290bool homogeneous(cvector<eqn>& e)
291{
292 for(cvector<eqn>::iterator p=e.begin(); p!=e.end(); ++p)
293 if (p->aug != 0)
294 return false;
295 return true;
296}

301{
302 return e.size()==2 && e.front().aug==0 && e.back().aug==0;

321/* Recalculate the equations in the form:
322 * theta[j] + post * theta[j+1] = aug + w * theta[0]
323 *
324 * Used as the first step in solve cyclic equations.
325 */
326cvector<weqn> recalc(cvector<eqn>& e)
327{
328 Int n=(Int) e.size();
329 cvector<weqn> we;
330 weqn lasteqn(0,1,0,0,1);
331 we.push_back(lasteqn); // As a placeholder.
332 for (Int j=1; j < n; j++) {
333 // Subtract a factor of the last equation so that the first entry is
334 // zero, then proceed to scale it.
335 eqn& q=e[j];
336 lasteqn=scale(weqn(0,q.piv-q.pre*lasteqn.post,q.post,
337 q.aug-q.pre*lasteqn.aug,-q.pre*lasteqn.w));
338 we.push_back(lasteqn);
339 }
340 // To keep all of the information enocoded in the linear equations, we need
341 // to augment the computation to replace our trivial start weqn with a
342 // real one. To do this, we take one more step in the iteration and
343 // compute the weqn for j=n, since n=0 (mod n).
344 {
345 eqn& q=e[0];
346 we.front()=scale(weqn(0,q.piv-q.pre*lasteqn.post,q.post,
347 q.aug-q.pre*lasteqn.aug,-q.pre*lasteqn.w));
348 }
349 return we;
350}

352double solveForTheta0(cvector<weqn>& we)
353{
354 // Solve for theta[0]=theta[n].
355 // How we do this is essentially to write out the first equation as:
356 //
357 // theta[n] = aug[0] + w[0]*theta[0] - post[0]*theta[1]
358 //
359 // and then use the next equation to substitute in for theta[1]:
360 //
361 // theta[1] = aug[1] + w[1]*theta[0] - post[1]*theta[2]
362 //
363 // and so on until we have an equation just in terms of theta[0] and
364 // theta[n] (which are the same theta).
365 //
366 // The loop invariant maintained is that after j iterations, we have
367 // theta[n]= a + b*theta[0] + c*theta[j]
368 Int n=(Int) we.size();
369 double a=0,b=0,c=1;
370 for (Int j=0;j<n;++j) {
371 weqn& q=we[j];
372 a+=c*q.aug;
373 b+=c*q.w;
374 c=-c*q.post;
375 }
376
377 // After the iteration we have
378 //
379 // theta[n] = a + b*theta[0] + c*theta[n]
380 //
381 // where theta[n]=theta[0], so
382 return a/(1.0-(b+c));
383}
384
385cvector<double> backsubCyclic(cvector<weqn>& we, double theta0)
386{
387 Int n=(Int) we.size();
388 cvector<double> thetas;
389 double lastTheta=theta0;
390 for (Int j=1;j<=n;++j)
391 {
392 weqn& q=we[n-j];
393 assert(q.pre == 0 && q.piv == 1);
394 double theta=-q.post*lastTheta+q.aug+q.w*theta0;
395 thetas.push_back(theta);
396 lastTheta=theta;
397 }
398 reverse(thetas.begin(),thetas.end());
399 return thetas;
400}

402// For the non-cyclic equations, do row operation to put the matrix into
403// reduced echelon form, ie. calculates equivalent equations but with pre=0 and
404// piv=1 for each eqn.
405struct ref : public knotprop<eqn> {
406 cvector<eqn>& e;
407 eqn lasteqn;
408
409 ref(knotlist& l, cvector<eqn>& e)
410 : knotprop<eqn>(l), e(e), lasteqn(0,1,0,0) {}
411
412 // Scale the equation so that the pivot (diagonal) entry is one, and save
413 // the new equation as lasteqn.
414 eqn scale(eqn q) {
415 assert(q.pre == 0 && q.piv != 0);
416 return lasteqn = eqn(0,1,q.post/q.piv,q.aug/q.piv);
417 }
418
419 eqn start(Int j) {
420 return scale(e[j]);
421 }
422 eqn mid(Int j) {
423 // Subtract a factor of the last equation so that the first entry is
424 // zero, then proceed to scale it.
425 eqn& q=e[j];
426 return scale(eqn(0,q.piv-q.pre*lasteqn.post,q.post,
427 q.aug-q.pre*lasteqn.aug));
428 }
429 // The end case is the same as the middle case.
430};

432// Once the matrix is in reduced echelon form, we can solve for the values by
433// back-substitution. This algorithm works from the bottom-up, so backCompute
434// must be used to get the answer.
435struct backsub : public knotprop<double> {
436 cvector<eqn>& e;
437 double lastTheta;
438
439 backsub(knotlist& l, cvector<eqn>& e)
440 : knotprop<double>(l), e(e) {}
441
442 double end(Int j) {
443 eqn& q=e[j];
444 assert(q.pre == 0 && q.piv == 1 && q.post == 0);
445 double theta=q.aug;
446 lastTheta=theta;
447 return theta;
448 }
449
450 double mid(Int j) {
451 eqn& q=e[j];
452 assert(q.pre == 0 && q.piv == 1);
453 double theta=-q.post*lastTheta+q.aug;
454 lastTheta=theta;
455 return theta;
456 }
457
458 // start is the same as mid.
459};

461// Once the equations have been determined, solve for the thetas.
462cvector<double> solveThetas(knotlist& l, cvector<eqn>& e)
463{
464 if (homogeneous(e))
465 // We are solving Ax=0, so a solution is zero for every theta.
466 return cvector<double>(e.size(),0);
467 else if (l.cyclic()) {
468 // The knotprop template is unusually unhelpful in this case, so I
469 // won't use it here. The algorithm breaks into three passes on the
470 // object. The old Asymptote code used a two-pass method, but I
471 // implemented this to stay closer to the MetaPost source code.
472 // This might be something to look at for optimization.
473 cvector<weqn> we=recalc(e);
474 INFO(we);
475 double theta0=solveForTheta0(we);
476 return backsubCyclic(we, theta0);
477 }
478 else { /* Non-cyclic case. */
479 /* First do row operations to get it into reduced echelon form. */
480 cvector<eqn> el=ref(l,e).compute();
481
482 /* Then, do back substitution. */
483 return backsub(l,el).backCompute();
484 }
485}

487// Once thetas have been solved, determine the first control point of every
488// join.
489struct postcontrolprop : public knotprop<pair> {
490 cvector<pair>& dz;
491 cvector<double>& psi;
492 cvector<double>& theta;
493
494 postcontrolprop(knotlist& l, cvector<pair>& dz,
495 cvector<double>& psi, cvector<double>& theta)
496 : knotprop<pair>(l), dz(dz), psi(psi), theta(theta) {}
497
498 double phi(Int j) {
499 /* The third angle: psi + theta + phi = 0 */
500 return -psi[j] - theta[j];
501 }
502
503 double vel(Int j) {
504 /* Use the standard velocity function. */
505 return velocity(theta[j],phi(j+1),l[j].tout);
506 }
507
508 // start is the same as mid.
509
510 pair mid(Int j) {
511 // Put a control point at the relative distance determined by the velocity,
512 // and at an angle determined by theta.
513 return l[j].z + vel(j)*expi(theta[j])*dz[j];
514 }
515
516 // The end postcontrol is the same as the last knot.
517 pair end(Int j) {
518 return l[j].z;
519 }
520};

522// Determine the first control point of every join.
523struct precontrolprop : public knotprop<pair> {
524 cvector<pair>& dz;
525 cvector<double>& psi;
526 cvector<double>& theta;
527
528 precontrolprop(knotlist& l, cvector<pair>& dz,
529 cvector<double>& psi, cvector<double>& theta)
530 : knotprop<pair>(l), dz(dz), psi(psi), theta(theta) {}
531
532 double phi(Int j) {
533 return -psi[j] - theta[j];
534 }
535
536 double vel(Int j) {
537 return velocity(phi(j),theta[j-1],l[j].tin);
538 }
539
540 // The start precontrol is the same as the first knot.
541 pair start(Int j) {
542 return l[j].z;
543 }
544 pair mid(Int j) {
545 return l[j].z - vel(j)*expi(-phi(j))*dz[j-1];
546 }
547
548 // end is the same as mid.
549};

597void encodeStraight(protopath& p, Int k, knotlist& l)
598{
599 pair a=l.front().z;
600 double at=l.front().tout.val;
601 pair b=l.back().z;
602 double bt=l.back().tin.val;
603 pair step=(b-a)/3.0;
604
605 if (at==1.0 && bt==1.0) {
606 p.straight(k)=true;
607 p.post(k)=a+step;
608 p.pre(k+1)=b-step;
609 p.point(k+1)=b;
610 }
611 else {
612 p.post(k)=a+step/at;
613 p.pre(k+1)=b-step/bt;
614 p.point(k+1)=b;
615 }
616}

618void solveSection(protopath& p, Int k, knotlist& l)
619{
620 if (l.length()>0) {
621 info(cerr, "solving section", l);
622
623 // Calculate useful properties.
624 cvector<pair> dz = dzprop(l) .compute();
625 cvector<double> d = dprop(l,dz).compute();
626 cvector<double> psi = psiprop(l,dz).compute();
627
628 INFO(dz); INFO(d); INFO(psi);
629
630 // Build and solve the linear equations for theta.
631 cvector<eqn> e = eqnprop(l,d,psi).compute();
632 INFO(e);
633
634 if (straightSection(e))
635 // Handle straight section as special case.
636 encodeStraight(p,k,l);
637 else {
638 cvector<double> theta = solveThetas(l,e);
639 INFO(theta);
640
641 // Calculate the control points.
642 cvector<pair> post = postcontrolprop(l,dz,psi,theta).compute();
643 cvector<pair> pre = precontrolprop(l,dz,psi,theta).compute();
644
645 // Encode the results into the protopath.
646 encodeControls(p,k,pre,l,post).exec();
647 }
648 }
649}

651// Find the first breakpoint in the knotlist, ie. where we can start solving a
652// non-cyclic section. If the knotlist is fully cyclic, then this returns
653// NOBREAK.
654// This must be called with a knot that has all of its implicit specifiers in
655// place.
656const Int NOBREAK=-1;
657Int firstBreakpoint(knotlist& l)
658{
659 for (Int j=0;j<l.size();++j)
660 if (!l[j].out->open())
661 return j;
662 return NOBREAK;
663}
664
665// Once a breakpoint, a, is found, find where the next breakpoint after it is.
666// This must be called with a knot that has all of its implicit specifiers in
667// place, so that breakpoint can be identified by either an in or out specifier
668// that is not open.
669Int nextBreakpoint(knotlist& l, Int a)
670{
671 // This is guaranteed to terminate if a is the index of a breakpoint. If the
672 // path is non-cyclic it will stop at or before the last knot which must be a
673 // breakpoint. If the path is cyclic, it will stop at or before looping back
674 // around to a which is a breakpoint.
675 Int j=a+1;
676 while (l[j].in->open())
677 ++j;
678 return j;
679}

681// Write out the controls for section of the form
682// a.. control b and c ..d
683void writeControls(protopath& p, Int a, knotlist& l)
684{
685 // By convention, the first point will already be encoded.
686 p.straight(a)=dynamic_cast<controlSpec *>(l[a].out)->straight;
687 p.post(a)=dynamic_cast<controlSpec *>(l[a].out)->cz;
688 p.pre(a+1)=dynamic_cast<controlSpec *>(l[a+1].in)->cz;
689 p.point(a+1)=l[a+1].z;
690}

692// Solves a path that has all of its specifiers laid out explicitly.
693path solveSpecified(knotlist& l)
694{
695 protopath p(l.size(),l.cyclic());
696
697 Int first=firstBreakpoint(l);
698 if (first==NOBREAK)
699 /* We are solving a fully cyclic path, so do it in one swoop. */
700 solveSection(p,0,l);
701 else {
702 // Encode the first point.
703 p.point(first)=l[first].z;
704
705 // If the path is cyclic, we should stop where we started (modulo the
706 // length of the path); otherwise, just stop at the end.
707 Int last=l.cyclic() ? first+l.length()
708 : l.length();
709 Int a=first;
710 while (a!=last) {
711 if (l[a].out->controlled()) {
712 assert(l[a+1].in->controlled());
713
714 // Controls are already picked, just write them out.
715 writeControls(p,a,l);
716 ++a;
717 }
718 else {
719 // Find the section a to b and solve it, putting the result (starting
720 // from index a into our protopath.
721 Int b=nextBreakpoint(l,a);
722 subknotlist section(l,a,b);
723 solveSection(p,a,section);
724 a=b;
725 }
726 }
727
728 // For a non-cyclic path, the end control points need to be set.
729 p.controlEnds();
730 }
731
732 return p.fix();
733}

735/* If a knot is open on one side and restricted on the other, this replaces the
736 * open side with a restriction determined by the restriction on the other
737 * side. After this, any knot will either have two open specs or two
738 * restrictions.
739 */
740struct partnerUp : public knoteffect {
741 partnerUp(knotlist& l)
742 : knoteffect(l) {}
743
744 void mid(Int j) {
745 knot& k=l[j];
746 if (k.in->open() && !k.out->open())
747 k.in=k.out->inPartner(k.z);
748 else if (!k.in->open() && k.out->open())
749 k.out=k.in->outPartner(k.z);
750 }
751};
752
753/* Ensures a non-cyclic path has direction specifiers at the ends, adding curls
754 * if there are none.
755 */
756void curlEnds(knotlist& l)
757{
758 static curlSpec endSpec;
759
760 if (!l.cyclic()) {
761 if (l.front().in->open())
762 l.front().in=&endSpec;
763 if (l.back().out->open())
764 l.back().out=&endSpec;
765 }
766}
767
768/* If a point occurs twice in a row in a knotlist, write in controls
769 * between the two knots at that point (unless it already has controls).
770 */
771struct controlDuplicates : public knoteffect {
772 controlDuplicates(knotlist& l)
773 : knoteffect(l) {}
774
775 void solo(Int) { /* One point ==> no duplicates */ }
776 // start is the same as mid.
777 void mid(Int j) {
778 knot &k1=l[j];
779 knot &k2=l[j+1];
780 if (!k1.out->controlled() && k1.z==k2.z) {
781 k1.out=k2.in=new controlSpec(k1.z,true);
782 }
783 }
784 void end(Int) { /* No next point to compare with. */ }
785};
786
787path solve(knotlist& l)
788{
789 if (l.empty())
790 return path();
791 else {
792 info(cerr, "input knotlist", l);
793 curlEnds(l);
794 controlDuplicates(l).exec();
795 partnerUp(l).exec();
796 info(cerr, "specified knotlist", l);
797 return solveSpecified(l);
798 }
799}
