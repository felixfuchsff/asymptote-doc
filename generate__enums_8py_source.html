38def create_enums(filename: str) -> List[Union[Tuple[str, int, str], Tuple[str, int]]]:
39 final_list = []
40 with io.open(filename, newline="", encoding="utf-8") as rawfile:
41 for line in rawfile.readlines():
42 if line.startswith("#") or line.strip() == "":
43 continue
44 raw_line = line.strip().split(",")
45 raw_str, raw_number = raw_line[0:2]
46 comment = None
47 if len(raw_line) >= 3:
48 comment = raw_line[-1]
49 final_list.append((raw_str.strip(), int(raw_number.strip()), comment))
50 else:
51 final_list.append((raw_str.strip(), int(raw_number.strip())))
52 return final_list
