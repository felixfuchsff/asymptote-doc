222block diamond(object body, pair center=(0,0),
223 pen fillpen=invisible, pen drawpen=currentpen,
224 real ds=5, real dw=1,
225 real height=20, real minwidth=minblockwidth,
226 real minheight=minblockheight)
227{
228 frame f=body.f;
229 pair m=min(f);
230 pair M=max(f);
231 pair bound=maxbound(M-m,(minwidth,minheight));
232
233 real e=ds;
234 real a=0.5bound.x-dw;
235 real b=0.5bound.y;
236 real c=b+height;
237
238 real arg=a^2+b^2+c^2-2b*c-e^2;
239 real denom=e^2-a^2;
240 real slope=arg >= 0 && denom != 0 ? (a*(c-b)-e*sqrt(arg))/denom : 1.0;
241 real d=abs(c/slope);
242
243 path shape=(2d,c)--(d,2c)--(0,c)--(d,0)--cycle;
244
245 block block;
246 block.draw=new frame(pen p) {
247 frame block;
248 filldraw(block,shape,fillpen,drawpen);
249 add(block,shift(-0.5*(M+m))*f,(d,c));
250 return block;
251 };
252 block.f_position=new pair(real x) {
253 return point(shape,x);
254 };
255 block.f_center=(point(shape,1).x,point(shape,0).y);

384path path(pair point[] ... flowdir dir[])
385{
386 path line=point[0];
387 pair current, prev=point[0];
388 for(int i=1; i < point.length; ++i) {
389 if(i-1 >= dir.length || dir[i-1] == Horizontal)
390 current=(point[i].x,point[i-1].y);
391 else
392 current=(point[i-1].x,point[i].y);
393
394 if(current != prev) {
395 line=line--current;
396 prev=current;

399 current=point[i];
400 if(current != prev) {
401 line=line--current;
402 prev=current;

419blockconnector blockconnector(picture pic, transform t, pen p=currentpen,
420 margin margin=PenMargin)
421{
422 return new block(block b1, block b2) {
423 if(b1.dirs.length == 0) {
424 if(abs(b1.center.y-b2.center.y) < sqrtEpsilon) {
425 // horizontally aligned
426 b1.dirs[0]=b1.center.x < b2.center.x ? right : left;
427 blockconnector(pic,t,p,margin)(b1,b2);
428 } else if(abs(b1.center.x-b2.center.x) < sqrtEpsilon) {
429 // vertically aligned
430 b1.dirs[0]=b1.center.y < b2.center.y ? up : down;
431 blockconnector(pic,t,p,margin)(b1,b2);
432 } else {
433 if(abs(b1.center.y-b2.center.y) < abs(b1.center.x-b2.center.x)) {
434 b1.dirs[0]=b1.center.x < b2.center.x ? right : left;
435 b1.dirs[1]=b1.center.y < b2.center.y ? up : down;
436 blockconnector(pic,t,p,margin)(b1,b2);
437 } else {
438 b1.dirs[0]=b1.center.y < b2.center.y ? up : down;
439 b1.dirs[1]=b1.center.x < b2.center.x ? right : left;
440 blockconnector(pic,t,p,margin)(b1,b2);
441 }
442 }
443 return b2;
444 }
445
446 // compute the link for given directions (and label if any)
447 pair[] dirs=copy(b1.dirs); // deep copy
448 pair current,prev;
449 pair dir=dirs[0];
450 if(dir == up) prev=b1.top(t);
451 if(dir == down) prev=b1.bottom(t);
452 if(dir == left) prev=b1.left(t);
453 if(dir == right) prev=b1.right(t);
454 path line=prev;
455 arrowbar arrow=b1.arrow;
456
457 int i;
458 for(i=1; i < dirs.length-1; ++i) {
459 if(abs(length(dirs[i-1])-1) < sqrtEpsilon)
460 current=prev+t*dirs[i-1]*defaultexcursion;
461 else
462 current=prev+t*dirs[i-1];
463
464 if(current != prev) {
465 line=line--current;
466 prev=current;
467 }
468 }
469 dir=dirs[dirs.length-1];
470 current=0;
471 if(dir == up) current=b2.bottom(t);
472 if(dir == down) current=b2.top(t);
473 if(dir == left) current=b2.right(t);
474 if(dir == right) current=b2.left(t);
475 if(abs(dirs[i-1].y) < sqrtEpsilon &&
476 abs(prev.x-current.x) > sqrtEpsilon) {
477 prev=(current.x,prev.y);
478 line=line--prev; // horizontal
479 } else if(abs(dirs[i-1].x) < sqrtEpsilon &&
480 abs(prev.y-current.y) > sqrtEpsilon) {
481 prev=(prev.x,current.y);
482 line=line--prev;
483 }
484 if(current != prev)
485 line=line--current;
486
487 draw(pic,b1.label,line,p,arrow,margin);
488
489 b1.label="";
490 b1.dirs.delete();
491 b1.arrow=None;
492 return b2;
493 };
