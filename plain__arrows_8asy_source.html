464void draw(picture pic=currentpicture, Label L=null, path g,
465 align align=NoAlign, pen p=currentpen, arrowbar arrow=None,
466 arrowbar bar=None, margin margin=NoMargin, Label legend=null,
467 marker marker=nomarker)
468{
469 // These if statements are ordered in such a way that the most common case
470 // (with just a path and a pen) executes the least bytecode.
471 if (marker == nomarker)
472 {
473 if (arrow == None && bar == None)
474 {
475 if (margin == NoMargin && size(nib(p)) == 0)
476 {
477 pic.addExactAbove(
478 new void(frame f, transform t, transform T, pair, pair) {
479 _draw(f,t*T*g,p);
480 });
481 pic.addPath(g,p);
482
483 // Jumping over else clauses takes time, so test if we can return
484 // here.
485 if (L == null && legend == null)
486 return;
487 }
488 else // With margin or polygonal pen.
489 {
490 _draw(pic, g, p, margin);
491 }
492 }
493 else /* arrow or bar */
494 {
495 // Note we are using & instead of && as both arrow and bar need to be
496 // called.
497 if (arrow(pic, g, p, margin) & bar(pic, g, p, margin))
498 _draw(pic, g, p, margin);
499 }
500
501 if(L != null && L.s != "") {
502 L=L.copy();
503 L.align(align);
504 L.p(p);
505 L.out(pic,g);
506 }
507
508 if(legend != null && legend.s != "") {
509 legend.p(p);
510 pic.legend.push(Legend(legend.s,legend.p,p,marker.f,marker.above));
511 }
512 }
513 else /* marker != nomarker */
514 {
515 if(marker != nomarker && !marker.above) marker.mark(pic,g);
516
517 // Note we are using & instead of && as both arrow and bar need to be
518 // called.
