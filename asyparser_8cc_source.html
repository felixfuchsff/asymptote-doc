68absyntax::file *doParse(size_t (*input) (char* bif, size_t max_size),
69 const string& filename, bool extendable=false)
70{
71 setlexer(input,filename);
72 absyntax::file *root = yyparse() == 0 ? absyntax::root : 0;
73 absyntax::root = 0;
74 yy::sbuf = 0;
75
76 if (!root) {
77 if (lexerEOF()) {
78 if (extendable) {
79 return 0;
80 } else {
81 // Have the lexer report the error.
82 reportEOF();
83 }
84 }
85
86 em.error(nullPos);
87 if(!interact::interactive)
88 error(filename);
89 else
90 throw handled_error();
91 }
92
93 return root;
94}

112absyntax::file *parseFile(const string& filename,
113 const char *nameOfAction)
114{
115#ifdef HAVE_LIBCURL
116 if(isURL(filename))
117 return parseURL(filename,nameOfAction);
118#endif
119
120 if(filename == "-")
121 return parseStdin();
122
123 string file = settings::locateFile(filename);
124
125 if(file.empty())
126 error(filename);
127
128 if(nameOfAction && settings::verbose > 1)
129 cerr << nameOfAction << " " << filename << " from " << file << endl;
130
131 debug(false);
132
133 std::filebuf filebuf;
134 if(!filebuf.open(file.c_str(),std::ios::in))
135 error(filename);
136
137#ifdef HAVE_SYS_STAT_H
138 // Check that the file is not a directory.
139 static struct stat buf;
140 if(stat(file.c_str(),&buf) == 0) {
141 if(S_ISDIR(buf.st_mode))
142 error(filename);
143 }
144#endif
145
146 // Check that the file can actually be read.
147 try {
148 filebuf.sgetc();
149 } catch (...) {
150 error(filename);
151 }
152
153 yy::sbuf = &filebuf;
154 return doParse(yy::stream_input,file);
155}

180bool readURL(stringstream& buf, const string& filename)
181{
182 CURL *curl=curl_easy_init();
183 if(settings::verbose > 3)
184 curl_easy_setopt(curl,CURLOPT_VERBOSE,true);
185#ifdef __MSDOS__
186 string cert=settings::getSetting<string>("sysdir")+settings::dirsep+
187 "ca-bundle.crt";
188 curl_easy_setopt(curl,CURLOPT_CAINFO,cert.c_str());
189#endif
190 curl_easy_setopt(curl,CURLOPT_URL,filename.c_str());
191 curl_easy_setopt(curl,CURLOPT_WRITEFUNCTION,curlCallback);
192 curl_easy_setopt(curl,CURLOPT_WRITEDATA,&buf);
193 curl_easy_setopt(curl,CURLOPT_NOPROGRESS,0);
194 curl_easy_setopt(curl,CURLOPT_XFERINFOFUNCTION,curlProgress);
195
196 CURLcode res=curl_easy_perform(curl);
197 curl_easy_cleanup(curl);
198
199 if(res != CURLE_OK) {
200 cerr << curl_easy_strerror(res) << endl;
201 return false;
202 }
203 string s=buf.str();
204 return !s.empty() && s != "404: Not Found";
205}
