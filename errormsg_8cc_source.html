25ostream& operator<< (ostream& out, const position& pos)
26{
27 if (!pos)
28 return out;
29
30 string filename=pos.file->name();
31
32 if(filename != "-" && !(settings::getSetting<bool>("quiet") ||
33 settings::getSetting<bool>("where"))) {
34 std::ifstream fin(filename.c_str());
35 string s;
36 size_t count=pos.line;
37 while(count > 0 && getline(fin,s)) {
38 count--;
39 }
40 s=std::regex_replace(s,std::regex("\t")," ");
41 out << s << endl;
42 for(size_t i=1; i < pos.column; ++i)
43 out << " ";
44 out << "^" << endl;
45 }
46
47 out << filename << ": ";
48 out << pos.line << "." << pos.column;
49
50 if(settings::xasy) {
51 camp::openpipeout();
52 fprintf(camp::pipeout,"Error\n");
53 fflush(camp::pipeout);
54 }
55
56 return out;
57}
