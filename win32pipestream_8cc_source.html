23void Win32IoPipeStream::open(
24 mem::vector<string> const& command, char const* hint,
25 char const* application, int out_fileno)
26{
27 if (out_fileno != STDOUT_FILENO && out_fileno != STDERR_FILENO)
28 {
29 camp::reportError("out_fileno must be stdout or stdin");
30 }
31
32 // creating pipes
33 SECURITY_ATTRIBUTES pipeSecurityAttr = {};
34 pipeSecurityAttr.nLength = sizeof(SECURITY_ATTRIBUTES);
35 pipeSecurityAttr.bInheritHandle = true;
36 pipeSecurityAttr.lpSecurityDescriptor = nullptr;
37
38 {
39 cw32::HandleRaiiWrapper processStdinRd;
40 cw32::checkResult(CreatePipe(processStdinRd.put(), &processStdinWr, &pipeSecurityAttr, 0));
41
42 cw32::HandleRaiiWrapper processOutWr;
43 cw32::checkResult(CreatePipe(&processOutRd, processOutWr.put(), &pipeSecurityAttr, 0));
44
45 // building command
46 string cmd= camp::w32::buildWindowsCmd(command);
47
48 STARTUPINFOA startInfo= {};
49 startInfo.cb= sizeof(startInfo);
50 startInfo.dwFlags= STARTF_USESTDHANDLES;
51 startInfo.hStdInput= processStdinRd.getHandle();
52 startInfo.hStdOutput= out_fileno == STDOUT_FILENO ? processOutWr.getHandle() : GetStdHandle(STD_OUTPUT_HANDLE);
53 startInfo.hStdError= out_fileno == STDERR_FILENO ? processOutWr.getHandle() : GetStdHandle(STD_ERROR_HANDLE);
54
55 auto const result= CreateProcessA(
56 nullptr,
57 cmd.data(),
58 nullptr, nullptr, true,
59 0,
60 nullptr, nullptr,
61 &startInfo,
62 &procInfo);
63 if (!result)
64 {
65 execError(command.at(0).c_str(), hint, application);
66 cw32::checkResult(result, "Cannot open application");
67 }
68 }
69
70 ZeroMemory(buffer, BUFFER_LEN);
71 Running=true;
72 pipeopen=true;
73 pipein=true;
74 block(false,true);
75}

78{
79 /*

141int Win32IoPipeStream::wait()
142{
143 DWORD retcode = 0;
144 if (procInfo.hProcess == nullptr)
145 {
146 return static_cast<int>(retcode);
147 }
148
149 switch(WaitForSingleObject(procInfo.hProcess, INFINITE))
150 {
151 case WAIT_OBJECT_0:
152 cw32::checkResult(GetExitCodeProcess(procInfo.hProcess, &retcode));
153 break;
154 default:
155 closeProcessHandles();
156 camp::reportError("Wait for process error");
157 break;
158 }
159
160 closeProcessHandles();
161 return static_cast<int>(retcode);
162}

201ssize_t Win32IoPipeStream::readbuffer()
202{
203 if (!(Running && pipeopen))
204 {
205 return 0;
206 }
207
208 DWORD nc;
209
210 if (!ReadFile(processOutRd, buffer, BUFFER_LEN - 1, &nc, nullptr))
211 {
212 if (GetLastError() != ERROR_BROKEN_PIPE)
213 {
214 // process could have exited
215 camp::reportError("read failed from pipe");
216 }
217 }
218
219 buffer[nc]=0;
220
221 if (nc > 0)
222 {
223 if (settings::verbose > 2)
224 {
225 cerr << buffer;
226 }
227 }
228 else if (procInfo.hProcess != nullptr)
229 {
230 switch (WaitForSingleObject(procInfo.hProcess, 0))
231 {
232 case WAIT_OBJECT_0:
233 {
234 closeProcessHandles();
235 Running=false;
236 break;
237 }
238 case WAIT_TIMEOUT:
239 break;
240 default:
241 camp::reportError("Waiting for process failed");
242 break;
243 }
244 }
245 return nc;
246}

259bool Win32IoPipeStream::tailequals(char const* buf, size_t len, char const* prompt, size_t plen)
260{
261 const char *a=buf+len;
262 const char *b=prompt+plen;
263 while(b >= prompt) {
264 if(a < buf) return false;
265 if(*a != *b) return false;
266 // Handle MSDOS incompatibility:
267 if(a > buf && *a == '\n' && *(a-1) == '\r') --a;
268 --a; --b;
269 }
270 return true;
271}
