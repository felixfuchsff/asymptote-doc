5// diffdiv(x,y) computes Newton's Divided Difference for
6// Lagrange interpolation with distinct values {x_0,..,x_n} in the array x

11//
12// fhorner(x,coeff) uses Horner's rule to compute the polynomial
13// a_0+a_1(x-x_0)+a_2(x-x_0)(x-x_1)+...+a_n(x-x_0)..(x-x_{n-1}),
14// where coeff={a_0,a_1,...,a_n}.

16// fspline does standard cubic spline interpolation of a function f

18// The points a=x_1 < x_2 < .. < x_n=b form the array x;

44// Evaluate p(x)=d0+(x-x0)(d1+(x-x1)+...+(d(n-1)+(x-x(n-1))*dn)))
45// via Horner's rule: n-1 multiplications, 2n-2 additions.
46fhorner fhorner(horner sh)
47{
48 int n=sh.x.length;
49 checklengths(n,sh.a.length);
50 return new real(real x) {
51 real s=sh.a[n-1];
52 for(int k=n-2; k >= 0; --k)
53 s=sh.a[k]+(x-sh.x[k])*s;
54 return s;
55 };
56}

58// Newton's Divided Difference method: n(n-1)/2 divisions, n(n-1) additions.
59horner diffdiv(real[] x, real[] y)
60{
61 int n=x.length;
62 horner s;
63 checklengths(n,y.length);
64 for(int i=0; i < n; ++i)
65 s.a[i]=y[i];
66 for(int k=0; k < n-1; ++k) {
67 for(int i=n-1; i > k; --i) {
68 s.a[i]=(s.a[i]-s.a[i-1])/(x[i]-x[i-k-1]);
69 }
70 }
71 s.x=x;
72 return s;
73}

75// Newton's Divided Difference for simple Hermite interpolation,
76// where one specifies both p(x_i) and p'(x_i).
77horner hdiffdiv(real[] x, real[] y, real[] dy)
78{
79 int n=x.length;
80 horner s;
81 checklengths(n,y.length);
82 checklengths(n,dy.length);
83 for(int i=0; i < n; ++i) {
84 s.a[2*i]=y[i];
85 s.a[2*i+1]=dy[i];
86 s.x[2*i]=x[i];
87 s.x[2*i+1]=x[i];
88 }
89
90 for(int i=n-1; i > 0; --i)
91 s.a[2*i]=(s.a[2*i]-s.a[2*i-2])/(x[i]-x[i-1]);
92
93 int stop=2*n-1;
94 for(int k=1; k < stop; ++k) {
95 for(int i=stop; i > k; --i) {
96 s.a[i]=(s.a[i]-s.a[i-1])/(s.x[i]-s.x[i-k-1]);
97 }
98 }
99 return s;
100}

104// piecewise Hermite interpolation:
105// return the piecewise polynomial p(x), where on [x_i,x_i+1], deg(p) <= 3,
106// p(x_i)=y_i, p(x_{i+1})=y_i+1, p'(x_i)=dy_i, and p'(x_{i+1})=dy_i+1.
107// Outside [x_1,x_n] the returned function is constant: y_1 on (infinity,x_1]
108// and y_n on [x_n,infinity).
109realfunction pwhermite(real[] x, real[] y, real[] dy)
110{
111 int n=x.length;
112 checklengths(n,y.length);
113 checklengths(n,dy.length);
114 if(n < 2) abort(morepoints);
115 if(!increasing(x,strict=true)) abort("array x is not strictly increasing");
116 return new real(real t) {
117 int i=search(x,t);
118 if(i == n-1) {
119 i=n-2;
120 t=x[n-1];
121 } else if(i == -1) {
122 i=0;
123 t=x[0];
124 }
125 real h=x[i+1]-x[i];
126 real delta=(y[i+1]-y[i])/h;
127 real e=(3*delta-2*dy[i]-dy[i+1])/h;
128 real f=(dy[i]-2*delta+dy[i+1])/h^2;
129 real s=t-x[i];
130 return y[i]+s*(dy[i]+s*(e+s*f));
131 };
132}
