20def process_configure_ac_file(f):
21 """
22 Returns a dictionary containing version information, if found
23 from the configure.ac file object
24 """
25 regexes_to_search = dict(REGEXES)
26 result = {}
27 for line in f:
28 if not regexes_to_search:
29 return result # done now, can return
30 regex_result = {key: val.match(line) for key, val in regexes_to_search.items()}
31 regex_result_not_none: dict[str, re.Match[str]] = {
32 key: val for key, val in regex_result.items() if val is not None
33 }
34 for key, match_obj in regex_result_not_none.items():
35 regexes_to_search.pop(key)
36 if key == "base":
37 result.update(
38 {
39 "name": match_obj.group(1),
40 "version-base": match_obj.group(2),
41 "issue-url": match_obj.group(3),
42 }
43 )
44 if key == "asygl":
45 result["asygl-version"] = match_obj.group(1)
46 return result
