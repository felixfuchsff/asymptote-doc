35pair min(pair[] z) {
36 pair m=(infinity, infinity);
37 for (int i=0; i<z.length; ++i) {
38 if (z[i].x < m.x)
39 m=(z[i].x,m.y);
40 if (z[i].y < m.y)
41 m=(m.x,z[i].y);
42 }
43 return m;
44}

46pair max(pair[] z) {
47 pair M=(-infinity, -infinity);
48 for (int i=0; i<z.length; ++i) {
49 if (z[i].x > M.x)
50 M=(z[i].x,M.y);
51 if (z[i].y > M.y)
52 M=(M.x,z[i].y);
53 }
54 return M;
55}

168 guide[] drawStep(picture pic, Placement p, guide[] ins) {
169 int i=0,j=0;
170
171 // Draw the component.
172 Component c=p.c;
173 //write("drawing "+c.symbol+" at place "+(string)p.place);
174 guide[] couts=c.draw(pic, ins[sequence(c.in)+p.place]);
175
176 pair M=max(endpoints(couts));
177
178 // Extend lines not in the component.
179 guide[] outs;
180 pair[] z=endpoints(ins);
181 while (i<p.place) {
182 outs.push(ins[i]..(z[i].x,M.y));
183 ++i;
184 }
185
186 outs.append(couts);
187 i+=c.in;
188
189 while (i<ins.length) {
190 outs.push(ins[i]..(z[i].x,M.y));
191 ++i;
192 }
193
194 return outs;
195 }

292 // Tests if the component at index 'start' can be moved to index 'end'
293 // without interfering with other components.
294 bool moveable(int start, int end) {
295 assert(start<places.length); assert(end<places.length);
296 if (start==end)
297 return true;
298 else if (end<start)
299 return moveable(end,start);
300 else {
301 assert(start<end);
302 Placement p=places[start].copy();
303 for (int step=start; step<end; ++step) {
304 Placement q=places[step+1];
305 if (q.place + q.c.in <= p.place)
306 // q is left of p - adjust for q renumbering strings.
307 p.place+=q.c.out-q.c.in;
308 else if (p.place + p.c.out <= q.place)
309 // q is right of p - nothing to do.
310 continue;
311 else
312 // q is directly on top of p
313 return false;
314 }
315 return true;
316 }

351 bool tryMatch(Braid sub, int place, int size, int[] acc) {
352 // If we've matched all the components, we've won.
353 if (acc.length >= sub.places.length)
354 return true;
355
356 // The next component to match.
357 Placement p=sub.places[acc.length];
358
359 // Start looking immediately after the last match.
360 for (int step=acc[acc.length-1]+1; step<this.places.length; ++step) {
361 Placement q=this.places[step];
362 // Check if the next component is in the set of strings used by the
363 // subbraid.
364 if (q.place + q.c.in > place && q.place < place + size) {
365 // It's in the window, so it must match the next component in the
366 // subbraid.
367 if (p.c==q.c && p.place+place==q.place) {
368 // A match - go on to the next component.
369 acc.push(step);
370 return tryMatch(sub, place, size, acc); // TODO: Adjust place/size.
371 }
372 else
373 return false;
374 }
375
376 // TODO: Adjust place and size.
377 }
378
379 // We've run out of components to match.
380 return false;
381 }

384 // This attempts to find a subbraid within the braid. It allows other
385 // components to be interspersed with the components of the subbraid so long
386 // as they don't occur on the same string as the ones the subbraid lies on.
387 // Returns null on failure.
388 int[] match(Braid sub, int place) {
389 for (int i=0; i<=this.places.length-sub.places.length; ++i) {
390 // Find where the first component of the subbraid matches and try to
391 // match the rest of the braid starting from there.
392 if (matchComponent(sub, 0, place, i)) {
393 int[] result;
394 result.push(i);
395 if (tryMatch(sub,place,sub.n,result))
396 return result;
397 }
398 }
399 return null;
400 }

402 // Equations {{{2
403 // Returns the string that 'place' moves to when going through the section
404 // with Placement p.
405 static int advancePast(Placement p, int place) {
406 // If it's to the left of the component, it is unaffected.
407 return place<p.place ? place :
408 // If it's to the right of the component, adjust the numbering due
409 // to the change of the number of strings in the component.
410 p.place+p.c.in <= place ? place - p.c.in + p.c.out :
411 // If it's in the component, ask the component to do the work.
412 p.place + p.c.connections[place-p.place];
413 }

415 // Adjust the place (at step 0) to the step given, to find which string it is
416 // on in that part of the diagram.
417 int advanceToStep(int step, int place) {
418 assert(place>=0 && place<n);
419 assert(step>=0 && step<places.length);
420
421 for (int i=0; i<step; ++i)
422 place=advancePast(places[i], place);
423
424 return place;
425 }

584struct Relation { // {{{1

590 string toFormula() {
591 return lhs.toFormula() + " = " + rhs.toFormula();

594 string linearName() {
595 assert(lhs.n==rhs.n);
596 assert(lsym!="");
597
598 string s=(inverted ? "-" : "") + lsym+"(";
599 for (int i=1; i<=lhs.n+1; ++i) {
600 if (i>1)
601 s+=",";
602 s+="x_"+(string)i;
603 }
604 return s+")";
605 }

608 assert(lhs.n==rhs.n);

610
611 string s=(inverted ? "minus" : "") + codename+"[";
612 for (int i=1; i<=lhs.n+1; ++i) {
613 if (i>1)
614 s+=", ";
615 s+="x"+(string)i+"_";
616 }
617 return s+"]";
618 }

621 return linearName() + " = " + lhs.toLinear() + rhs.toLinear(true);

653Braid apply(Relation r, Braid b, int step, int place) {
654 bool valid=b.exactMatch(r.lhs,place,step);
655 if (valid) {
656 Braid result=new Braid;
657 result.n=b.n;
658 for (int i=0; i<step; ++i)
659 result.places.push(b.places[i]);
660 result.add(r.rhs,place);
661 for (int i=step+r.lhs.places.length; i<b.places.length; ++i)
662 result.places.push(b.places[i]);
663 return result;
664 }
665 else {
666 write("Invalid match!");
667 return null;
668 }

673// Draw a number of frames in a nice circular arrangement.
674picture tableau(frame[] cards, bool number=false) {
675 int n=cards.length;
676
677 // Calculate the max height and width of the frames (assuming min(f)=(0,0)).
678 pair M=(0,0);
679 for (int i=0; i<n; ++i) {
680 pair z=max(cards[i]);
681 if (z.x > M.x)
682 M=(z.x,M.y);
683 if (z.y > M.y)
684 M=(M.x,z.y);
685 }
686
687 picture pic;
688 real xpad=2.0, ypad=1.3;
689 void place(int index, real row, real column) {
690 pair z=((M.x*xpad)*column,(M.y*ypad)*row);
691 add(pic, cards[index], z);
692 if (number) {
693 label(pic,(string)index, z+(0.5M.x,0), S);
694 }
695 }
696
697 // Handle small collections.
698 if (n<=4) {
699 for (int i=0; i<n; ++i)
700 place(i,0,i);
701 }
702 else {
703 int rows=quotient(n-1,2), columns=3;
704
705 // Add the top middle card.
706 place(0,rows-1,1);
707
708 // place cards down the right side.
709 for (int i=1; i<rows; ++i)
710 place(i, rows-i,2);
711
712 // place cards at the bottom.
713 if (n%2==0) {
714 place(rows,0,2);
715 place(rows+1,0,1);
716 place(rows+2,0,0);
717 }
718 else {
719 place(rows,0,1.5);
720 place(rows+1,0,0.5);
721 }
722
723 // place cards up the left side.
724 for (int i=1; i<rows; ++i)
725 place(i+n-rows,i,0);
726 }
727
728 return pic;
