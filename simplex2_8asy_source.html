1/*****
2 * simplex.asy
3 * Andy Hammerlindl 2004/07/27
4 *
5 * Solves the two-variable linear programming problem using the simplex method.
6 * This problem is specialized in that the second variable, "b", does not have
7 * a non-negativity condition, and the first variable, "a", is the quantity
8 * being maximized.
9 * Correct execution of the algorithm also assumes that the coefficient of "b"
10 * will be +1 or -1 in every added restriction, and that the problem can be
11 * initialized to a valid state by pivoting b with one of the slack
12 * variables. This assumption may in fact be incorrect.
13 *****/

50 // Pivot the variable v[col] with vp.
51 void pivot(int col, var vp)
52 {
53 var vc=v[col];
54
55 // Recalculate rows v[col] and vp for the pivot-swap.
56 row rvc = rows[vc], rvp = rows[vp];
57 real factor=1/rvp.t[col]; // NOTE: Handle rvp.t[col] == 0 case.
58 rvc.c=-rvp.c*factor;
59 rvp.c=0;
60 rvc.t=-rvp.t*factor;
61 rvp.t *= 0;
62 rvc.t[col]=factor;
63 rvp.t[col]=1;
64
65 var a=min(vc,vp);
66 var b=max(vc,vp);
67
68 // Recalculate the rows other than the two used for the above pivot.
69 for (var i = 0; i < a; ++i) {
70 row r=rows[i];
71 real m = r.t[col];
72 r.c += m*rvc.c;
73 r.t += m*rvc.t;
74 r.t[col]=m*factor;
75 }
76 for (var i = a+1; i < b; ++i) {
77 row r=rows[i];
78 real m = r.t[col];
79 r.c += m*rvc.c;
80 r.t += m*rvc.t;
81 r.t[col]=m*factor;
82 }
83 for (var i = b+1; i < rows.length; ++i) {
84 row r=rows[i];
85 real m = r.t[col];
86 r.c += m*rvc.c;
87 r.t += m*rvc.t;
88 r.t[col]=m*factor;
89 }
90
91 // Relabel the vars.
92 v[col] = vp;
93 }

95 // As b does not have a non-negativity condition, it must initially be
96 // pivoted out for a variable that does. This selects the initial
97 // variable to pivot with b. It also assumes that there is a valid
98 // solution with a == 0 to the linear programming problem, and if so, it
99 // picks a pivot to get to that state. In our case, a == 0 corresponds to
100 // a picture with the user coordinates shrunk down to zero, and if that
101 // doesn't fit, nothing will.
102 //
103 // If b has a minimal value, choose a pivot that will give b its minimal
104 // value. Otherwise, if b has maximal value, choose a pivot to give b its
105 // maximal value.
106 var initVar()
107 {
108 real min;
109 var argmin;
110 var i=2;
111 for (; i < rows.length; ++i) {
112 row r=rows[i];
113 if (r.t[VAR_B] > 0) {
114 min=r.c/r.t[VAR_B];
115 argmin=i;
116 break;
117 }
118 }
119 for (; i < rows.length; ++i) {
120 row r=rows[i];
121 if (r.t[VAR_B] > 0) {
122 real val=r.c/r.t[VAR_B];
123 if (val < min) {
124 min=val;
125 argmin=i;
126 }
127 }
128 }
129
130 if(argmin != 0) return argmin;
131
132 real max;
133 var argmax;
134 var i=2;
135 for (; i < rows.length; ++i) {
136 row r=rows[i];
137 if (r.t[VAR_B] < 0) {
138 max=r.c/r.t[VAR_B];
139 argmax=i;
140 break;
141 }
142 }
143
144 for (; i < rows.length; ++i) {
145 row r=rows[i];
146 if (r.t[VAR_B] < 0) {
147 real val=r.c/r.t[VAR_B];
148 if (val > max) {
149 max=val;
150 argmax=i;
151 }
152 }
153 }
154
155 if(argmax != 0) return argmax;
156
157 return UNBOUNDED;
158 }

160 // Initialize the linear program problem by moving into an acceptable state
161 // this assumes that b is unrestrained and is the second variable.
162 // NOTE: Works in limited cases, may be bug-ridden.
163 void init()
164 {
165 // Find the lowest constant term in the equations.
166 var lowest = 0;
167 for (var i = 2; i < rows.length; ++i) {
168 if (rows[i].c < rows[lowest].c)
169 lowest = i;
170 }
171
172 // Pivot if necessary.
173 if (lowest != 0)
174 pivot(VAR_B, lowest);
175 }

185 // Select the new variable associated with a pivot on the column given.
186 // Returns UNBOUNDED if the space is unbounded.
187 var selectVar(int col)
188 {
189 // We assume that the first two vars (a and b) once swapped out, won't be
190 // swapped back in. This finds the variable which gives the tightest
191 // non-negativity condition restricting our optimization. This turns
192 // out to be the max of c/t[col]. Note that as c is positive, and
193 // t[col] is negative, all c/t[col] will be negative, so we are finding
194 // the smallest in magnitude.
195 var vp=UNBOUNDED;
196 real max=0;
197 int i=2;
198 for (; i < rows.length; ++i) {
199 row r=rows[i];
200 if(r.c < 0) r.c=0; // Fix any numerical precision error
201 if(r.t[col] < 0) {
202 max=r.c/r.t[col]; vp=i;
203 break;
204 }
205 }
206 for (; i < rows.length; ++i) {
207 row r=rows[i];
208 if(r.c < 0) r.c=0; // Fix any numerical precision error
209 if(r.c < max*r.t[col]) {
210 max=r.c/r.t[col]; vp=i;
211 }
212 }
213
214 return vp;
215 }

232 // used.

234
235 var vc = v[col];
236 row rvc = rows[vc];

238 if (rvc.c != 0)

259 // Perform the algorithm to find the optimal solution. Returns OPTIMAL,
260 // UNBOUNDED, or INVALID (if no solution is possible).
261 int optimize()
262 {
263 // Put into a valid state to begin and pivot b out.
264 var iv=initVar();
265 if (iv == UNBOUNDED)
266 return iv;
267 pivot(VAR_B, iv);
268
269 if (!valid())
270 return INVALID;
271
272 while(true) {
273 int col = selectColumn();
274
275 if (col == OPTIMAL)
276 return col;
277 var vp = selectVar(col);
278
279 if (vp == UNBOUNDED)
280 return vp;
281
282 pivot(col, vp);
283 }
284
285 // Shouldn't reach here.
286 return INVALID;
287 }
