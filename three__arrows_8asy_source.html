1// A transformation that bends points along a path
2transform3 bend(path3 g, real t)
3{
4 triple dir=dir(g,t);
5 triple a=point(g,0), b=postcontrol(g,0);
6 triple c=precontrol(g,1), d=point(g,1);
7 triple dir1=b-a;
8 triple dir2=c-b;
9 triple dir3=d-c;
10
11 triple u = unit(cross(dir1,dir3));
12 real eps=1000*realEpsilon;
13 if(abs(u) < eps) {
14 u = unit(cross(dir1,dir2));
15 if(abs(u) < eps) {
16 u = unit(cross(dir2,dir3));
17 if(abs(u) < eps)
18 // linear segment: use any direction perpendicular to initial direction
19 u = perp(dir1);
20 }
21 }
22 u = unit(perp(u,dir));
23
24 triple w=cross(dir,u);
25 triple q=point(g,t);
26 return new real[][] {
27 {u.x,w.x,dir.x,q.x},
28 {u.y,w.y,dir.y,q.y},
29 {u.z,w.z,dir.z,q.z},
30 {0,0,0,1}
31 };
32}

51// Refine a noncyclic path3 g so that it approaches its endpoint in
52// geometrically spaced steps.
53path3 approach(path3 p, int n, real radix=3)
54{
55 guide3 G;
56 real L=length(p);
57 real tlast=0;
58 real r=1/radix;
59 for(int i=1; i < n; ++i) {
60 real t=L*(1-r^i);
61 G=G&subpath(p,tlast,t);
62 tlast=t;
63 }
64 return G&subpath(p,tlast,L);
65}
