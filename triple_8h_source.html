139 friend void boundstriples(triple& Min, triple& Max, size_t n, const triple* v)
140 {
141 if(n==0 || v==NULL)
142 return;
143
144 double x,y,z;
145 double X,Y,Z;
146
147 X=x=v[0].getx();
148 Y=y=v[0].gety();
149 Z=z=v[0].getz();
150 for(size_t i=1; i < n; ++i) {
151 const double vx=v[i].getx();
152 x=fmin(x,vx); X=fmax(X,vx);
153 const double vy=v[i].gety();
154 y=fmin(y,vy); Y=fmax(Y,vy);
155 const double vz=v[i].getz();
156 z=fmin(z,vz); Z=fmax(Z,vz);
157 }
158
159 Min.set(x,y,z);
160 Max.set(X,Y,Z);
161 }

163 friend void ratiotriples(pair &b, double (*m)(double, double), bool &first,
164 size_t n, const triple* v)
165 {
166 if(n==0 || v==NULL)
167 return;
168
169 if(first) {
170 first=false;
171 const triple& v0=v[0];
172 b=pair(v0.x/v0.z,v0.y/v0.z);
173 }
174
175 double x=b.getx();
176 double y=b.gety();
177 for(size_t i=0; i < n; ++i) {
178 const triple& vi = v[i];
179 x=m(x,vi.x/vi.z);
180 y=m(y,vi.y/vi.z);
181 }
182 b=pair(x,y);
183 }

396// return the maximum distance squared of points c0 and c1 from
397// the respective internal control points of z0--z1.
398inline double Straightness(const triple& z0, const triple& c0,
399 const triple& c1, const triple& z1)
400{
401 triple v=third*(z1-z0);
402 return std::max(abs2(c0-v-z0),abs2(z1-v-c1));
403}

405// Return one ninth of the relative flatness squared of a--b and c--d.
406inline double Flatness(const triple& a, const triple& b, const triple& c,
407 const triple& d)
408{
409 static double ninth=1.0/9.0;
410 triple u=b-a;
411 triple v=d-c;
412 return ninth*std::max(abs2(cross(u,unit(v))),abs2(cross(v,unit(u))));
413}
