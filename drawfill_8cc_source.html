15void drawAxialShade::palette(psfile *out)
16{
17 pentype.convert();
18 penb.convert();
19
20 colorspace=(ColorSpace) max(pentype.colorspace(),penb.colorspace());
21
22 switch(colorspace) {
23 case RGB:
24 {
25 if (pentype.grayscale()) pentype.greytorgb();
26 else if (penb.grayscale()) penb.greytorgb();
27 break;
28 }
29
30 case CMYK:
31 {
32 if (pentype.grayscale()) pentype.greytocmyk();
33 else if (penb.grayscale()) penb.greytocmyk();
34
35 if (pentype.rgb()) pentype.rgbtocmyk();
36 else if (penb.rgb()) penb.rgbtocmyk();
37 break;
38 }
39 default:
40 break;
41 }
42
43 out->gsave();
44}
