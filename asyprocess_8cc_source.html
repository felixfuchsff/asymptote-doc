90// A process environment. Basically this just serves as short-hand to start a
91// new global environment (genv) and new process data at the same time. When
92// it goes out of scope, the process data is popped off the stack. This also
93// ensures that the data is popped even if an exception is thrown.
94class penv {
95 genv *_ge;
96 processDataStruct _pd;
97
98 // Do not let this object be dynamically allocated.
99 void *operator new(size_t);
100
101public:
102 penv() : _ge(0), _pd() {
103 // Push the processData first, as it needs to be on the stack before genv
104 // is initialized.
105 _pd.topPos = nullPos;
106 processDataStack.push(&_pd);
107 _ge = new genv;
108 }
109
110 virtual ~penv() {
111 processDataStack.pop();
112 delete _ge;
113 }
114
115 genv &ge() { return *_ge; }
116
117 processDataStruct *pd() { return &_pd; }
118};

159// How to run a runnable in runnable-at-a-time mode.
160bool runRunnable(runnable *r, coenv &e, istack &s, transMode tm=TRANS_NORMAL) {
161 e.e.beginScope();
162
163 lambda *codelet= tm==TRANS_INTERACTIVE ?
164 interactiveRunnable(r).transAsCodelet(e) :
165 r->transAsCodelet(e);
166 em.sync();
167 if(!em.errors()) {
168 if(getSetting<bool>("translate")) print(cout,codelet->code);
169 s.run(codelet);
170
171 // Commits the changes made to the environment.
172 e.e.collapseScope();
173 } else {
174 e.e.endScope(); // Remove any changes to the environment.
175
176 // Should an interactive error hurt the status?
177 em.statusError();
178
179 return false;
180 }
181 return true;
182}

204void icore::doRun(bool purge, transMode tm) {
205 em.sync();
206 if(em.errors())
207 return;
208
209 try {
210 if(purge) run::purge();
211
212 penv pe;
213 env base_env(pe.ge());
214 coder base_coder(nullPos, "icore::doRun");
215 coenv e(base_coder,base_env);
216
217 vm::interactiveStack s;
218 s.setInitMap(pe.ge().getInitMap());
219 s.setEnvironment(&e);
220
221 preRun(e,s);
222
223 if(purge) run::purge();
224
225 // Now that everything is set up, run the core.
226 run(e,s,tm);
227
228 postRun(e,s);
229
230 } catch(std::bad_alloc&) {
231 outOfMemory();
232 } catch(quit const&) {
233 // Exception to quit running the current code. Nothing more to do.
234 } catch(handled_error const&) {
235 em.statusError();
236 }
237
238 run::cleanup();
239
240 em.clear();
241}

362// cleanLine changes a C++ style comment (//) into a C-style comment (/* */) so
363// that comments don't absorb subsequent lines of code when multiline input is
364// collapsed to a single line in the history.
365//
366// ex. if (x==1) // test x
367// x=2;
368// becomes
369// if (x==1) /* test x */ x=2 (all on one line)
370//
371// cleanLine is a mess because we have to check that the // is not in a string
372// or c-style comment, which entails re-inventing much of the lexer. The
373// routine handles most cases, but multiline strings lose their newlines when
374// recorded in the history.
375typedef string::size_type size_type;
376const size_type npos=string::npos;
377
378inline size_type min(size_type a, size_type b) {
379 return a < b ? a : b;
380}
381
382// If start is an offset somewhere within a string, this returns the first
383// offset outside of the string. sym is the character used to start the string,
384// ' or ".
385size_type endOfString(const char sym, const string line, size_type start) {
386 size_type endString=line.find(sym, start);
387 if (endString == npos)
388 return npos;
389
390 size_type escapeInString=min(line.find(string("\\")+sym, start),
391 line.find("\\\\", start));
392 if (endString < escapeInString)
393 return endString+1;
394 else
395 return endOfString(sym, line, escapeInString+2);
396}
397
398// If start is an offset somewhere within a C-style comment, this returns the
399// first offset outside of the comment.
400size_type endOfComment(const string line, size_type start) {
401 size_type endComment=line.find("*/", start);
402 if (endComment == npos)
403 return npos;
404 else
405 return endComment+2;
406}
407
408// Find the start of a string literal in the line.
409size_type stringPos(const string line, size_type start) {
410 if (start == npos)
411 return npos;
412
413 size_type pos=line.find_first_of("\'\"", start);
414 if (pos == npos)
415 return npos;
416
417 // Skip over comments /* */ and ignore anything after //
418 size_type startComment=line.find("/*", start);
419 size_type startLineComment=line.find("//", start);
420
421 if (min(startComment,startLineComment) < pos)
422 return stringPos(line,
423 startComment < startLineComment ?
424 endOfComment(line, startComment+2) :
425 npos);
426 else
427 // Nothing to skip over - the symbol actually starts a string.
428 return pos;
429}
430
431// A multiline string should retain its newlines when collapsed to a single
432// line. This converts
433// 'hello
434// there'
435// to
436// 'hello\nthere'
437// and
438// "hello
439// there"
440// to
441// "hello" '\n' "there"
442// If the line doesn't end mid-string, this adds a space to the end to preserve
443// whitespace, since it is the last function to touch the line.
444string endString(const string line, size_type start) {
445 assert(start!=npos);
446
447 size_type pos=stringPos(line, start);
448
449 if (pos==npos)
450 // String ends in normal code.
451 return line+" ";
452 else {
453 char sym=line[pos];
454 size_type eos=endOfString(sym, line, pos+1);
455 if (eos==npos) {
456 // Line ends while in a string, attach a newline symbol.
457 switch (line[pos]) {
458 case '\'':
459 return line+"\\n";
460 case '\"':
461 return line+"\" \'\\n\' \"";
462 default:
463 assert(False);
464 return line;
465 }
466 }
467 else {
468 return endString(line, eos+1);
469 }
470 }
471}
472
473// Find the first // that isn't in a C-style comment or a string.
474size_type slashPos(const string line, size_type start) {
475 if (start == npos)
476 return npos;
477
478 size_type pos=line.find("//", start);
479 if (pos == npos)
480 return npos;
481
482 // Skip over comments /* */ and strings both " " and ' '
483 size_type startComment=line.find("/*", start);
484 size_type startString=line.find_first_of("\'\"", start);
485
486 if (min(startComment,startString) < pos)
487 return slashPos(line,
488 startComment < startString ?
489 endOfComment(line, startComment+2) :
490 endOfString(line[startString], line, startString+1));
491 else
492 // Nothing to skip over - the // actually starts a comment.
493 return pos;
494}
495
496string endCommentOrString(const string line) {
497 size_type pos=slashPos(line, 0);
498 if (pos == npos)
499 return endString(line, 0);
500 else {
501 string sub=line;
502
503 // Replace the first // by /*
504 sub[pos+1]='*';
505
506 // Replace any */ in the comment by *!
507 while ((pos = line.find("*/", pos+2)) != npos)
508 sub[pos+1]='!';
509
510 // Tack on a */ at the end.
511 sub.append(" */ ");
512 return sub;
513 }
514}

516bool isSlashed(const string line) {
517 // NOTE: This doesn't fully handle escaped slashed in a string literal.
518 unsigned n=line.size();
519 return n > 0 ? line[line.size()-1] == '\\' : false;
520}
521string deslash(const string line) {
522 return isSlashed(line) ? line.substr(0,line.size()-1) : line;
523}

525// This transforms a line in to the history, so that when more code is added
526// after it, the code behaves the same as if there was a newline between the
527// two lines. This involves changing // style comments to /* */ style comments,
528// and adding explicit newlines to multiline strings.
529string cleanLine(const string line) {
530 // First remove a trailing slash, if there is one.
531 return endCommentOrString(deslash(line));
532}

547 // Commands are chopped into the starting word and the rest of the line.
548 struct commandLine {
549 string line;
550 string word;
551 string rest;
552
553 commandLine(string line) : line(line) {
554 string::iterator c=line.begin();
555
556 // Skip leading whitespace
557 while (c != line.end() && isspace(*c))
558 ++c;
559
560 // Only handle identifiers starting with a letter.
561 if (c != line.end() && isalpha(*c)) {
562 // Store the command name.
563 while (c != line.end() && (isalnum(*c) || *c=='_')) {
564 word.push_back(*c);
565 ++c;
566 }
567 }
568
569 // Copy the rest to rest.
570 while (c != line.end()) {
571 rest.push_back(*c);
572 ++c;
573 }
574
575#if 0
576 cerr << "line: " << line << endl;
577 cerr << "word: " << word << endl;
578 cerr << "rest: " << rest << endl;
579 cerr << "simple: " << simple() << endl;
580#endif
581 }
582
583 // Simple commands have at most spaces or semicolons after the command word.
584 bool simple() {
585 for (string::iterator c=rest.begin(); c != rest.end(); ++c)
586 if (!isspace(*c) && *c != ';')
587 return false;
588 return true;
589 }
590 };

700 // Get input from the interactive prompt. Such input may be over several
701 // user-typed lines if he/she ends a line a with backslash to continue input
702 // on the next line. If continuation is true, the input started on a previous
703 // line and is being continued (either because of a backslash or the parser
704 // detecting it in multiline mode).
705 string getline(bool continuation) {
706 string prompt;
707 if(!settings::xasy)
708 prompt=getSetting<string>(continuation ? "prompt2" : "prompt");
709 string line=interact::simpleline(prompt);
710
711 if (continuation)
712 addToLastLine(line);
713 else
714 addToHistory(line);
715
716 // If the line ends in a slash, get more input.
717 return isSlashed(line) ? line+"\n"+getline(true) :
718 line;
719 }

721 // Continue taking input for a line until it properly parses, or a syntax
722 // error occurs. Returns the parsed code on success, and throws a
723 // handled_error exception on failure.
724 block *parseExtendableLine(string line) {
725 block *code=parser::parseString(line, "-", true);
726 if (code) {
727 return code;
728 } else {
729 string nextline=getline(true);
730 return parseExtendableLine(line+"\n"+nextline);
731 }
732 }

734 // Continue taking input until a termination command is received from xasy.
735 block *parseXasyLine(string line) {
736 const string EOT="\x04\n";
737 string s;
738 while((s=getline(true)) != EOT)
739 line += s;
740 return parser::parseString(line, "-", true);
741 }

743 void runLine(coenv &e, istack &s, string line) {
744 try {
745 if(getSetting<bool>("multiline")) {
746 block *code=parseExtendableLine(line);
747
748 icode i(code);
749 i.run(e,s,TRANS_INTERACTIVE);
750 } else if(settings::xasy) {
751 block *code=parseXasyLine(line);
752
753 icode i(code);
754 i.run(e,s,TRANS_INTERACTIVE);
755 } else {
756 // Add a semi-colon to the end of the line if one is not there. Do this
757 // to the history as well, so the transcript can be run as regular asy
758 // code. This also makes the history work correctly if the multiline
759 // setting is changed within an interactive session.
760 // It is added to the history at the last possible moment to avoid
761 // tampering with other features, such as using a slash to extend a
762 // line.
763 terminateLastHistoryLine();
764 istring i(terminateLine(line), "-");
765 i.run(e,s,TRANS_INTERACTIVE);
766 }
767
768 run::updateFunction(&s);
769 uptodate=false;
770
771 } catch(handled_error const&) {
772 vm::indebugger=false;
773 } catch(interrupted&) {
774 // Turn off the interrupted flag.
775 em.Interrupt(false);
776 uptodate=true;
777 cout << endl;
778 } catch(quit&) {
779 }

782 em.clear();
