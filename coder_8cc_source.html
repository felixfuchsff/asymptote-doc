137void coder::encodePop()
138{
139 if (isStatic() && !isTopLevel()) {
140 assert(parent);
141 parent->encodePop();
142 }
143 else {
144#ifdef COMBO
145 vm::program::label end = program->end();
146 --end;
147 inst& lastInst = *end;
148 if (lastInst.op == inst::varsave) {
149 lastInst.op = inst::varpop;
150 return;
151 }
152 if (lastInst.op == inst::fieldsave) {
153 lastInst.op = inst::fieldpop;
154 return;
155 }
156 // TODO: push+pop into no op.
157#endif
158
159 // No combo applicable. Just encode a usual pop.
160 encode(inst::pop);
161 }
162}

165bool coder::encode(frame *f)
166{
167 frame *toplevel = getFrame();
168
169 if (f == 0) {
170 encode(inst::constpush,(item)0);
171 return true;
172 }
173 else if (f == toplevel) {
174 encode(inst::pushclosure);
175 return true;
176 }
177 else {
178 encode(inst::varpush,toplevel->parentIndex());
179 return encode(f, toplevel->getParent());
180 }
181}

183bool coder::encode(frame *dest, frame *top)
184{
185 if (dest == 0) {
186 // Change to encodePop?
187 encode(inst::pop);
188 encode(inst::constpush,(item)0);
189 }
190 else {
191 frame *level = top;
192 while (level != dest) {
193 if (level == 0) {
194 // Frame request was in an improper scope.
195 return false;
196 }
197
198 encode(inst::fieldpush, level->parentIndex());
199
200 level = level->getParent();
201 }
202 }
203
204 //cerr << "succeeded\n";
205 return true;
206}

235label coder::defLabel(label label)
236{
237 if (isStatic())
238 return parent->defLabel(label);
239
240 //cout << "defining label " << label << endl;
241
242 assert(!label->location.defined());
243 //vm::program::label here = program->end();
244 label->location = program->end();
245 assert(label->location.defined());
246
247 if (label->firstUse.defined()) {
248 replaceEmptyJump(label->firstUse, program->end());
249 //vm::printInst(cout, label->firstUse, program->begin());
250 //cout << endl;
251
252 if (label->moreUses) {
253 typedef label_t::useVector useVector;
254 useVector& v = *label->moreUses;
255 for (useVector::iterator p = v.begin(); p != v.end(); ++p) {
256 replaceEmptyJump(*p, program->end());
257 }
258 }
259 }
260
261 return label;

265void coder::useLabel(inst::opcode op, label label)
266{
267 if (isStatic())
268 return parent->useLabel(op,label);
269
270 if (label->location.defined()) {
271 encode(op, label->location);
272 } else {
273 if (label->firstUse.defined()) {
274 // Store additional uses in the moreUses array.
275 if (!label->moreUses)
276 label->moreUses = new label_t::useVector;
277 label->moreUses->push_back(encodeEmptyJump(op));
278 }
279 else {
280 label->firstUse = encodeEmptyJump(op);
281 assert(label->firstUse.defined());
282 assert(!label->location.defined());
283 }
284 }

326// When translating the function is finished, this ties up loose ends
327// and returns the lambda.
328vm::lambda *coder::close() {
329 // These steps must be done dynamically, not statically.
330 sord = EXPLICIT_DYNAMIC;
331 sord_stack.push(sord);
332
333 // Add a return for void types; may be redundant.
334 if (funtype->result->kind == types::ty_void)
335 encode(inst::ret);
336
337 l->code = program;
338
339 l->parentIndex = level->parentIndex();
340
341 l->framesize = level->size();
342
343 sord_stack.pop();
344 sord = sord_stack.top();
345
346 return l;
347}

375bool coder::encodeParent(position pos, trans::tyEntry *ent) {
376 record *r = dynamic_cast<record *>(ent->t);
377 if (!r) {
378 em.compiler(pos);
379 em << "type '" << *ent->t << "' is not a structure";
380 return false;
381 }
382 assert(r);
383
384 // The level needed on which to allocate the record.
385 frame *level = r->getLevel()->getParent();
386
387 if (ent->v) {
388 // Put the record on the stack. For instance, in code like
389 // access imp;
390 // new imp.t;
391 // we are putting the instance of imp on the stack, so we can use it to
392 // allocate an instance of imp.t.
393 ent->v->encode(trans::READ, pos, *this);
394
395 // Adjust to the right frame. For instance, in the last new in
396 // struct A {
397 // struct B {
398 // static struct C {}
399 // }
400 // B b=new B;
401 // }
402 // A a=new A;
403 // new a.b.C;
404 // we push a.b onto the stack, but need a as the enclosing frame for
405 // allocating an instance of C.
406 record* q= dynamic_cast<record*>(ent->v->getType());
407 assert(q);
408 return encode(level, q->getLevel());
409 } else
410 return encode(level);
411}
