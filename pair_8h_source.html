1/*****
2 * pair.h
3 * Andy Hammerlindl 2002/05/16
4 *
5 * Stores a two-dimensional point similar to the pair type in MetaPost.
6 * In some cases, pairs behave as complex numbers.
7 *
8 * A pair is a guide as a pair alone can be used to describe a path.
9 * The solve and subsolve methods are fairly straight forward as solve
10 * returns a path with just the pair and subsolve just adds the pair to
11 * the structure.

254// Calculates exp(i * theta), useful for unit vectors.
255inline pair expi(double theta)
256{
257 if(theta == 0.0) return pair(1.0,0.0); // Frequently occurring case
258 return pair(cos(theta),sin(theta));
259}

261// Complex exponentiation
262inline pair pow(const pair& z, const pair& w)
263{
264 double u=w.getx();
265 double v=w.gety();
266 if(z == 0.0) return w == 0.0 ? 1.0 : 0.0;
267 double logr=0.5*log(z.abs2());
268 double th=z.angle();
269 return exp(logr*u-th*v)*expi(logr*v+th*u);
270}
