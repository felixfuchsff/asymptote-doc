17rmf[] rmf(path3 g, real[] t, triple perp=O)
18{
19 triple T=dir(g,0);
20 triple Tp=abs(perp) < sqrtEpsilon ? perp(T) : unit(perp);
21 rmf[] R=new rmf[t.length];
22 R[0]=rmf(point(g,0),Tp,T);
23 for(int i=1; i < t.length; ++i) {
24 rmf Ri=R[i-1];

27 triple v1=p-Ri.p;

32 triple tp=ti-2*dot(u1,ti)*u1;
33 ti=dir(g,t);

35 triple u2=unit(ti-tp);
36 rp=rp-2*dot(u2,rp)*u2;
37 R[i]=rmf(p,unit(rp),unit(ti));
38 } else
39 R[i]=R[i-1];
40 }

44rmf[] rmf(triple z0, triple c0, triple c1, triple z1, real[] t, triple perp=O)
45{
46 static triple s0;
47
48 real norm=sqrtEpsilon*max(abs(z0),abs(c0),abs(c1),abs(z1));
49
50 // Special case of dir for t in (0,1].
51 triple dir(real t) {
52 if(t == 1) {
53 triple dir=z1-c1;
54 if(abs(dir) > norm) return unit(dir);
55 dir=2.0*c1-c0-z1;
56 if(abs(dir) > norm) return unit(dir);
57 return unit(z1-z0+3.0*(c0-c1));
58 }
59 triple a=z1-z0+3.0*(c0-c1);
60 triple b=2.0*(z0+c1)-4.0*c0;
61 triple c=c0-z0;
62 triple dir=a*t*t+b*t+c;
63 if(abs(dir) > norm) return unit(dir);
64 dir=2.0*a*t+b;
65 if(abs(dir) > norm) return unit(dir);
66 return unit(a);
67 }
68
69 triple T=c0-z0;
70 if(abs(T) < norm) {
71 T=z0-2*c0+c1;
72 if(abs(T) < norm)
73 T=z1-z0+3.0*(c0-c1);
74 }
75 T=unit(T);
76 triple Tp=perp == O ? cross(s0,T) : perp;
77 Tp=abs(Tp) < sqrtEpsilon ? perp(T) : unit(Tp);
78 rmf[] R=new rmf[t.length];
79 R[0]=rmf(z0,Tp,T);
80
81 for(int i=1; i < t.length; ++i) {
82 rmf Ri=R[i-1];
83 real t=t[i];
84 triple p=bezier(z0,c0,c1,z1,t);
85 triple v1=p-Ri.p;
86 if(v1 != O) {
87 triple r=Ri.r;
88 triple u1=unit(v1);
89 triple ti=Ri.t;
90 triple tp=ti-2*dot(u1,ti)*u1;
91 ti=dir(t);
92 triple rp=r-2*dot(u1,r)*u1;
93 triple u2=unit(ti-tp);
94 rp=rp-2*dot(u2,rp)*u2;
95 R[i]=rmf(p,unit(rp),unit(ti));
96 } else
97 R[i]=R[i-1];
98 }
99 s0=R[t.length-1].s;
100 return R;
101}
