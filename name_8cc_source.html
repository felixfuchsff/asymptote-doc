33void name::forceEquivalency(action act, coenv &e,
34 types::ty *target, types::ty *source)
35{
36 if (act == READ)
37 e.implicitCast(getPos(), target, source);
38 else if (!equivalent(target, source)) {
39 em.compiler(getPos());
40 em << "type mismatch in variable: "
41 << *target
42 << " vs " << *source;
43 }

72{

85void simpleName::varTrans(action act, coenv &e, types::ty *target)
86{

89 if (v) {

156record *qualifiedName::castToRecord(types::ty *t)
157{
158 switch (t->kind) {
159 case ty_overloaded:
160 em.compiler(qualifier->getPos());
161 em << "name::getType returned overloaded";
162 return 0;
163 case ty_record:
164 return (record *)t;
165 case ty_error:
166 return 0;
167 default:
168 em.error(qualifier->getPos());
169 em << "type '" << *t << "' is not a structure";
170 return 0;
171 }
172}

174bool qualifiedName::varTransVirtual(action act, coenv &e,
175 types::ty *target, types::ty *qt)
176{
177 varEntry *v = qt->virtualField(id, target->getSignature());
178 if (v) {
179 // Push qualifier onto stack.
180 qualifier->varTrans(READ, e, qt);
181
182 v->encode(act, getPos(), e.c);
183
184 // A virtual field was used.
185 return true;
186 }
187
188 // No virtual field.
189 return false;
190}

192void qualifiedName::varTransField(action act, coenv &e,
193 types::ty *target, record *r)
194{
195 varEntry *v = r->e.lookupVarByType(id, target);
196
197 if (v) {
198 frame *f = qualifier->frameTrans(e);
199 if (f)
200 v->encode(act, getPos(), e.c, f);
201 else
202 v->encode(act, getPos(), e.c);
203
204 forceEquivalency(act, e, target, v->getType());
205 }
206 else {
207 em.error(getPos());
208 em << "no matching field of name \'" << id << "\' in \'" << *r << "\'";
209 }
210}

212void qualifiedName::varTrans(action act, coenv &e, types::ty *target)
213{
214 types::ty *qt = qualifier->getType(e);
215
216 // Use virtual fields if applicable.
217 if (varTransVirtual(act, e, target, qt))
218 return;
219
220 record *r = castToRecord(qt);
221 if (r)
222 varTransField(act, e, target, r);

225types::ty *qualifiedName::varGetType(coenv &e)
226{
227 types::ty *qt;
228 { // Suppress errors while calling qualifier->getType.
229 auto modeGuard = em.modeGuard(ErrorMode::SUPPRESS);
230 qt = qualifier->getType(e);
231 }
232
233 // Look for virtual fields.
234 types::ty *t = qt->virtualFieldGetType(id);
235 if (t)
236 return t;
237
238 record *r;
239 { // Suppress errors while calling castToRecord.
240 auto modeGuard = em.modeGuard(ErrorMode::SUPPRESS);
241 r = castToRecord(qt);
242 }
243 return r ? r->e.varGetType(id) : 0;

272types::ty *qualifiedName::typeTrans(coenv &e)
273{
274 types::ty *rt = qualifier->getType(e);
275
276 record *r = castToRecord(rt);
277 if (!r)
278 return primError();
279
280 tyEntry *ent = r->e.lookupTyEntry(id);
281 if (ent) {
282 if (!em.isSuppressed())
283 ent->reportPerm(READ, getPos(), e.c);
284 return ent->t;
285 }
286 else {
287 em.error(getPos());
288 em << "no matching field or type of name \'" << id << "\' in \'"
289 << *r << "\'";
290 return primError();
291 }

294tyEntry *qualifiedName::tyEntryTrans(coenv &e)
295{
296 types::ty *rt = qualifier->getType(e);
297
298 record *r = castToRecord(rt);
299 if (!r)
300 return new tyEntry(primError(), nullptr, nullptr, nullPos);
301
302 tyEntry *ent = r->e.lookupTyEntry(id);
303 if (!ent) {
304 em.error(getPos());
305 em << "no matching type of name \'" << id << "\' in \'"
306 << *r << "\'";
307 return new tyEntry(primError(), nullptr, nullptr, nullPos);
308 }
309 ent->reportPerm(READ, getPos(), e.c);
310
311 return trans::qualifyTyEntry(qualifier->getVarEntry(e), ent);
312}

314frame *qualifiedName::tyFrameTrans(coenv &e)
315{
316 frame *f=qualifier->frameTrans(e);
317 tyEntry *ent = e.e.lookupTyEntry(id);
318 if (ent && ent->t->kind==types::ty_record && ent->v) {
319 if (f)
320 ent->v->encode(READ, getPos(), e.c, f);
321 else
322 ent->v->encode(READ, getPos(), e.c);
323 return ent->v->getLevel();
324 }
325 else
326 return f;

337AsymptoteLsp::SymbolLit qualifiedName::getLit() const
338{
339 std::vector<std::string> accessors;
340 name const* currentScope = this->qualifier;
341
342 while (auto* qn = dynamic_cast<qualifiedName const*>(currentScope))
343 {
344 accessors.push_back(static_cast<std::string>(qn->getName()));
345 currentScope = qn->qualifier;
346 }
347 accessors.push_back(static_cast<std::string>(currentScope->getName()));
348
349 return AsymptoteLsp::SymbolLit(static_cast<std::string>(id), std::move(accessors));
350}
