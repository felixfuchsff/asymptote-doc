104int[][] frequency(real[] x, real[] y, pair a, pair b, int nx, int ny=nx)
105{
106 int[][] freq=new int[nx][];
107 for(int i=0; i < nx; ++i)
108 freq[i]=sequence(new int(int x) {return 0;},ny);

214// Method of Shimazaki and Shinomoto for selecting the optimal number of bins.
215// Shimazaki H. and Shinomoto S., A method for selecting the bin size of a
216// time histogram, Neural Computation (2007), Vol. 19(6), 1503-1527.
217// cf. http://www.ton.scphys.kyoto-u.ac.jp/~hideaki/res/histogram.html
218int bins(real[] data, int max=100)
219{
220 real m=min(data);
221 real M=max(data)*(1+epsilon);
222 real n=data.length;
223 int bins=1;
224 real minC=2n-n^2; // Cost function for N=1.
225 for(int N=2; N <= max; ++N) {
226 real C=N*(2n-sum(frequency(data,m,M,N)^2));
227 if(C < minC) {
228 minC=C;
229 bins=N;
230 }
231 }
232
233 return bins;
234}

269// Do a least-squares fit of data in real arrays x and y to the line y=m*x+b
270linefit leastsquares(real[] x, real[] y)
271{
272 linefit L;
273 int n=x.length;
274 if(n == 1) abort("Least squares fit requires at least 2 data points");
275 real sx=sum(x);
276 real sy=sum(y);
277 real sxx=n*sum(x^2)-sx^2;
278 real sxy=n*sum(x*y)-sx*sy;
279 L.m=sxy/sxx;
280 L.b=(sy-L.m*sx)/n;
281 if(n > 2) {
282 real syy=n*sum(y^2)-sy^2;
283 if(sxx == 0 || syy == 0) return L;
284 L.r=sxy/sqrt(sxx*syy);
285 real arg=syy-sxy^2/sxx;
286 if(arg <= 0) return L;
287 real s=sqrt(arg/(n-2));
288 L.dm=s*sqrt(1/sxx);
289 L.db=s*sqrt(1+sx^2/sxx)/n;
290 }
291 return L;
292}

294// Do a least-squares fit of data in real arrays x and y weighted by w
295// to the line y=m*x+b, by minimizing sum(w*(y-m*x-b)^2).
296linefit leastsquares(real[] x, real[] y, real[] w)
297{

299 int n=x.length;
300 if(n == 1) abort("Least squares fit requires at least 2 data points");
301 real sx=sum(w*x);
302 real sy=sum(w*y);
303 real W=sum(w);
304 real sxx=W*sum(w*x^2)-sx^2;
