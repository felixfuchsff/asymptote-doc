148void texfile::beginlayer(string psname, bool postscript)
149{
150#ifdef _WIN32
151 backslashToSlash(psname);
152#endif
153 if(box.right > box.left && box.top > box.bottom) {
154 if(postscript) {
155 if(settings::context(texengine))
156 *out << "\\externalfigure[" << psname << "]%" << newl;
157 else {
158 *out << "{\\catcode`\"=12%" << newl
159 << "\\includegraphics";
160 string name=stripExt(psname);
161 if(inlinetex) {
162 size_t pos=name.rfind("-");
163 if(pos < string::npos) name="\\ASYprefix\\jobname"+name.substr(pos);
164 } else {
165 if(!pdf) name=psname;
166 }
167
168 if(!pdf)
169 *out << "[bb=" << box.left << " " << box.bottom << " "
170 << box.right << " " << box.top << "]";
171 *out << "{" << name << "}%" << newl << "}%" << newl;
172 }
173 if(!inlinetex)
174 *out << "\\kern " << (box.left-box.right)*ps2tex << "pt%" << newl;
175 } else {
176 *out << "\\leavevmode\\vbox to " << (box.top-box.bottom)*ps2tex
177 << "pt{}%" << newl;
178 if(inlinetex)
179 *out << "\\kern " << (box.right-box.left)*ps2tex << "pt%" << newl;
180 }
181 }

283void texfile::put(const string& label, const transform& T, const pair& z,
284 const pair& align)
285{
286 double sign=pdf ? 1.0 : -1.0;
287
288 if(label.empty()) return;
289
290 bool trans=!T.isIdentity();
291
292 *out << "\\ASYalign";
293 if(trans) *out << "T";
294 *out << "(" << (z.getx()-hoffset())*ps2tex
295 << "," << (z.gety()-voffset())*ps2tex
296 << ")(" << align.getx()
297 << "," << align.gety()
298 << ")";
299 if(trans)
300 *out << "{" << T.getxx() << " " << sign*T.getyx()
301 << " " << sign*T.getxy() << " " << T.getyy() << "}";
302 *out << "{" << label << "}%" << newl;
303}

486 size_t n=linetype->pattern.size();

488 *out << " stroke-dasharray='";

491 *out << "," << qtfix(vm::read<double>(linetype->pattern,i)*ps2tex);
492 *out << "'";
493 }
494 }
495
496 if(linetype->offset != lastlinetype->offset)
497 *out << " stroke-dashoffset='" << linetype->offset*ps2tex << "'";

556// Return the point on the line through p and q that is closest to z.
557pair closest(pair p, pair q, pair z)
558{
559 pair u=q-p;
560 double denom=dot(u,u);
561 return denom == 0.0 ? p : p+dot(z-p,u)/denom*u;
562}

564void svgtexfile::gouraudshade(const pen& p0, const pair& z0,
565 const pen& p1, const pair& z1,
566 const pen& p2, const pair& z2)
567{
568 string hex[]={rgbhex(p0),rgbhex(p1),rgbhex(p2)};
569
570 *out << "<defs>" << nl;
571
572 pair Z0=(z0-offset)*ps2tex;
573 pair Z1=(z1-offset)*ps2tex;
574 pair Z2=(z2-offset)*ps2tex;
575
576 pair Z[]={Z0,Z1,Z2};
577
578 for(size_t k=0; k < 3; ++k) {
579 pair z=Z[k];
580 pair opp=closest(Z[(k+1) % 3],Z[(k+2) % 3],z);
581 *out << "<linearGradient id='grad-" << gouraudcount << "-" << k
582 << "' gradientUnits='userSpaceOnUse'" << nl
583 << " x1='" << z.getx() << "' y1='" << -z.gety()
584 << "' x2='" << opp.getx() << "' y2='" << -opp.gety()
585 << "'>" << nl
586 << "<stop offset='0' stop-color='#" << hex[k]
587 << "' stop-opacity='1'/>" << nl
588 << "<stop offset='1' stop-color='#" << hex[k]
589 << "' stop-opacity='0'/>" << nl
590 << "</linearGradient>" << nl;
591 }
592
593 *out << "<polygon points='"
594 << Z0.getx() << "," << -Z0.gety() << " "
595 << Z1.getx() << "," << -Z1.gety() << " "
596 << Z2.getx() << "," << -Z2.gety() << "'"
597 << " id='triangle-" << gouraudcount << "' />" << nl;
598
599 for(unsigned vertex=0; vertex < 3; ++vertex)
600 *out << "<use xlink:href='#triangle-" << gouraudcount
601 << "' fill='url(#grad-" << gouraudcount << "-" << vertex
602 << ")' id='triangle-" << gouraudcount << "-" << vertex << "' />"
603 << nl;
604
605 *out << "<filter id='Gouraud-" << gouraudcount << "'>" << nl;
606
607 for(unsigned vertex=0; vertex < 3; ++vertex)
608 *out << "<feImage xlink:href='#triangle-" << gouraudcount << "-" << vertex
609 << "' result='layer" << vertex << "' x='0' y='0' />" << nl;
610
611 *out << "<feComposite in='layer0' in2='layer1' operator='arithmetic' k1='0' k2='1' k3='1' k4='0' result='temp'/>" << nl
612 << "<feComposite in='temp' in2='layer2' operator='arithmetic' k1='0' k2='1' k3='1' k4='0' result='temp2' />" << nl
613 << "<feComposite in='temp2' in2='SourceGraphic' operator='arithmetic' k1='0' k2='1' k3='1' k4='0'/>" << nl
614 << "</filter>" << nl
615 << "</defs>" << nl
616 << "<rect width='100\\percent' height='100\\percent' fill='none' ";
617 *out << " filter='url(#Gouraud-" << gouraudcount << ")'"
618 << "/>" << nl;
619
620 ++gouraudcount;
621}

632void svgtexfile::gouraudshade(const pen& pentype,
633 const array& pens, const array& vertices,
634 const array& edges)
635{
636 size_t size=pens.size();
637 if(size == 0) return;
638
639 endclip(pentype);
640 beginspecial();
641 begintransform();
642
643 pen *p0=NULL,*p1=NULL,*p2=NULL;
644 pair z0,z1,z2;
645
646 for(size_t i=0; i < size; i++) {
647 Int edge=read<Int>(edges,i);
648
649 switch(edge) {
650 case 0:
651 p0=read<pen *>(pens,i);
652 z0=read<pair>(vertices,i);
653 ++i;
654 if(i < size) {
655 p1=read<pen *>(pens,i);
656 z1=read<pair>(vertices,i);
657 ++i;
658 if(i < size) {
659 p2=read<pen *>(pens,i);
660 z2=read<pair>(vertices,i);
661 }
662 }
663 break;
664 case 1:
665 p0=read<pen *>(pens,i);
666 z0=read<pair>(vertices,i);
667 break;
668 case 2:
669 p1=read<pen *>(pens,i);
670 z1=read<pair>(vertices,i);
671 break;
672 default:
673 break;
674 }
675 if(p0 == NULL || p1 == NULL || p2 == NULL)
676 reportError("invalid edge flag");
677 gouraudshade(*p0,z0,*p1,z1,*p2,z2);
678 }
679 endtransform();
680 endspecial();
681}
