151 // Add variables and types of name src from another environment under the
152 // name dest in this environment.
153 Added *add(symbol src, symbol dest,
154 protoenv &source, varEntry *qualifier, coder &c)
155 {
156 Added *retv = new Added();
157 retv->typeAdded=te.add(src, dest, source.te, qualifier, c);
158 bool varAdded=ve.add(src, dest, source.ve, qualifier, c, &retv->varsAdded);
