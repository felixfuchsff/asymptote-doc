184inline void splitCubic(solvedKnot3 sn[], double t, const solvedKnot3& left_,
185 const solvedKnot3& right_)
186{
187 solvedKnot3 &left=(sn[0]=left_), &mid=sn[1], &right=(sn[2]=right_);
188 if(left.straight) {
189 mid.point=split(t,left.point,right.point);
190 triple deltaL=third*(mid.point-left.point);
191 left.post=left.point+deltaL;
192 mid.pre=mid.point-deltaL;
193 triple deltaR=third*(right.point-mid.point);
194 mid.post=mid.point+deltaR;
195 right.pre=right.point-deltaR;
196 mid.straight=true;
197 } else {
198 triple x=split(t,left.post,right.pre); // m1
199 left.post=split(t,left.point,left.post); // m0
200 right.pre=split(t,right.pre,right.point); // m2
201 mid.pre=split(t,left.post,x); // m3
202 mid.post=split(t,x,right.pre); // m4
203 mid.point=split(t,mid.pre,mid.post); // m5
204 }
205}

207path3 path3::subpath(double a, double b) const
208{
209 if(empty()) return path3();
210
211 if (a > b) {
212 const path3 &rp = reverse();
213 Int len=length();
214 return rp.subpath(len-a, len-b);
215 }
216
217 solvedKnot3 aL, aR, bL, bR;
218 if (!cycles) {
219 if (a < 0) {
220 a = 0;
221 if (b < 0)
222 b = 0;
223 }
224 if (b > n-1) {
225 b = n-1;
226 if (a > b)
227 a = b;
228 }
229 aL = nodes[(Int)floor(a)];
230 aR = nodes[(Int)ceil(a)];
231 bL = nodes[(Int)floor(b)];
232 bR = nodes[(Int)ceil(b)];
233 } else {
234 if(run::validInt(a) && run::validInt(b)) {
235 aL = nodes[imod((Int) floor(a),n)];
236 aR = nodes[imod((Int) ceil(a),n)];
237 bL = nodes[imod((Int) floor(b),n)];
238 bR = nodes[imod((Int) ceil(b),n)];
239 } else reportError("invalid path3 index");
240 }
241
242 if (a == b) return path3(point(a));
243
244 solvedKnot3 sn[3];
245 path3 p = subpath(Ceil(a), Floor(b));
246 if (a > floor(a)) {
247 if (b < ceil(a)) {
248 splitCubic(sn,a-floor(a),aL,aR);
249 splitCubic(sn,(b-a)/(ceil(b)-a),sn[1],sn[2]);
250 return path3(sn[0],sn[1]);
251 }
252 splitCubic(sn,a-floor(a),aL,aR);
253 p=concat(path3(sn[1],sn[2]),p);
254 }
255 if (ceil(b) > b) {
256 splitCubic(sn,b-floor(b),bL,bR);
257 p=concat(p,path3(sn[0],sn[1]));
258 }
259 return p;
260}

325// Return f evaluated at controlling vertex of bounding box of convex hull for
326// similiar-triangle transform x'=x/z, y'=y/z, where z < 0.
327double ratiobound(triple z0, triple c0, triple c1, triple z1,
328 double (*m)(double, double),
329 double (*f)(const triple&))
330{
331 double MX=m(m(m(-z0.getx(),-c0.getx()),-c1.getx()),-z1.getx());
332 double MY=m(m(m(-z0.gety(),-c0.gety()),-c1.gety()),-z1.gety());
333 double Z=m(m(m(z0.getz(),c0.getz()),c1.getz()),z1.getz());
334 double MZ=m(m(m(-z0.getz(),-c0.getz()),-c1.getz()),-z1.getz());
335 return m(f(triple(-MX,-MY,Z)),f(triple(-MX,-MY,-MZ)));
336}

338double bound(triple z0, triple c0, triple c1, triple z1,
339 double (*m)(double, double),
340 double (*f)(const triple&), double b, double fuzz, int depth)
341{
342 b=m(b,m(f(z0),f(z1)));
343 if(m(-1.0,1.0)*(b-ratiobound(z0,c0,c1,z1,m,f)) >= -fuzz || depth == 0)
344 return b;
345
346 --depth;
347 fuzz *= 2;
348
349 triple m0=0.5*(z0+c0);
350 triple m1=0.5*(c0+c1);
351 triple m2=0.5*(c1+z1);
352 triple m3=0.5*(m0+m1);
353 triple m4=0.5*(m1+m2);
354 triple m5=0.5*(m3+m4);
355
356 // Check both Bezier subpaths.
357 b=bound(z0,m0,m3,m5,m,f,b,fuzz,depth);
358 return bound(m5,m4,m2,z1,m,f,b,fuzz,depth);
359}

399// integration.
400double arcLength(const triple& z0, const triple& c0, const triple& c1,
401 const triple& z1)
402{

404 derivative(a,b,c,z0,c0,c1,z1);

411double path3::cubiclength(Int i, double goal) const
412{
413 const triple& z0=point(i);
414 const triple& z1=point(i+1);
415 double L;
416 if(straight(i)) {
417 L=(z1-z0).length();
418 return (goal < 0 || goal >= L) ? L : -goal/L;
419 }
420
421 double integral=arcLength(z0,postcontrol(i),precontrol(i+1),z1);
422
423 L=3.0*integral;
424 if(goal < 0 || goal >= L) return L;
425
426 double t=goal/L;
427 goal *= third;
428 static double dxmin=sqrt(DBL_EPSILON);
429 if(!unsimpson(goal,ds,0.0,t,100.0*DBL_EPSILON,integral,1.0,dxmin))
430 reportError("nesting capacity exceeded in computing arctime");
431 return -t;
432}

447double path3::arctime(double goal) const
448{
449 if (cycles) {
450 if (goal == 0 || cached_length == 0) return 0;
451 if (goal < 0) {
452 const path3 &rp = this->reverse();
453 double result = -rp.arctime(-goal);
454 return result;
455 }
456 if (cached_length > 0 && goal >= cached_length) {
457 Int loops = (Int)(goal / cached_length);
458 goal -= loops*cached_length;
459 return loops*n+arctime(goal);
460 }
461 } else {
462 if (goal <= 0)
463 return 0;
464 if (cached_length > 0 && goal >= cached_length)
465 return n-1;
466 }
467
468 double l,L=0;
469 for (Int i = 0; i < n-1; i++) {
470 l = cubiclength(i,goal);
471 if (l < 0)
472 return (-l+i);
473 else {
474 L += l;
475 goal -= l;
476 if (goal <= 0)
477 return i+1;
478 }
479 }
480 if (cycles) {
481 l = cubiclength(n-1,goal);
482 if (l < 0)
483 return -l+n-1;
484 if (cached_length > 0 && cached_length != L+l) {
485 reportError("arclength != length.\n"
486 "path3::arclength(double) must have broken semantics.\n"
487 "Please report this error.");
488 }
489 cached_length = L += l;
490 goal -= l;
491 return arctime(goal)+n;
492 }
493 else {
494 cached_length = L;
495 return length();
496 }
497}

503// Return all intersection times of path3 g with the triple v.
504void intersections(std::vector<double>& T, const path3& g, const triple& v,
505 double fuzz)
506{
507 double fuzz2=fuzz*fuzz;
508 Int n=g.length();
509 bool cycles=g.cyclic();
510 for(Int i=0; i < n; ++i) {
511 // Check all directions to circumvent degeneracy.
512 std::vector<double> r;
513 roots(r,g.point(i).getx(),g.postcontrol(i).getx(),
514 g.precontrol(i+1).getx(),g.point(i+1).getx(),v.getx());
515 roots(r,g.point(i).gety(),g.postcontrol(i).gety(),
516 g.precontrol(i+1).gety(),g.point(i+1).gety(),v.gety());
517 roots(r,g.point(i).getz(),g.postcontrol(i).getz(),
518 g.precontrol(i+1).getz(),g.point(i+1).getz(),v.getz());
519
520 size_t m=r.size();
521 for(size_t j=0 ; j < m; ++j) {
522 double t=r[j];
523 if(t >= -Fuzz2 && t <= 1.0+Fuzz2) {
524 double s=i+t;
525 if((g.point(s)-v).abs2() <= fuzz2) {
526 if(cycles && s >= n-Fuzz2) s=0;
527 T.push_back(s);
528 }
529 }
530 }
531 }
532}

591 std::vector<double>& T, path3& p, path3& q,

598 Int lp=p.length();

601 intersections(T1,S1,q,p.point(lp),fuzz);

604 }
605
606 Int lq=q.length();
607 if(lq == 0 && exact) {
608 std::vector<double> S1,T1;
609 intersections(S1,T1,p,q.point(lq),fuzz);
610 add(s,t,S,T,S1,T1,p,q,fuzz2,single);

612 }

614 triple maxp=p.max();
615 triple minp=p.min();
616 triple maxq=q.max();
617 triple minq=q.min();
618
619 if(maxp.getx()+fuzz >= minq.getx() &&
620 maxp.gety()+fuzz >= minq.gety() &&
621 maxp.getz()+fuzz >= minq.getz() &&
622 maxq.getx()+fuzz >= minp.getx() &&
623 maxq.gety()+fuzz >= minp.gety() &&
624 maxq.getz()+fuzz >= minp.getz()) {
625 // Overlapping bounding boxes
626
627 --depth;
628// fuzz *= 2;

630 if((maxp-minp).length()+(maxq-minq).length() <= fuzz || depth == 0) {
631 if(single) {
632 s=0.5;
633 t=0.5;
634 } else {
635 S.push_back(0.5);
636 T.push_back(0.5);
637 }
638 return true;
639 }
640
641 path3 p1,p2;
642 double pscale,poffset;

653 return S1.size() > 0;
654 }

656 } else {

659 p2=p.subpath(tp,lp);
660 poffset=tp;

662 }

665 double qscale,qoffset;

668 if(lq == 1) q.halve(q1,q2);
669 if(lq == 0 || q1 == q || q2 == q) {
670 intersections(S1,T1,p,q.point((Int) 0),fuzz);
671 add(s,t,S,T,S1,T1,p,q,fuzz2,single);
672 return S1.size() > 0;
673 }
674 qscale=qoffset=0.5;
675 } else {
676 Int tq=lq/2;
677 q1=q.subpath(0,tq);
678 q2=q.subpath(tq,lq);
679 qoffset=tq;
680 qscale=1.0;
681 }

683 bool Short=lp == 1 && lq == 1;
684
685 static size_t maxcount=9;
686 size_t count=0;
687
688 if(intersections(s,t,S1,T1,p1,q1,fuzz,single,exact,depth)) {
689 add(s,t,S,T,S1,T1,pscale,qscale,0.0,0.0,p,q,fuzz2,single);
690 if(single || depth <= mindepth)
691 return true;
692 count += S1.size();
693 if(Short && count > maxcount) return true;
