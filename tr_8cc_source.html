109static void Setup(TRcontext *tr)
110{
111 if (!tr)
112 return;
113
114 tr->Columns = (tr->ImageWidth + tr->TileWidthNB - 1) / tr->TileWidthNB;
115 tr->Rows = (tr->ImageHeight + tr->TileHeightNB - 1) / tr->TileHeightNB;
116 tr->CurrentTile = 0;
117
118 assert(tr->Columns >= 0);
119 assert(tr->Rows >= 0);
120}

147{
148 if (!tr)

152 assert(width >= 1);
153 assert(height >= 1);

155 assert(height >= 2*border+1);
156
157 tr->TileBorder = border;
158 tr->TileWidth = width;
159 tr->TileHeight = height;
160 tr->TileWidthNB = width - 2 * border;
161 tr->TileHeightNB = height - 2 * border;

289void trPerspective(TRcontext *tr,
290 GLdouble fovy, GLdouble aspect,
291 GLdouble zNear, GLdouble zFar )
292{
293 GLdouble xmin, xmax, ymin, ymax;
294 static const double halfradians=acos(-1.0)/360.0;
295 ymax = zNear * tan(fovy * halfradians);
296 ymin = -ymax;
297 xmin = ymin * aspect;
298 xmax = ymax * aspect;
299 trFrustum(tr, xmin, xmax, ymin, ymax, zNear, zFar);
300}

303void trBeginTile(TRcontext *tr)
304{
305 GLint tileWidth, tileHeight, border;
306 GLdouble left, right, bottom, top;
307
308 if (!tr)
309 return;
310
311 if (tr->CurrentTile <= 0) {
312 Setup(tr);
313 /* Save user's viewport, will be restored after last tile rendered */
314 glGetIntegerv(GL_VIEWPORT, tr->ViewportSave);
315 }
316
317 /* which tile (by row and column) we're about to render */
318 if (tr->RowOrder==TR_BOTTOM_TO_TOP) {
319 tr->CurrentRow = tr->CurrentTile / tr->Columns;
320 tr->CurrentColumn = tr->CurrentTile % tr->Columns;
321 }
322 else if (tr->RowOrder==TR_TOP_TO_BOTTOM) {
323 tr->CurrentRow = tr->Rows - (tr->CurrentTile / tr->Columns) - 1;
324 tr->CurrentColumn = tr->CurrentTile % tr->Columns;
325 }
326 else {
327 /* This should never happen */
328 abort();
329 }
330 assert(tr->CurrentRow < tr->Rows);
331 assert(tr->CurrentColumn < tr->Columns);
332
333 border = tr->TileBorder;
334
335 /* Compute actual size of this tile with border */
336 if (tr->CurrentRow < tr->Rows-1)
337 tileHeight = tr->TileHeight;
338 else
339 tileHeight = tr->ImageHeight - (tr->Rows-1) * (tr->TileHeightNB) + 2 * border;
340
341 if (tr->CurrentColumn < tr->Columns-1)
342 tileWidth = tr->TileWidth;
343 else
344 tileWidth = tr->ImageWidth - (tr->Columns-1) * (tr->TileWidthNB) + 2 * border;
345
346 /* Save tile size, with border */
347 tr->CurrentTileWidth = tileWidth;
348 tr->CurrentTileHeight = tileHeight;
349
350 glViewport(0, 0, tileWidth, tileHeight); /* tile size including border */
351
352 /* compute projection parameters */
353 left = tr->Left + (tr->Right - tr->Left)
354 * (tr->CurrentColumn * tr->TileWidthNB - border) / tr->ImageWidth;
355 right = left + (tr->Right - tr->Left) * tileWidth / tr->ImageWidth;
356 bottom = tr->Bottom + (tr->Top - tr->Bottom)
357 * (tr->CurrentRow * tr->TileHeightNB - border) / tr->ImageHeight;
358 top = bottom + (tr->Top - tr->Bottom) * tileHeight / tr->ImageHeight;
359
360 if (tr->Perspective)
361 frustum(left, right, bottom, top, tr->Near, tr->Far);
362 else
363 ortho(left, right, bottom, top, tr->Near, tr->Far);
364}

368int trEndTile(TRcontext *tr)
369{
370 GLint prevRowLength, prevSkipRows, prevSkipPixels;
371
372 if (!tr)
373 return 0;
374
375 assert(tr->CurrentTile>=0);
376
377 /* be sure OpenGL rendering is finished */
378 glFlush();
379
380 /* save current glPixelStore values */
381 glGetIntegerv(GL_PACK_ROW_LENGTH, &prevRowLength);
382 glGetIntegerv(GL_PACK_SKIP_ROWS, &prevSkipRows);
383 glGetIntegerv(GL_PACK_SKIP_PIXELS, &prevSkipPixels);
384 /*glGetIntegerv(GL_PACK_ALIGNMENT, &prevAlignment);*/
385
386 if (tr->TileBuffer) {
387 GLint srcX = tr->TileBorder;
388 GLint srcY = tr->TileBorder;
389 GLint srcWidth = tr->TileWidthNB;
390 GLint srcHeight = tr->TileHeightNB;
391 glReadPixels(srcX, srcY, srcWidth, srcHeight,
392 tr->TileFormat, tr->TileType, tr->TileBuffer);
393 }
394
395 if (tr->ImageBuffer) {
396 GLint srcX = tr->TileBorder;
397 GLint srcY = tr->TileBorder;
398 GLint srcWidth = tr->CurrentTileWidth - 2 * tr->TileBorder;
399 GLint srcHeight = tr->CurrentTileHeight - 2 * tr->TileBorder;
400 GLint destX = tr->TileWidthNB * tr->CurrentColumn;
401 GLint destY = tr->TileHeightNB * tr->CurrentRow;
402
403 /* setup pixel store for glReadPixels */
404 glPixelStorei(GL_PACK_ROW_LENGTH, tr->ImageWidth);
405 glPixelStorei(GL_PACK_SKIP_ROWS, destY);
406 glPixelStorei(GL_PACK_SKIP_PIXELS, destX);
407 /*glPixelStorei(GL_PACK_ALIGNMENT, 1);*/
408
409 /* read the tile into the final image */
410 glReadPixels(srcX, srcY, srcWidth, srcHeight,
411 tr->ImageFormat, tr->ImageType, tr->ImageBuffer);
412 }
413
414 /* restore previous glPixelStore values */
415 glPixelStorei(GL_PACK_ROW_LENGTH, prevRowLength);
416 glPixelStorei(GL_PACK_SKIP_ROWS, prevSkipRows);
417 glPixelStorei(GL_PACK_SKIP_PIXELS, prevSkipPixels);
418 /*glPixelStorei(GL_PACK_ALIGNMENT, prevAlignment);*/
419
420 /* increment tile counter, return 1 if more tiles left to render */
421 tr->CurrentTile++;
422 if (tr->CurrentTile >= tr->Rows * tr->Columns) {
423 /* restore user's viewport */
424 glViewport(tr->ViewportSave[0], tr->ViewportSave[1],
425 tr->ViewportSave[2], tr->ViewportSave[3]);
426 tr->CurrentTile = -1; /* all done */
427 return 0;
428 }
429 else
430 return 1;
431}
