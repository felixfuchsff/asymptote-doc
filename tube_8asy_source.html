39// A 3D version of roundedpath(path, real).
40path3 roundedpath(path3 A, real r)
41{
42 // Author of this routine: Jens Schwaiger
43 guide3 rounded;
44 triple before, after, indir, outdir;
45 int len=length(A);
46 bool cyclic=cyclic(A);
47 if(len < 2) {return A;};
48 if(cyclic) {rounded=point(point(A,0)--point(A,1),r);}
49 else {rounded=point(A,0);}
50 for(int i=1; i < len; i=i+1) {
51 before=point(point(A,i)--point(A,i-1),r);
52 after=point(point(A,i)--point(A,i+1),r);
53 indir=dir(point(A,i-1)--point(A,i),1);
54 outdir=dir(point(A,i)--point(A,i+1),1);
55 rounded=rounded--before{indir}..{outdir}after;
56 }
57 if(cyclic) {
58 before=point(point(A,0)--point(A,len-1),r);
59 indir=dir(point(A,len-1)--point(A,0),1);
60 outdir=dir(point(A,0)--point(A,1),1);
61 rounded=rounded--before{indir}..{outdir}cycle;
62 } else rounded=rounded--point(A,len);
63
64 return rounded;
65}

67real[] sample(path3 g, real r, real relstep=0)
68{
69 real[] t;
70 int n=length(g);
71 if(relstep <= 0) {
72 for(int i=0; i < n; ++i)
73 render(subpath(g,i,i+1),r,new void(path3, real s) {t.push(i+s);});
74 t.push(n);
75 } else {
76 int nb=ceil(1/relstep);
77 relstep=n/nb;
78 for(int i=0; i <= nb; ++i)
79 t.push(i*relstep);
80 }
81 return t;
82}

84real degrees(rmf a, rmf b)
85{
86 real d=degrees(acos1(dot(a.r,b.r)));
87 real dt=dot(cross(a.r,b.r),a.t);
