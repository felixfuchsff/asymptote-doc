37types::ty *newRecordExp::transFromTyEntry(position pos, coenv &e,
38 trans::tyEntry *ent)
39{
40 types::ty *t = ent->t;
41 if (t->kind == ty_error)
42 return t;
43 else if (t->kind != ty_record) {
44 em.error(pos);
45 em << "type '" << *t << "' is not a structure";
46 return primError();
47 }
48
49 // Put the enclosing frame on the stack.
50 if (!e.c.encodeParent(pos, ent)) {
51 em.error(pos);
52 em << "allocation of struct '" << *t << "' is not in a valid scope";
53 return primError();
54 }
55
56 record *r = dynamic_cast<record *>(t);
57 assert(r);
58
59 // Encode the allocation.
60 e.c.encode(inst::makefunc,r->getInit());
61 e.c.encode(inst::popcall);
62
63 return t;
64}

94types::ty *newArrayExp::trans(coenv &e)
95{
96 types::ty *c = celltype->trans(e);
97 if (c->kind == ty_void) {
98 em.error(getPos());
99 em << "cannot declare array of type void";
100 return primError();
101 }
102
103 if (dims)
104 c = dims->truetype(c);
105
106 if (ai) {
107 ai->transToType(e, c);
108 return c;
109 } else if (dimexps || dims) {
110 if (dimexps) {
111 for (size_t i = 0; i < dimexps->size(); ++i) {
112 (*dimexps)[i]->transToType(e, types::primInt());
113 c = new types::array(c);
114 }
115 }
116 if (dims) {
117 for (size_t i = 0; i < dims->size(); ++i) {
118 e.c.encode(inst::intpush,0);
119 }
120 }
121 e.c.encode(inst::intpush,
122 (Int) ((dimexps ? dimexps->size():0)
123 + (dims ? dims->size():0)));
124 e.c.encode(inst::builtin, run::newDeepArray);
125
126 return c;
127 } else {
128 em.compiler(getPos());
129 em << "new array expression must have either dims or dimexps";
130 return primError();
131 }
132}

134types::ty *newArrayExp::getType(coenv &e)
135{
136 types::ty *c = celltype->trans(e);
137 if (c->kind == ty_void) {
138 return primError();

140 if (dims)
141 c = dims->truetype(c);
142

144 Int depth = (Int)dimexps->size();
145 while (depth > 0) {

147 depth--;
148 }
149 }
