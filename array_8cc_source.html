24inline size_t sliceIndex(Int in, size_t len) {
25 if (in < 0)
26 // The Python behaviour here would simply be
27 // in += len;
28 // but this is inconsistent with Asymptote issuing an error for A[-1] if A
29 // is a non-cyclic array, so we also issue an error here.
30 vm::error("invalid negative index in slice of non-cyclic array");
31 if (in < 0)
32 return 0;
33 size_t index = (size_t)in;
34 return index < len ? index : len;

37array *array::slice(Int left, Int right)
38{
39 checkBackSlice(left, right);
40
41 if (left == right)
42 return new array();
43
44 size_t length=size();
45 if (length == 0)
46 return new array();
47
48 if (cycle) {
49 size_t resultLength = (size_t)(right - left);
50 array *result = new array(resultLength);
51
52 size_t i = (size_t)imod(left, length), ri = 0;
53 while (ri < resultLength) {
54 (*result)[ri] = (*this)[i];
55
56 ++ri;
57 ++i;
58 if (i >= length)
59 i -= length;
60 }
61
62 return result;
63 }
64 else { // Non-cyclic
65 size_t l = sliceIndex(left, length);
66 size_t r = sliceIndex(right, length);
67
68 size_t resultLength = r - l;
69 array *result = new array(resultLength);
70
71 std::copy(this->begin()+l, this->begin()+r, result->begin());
72
73 return result;
74 }

77void array::setNonBridgingSlice(size_t l, size_t r, mem::vector<item> *a)
78{
79 assert(0 <= l);
80 assert(l <= r);
81
82 size_t const sliceLength=r-l;
83
84 size_t asize=a->size();
85 if (asize == sliceLength) {
86 // In place
87 std::copy(a->begin(), a->end(), this->begin()+l);
88 }
89 else if (asize < sliceLength) {
90 // Shrinking
91 std::copy(a->begin(), a->end(), this->begin()+l);
92 this->erase(this->begin()+(l+a->size()), this->begin()+r);
93 }
94 else {
95 // Expanding
96 // NOTE: As a speed optimization, we could check capacity() to see if the
97 // array can fit the new entries, and build the new array from scratch
98 // (using swap()) if a new allocation is necessary.
99 std::copy(a->begin(), a->begin()+sliceLength, this->begin()+l);
100 this->insert(this->begin()+r, a->begin()+sliceLength, a->end());
101 }
102}

115void array::setSlice(Int left, Int right, array *a)
116{
117 checkBackSlice(left, right);
118
119 // If we are slicing an array into itself, slice in a copy instead, to ensure
120 // the proper result.
121 mem::vector<item> *v = (a == this) ? new mem::vector<item>(*a) : a;
122
123 size_t length=size();
124 if (cycle) {
125 if (right == left) {
126 // Notice that assigning to the slice A[A.length:A.length] is the same as
127 // assigning to the slice A[0:0] for a cyclic array.
128 size_t l = (size_t)imod(left, length);
129 setNonBridgingSlice(l, l, v);
130 }
131 else {
132 if (left + (Int) length < right)
133 vm::error("assigning to cyclic slice with repeated entries");
134
135 size_t l = (size_t)imod(left, length);
136
137 // Set r to length instead of zero, so that slices that go to the end of
138 // the array are properly treated as non-bridging.
139 size_t r = (size_t)imod(right, length);
140 if (r == 0)
141 r = length;
142
143 if (l < r)
144 setNonBridgingSlice(l, r, v);
145 else {
146 if (r + length - l == v->size())
147 setBridgingSlice(l, r, v);
148 else
149 vm::error("assignment to cyclic slice is not well defined");
150 }
151 }
152 }
153 else {
154 size_t l=sliceIndex(left, length);
155 size_t r=sliceIndex(right, length);
156
157 setNonBridgingSlice(l, r, v);
158 }
159}
