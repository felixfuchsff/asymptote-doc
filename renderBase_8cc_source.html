84double AsyRender::getRenderResolution(triple Min) const
85{
86 double prerender = settings::getSetting<double>("prerender");
87
88 if (prerender <= 0.0)
89 return 0.0;
90
91 prerender = 1.0 / prerender;
92 double perspective = orthographic || Zmax == 0.0 ? 0.0 : 1.0 / Zmax;
93 double s = perspective ? Min.getz() * perspective : 1.0;
94 triple b(Xmin, Ymin, Zmin);
95 triple B(Xmax, Ymax, Zmax);
96 pair size3(s * (B.getx() - b.getx()), s * (B.gety() - b.gety()));
97 pair size2(Width, Height);
98 return prerender * size3.length() / size2.length();
99}

102void AsyRender::setDimensions(int Width, int Height, double X, double Y)
103{
104 double aspect = ((double) Width) / Height;
105 double xshift = (X / (double) Width + Shift.getx() * Xfactor) * Zoom;
106 double yshift = (Y / (double) Height + Shift.gety() * Yfactor) * Zoom;
107 double zoominv = 1.0 / Zoom;
108 if (orthographic) {
109 double xsize = Xmax - Xmin;
110 double ysize = Ymax - Ymin;
111 if (xsize < ysize * aspect) {
112 double r = 0.5 * ysize * aspect * zoominv;
113 double X0 = 2.0 * r * xshift;
114 double Y0 = ysize * zoominv * yshift;
115 xmin = -r - X0;
116 xmax = r - X0;
117 ymin = Ymin * zoominv - Y0;
118 ymax = Ymax * zoominv - Y0;
119 } else {
120 double r = 0.5 * xsize * zoominv / aspect;
121 double X0 = xsize * zoominv * xshift;
122 double Y0 = 2.0 * r * yshift;
123 xmin = Xmin * zoominv - X0;
124 xmax = Xmax * zoominv - X0;
125 ymin = -r - Y0;
126 ymax = r - Y0;
127 }
128 } else {
129 double r = H * zoominv;
130 double rAspect = r * aspect;
131 double X0 = 2.0 * rAspect * xshift;
132 double Y0 = 2.0 * r * yshift;
133 xmin = -rAspect - X0;
134 xmax = rAspect - X0;
135 ymin = -r - Y0;
136 ymax = r - Y0;
137 }
138}

247projection AsyRender::camera(bool user)
248{
249 camp::Triple vCamera, vUp, vTarget;
250
251 double cz = 0.5 * (Zmin + Zmax);
252
253 double *Rotate = value_ptr(rotateMat);
254
255 if(user) {
256 double shift[]={0.0,0.0,0.0,0.0};
257 for(int i=0; i < 3; ++i) {
258 double sumCamera=0.0, sumTarget=0.0, sumUp=0.0;
259 int i4=4*i;
260 shift[3]=T[i4+2]*cz;
261 for(int j=0; j < 4; ++j) {
262 int j4=4*j;
263 double R0=Rotate[j4];
264 double R1=Rotate[j4+1];
265 double R2=Rotate[j4+2];
266 double R3=Rotate[j4+3];
267 double T4ij=T[i4+j]+shift[j]; // T -> T*shift(0,0,cz);
268 sumCamera += T4ij*(R3-cx*R0-cy*R1-cz*R2);
269 sumUp += Tup[i4+j]*R1;
270 sumTarget += T4ij*(R3-cx*R0-cy*R1);
271 }
272 vCamera[i]=sumCamera;
273 vUp[i]=sumUp;
274 vTarget[i]=sumTarget;
275 }
276 } else {
277 for(int i=0; i < 3; ++i) {
278 int i4=4*i;
279 double R0=Rotate[i4];
280 double R1=Rotate[i4+1];
281 double R2=Rotate[i4+2];
282 double R3=Rotate[i4+3];
283 vCamera[i]=R3-cx*R0-cy*R1-cz*R2;
284 vUp[i]=R1;
285 vTarget[i]=R3-cx*R0-cy*R1;
286 }
287 }
288
289 return projection(orthographic, vCamera, vUp, vTarget, Zoom,
290 2.0*atan(tan(0.5*Angle)/Zoom)/radians,
291 pair(X/Width+Shift.getx(),
292 Y/Height+Shift.gety()));
293}

349{
350 double zoomFactor = settings::getSetting<double>("zoomfactor");
351 if (zoomFactor > 0.0) {
352 double zoomStep = settings::getSetting<double>("zoomstep");

791void AsyRender::onCursorPos(double xpos, double ypos)
792{
793 if (lastAction == "rotate") {

910void AsyRender::initDisplay(int contentWidth, int contentHeight)
911{
912 // Compute expand/fullWidth/fullHeight (unscaled content dimensions).
913 double expand = settings::getSetting<double>("render");
914 if (expand < 0)
915 expand *= (Format.empty() || Format == "eps" || Format == "pdf") ? -2.0 : -1.0;
916 if (antialias) expand *= 2.0;
917
918 fullWidth = (int) std::ceil(expand * contentWidth);
919 fullHeight = (int) std::ceil(expand * contentHeight);
920
921 oWidth = contentWidth;
922 oHeight = contentHeight;
923
924 GLFWmonitor* monitor = NULL;
925 glfwInit();
926
927 devicePixelRatio = settings::getSetting<double>("devicepixelratio");
928 monitor = glfwGetPrimaryMonitor();
929 if (monitor) {
930 int mx, my;
931 glfwGetMonitorWorkarea(monitor, &mx, &my, &screenWidth, &screenHeight);
932 if (devicePixelRatio <= 0.0) {
933 float sx = 1.0f, sy = 1.0f;
934 glfwGetMonitorContentScale(monitor, &sx, &sy);
935 devicePixelRatio = std::max(sx, sy);
936 }
937 } else {
938 screenWidth = fullWidth;
939 screenHeight = fullHeight;
940 }
941
942 oldWidth = (int) std::ceil(contentWidth * devicePixelRatio);
943 oldHeight = (int) std::ceil(contentHeight * devicePixelRatio);
944
945 int w = std::min(oldWidth, screenWidth);
946 int h = std::min(oldHeight, screenHeight);
947
948 fitAspect(w, h);
949
950 Width = w;
951 Height = h;
952
953 home();
954
955 ArcballFactor = 1 + 8.0 * hypot(Margin.getx(), Margin.gety()) / hypot(Width, Height);
956}
