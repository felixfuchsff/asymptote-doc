5 * Using
6 *
7 * void operator init(<args>)
8 *
9 * as a field in the definition structure named Foo implicitly creates a
10 * function that could be explicitly defined with code similar to
11 *
12 * static Foo Foo(<args>) {
13 * Foo a=new Foo;

16 * }

18 * This function is usable within further code in the structure definition and,

53// Given the coenv of the body of the constructor, encode the necessary
54// instructions to make a new initialized object.
55void transConstructorBody(position pos, coenv &e, record *r, varEntry *init)
56{
57 assert(r);
58 assert(init);
59
60 // Create a varEntry to hold the new object. Foo a;
61 varEntry *v=makeVarEntry(pos, e, 0 /* not a field */, r);
62
63 // Initialize the object. a=new Foo;
64 newRecordExp::transFromTyEntry(pos, e, new tyEntry(r, 0, 0, nullPos));
65 v->encode(WRITE, pos, e.c);
66 e.c.encodePop();
67
68 // Push the args onto the stack.
69 size_t numArgs=init->getSignature()->getNumFormals();
70 for (size_t i=0; i<numArgs; ++i) {
71 access *a=e.c.accessFormal((Int)i);
72 a->encode(READ, pos, e.c);
73 }
74
75 // Push the object on the stack.
76 v->encode(READ, pos, e.c);
77
78 // Call the 'operator init' field of the object.
79 init->encode(action::CALL, pos, e.c, v->getLevel());
80
81 // Push the object again.
82 v->encode(READ, pos, e.c);
83
84 // Return the initialized object.
85 e.c.encode(inst::ret);
86}

88varEntry *constructorFromInitializer(position pos, coenv &e, record *r,
89 varEntry *init)
90{
91 assert(r);
92
93 types::function *ft=new types::function(r, init->getSignature());
94
95 ostringstream out;
96 ft->printVar(out, symbol::trans("<constructor>"));
97
98 // Create a new function environment.
99 coder fc = e.c.newFunction(pos, out.str(), ft);
100 coenv fe(fc,e.e);
101
102 // Translate the function.
103 fe.e.beginScope();
104
105 transConstructorBody(pos, fe, r, init);
106
107 fe.e.endScope();
108
109 // Put an instance of the new function on the stack.
110 vm::lambda *l = fe.c.close();
111 e.c.encode(inst::pushclosure);
112 e.c.encode(inst::makefunc, l);
113
114 // Save it into a varEntry.
115 varEntry *v=makeVarEntry(pos, e, r, ft);
116 v->encode(WRITE, pos, e.c);
117 e.c.encodePop();
118
119 return v;
120}

122void addConstructorFromInitializer(position pos, coenv &e, record *r,
123 varEntry *init)
124{
125 assert(r);
126
127 // Constructors are declared statically.
128 e.c.pushModifier(EXPLICIT_STATIC);
129
130 varEntry *v=constructorFromInitializer(pos, e, r, init);
131
132 // Add the constructor function under the same name as the record.
133 addVar(e, r, v, r->getName());
134
135 // Add to the "post definition environment" of the record, so it will also be
136 // added to the enclosing scope when the record definition ends.
137 r->postdefenv.addVar(r->getName(), v);
138
139 e.c.popModifier();
140}
