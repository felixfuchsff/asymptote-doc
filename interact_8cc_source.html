155string simpleline(string prompt) {
156 // Rebind tab key, as the setting tabcompletion may be changed at runtime.
157 pre_readline();
158
159 Signal(SIGINT,SIG_IGN);
160 // Get a line from the user.
161 char *line=Readline(prompt.c_str());
162 Signal(SIGINT,interruptHandler);
163
164 // Reset scroll count.
165 interact::lines=0;
166 interact::query=tty;
167
168 // Ignore keyboard interrupts while taking input.
169 errorstream::interrupt=false;
170
171 if(line) {
172 string s=line;
173 free(line);
174 return s;
175 } else {
176#if defined(HAVE_READLINE) && defined(HAVE_LIBCURSES)
177 if(!tty || getSetting<bool>("exitonEOF"))
178#endif

180 cout << endl;
