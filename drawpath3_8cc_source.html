57void drawPath3::render(double size2, const triple& b, const triple& B,
58 double perspective, bool remesh)
59{
60#ifdef HAVE_RENDERER
61 if(invisible) return;
62
63 setcolors(diffuse,emissive,specular,shininess,metallic,fresnel0,true);
64
65 bool offscreen;
66 if(billboard) {
67 drawElement::centerIndex=centerIndex;
68 offscreen=bbox2(Min,Max,center).offscreen();
69 } else
70 offscreen=bbox2(Min,Max).offscreen();
71
72 if(offscreen) { // Fully offscreen
73 R.Onscreen=false;
74 R.data.clear();
75 R.notRendered();
76 return;
77 }
78
79 triple controls[]={g.point((Int) 0),g.postcontrol((Int) 0),
80 g.precontrol((Int) 1),g.point((Int) 1)};
81 triple *Controls;
82 triple Controls0[4];
83 if(billboard) {
84 Controls=Controls0;
85 for(size_t i=0; i < 4; i++)
86 Controls[i]=billboardTransform(center,controls[i]);
87 } else {
88 Controls=controls;
89 if(!remesh && R.Onscreen) { // Fully onscreen; no need to re-render
90 R.append();
91 return;
92 }
93 }
94
95 double s=perspective ? Min.getz()*perspective : 1.0; // Move to glrender
96
97 const pair size3(s*(B.getx()-b.getx()),s*(B.gety()-b.gety()));
98
99 R.queue(controls,straight,size3.length()/size2);
100#endif
101}

148void drawNurbsPath3::ratio(const double* t, pair &b, double (*m)(double, double),
149 double, bool &first)
150{
151 triple* Controls;
152 if(t == NULL) Controls=controls;
153 else {
154 Controls=new triple[n];
155 for(size_t i=0; i < n; i++)
156 Controls[i]=t*controls[i];
157 }
158
159 if(first) {
160 first=false;
161 triple v=Controls[0];
162 b=pair(xratio(v),yratio(v));
163 }
164
165 double x=b.getx();
166 double y=b.gety();
167 for(size_t i=0; i < n; ++i) {
168 triple v=Controls[i];
169 x=m(x,xratio(v));
170 y=m(y,yratio(v));
171 }
172 b=pair(x,y);
173
174 if(t != NULL)
175 delete[] Controls;
176}
