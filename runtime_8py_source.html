29# Convert parameters into stack pop code in correct order
30def c_params(params_list):
31 global STACK
32 result = []
33 # Reverse order for correct push/pop
34 for p in reversed(params_list):
35 match = re.search(
36 r"^\s*" # optional whitespace
37 r"(explicit)*" # optional explicit
38 r"\s*"
39 r"(\w*(?:\s*\*)?)" # type
40 r"\s*"
41 r"(\w*)" # parameter name
42 r"(=*)([\w.+\-]*)", # optional default value
43 p,
44 )
45 if not match:
46 continue
47 _, type_str, name, eqsign, val = match.groups()
48 template = "" if type_str == "item" else f"<{type_str}>"
49 def_val = f",{val}" if eqsign else ""
50 r = f" {type_str} {name}=vm::pop{template}({STACK}{def_val});\n"
51 result.append(r)
52 return result

63def symbolize(name, opsymbols):
64 # If name is alphanumeric/underscore, just SYM(...) it
65 if re.search(r"^[A-Za-z0-9_]+$", name):
66 return f"SYM({name})"
67 # If the name is recognized in opsymbols, substitute
68 if name in opsymbols:
69 return opsymbols[name]
70 # Check if it is "operator smth" form
71 opm = re.search(r"operator\s+(\w+)", name)
72 if opm and opm.group(1) in opsymbols:
73 return opsymbols[opm.group(1)]
74 # Otherwise, generate a trans(...) symbol
75 return f'symbol::trans("{name}")'

78def asy_params(params_string, filename, line, type_map, opsymbols):
79 # Split parameters by comma
80 params_list = re.split(r",\s*", params_string) if params_string else []
81 result = []
82 for p in params_list:
83 # (explicit)*\s*(\w*(?:\s*\*)?) => optional 'explicit', type, pointer
84 # (\w*) => parameter name, possibly something like "x"
85 # (=*) => optional equals sign for default parameter
86 match = re.search(
87 r"^\s*(explicit)*\s*" # optional explicit
88 r"(\w*(?:\s*\*)?)\s*" # type or pointer type
89 r"(\w*)" # name
90 r"(=*)", # if '=' is present, there is a default value
91 p,
92 )
93 if not match:
94 continue
95 explicit, t, n, d = match.groups()
96 t = clean_type(t)
97 if t not in type_map:
98 Er.assoc_error(filename, line, t)
99 has_def = "true" if d else "false"
100 is_ex = "true" if explicit else "false"
101 result.append(
102 "formal("
103 f"{type_map[t]}, {symbolize(n.lower(), opsymbols)}, {has_def}, {is_ex})"
104 )
105 return result

150def parse_types(
151 data: str, input_filename: str, input_start_line: int
152) -> Tuple[Dict, int]:
153 """Return (new_entries, lines_consumed)."""
154 entries: Dict = {}
155 lines = data.split("\n")
156 for line_no, line in enumerate(lines, start=input_start_line):
157 # Remove // comments in each line
158 trimmed = re.sub(r"//.*", "", line)
159
160 # Skip empty lines
161 if re.fullmatch(r"\s*", trimmed):
162 continue
163 # Regex for matching something like "item => ITEMVAL"
164 # (\w*(?:\s*\*)?) matches a type or pointer (e.g. "int", "item*", etc.)
165 # \s*=>\s* captures the '=>' then
166 # (.*) grabs the rest of the line as the code
167 match = re.search(
168 r"(\w*(?:\s*\*)?)" # Type or pointer type
169 r"\s*=>\s*" # =>
170 r"(.*)", # Everything else
171 trimmed,
172 )
173 if not match:
174 Er.report_error(input_filename, line_no, "bad type declaration")
175 continue
176 t, code = match.groups()
177 entries[clean_type(t)] = code
178 return entries, data.count("\n")

185class FunctionData:
186 # This regex attempts to parse a function definition in the form:
187 # optional comments, return type, asy function name,
188 # optional C++ function name, parameters, code
189 _pat = re.compile(
190 # pylint: disable=line-too-long
191 r"^((?:\s*//[^\n]*\n)*)" # $1: capture comment lines starting with //
192 r"\s*"
193 r"(\w*(?:\s*\*)?)" # $2: return type (e.g., 'int', 'item*', etc.)
194 r"\s*"
195 r"([^(:]*)" # $3: read to the first colon or open parenthesis (asy function name)
196 r"\:*"
197 r"([^(]*)" # $4: read to the first open parenthesis (optional c++ function name)
198 r"\s*"
199 r"\‍(([\w\s*,=.+\-]*)\‍)" # $5: parameters list inside parentheses
200 r"\s*"
201 r"\{(.*)\}", # $6: function body up to last closing brace in section
202 re.DOTALL, # allow . to match newlines (for $6)
203 )

205 def __init__(
206 self,
207 section: str,
208 prefix: str,
209 function_count: int,
210 header_lines: List,
211 ):
212 global STACK
213 md = self._pat.search(section)
214 if not md:
215 raise MatchNotFoundError
216 comments, return_type, name, cname, params, code = md.groups()
217 # Insert a fallback cName if needed
218 if cname:
219 header_lines.append(f"void {cname}(vm::stack *);\n")
220 else:
221 cname = f"gen_{prefix}{function_count}"
222 # Added newlines here would mess up the line count
223 assert cname.count("\n") == 0
224
225 # Clean up types
226 return_type = clean_type(return_type)
227 # If there's "Operator", remove it:
228 name = re.sub(r"Operator\s*", "", name)
229 # Replace 'return X;' with push
230 qualifier = "" if return_type == "item" else f"<{return_type}>"
231 code = re.sub(
232 r"\breturn\s+([^;]*);", # read until the next semicolon
233 rf"{{{STACK}->push{qualifier}(\1); return;}}",
234 code,
235 )
236
237 self.comments: str = comments
238 self.return_type: str = return_type
239 self.name: str = name
240 self.cname: str = cname
241 self.params_string: str = params
242 self.code: str = code

276 def write_cc(self, f_out: TextIO, d: RunData, in_line_counter: int) -> None:
277 global STACK
278 # Write out any preceding comments
279 f_out.write(self.comments)
280 in_line_counter += self.comments.count("\n")
281 f_out.write(f'#line {in_line_counter} "{d.srcTemplateDir}/{d.prefix}.in"\n')
282
283 # Split the parameter list by commas (ignoring whitespace after each comma)
284 param_list = re.split(r",\s*", self.params_string) if self.params_string else []
285 # Build param popping lines
286 param_code = "".join(c_params(param_list))
287
288 # Write out the function prototype as a comment
289 prototype = f"{self.return_type} {self.name}({self.params_string});"
290 in_line_counter += prototype.count("\n") + 1
291 if self.name:
292 prototype = clean_params(prototype)
293 assert prototype.count("\n") == 0
294 f_out.write(f"// {prototype}\n")
295
296 # Actual function definition in prefix.cc
297 param_name = STACK if self.return_type != "void" or param_list else ""
298 f_out.write(f"void {self.cname}(stack *{param_name})\n")
299 f_out.write("{\n")
300 assert not param_code or param_code[-1] == "\n"
301 f_out.write(param_code)
302 f_out.write(f'#line {in_line_counter} "{d.srcTemplateDir}/{d.prefix}.in"')
303 assert self.code[0] == "\n"
304 f_out.write(self.code)
305 f_out.write("}\n\n")
