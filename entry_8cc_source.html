143void core_venv::initTable(size_t capacity) {
144 // Assert that capacity is a power of two.
145 assert((capacity & (capacity-1)) == 0);
146
147 this->capacity = capacity;
148 size = 0;
149 mask = capacity - 1;
150 table = new (UseGC) cell[capacity];
151 memset(table, 0, sizeof(cell) * capacity);
152}

154void core_venv::clear() {
155 if (size != 0) {
156 memset(table, 0, sizeof(cell) * capacity);
157 size = 0;
158 }
159}

161void core_venv::resize() {
162 size_t oldCapacity = capacity;
163 size_t oldSize = size;
164 cell *oldTable = table;
165
166 initTable(capacity * 4);
167
168 for (size_t i = 0; i < oldCapacity; ++i) {
169 cell& b = oldTable[i];
170 if (!b.empty() && !b.isATomb()) {
171 varEntry *old = store(b.name, b.ent);
172
173 // We should not be shadowing anything when reconstructing the table.
174 DEBUG_CACHE_ASSERT(old == 0);
175 }
176 }
177
178 assert(size == oldSize);
179
180#if DEBUG_CACHE
181 confirm_size();
182
183 for (size_t i = 0; i < oldCapacity; ++i) {
184 cell& b = oldTable[i];
185 if (!b.empty() && !b.isATomb()) {
186 assert(lookup(b.name, b.ent->getType()) == b.ent);
187 }
188 }
189#endif
190
191 //cout << "resized from " << oldCapacity << " to " << capacity << endl;
192}

202void core_venv::confirm_size() {
203 size_t sum = 0;
204 for (size_t i = 0; i < capacity; ++i) {
205 cell& b = table[i];
206 if (!b.empty() && !b.isATomb())
207 ++sum;
208 }

212varEntry *core_venv::storeNew(cell& cell, symbol name, varEntry *ent) {
213 // Store the value into the cell.
214 cell.storeNew(name, ent);
215
216 // We now have a new entry. Update the size.
217 ++size;
218
219 // Check if the hash table has grown too big and needs to be expanded.
220 if (2*size > capacity)
221 resize();
222
223 // Nothing is shadowed.
224 return 0;
225}

451void venv::checkName(symbol name)
452{
453#if 0
454 // This size test is too slow, even for DEBUG_CACHE.
455 core.confirm_size();
456#endif
457
458 // Get the type, and make it overloaded if it is not (for uniformity).
459 overloaded o;
460 ty *t = getType(name);
461 if (!t)
462 t = &o;
463 if (!t->isOverloaded()) {
464 o.add(t);
465 t = &o;
466 }
467 assert(t->isOverloaded());
468
469 size_t maxFormals = names[name].maxFormals;
470
471 size_t size = 0;
472 for (ty_iterator i = t->begin(); i != t->end(); ++i) {
473 assert(numFormals(*i) <= maxFormals);
474 varEntry *v = lookByType(name, *i);
475 assert(v);
476 assert(equivalent(v->getType(), *i));
477 ++size;
478 }
479
480 size_t matches = 0;
481 core_venv::const_iterator end = core.end();
482 for (core_venv::const_iterator p = core.begin(); p != end; ++p) {
483 if (p->name == name) {
484 ++matches;
485
486 varEntry *v=p->ent;
487 assert(v);
488 assert(equivalent(t, v->getType()));
489 }
490 }
491 assert(matches == size);
492}

509void venv::namevalue::addType(ty *s) {
510 RIGHTKIND(t);
511
512#ifdef DEBUG_CACHE
513 assert(!s->isOverloaded());
514#endif
515
516 if (t == 0) {
517 maxFormals = numFormals(s);
518 t = s;
519 } else {
520 if (!t->isOverloaded())
521 t = new overloaded(t);
522
523#ifdef DEBUG_CACHE
524 assert(t->isOverloaded());
525 assert(!equivalent(t, s));
526#endif
527
528 ((overloaded *)t)->add(s);
529
530 size_t n = numFormals(s);
531 if (n > maxFormals)
532 maxFormals = n;
533 }
534
535 RIGHTKIND(t);
536}

609void venv::remove(const addition& a) {
610 CHECKNAME(a.name);
611
612 if (a.shadowed) {
613 varEntry *popEnt = core.store(a.name, a.shadowed);
614
615 DEBUG_CACHE_ASSERT(popEnt);
616
617 // Unshadow the previously shadowed varEntry.
618 names[a.name].replaceType(a.shadowed->getType(), popEnt->getType());
619 }
620 else {
621 // Remove the (name,sig) key completely.
622#if DEBUG_CACHE
623 varEntry *popEnt = core.lookup(a.name, a.t);
624 assert(popEnt);
625 names[a.name].popType(popEnt->getType());
626#else
627 names[a.name].popType();
628#endif
629
630 core.remove(a.name, a.t);
631 }
632
633 CHECKNAME(a.name);
634}

645void venv::endScope() {
646 if (scopesizes.empty()) {
647 // The corresponding beginScope happened when the venv was empty, so
648 // clear the hash tables to return to that state.
649 core.clear();
650 names.clear();
651
652 assert(empty_scopes > 0);
653 --empty_scopes;
654 } else {
655 size_t scopesize = scopesizes.top();
656 assert(additions.size() >= scopesize);
657 while (additions.size() > scopesize) {
658 remove(additions.top());
659 additions.pop();
660 }
661 scopesizes.pop();
662 }
663}

681void venv::enter(symbol name, varEntry *v)
682{
683 CHECKNAME(name);
684
685 // Store the new variable. If it shadows an older variable, that varEntry
686 // will be returned.
687 varEntry *shadowed = core.store(name, v);
688
689 ty *t = v->getType();
690
691 // Record the addition, so it can be undone during endScope.
692 if (!scopesizes.empty())
693 additions.push(addition(name, t, shadowed));
694
695 if (shadowed)
696 // The new value shadows an old value. They have the same signature, but
697 // possibly different return types. If necessary, update the type stored
698 // by name.
699 names[name].replaceType(t, shadowed->getType());
700 else
701 // Add to the names hash table.
702 names[name].addType(t);
703
704 CHECKNAME(name);
705}

708varEntry *venv::lookBySignature(symbol name, signature *sig) {
709 // Rest arguments are complicated and rare. Don't handle them here.
710 if (sig->hasRest())
711 return 0;
712
713 // Likewise with the special operators.
714 if (name.special())
715 return 0;
716
717 namevalue& nv = names[name];
718
719 // Avoid ambiguities with default parameters.
720 if (nv.maxFormals != sig->getNumFormals())
721 return 0;
722
723 // See if this exactly matches a function in the table.
724 varEntry *ve = core.lookupNonSpecial(name, sig);
725
726 if (!ve)
727 return 0;
728
729 // Keyword-only arguments may cause matching to fail.
730 if (ve->getSignature()->numKeywordOnly > 0)
731 return 0;
732
733 // At this point, any function with an equivalent signature will be equal
734 // to the result of the normal overloaded function resolution. We may
735 // safely return it.
736 return ve;
737}

761bool venv::add(
762 symbol src, symbol dest, venv& source, varEntry *qualifier, coder &c,
763 mem::vector<varEntry*> *addedVec
764) {
765 ty *t=source.getType(src);
766
767 if (!t)
768 return false;
769
770 if (t->isOverloaded()) {
771 bool added=false;
772 for (ty_iterator i = t->begin(); i != t->end(); ++i)
773 {
774 varEntry *v=source.lookByType(src, *i);
775 if (v->checkPerm(READ, c)) {
776 if (permission perm = c.getPermission(); perm != PUBLIC) {
777 // Add an additional restriction to v based on c.getPermission().
778 v = new varEntry(*v, perm, c.thisType());
779 }
780 varEntry *qve=qualifyVarEntry(qualifier, v);
781 enter(dest, qve);
782 if (addedVec != nullptr) {
783 addedVec->push_back(qve);
784 }
785 added=true;
786 }
787 }
788 return added;
789 }
790 else {
791 varEntry *v=source.lookByType(src, t);
792 if (!v->checkPerm(READ, c)) {
793 return false;
794 }
795 if (permission perm = c.getPermission(); perm != PUBLIC) {
796 // Add an additional restriction to v based on c.getPermission().
797 v = new varEntry(*v, perm, c.thisType());
798 }
799 varEntry *qve=qualifyVarEntry(qualifier, v);
800 enter(dest, qve);
801 if (addedVec != nullptr) {
802 addedVec->push_back(qve);
803 }
804 return true;
805 }
