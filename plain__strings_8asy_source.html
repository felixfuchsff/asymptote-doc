89string[] split(string s, string delimiter="")

93 prune=true;
94 delimiter=" ";
95 }
96
97 string[] S;
98 int last=0;
99 int i;
100 int N=length(delimiter);
101 int n=length(s);
102 while((i=find(s,delimiter,last)) >= 0) {
103 if(i > last || (i == last && !prune))
104 S.push(substr(s,last,i-last));
105 last=i+N;

107 if(n > last || (n == last && !prune))
108 S.push(substr(s,last,n-last));
