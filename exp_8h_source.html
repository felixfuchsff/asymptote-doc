81 // Translates the expression and returns the resultant type.
82 // For some expressions, this will be ambiguous and return an error.
83 // Trans may only return ty_error, if it (or one of its recursively
84 // called children in the syntax tree) reported an error to em.
85 virtual types::ty *trans(coenv &) = 0;
86
87 // getType() figures out the type of the expression without translating
88 // the code into the virtual machine language or reporting errors to em.
89 // This must follow a few rules to ensure proper translation:
90 // 1. If this returns a valid type, t, trans(e) must return t or
91 // report an error, and transToType(e, t) must run either reporting
92 // an error or reporting no error and yielding the same result as
93 // trans(e).
94 // 2. If this returns a superposition of types (ie. for overloaded
95 // functions), trans must not return a singular type, and every
96 // type in the superposition must run without error properly
97 // if fed to transAsType(e, t).
98 // 3. If this returns ty_error, then so must a call to trans(e) and any
99 // call to trans, transAsType, or transToType must report an error
100 // to em.
101 // 4. Any call to transAsType(e, t) with a type that is not returned by
102 // getType() (or one of the subtypes in case of a superposition)
103 // must report an error.
104 // Any call to transToType(e, t) with a type that is not returned by
105 // getType() (or one of the subtypes in case of a superposition)
106 // or any type not implicitly castable from the above must report an
107 // error.
108 virtual types::ty *getType(coenv &) = 0;
109
110 // This is an optimization which works in some cases to by-pass the slow
111 // overloaded function resolution provided by the application class.
112 //
113 // If an expression is called with arguments given by sig, getCallee must
114 // either return 0 (the default), or if it returns a varEntry, the varEntry
115 // must correspond to the function which would be called after normal
116 // function resolution.
117 //
118 // The callee must produce no side effects as there are no guarantees when
119 // the varEntry will be translated.
120 virtual trans::varEntry *getCallee(coenv &e, types::signature *sig) {
121//#define DEBUG_GETAPP
122#if DEBUG_GETAPP
123 cout << "exp fail" << endl;
124 cout << "exp fail at " << getPos() << endl;
125 prettyprint(cout, 2);
126#endif
127 return 0;
128 }
129
130 // Same result as getType, but caches the result so that subsequent
131 // calls are faster. For this to work correctly, the expression should
132 // only be used in one place, so the environment doesn't change between
133 // calls.
134 virtual types::ty *cgetType(coenv &e) {
135#ifdef DEBUG_CACHE
136 testCachedType(e);
137#endif
138 return ct ? ct : ct = getType(e);
139 }
140
141 void testCachedType(coenv &e);
142
143 // The expression is being written. Translate code such that the value
144 // (represented by the exp value) is stored into the address represented by
145 // this expression.
146 // In terms of side-effects, this expression must be evaluated (once) before
147 // value is evaluated (once).
148 virtual void transWrite(coenv &e, types::ty *t, exp *value) {
149 em.error(getPos());
150 em << "expression cannot be used as an address";
151
152 // Translate the value for errors.
153 value->transToType(e, t);
154 }
155
156 // Translates code for calling a function. The arguments, in the order they
157 // appear in the function's signature, must all be on the stack.
158 virtual void transCall(coenv &e, types::ty *target);
159
160 // transConditionalJump must produce code equivalent to the following:
161 // Evaluate the expression as a boolean. If the result equals cond, jump to
162 // the label dest, otherwise do not jump. In either case, no value is left
163 // on the stack.
164 virtual void transConditionalJump(coenv &e, bool cond, label dest);
165
166 // This is used to ensure the proper order and number of evaluations. When
167 // called, it immediately translates code to perform the side-effects
168 // consistent with a corresponding call to transAsType(e, target).
169 //
170 // The return value, called an evaluation for lack of a better name, is
171 // another expression that responds to the trans methods exactly as would the
172 // original expression, but without producing side-effects. It is also no
173 // longer overloaded, due to the resolution effected by giving a target type
174 // to evaluate().
175 //
176 // The methods transAsType, transWrite, and transCall of the evaluation must
177 // be called with the same target type as the original call to evaluate.
178 // When evaluate() is called during the translation of a function, that
179 // function must still be in translation when the evaluation is translated.
180 //
181 // The base implementation uses a tempExp (see below). This is
182 // sufficient for most expressions.
183 virtual exp *evaluate(coenv &e, types::ty *target);
184
185 // NOTE: could add a "side-effects" method which says if the expression has
186 // side-effects. This might allow some small optimizations in translating.
187};
