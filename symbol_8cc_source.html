23struct symbolRecord {
24 // When a symbol is entered into the table, its hash is computed. If the
25 // corresponding entry in the table is full, this value is incremented until
26 // an empty slot is found. hashplus stores the end value.
27 // Each symbol has a unique hashplus value, even if there is a collision in
28 // the original hashing function.
29 uint hashplus;
30
31 // Whether the cell of the table is empty, in use, or a "skip" entry due to
32 // a resizing of the table.
33 unsigned char flag;
34
35 // Pointer to a copy of the string (allocated on the heap). This string
36 // will never be deallocated. Symbols, in essence, last forever.
37 char *s;
38};
39
40// The table size must be a power of two so that (h % tableSize) can be
41// replaced by (h & tableMask). 1 << 15 was chosen based on the number of
42// unique symbols (roughly 4000) which occurred in all of the base modules.
43const size_t SYMBOL_TABLE_BASE_CAPACITY = 1 << 15;
44symbolRecord baseSymbolTable[SYMBOL_TABLE_BASE_CAPACITY];
45
46symbolRecord *table = baseSymbolTable;
47size_t tableCapacity = SYMBOL_TABLE_BASE_CAPACITY;
48uint tableMask = 0;
49size_t tableSize = 0;
50
51symbolRecord &recordByHashplus(uint h)
52{
53 return table[h & tableMask];
54}

64void initTable() {
65 tableMask = (uint)(tableCapacity - 1);
66 tableSize = 0;
67
68 // Set every entry to empty. (Is this faster than memsetting the whole
69 // thing?)
70 for (size_t i = 0; i < tableCapacity; ++i)
71 table[i].flag = 0;
72
73 // The zeroth entry is reserved for the "null" symbol.
74 if (table == baseSymbolTable) {
75 table[0].flag = USED;
76 table[0].s = new char[strlen(nullsymstr) + 1];
77 strcpy(table[0].s, nullsymstr);
78 ++tableSize;
79
80 symbol::nullsym.hashplus = 0;
81
82 symbol::initsym = symbol::opTrans("init");
83 symbol::castsym = symbol::opTrans("cast");
84 symbol::ecastsym = symbol::opTrans("ecast");
85 }
86}

88// Hashing constants found experimentally to reduce collision (a little).
89const uint A = 25191, B = 16342, C = 1746, D = 18326;
90
91// Hash the string into an integer. Experimental testing has shown that
92// hashing only the first few letters seems to be faster than hashing deeper
93// into the string, even though this approach causes more hash collisions.
94uint hash(const char *s, size_t len)
95{
96 uint h = s[0];
97 if (len == 2)
98 return h;
99 h += A*s[1];
100 if (len == 3)
101 return h;
102 h += B*s[2];
103 if (len == 4)
104 return h;
105 h += C*s[3];
106 if (len == 5)
107 return h;
108 h += D*s[4];
109 return h+len;
110}

112/* Under normal circumstances, the initial table should be large enough for
113 * all of the symbols used and will never be resized. Just in case the
114 * program encounters a large number of distinct symbols, we implement
115 * resizing of the table.
116 */
117void resizeTable() {
118 symbolRecord *oldTable = table;
119 size_t oldSize = tableSize;
120 size_t oldCapacity = tableCapacity;
121
122 tableCapacity *= 4;
123 table = new symbolRecord[tableCapacity];
124 initTable();
125
126 // The null symbol is a special case.
127 table[0] = oldTable[0];
128 ++tableSize;
129
130#if 0
131 printf("old:\n");
132 for (size_t i = 0; i < oldCapacity; ++i) {
133 symbolRecord &r = oldTable[i];
134
135 if (r.flag != USED)
136 continue;
137
138 printf(" %u -> %s\n", r.hashplus, r.s);
139 }
140#endif
141
142 for (size_t i = 1; i < oldCapacity; ++i) {
143 symbolRecord &r = oldTable[i];
144
145 if (r.flag != USED)
146 continue;
147
148 // Entries that were skipped over when this symbol was entered into the
149 // old hash table may not appear in the same spot in the new hash table.
150 // Put "SKIP" entries in their place, so that the symbol will still be
151 // found.
152 for (uint h = hash(r.s, strlen(r.s)+1); h < r.hashplus; ++h) {
153 symbolRecord &skipr = recordByHashplus(h);
154 if (skipr.flag == 0)
155 skipr.flag = SKIP;
156 }
157
158 // Enter the symbol in its spot.
159 symbolRecord &newr = recordByHashplus(r.hashplus);
160 assert(newr.flag != USED);
161 newr.flag = USED;
162 newr.hashplus = r.hashplus;
163 newr.s = r.s;
164 ++tableSize;
165 }
166
167#if 0
168 printf("new:\n");
169 for (size_t i = 0; i < tableCapacity; ++i) {
170 symbolRecord &r = table[i];
171
172 if (r.flag != USED)
173 continue;
174
175 printf(" %u -> %s\n", r.hashplus, r.s);
176 }
177#endif
178
179 assert(tableSize == oldSize);
180
181 // Debugging resize.
182 for (size_t i = 1; i < oldCapacity; ++i) {
183 symbolRecord &r = oldTable[i];
184
185 if (r.flag != USED)
186 continue;
187
188 symbolRecord &newr = recordByHashplus(r.hashplus);
189 assert(newr.hashplus == r.hashplus);
190 assert(newr.flag != 0);
191 assert(newr.flag != SKIP);
192 assert(newr.flag == USED);
193 assert(newr.s = r.s);
194 if (strncmp(r.s, "gensym", 6) != 0)
195 assert(symbol::rawTrans(r.s, strlen(r.s)+1).hashplus == r.hashplus);
196 }
197
198#if 0
199 // Diagnostics.
200 uint empty=0, used=0, skip=0;
201 for (size_t i = 0; i < tableCapacity; ++i) {
202 symbolRecord &r = table[i];
203 if (r.flag == 0) ++empty;

205 else if (r.flag == SKIP) ++skip;

223// Handles the insertion of a new symbol into a table the has been resized (or
224// needs resizing).
225symbol advancedInsert(const char *s, size_t len)
226{
227 if (2*tableSize >= tableCapacity)
228 resizeTable();
229
230 uint hashplus = hash(s, len);
231
232#if 1
233 assert(s != 0);
234 assert(len > 0);
235 assert(2*tableSize <= tableCapacity);
236#endif
237
238 // We know the symbol is not in the table. Just search for the first unused
239 // entry (either empty or a skip entry) and insert there.
240 for (;;) {
241 symbolRecord &r = recordByHashplus(hashplus);
242
243 if (r.flag != USED) {
244 r.flag = USED;
245 r.s = new char[len];
246 memcpy(r.s, s, len);
247 assert(r.s[len-1] == '\0');
248 r.hashplus = hashplus;
249
250 ++tableSize;
251 assert(2*tableSize <= tableCapacity);
252
253 return symbolize(hashplus);
254 }
255
256 ++hashplus;
257 }
258
259 assert("Unreachable code" == 0);
260 return symbol::nullsym;
261}

270symbol symbol::rawTrans(const char *s, size_t len)
271{
272 uint hashplus = sym::hash(s, len);
273
274#if 1
275 assert(s != 0);
276 assert(len > 0);
277 assert(2*tableSize <= tableCapacity);
278#endif
279
280 // Search through the table till we find the symbol already translated or
281 // an empty field.
282 for (;;) {
283 symbolRecord &r = recordByHashplus(hashplus);
284
285 // Translating pre-existing symbols is more common, so check for it first.
286 if (r.hashplus == hashplus &&
287 r.flag == USED &&
288 strncmp(r.s, s, len) == 0) {
289 return symbolize(hashplus);
290 }
291
292 // Then check for an empty entry, in which case the entry is added.
293 if (r.flag == 0) {
294 // Test if the table needs resizing before entering a new symbol, or if
295 // the table has already been resized. In either case, the symbol will
296 // be added to a resized table which may contain skip entries, and a
297 // more involved insertion routine is needed.
298 if (2*tableSize >= SYMBOL_TABLE_BASE_CAPACITY)
299 return advancedInsert(s, len);
300
301 r.flag = USED;
302 r.s = new char[len];
303 memcpy(r.s, s, len);
304 assert(r.s[len-1] == '\0');
305 r.hashplus = hashplus;
306
307 ++tableSize;
308 assert(2*tableSize <= tableCapacity);
309
310 return symbolize(hashplus);
311 }
312
313 // A case where a different symbol is in the spot, continue along the
314 // table.
315 ++hashplus;
316 }
317
318 assert("Unreachable code" == 0);
319 return symbol::nullsym;
320}
