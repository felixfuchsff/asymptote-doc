62template<class T>
63void texpreamble(T& out, mem::list<string>& preamble=processData().TeXpreamble,
64 bool pipe=false, bool ASYbox=true)
65{
66 texuserpreamble(out,preamble,pipe);
67 string texengine=settings::getSetting<string>("tex");
68 string outPath=stripFile(settings::outname());
69 if(settings::context(texengine))
70 out << "\\disabledirectives[system.errorcontext]%" << newl;
71 if(ASYbox)
72 out << "\\newbox\\ASYbox" << newl
73 << "\\newdimen\\ASYdimen" << newl;
74 out << "\\def\\ASYprefix{" << escapeCharacters(outPath, latexCharacters) << "}" << newl
75 << "\\long\\def\\ASYbase#1#2{\\leavevmode\\setbox\\ASYbox=\\hbox{#1}%"
76 << "\\ASYdimen=\\ht\\ASYbox%" << newl
77 << "\\setbox\\ASYbox=\\hbox{#2}\\lower\\ASYdimen\\box\\ASYbox}" << newl;
78 if(!pipe)
79 out << "\\long\\def\\ASYaligned(#1,#2)(#3,#4)#5#6#7{\\leavevmode%" << newl
80 << "\\setbox\\ASYbox=\\hbox{#7}%" << newl
81 << "\\setbox\\ASYbox\\hbox{\\ASYdimen=\\ht\\ASYbox%" << newl
82 << "\\advance\\ASYdimen by\\dp\\ASYbox\\kern#3\\wd\\ASYbox"
83 << "\\raise#4\\ASYdimen\\box\\ASYbox}%" << newl
84 << "\\setbox\\ASYbox=\\hbox{#5\\wd\\ASYbox 0pt\\dp\\ASYbox 0pt\\ht\\ASYbox 0pt\\box\\ASYbox#6}%" << newl
85 << "\\hbox to 0pt{\\kern#1pt\\raise#2pt\\box\\ASYbox\\hss}}%" << newl
86 << "\\long\\def\\ASYalignT(#1,#2)(#3,#4)#5#6{%" << newl
87 << "\\ASYaligned(#1,#2)(#3,#4){%" << newl
88 << settings::beginlabel(texengine) << "%" << newl
89 << "}{%" << newl
90 << settings::endlabel(texengine) << "%" << newl
91 << "}{#6}}" << newl
92 << "\\long\\def\\ASYalign(#1,#2)(#3,#4)#5{"
93 << "\\ASYaligned(#1,#2)(#3,#4){}{}{#5}}" << newl
94 << settings::rawpostscript(texengine) << newl;
95}

98template<class T>
99void dvipsfix(T &out)
100{
101 if(!settings::pdf(settings::getSetting<string>("tex"))) {
102 out << "\\makeatletter" << newl
103 << "\\def\\Ginclude@eps#1{%" << newl
104 << " \\message{<#1>}%" << newl
105 << " \\bgroup" << newl
106 << " \\def\\@tempa{!}%" << newl
107 << " \\dimen@\\Gin@req@width" << newl
108 << " \\dimen@ii.1bp%" << newl
109 << " \\divide\\dimen@\\dimen@ii" << newl
110 << " \\@tempdima\\Gin@req@height" << newl
111 << " \\divide\\@tempdima\\dimen@ii" << newl
112 << " \\special{PSfile=#1\\space" << newl
113 << " llx=\\Gin@llx\\space" << newl
114 << " lly=\\Gin@lly\\space" << newl
115 << " urx=\\Gin@urx\\space" << newl
116 << " ury=\\Gin@ury\\space" << newl
117 << " \\ifx\\Gin@scalex\\@tempa\\else rwi=\\number\\dimen@\\space\\fi" << newl
118 << " \\ifx\\Gin@scaley\\@tempa\\else rhi=\\number\\@tempdima\\space\\fi" << newl
119 << " \\ifGin@clip clip\\fi}%" << newl
120 << " \\egroup}" << newl
121 << "\\makeatother" << newl;
122 }

125template<class T>
126void texdefines(T& out, mem::list<string>& preamble=processData().TeXpreamble,
127 bool pipe=false)
128{
129 string texengine=settings::getSetting<string>("tex");
130 bool latex=settings::latex(texengine);
131 bool inlinetex=settings::getSetting<bool>("inlinetex");
132 if(pipe || !inlinetex) {
133 bool lua=settings::lua(texengine);
134 if(latex) {
135 if(lua) {
136 out << "\\edef\\pdfpageattr{\\pdfvariable pageattr}" << newl
137 << "\\ifx\\pdfpagewidth\\undefined\\let\\pdfpagewidth\\paperwidth"
138 << "\\fi" << newl
139 << "\\ifx\\pdfpageheight\\undefined\\let\\pdfpageheight"
140 << "\\paperheight"
141 << "\\fi" << newl
142 << "\\usepackage{graphicx}" << newl;
143 } else {
144 out << "\\let\\paperwidthsave\\paperwidth\\let\\paperwidth\\undefined"
145 << newl
146 << "\\usepackage{graphicx}" << newl
147 << "\\let\\paperwidth\\paperwidthsave" << newl;
148 }
149 } else {
150 if(lua) {
151 out << "\\edef\\pdfpageattr{\\pdfvariable pageattr}" << newl
152 << "\\ifx\\pdfpagewidth\\undefined\\let\\pdfpagewidth\\pagewidth"
153 << "\\fi" << newl
154 << "\\ifx\\pdfpageheight\\undefined\\let\\pdfpageheight"
155 << "\\pageheight"
156 << "\\fi" << newl;
157 }
158 }
159 texpreamble(out,preamble,pipe);
160 }
161
162 if(pipe) {
163 // Make tex pipe aware of a previously generated aux file.
164 string name=auxname(settings::outname(),"aux");
165 std::ifstream fin(name.c_str());
166 if(fin) {
167 std::ofstream fout("texput.aux");
168 string s;
169 while(getline(fin,s))
170 fout << s << endl;
171 }
172 }
173
174 if(latex) {
175 if(!inlinetex) {
176 dvipsfix(out);
177 }
178 if(pipe) {
179 out << "\\begin{document}" << newl;
180 latexfontencoding(out);
181 }
182 } else if(!settings::context(texengine)) {
183 out << "\\input graphicx" << newl // Fix miniltx path parsing bug:
184 << "\\makeatletter" << newl
185 << "\\def\\filename@parse#1{%" << newl
186 << " \\let\\filename@area\\@empty" << newl
187 << " \\expandafter\\filename@path#1/\\\\}" << newl
188 << "\\def\\filename@path#1/#2\\\\{%" << newl
189 << " \\ifx\\\\#2\\\\%" << newl
190 << " \\def\\reserved@a{\\filename@simple#1.\\\\}%" << newl
191 << " \\else" << newl
192 << " \\edef\\filename@area{\\filename@area#1/}%" << newl
193 << " \\def\\reserved@a{\\filename@path#2\\\\}%" << newl
194 << " \\fi" << newl
195 << " \\reserved@a}" << newl
196 << "\\makeatother" << newl;
197 dvipsfix(out);
198
199 if(!pipe)
200 out << "\\input picture" << newl;
201 }
202}

204template<class T>
205bool setlatexfont(T& out, const pen& p, const pen& lastpen)
206{
207 if(p.size() != lastpen.size() || p.Lineskip() != lastpen.Lineskip()) {
208 out << "\\fontsize{" << p.size()*settings::ps2tex << "}{"
209 << p.Lineskip()*settings::ps2tex << "}\\selectfont%" << newl;
210 return true;
211 }
212 return false;
213}
