96void printInst(ostream& out, const program::label& code,
97 const program::label& base)
98{
99 out.width(4);
100 out << offset(base,code) << " ";
101
102 Int i = (Int)code->op;
103
104 if (i < 0 || i >= numOps) {
105 out << "<<invalid op>>" << i;
106 return;
107 }
108 out << opnames[i];
109
110 switch (optypes[i]) {
111 case 'n':
112 {
113 out << " " << get<Int>(*code);
114 break;
115 }
116
117 case 't':
118 {
119 item c = code->ref;
120 out << " " << c;
121 break;
122 }
123
124 case 'b':
125 {
126#ifdef DEBUG_BLTIN
127 string s=lookupBltin(get<bltin>(*code));
128 out << " " << (!s.empty() ? s : "<unnamed>") << " ";
129#endif
130 break;
131 }
132
133 case 'o':
134 {
135 char f = out.fill('0');
136 out << " i";
137 out.width(4);
138 out << offset(base,get<program::label>(*code));
139 out.fill(f);
140 break;
141 }
142
143 case 'l':
144 {
145#ifdef DEBUG_FRAME
146 out << " " << get<lambda*>(*code)->name << " ";
147#endif
148 break;
149 }
150
151 default: {
152 /* nothing else to do */
153 break;
154 }
155 };
156}
