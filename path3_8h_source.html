162 inline double norm(const triple& z0, const triple& c0, const triple& c1,
163 const triple& z1) const {
164 return Fuzz2*camp::max((c0-z0).abs2(),
165 camp::max((c1-z0).abs2(),(z1-z0).abs2()));
166 }

168 triple predir(Int t, bool normalize=true) const {
169 if(!cycles && t <= 0) return triple(0,0,0);
170 triple z1=point(t);
171 triple c1=precontrol(t);
172 triple dir=3.0*(z1-c1);
173 if(!normalize) return dir;
174 triple z0=point(t-1);
175 triple c0=postcontrol(t-1);
176 double epsilon=norm(z0,c0,c1,z1);
177 if(dir.abs2() > epsilon) return unit(dir);
178 dir=2.0*c1-c0-z1;
179 if(dir.abs2() > epsilon) return unit(dir);
180 return unit(z1-z0+3.0*(c0-c1));
181 }
182
183 triple postdir(Int t, bool normalize=true) const {
184 if(!cycles && t >= n-1) return triple(0,0,0);
185 triple c0=postcontrol(t);
186 triple z0=point(t);
187 triple dir=3.0*(c0-z0);
188 triple z1=point(t+1);
189 triple c1=precontrol(t+1);
190 double epsilon=norm(z0,c0,c1,z1);
191 if(!normalize) return dir;
192 if(dir.abs2() > epsilon) return unit(dir);
193 dir=z0-2.0*c0+c1;
194 if(dir.abs2() > epsilon) return unit(dir);
195 return unit(z1-z0+3.0*(c0-c1));
196 }

198 triple dir(Int t, Int sign, bool normalize=true) const {
199 if(sign == 0) {
200 triple v=predir(t,normalize)+postdir(t,normalize);
201 return normalize ? unit(v) : 0.5*v;
202 }
203 if(sign > 0) return postdir(t,normalize);
204 return predir(t,normalize);
205 }
206
207 triple dir(double t, bool normalize=true) const {
208 if(!cycles) {
209 if(t <= 0) return postdir((Int) 0,normalize);
210 if(t >= n-1) return predir(n-1,normalize);
211 }
212 Int i=Floor(t);
213 t -= i;
214 if(t == 0) return dir(i,0,normalize);
215 triple z0=point(i);
216 triple c0=postcontrol(i);
217 triple c1=precontrol(i+1);
218 triple z1=point(i+1);
219 triple a=3.0*(z1-z0)+9.0*(c0-c1);
220 triple b=6.0*(z0+c1)-12.0*c0;
221 triple c=3.0*(c0-z0);
222 triple dir=a*t*t+b*t+c;
223 if(!normalize) return dir;
224 double epsilon=norm(z0,c0,c1,z1);
225 if(dir.abs2() > epsilon) return unit(dir);
226 dir=2.0*a*t+b;
227 if(dir.abs2() > epsilon) return unit(dir);
228 return unit(a);
229 }
230
231 triple postaccel(Int t) const {
232 if(!cycles && t >= n-1) return triple(0,0,0);
233 triple z0=point(t);
234 triple c0=postcontrol(t);
235 triple c1=precontrol(t+1);
236 return 6.0*(z0+c1)-12.0*c0;
237 }
238
239 triple preaccel(Int t) const {
240 if(!cycles && t <= 0) return triple(0,0,0);
241 triple z0=point(t-1);
242 triple c0=postcontrol(t-1);
243 triple c1=precontrol(t);
244 triple z1=point(t);
245 return 6.0*(z1+c0)-12.0*c1;
246 }
247
248 triple accel(Int t, Int sign) const {
249 if(sign == 0) return 0.5*(preaccel(t)+postaccel(t));
250 if(sign > 0) return postaccel(t);
251 return preaccel(t);
252 }
253
254 triple accel(double t) const {
255 if(!cycles) {
256 if(t <= 0) return postaccel((Int) 0);
257 if(t >= n-1) return preaccel(n-1);
258 }
259 Int i=Floor(t);
260 t -= i;
261 if(t == 0) return 0.5*(postaccel(i)+preaccel(i));
262 triple z0=point(i);
263 triple c0=postcontrol(i);
264 triple c1=precontrol(i+1);
265 triple z1=point(i+1);
266 return 6.0*t*(z1-z0+3.0*(c0-c1))+6.0*(z0+c1)-12.0*c0;
267 }
