132 size_t weightsize=checkArray(weight);
133
134 string wrongsize="Inconsistent NURBS data";
135 n=checkArray(&g);
136
137 if(n == 0 || (weightsize != 0 && weightsize != n))
138 reportError(wrongsize);
139
140 controls=new(UseGC) triple[n];
141
142 size_t k=0;
143 for(size_t i=0; i < n; ++i)
144 controls[k++]=vm::read<triple>(g,i);
145
146 if(weightsize > 0) {
147 size_t k=0;
148 weights=new(UseGC) double[n];
149 for(size_t i=0; i < n; ++i)
150 weights[k++]=vm::read<double>(weight,i);
151 } else weights=NULL;
152
153 size_t nknots=checkArray(knot);
154
155 if(nknots <= n+1 || nknots > 2*n)
156 reportError(wrongsize);
157
158 degree=nknots-n-1;
159
160 run::copyArrayC(knots,knot,0,NoGC);
161
162#ifdef HAVE_LIBGLM
163 Controls=NULL;
164#endif
165 }
