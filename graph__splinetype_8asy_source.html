34// Standard cubic spline interpolation with not-a-knot condition:
35// s'''(x_2^-)=s'''(x_2^+) et s'''(x_(n_2)^-)=s'''(x_(n-2)^+)
36// if n=2, linear interpolation is returned
37// if n=3, an interpolation polynomial of degree <= 2 is returned:
38// p(x_1)=y_1, p(x_2)=y_2, p(x_3)=y_3
39real[] notaknot(real[] x, real[] y)

43 checkincreasing(x);
44 real[] d;
45 if(n > 3) {
46 real[] a=new real[n];
47 real[] b=new real[n];
48 real[] c=new real[n];
49 real[] g=new real[n];
50 b[0]=x[2]-x[1];
51 c[0]=x[2]-x[0];
52 a[0]=0;
53 g[0]=((x[1]-x[0])^2*(y[2]-y[1])/b[0]+b[0]*(2*b[0]+3*(x[1]-x[0]))*
54 (y[1]-y[0])/(x[1]-x[0]))/c[0];

56 a[i]=x[i+1]-x[i];
57 c[i]=x[i]-x[i-1];
58 b[i]=2*(a[i]+c[i]);
59 g[i]=3*(c[i]*(y[i+1]-y[i])/a[i]+a[i]*(y[i]-y[i-1])/c[i]);

65 b[n-1]*(2*b[n-1]+3(x[n-1]-x[n-2]))*
66 (y[n-1]-y[n-2])/(x[n-1]-x[n-2]))/a[n-1];
67 d=tridiagonal(a,b,c,g);
68 } else if(n == 2) {
69 real val=(y[1]-y[0])/(x[1]-x[0]);

71 } else if(n == 3) {

188// Modified MATLAB code
189// [1] Fritsch, F. N. and R. E. Carlson,
190// "Monotone Piecewise Cubic Interpolation,"

194real[] monotonic(real[] x, real[] y)
195{
196 int n=x.length;
197 checklengths(n,y.length);
198 checkincreasing(x);
199 real[] d=new real[n];
200 if(n > 2) {
201 real[] h=new real[n-1];
202 real[] del=new real[n-1];
203 for(int i=0; i < n-1; ++i) {
204 h[i]=x[i+1]-x[i];
205 del[i]=(y[i+1]-y[i])/h[i];
206 }
207 int j=0;
208 int k[]=new int[];
209 for(int i=0; i < n-2; ++i)
210 if((sgn(del[i])*sgn(del[i+1])) > 0) {k[j]=i; j=j+1;}
211
212 real[] hs=new real[j];
213 for(int i=0; i < j; ++i) hs[i]=h[k[i]]+h[k[i]+1];
214 real w1[]=new real[j];
215 real w2[]=new real[j];
216 real dmax[]=new real[j];
217 real dmin[]=new real[j];
218 for(int i=0; i < j; ++i) {
219 w1[i]=(h[k[i]]+hs[i])/(3*hs[i]);
220 w2[i]=(h[k[i]+1]+hs[i])/(3*hs[i]);
221 dmax[i]=max(abs(del[k[i]]),abs(del[k[i]+1]));
222 dmin[i]=min(abs(del[k[i]]),abs(del[k[i]+1]));
223 }
224 for(int i=0; i < n; ++i) d[i]=0;
225 for(int i=0; i < j; ++i)
226 d[k[i]+1]=dmin[i]/(w1[i]*(del[k[i]]/dmax[i])+w2[i]*(del[k[i]+1]/dmax[i]));
227 d[0]=((2*h[0]+h[1])*del[0]-h[0]*del[1])/(h[0]+h[1]);
228 if(sgn(d[0]) != sgn(del[0])) {d[0]=0;}
229 else if((sgn(del[0]) != sgn(del[1])) && (abs(d[0]) > abs(3*del[0])))
230 d[0]=3*del[0];
231
232 d[n-1]=((2*h[n-2]+h[n-3])*del[n-2]-h[n-2]*del[n-2])/(h[n-2]+h[n-3]);

234 else if((sgn(del[n-2]) != sgn(del[n-3])) &&
235 (abs(d[n-1]) > abs(3*del[n-2])))
236 d[n-1]=3*del[n-2];
237 } else if(n == 2) {
238 d[0]=d[1]=(y[1]-y[0])/(x[1]-x[0]);
239 } else abort(morepoints);
240 return d;
241}
