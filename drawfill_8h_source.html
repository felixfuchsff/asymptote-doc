54public:

76 if(pentype.invisible() || empty()) return true;

98

100 out->gsave();
101 }

103 void beginshade(psfile *out) {
104 out->beginlatticeshade(pens,bpath);
105 }
106
107 void shade(psfile *out) {
108 bbox b;
109 for(size_t i=0; i < size; i++) {
110 path p=vm::read<path>(P,i).transformed(inverse(T));
111 if(stroke)
112 drawPathPenBase::strokebounds(b,p);
113 else
114 b += p.bounds();
115 }
116 out->latticeshade(pens,T*matrix(b.Min(),b.Max()));
117 }
