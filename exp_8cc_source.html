48void exp::transToType(coenv &e, types::ty *target)
49{
50 types::ty *ct=cgetType(e);
51
52 if (equivalent(target, ct)) {
53 transAsType(e, target);
54 return;
55 }
56
57 // See if the cast can be handled by the fastLookupCast method, which does
58 // less memory allocation.
59 if (ct->kind != ty_overloaded &&
60 ct->kind != ty_error &&
61 target->kind != ty_error) {
62 access *a = e.e.fastLookupCast(target, ct);
63 if (a) {
64 transAsType(e, ct);
65 a->encode(trans::CALL, getPos(), e.c);
66 return;
67 }
68 }
69
70 types::ty *source = e.e.castSource(target, ct, symbol::castsym);
71 if (source==0) {
72 if (target->kind != ty_error) {
73 types::ty *sources=cgetType(e);
74 em.error(getPos());
75
76 em << "cannot cast ";
77 if (sources->kind==ty_overloaded)
78 em << "expression";
79 else
80 em << "'" << *sources << "'";
81 em << " to '" << *target << "'";
82 }
83 }
84 else if (source->kind==ty_overloaded) {
85 if (target->kind != ty_error) {
86 em.error(getPos());
87 em << "expression is ambiguous in cast to '" << *target << "'";
88 }
89 }
90 else {
91 transAsType(e, source);
92 e.implicitCast(getPos(), target, source);
93 }
94}

320types::ty *subscriptExp::trans(coenv &e)
321{
322 if (!isAnArray(e, object)) {
323 callExp *call = buildSubscriptReadCall(object, index);
324 return call->trans(e);
325 }
326
327 array *a = transArray(e);
328 if (!a)
329 return primError();
330
331 if (isAnArray(e, index)) {
332 index->transToType(e, types::IntArray());
333 e.c.encode(inst::builtin, run::arrayIntArray);
334 return getArrayType(e);
335 }
336 else {
337 index->transToType(e, types::primInt());
338 e.c.encode(inst::builtin,
339 a->celltype->kind==ty_array ? run::arrayArrayRead :
340 run::arrayRead);
341 return a->celltype;
342 }

360void subscriptExp::transWrite(coenv &e, types::ty *t, exp *value)
361{
362 if (!isAnArray(e, object)) {
363 // Find the types of object and index.
364 types::ty *objectType = getObjectType(e);
365 assert(objectType);
366 types::ty *indexType = objectType->keyType();
367 if (!indexType || indexType->kind == ty_error) {
368 em.error(object->getPos());
369 em << "object does not have operator[]";
370 return;
371 }
372 // Evaluate them to control the order in which side effects occur.
373 // We need value evaluated because we use it twice. We need the other two
374 // because any side effects from their translation should occur before the
375 // side effects from translating value.
376 exp *objectEvaluated = object->evaluate(e, objectType);
377 exp *indexEvaluated = index->evaluate(e, indexType);
378 exp *valueEvaluated = value->evaluate(e, t);
379 // Call object.operator[=](index, value).
380 callExp* call= buildSubscriptWriteCall(
381 objectEvaluated, indexEvaluated, valueEvaluated
382 );
383 call->trans(e);
384 // Push the value back on the stack as the result of the assignment.
385 valueEvaluated->transAsType(e, t);
386 return;
387 }
388
389 // Put array, index, and value on the stack in that order, then call
390 // arrayWrite.
391 array *a = transArray(e);
392 if (!a)
393 return;
394
395 if (!equivalent(a->celltype, t))
396 {
397 em.error(getPos());
398 em << "array expression cannot be used as an address";
399
400 // Translate the value for errors.
401 value->transToType(e, t);
402 return;
403 }
404
405 index->transToType(e, types::primInt());
406
407 value->transToType(e, t);
408
409 e.c.encode(inst::builtin, run::arrayWrite);
410}

412exp *subscriptExp::evaluate(coenv &e, types::ty *)
413{
414 types::ty *base = object->cgetType(e)->signatureless();
415 if (!base) {
416 em.error(object->getPos());
417 em << "object to index cannot be resolved";
418 return exp::evaluate(e, primError());
419 }
420 types::ty *indexType = base->keyType();
421 if (indexType->kind == ty_error) {
422 em.error(object->getPos());
423 em << "object does not have operator[=] set up correctly";
424 return exp::evaluate(e, primError());
425 }
426 // Force object and index to be evaluated in the correct order.
427 // (Note that in C++, the order of evaluation of function arguments is
428 // unspecified.)
429 exp *a = object->evaluate(e, base);
430 exp *b = index->evaluate(e, indexType);
431 return new subscriptExp(getPos(), a, b);
432}

541// From a possibly overloaded type, if there is a unique function type, return
542// it, otherwise 0.
543types::ty *uniqueFunction(types::ty *t) {
544 if (t->kind == types::ty_function)
545 return t;
546
547 if (t->isOverloaded()) {
548 types::ty *ft = 0;
549 for (ty_iterator i = t->begin(); i != t->end(); ++i)
550 {
551 if ((*i)->kind != types::ty_function)
552 continue;
553
554 if (ft) {
555 // Multiple function types.
556 return 0;
557 }
558
559 ft = *i;
560 }
561
562 return ft;
563 }
564
565 // Not a function.
566 return 0;
567}
568
569// From two possibly overloaded types, if there is a unique function type
570// common to both, return it, otherwise 0.
571types::ty *uniqueFunction(types::ty *t1, types::ty *t2) {
572 if (t1->kind == types::ty_function)
573 return equivalent(t1, t2) ? t1 : 0;
574
575 if (t1->isOverloaded()) {
576 types::ty *ft = 0;
577 for (ty_iterator i = t1->begin(); i != t1->end(); ++i)
578 {
579 if ((*i)->kind != types::ty_function)
580 continue;
581
582 if (!equivalent(*i, t2))
583 continue;
584
585 if (ft) {
586 // Multiple function types.
587 return 0;
588 }
589
590 ft = *i;
591 }
592
593 return ft;
594 }
595
596 // Not a function.
597 return 0;
598}

607types::ty *equalityExp::trans(coenv &e) {
608 // First, try to handle by normal function resolution.
609 types::ty *t = callExp::getType(e);
610 assert(t);
611 if (t->kind != ty_error)
612 return callExp::trans(e);
613
614 // Then, check for the function equality case.
615 exp *left = (*this->args)[0].val;
616 exp *right = (*this->args)[1].val;
617
618 types::ty *lt = left->getType(e);
619 types::ty *rt = right->getType(e);
620
621 // TODO: decide what null == null should do.
622
623 // Check for function == null and null == function
624 types::ty *ft = 0;
625 if (rt->kind == types::ty_null)
626 ft = uniqueFunction(lt);
627 else if (lt->kind == types::ty_null)
628 ft = uniqueFunction(rt);
629 else
630 ft = uniqueFunction(lt, rt);
631
632
633 if (ft) {
634 assert(ft->kind == ty_function);
635
636 left->transToType(e, ft);
637 right->transToType(e, ft);
638 e.c.encode(inst::builtin, bltinFromName(callee->getName()));
639
640 return primBoolean();
641 } else {
642 // Let callExp report a "no such function" error.
643 types::ty *t = callExp::trans(e);
644 assert(t->kind == ty_error);
645 return t;
646 }
647}

838signature *callExp::argTypes(coenv &e, bool *searchable)
839{
840 signature *source=new signature;
841
842 // The signature is searchable unless one of the arguments is overloaded or
843 // named.
844 *searchable = true;
845
846 size_t n = args->size();
847
848 for (size_t i = 0; i < n; i++) {
849 if(string(args->args[i].name) == "KEY") {
850 stringExp *s=dynamic_cast<stringExp*>(args->args[i].val);
851 if(s) {
852 if(getPos().filename() == processData().fileName) {
853 processDataStruct *P=&processData();
854 P->xkey[getPos().shift(P->xmapCount).LineColumn()]=
855 Strdup(s->getString());
856 }
857 args->args.erase(args->args.begin()+i);
858 --n;
859 if(i == n) break;
860 }
861 }
862
863 argument a=(*args)[i];
864 types::ty *t = a.val->cgetType(e);
865 if (t->kind == types::ty_error)
866 return 0;
867 if (t->kind == types::ty_overloaded || a.name)
868 *searchable = false;
869 source->add(types::formal(t,a.name));
870 }
871
872 if (args->rest.val) {
873 argument a=args->rest;
874 types::ty *t = a.val->cgetType(e);
875 if (t->kind == types::ty_error)
876 return 0;
877 if (t->kind == types::ty_overloaded || a.name)
878 *searchable = false;
879 source->addRest(types::formal(t,a.name));
880 }
881
882 return source;
883}

886 bool tacit) {
887 app_list l=multimatch(e.e, o, source, *args);
888
889 if (l.empty()) {
890 //cerr << "l is empty\n";
891 if (!tacit) {
892 em.error(getPos());
893
894 symbol s = callee->getName();
895 if (s)
896 em << "no matching function \'" << s;
897 else
898 em << "no matching function for signature \'";
899 em << *source << "\'";
900 }
901
902 return 0;
903 }
904 else if (l.size() > 1) { // This may take O(n) time.
905 //cerr << "l is full\n";
906 if (!tacit) {
907 em.error(getPos());
908
909 symbol s = callee->getName();
910 if(s)
911 em << "call of function \'" << s;

913 em << "call with signature \'";
914 em << *source << "\' is ambiguous:\n\n";

916 em << *(*p)->getType() << "\n";
917 }
918
919 return 0;
920 }
921 else {
922 //cerr << "l is singleton\n";
923 return l.front();
924 }

980types::ty *callExp::cacheAppOrVarEntry(coenv &e, bool tacit)
981{
982 assert(cachedVarEntry == 0 && cachedApp == 0);
983
984 // First figure out the signature of what we want to call.
985 bool searchable;
986 signature *source=argTypes(e, &searchable);
987
988#ifdef DEBUG_GETAPP /* {{{ */
989 cout << "getApp for ";
990 if (callee->getName())
991 cout << *callee->getName();
992 else
993 cout << "unnamed";
994 cout << " at " << getPos() << endl;
995 cout << "searchable: " << searchable << endl;
996#endif /* }}} */
997
998 if (!source) {
999 return primError();
1000 }
1001
1002 // An attempt at speeding up compilation: See if the source arguments match
1003 // the (possibly overloaded) function exactly.
1004 if (searchable) {
1005 varEntry *ve = callee->getCallee(e, source);
1006
1007#ifdef DEBUG_GETAPP
1008 cout << "guessed: " << (ve!=0) << endl;
1009#endif
1010
1011 if (ve) {
1012 cachedVarEntry = ve;
1013#ifndef DEBUG_CACHE
1014 // Normally DEBUG_CACHE is not defined and we return here for efficiency
1015 // reasons. If DEBUG_CACHE is defined, we instead proceed to resolve
1016 // the function by the normal techniques and make sure we get the same
1017 // result.
1018 return ((function *)ve->getType())->getResult();
1019#endif
1020 }
1021 }
1022
1023 // Figure out what function types we can call.
1024 types::ty *ft = callee->cgetType(e);
1025
1026#ifdef DEBUG_GETAPP
1027 string name = callee->getName() ? string(*callee->getName()) :
1028 string("unnamed");
1029 if (!callee->getName())
1030 cout << getPos() << endl;
1031#endif
1032
1033 switch (ft->kind) {
1034 case ty_error:
1035 if (!tacit)
1036 // Report callee errors.
1037 callee->trans(e);
1038 break;
1039
1040 case ty_function:
1041 //cout << "name " << name << endl;
1042 cachedApp = application::match(e.e, (function *)ft, source, *args);
1043 if (!cachedApp && !tacit)
1044 reportMismatch((function *)ft, source);
1045 break;
1046
1047 case ty_overloaded: {
1048#ifdef DEBUG_GETAPP
1049 int size = ((overloaded *)ft)->sub.size();
1050 for (int i = 0; i < size; ++i) cout << "name " << name << endl;
1051#endif
1052 cachedApp = resolve(e, (overloaded *)ft, source, tacit);
1053 break;
1054 }
1055
1056 default:
1057 if (!tacit)
1058 reportNonFunction();
1059 break;
1060 }
1061
1062#ifdef DEBUG_GETAPP
1063 cout << name << " " << *source << " --> "
1064 << *cachedApp->getType()->getSignature() << endl;
1065#endif
1066
1067#if DEBUG_CACHE
1068 // Make sure cachedVarEntry is giving us the right function.
1069 if (cachedVarEntry)
1070 assert(equivalent(cachedVarEntry->getType(), cachedApp->getType()));
1071#endif
1072
1073 // getType relies on this method for the type.
1074 return cachedApp ? cachedApp->getType()->getResult() : primError();
1075}

1077types::ty *callExp::transPerfectMatch(coenv &e) {
1078 // The varEntry of the callee. (No longer needed after translation.)
1079 varEntry *ve = cachedVarEntry;
1080 cachedVarEntry = 0;
1081 assert(ve);
1082
1083 // Translate the arguments in turn.
1084 for (size_t i = 0; i < args->size(); ++i)
1085 (*args)[i].val->trans(e);
1086 if (args->rest.val)
1087 args->rest.val->trans(e);
1088
1089 // Call the function.
1090 ve->encode(trans::CALL, getPos(), e.c);
1091
1092 // That's it. Return the return type of the function.
1093 return ct ? ct : dynamic_cast<function *>(ve->getType())->getResult();
1094}
1095
1096types::ty *callExp::trans(coenv &e)
1097{
1098 if (cachedVarEntry == 0 && cachedApp == 0)
1099 cacheAppOrVarEntry(e, false);
1100
1101 if (cachedVarEntry)
1102 return transPerfectMatch(e);
1103
1104 // The cached data is no longer needed after translation, so let it be
1105 // garbage collected.
1106 application *a = cachedApp;
1107 cachedApp=0;
1108
1109 if (!a) {
1110 reportArgErrors(e);
1111 return primError();
1112 }
1113
1114 // To simulate left-to-right order of evaluation, produce the
1115 // side-effects for the callee.
1116 assert(a);
1117 function *t=a->getType();
1118 assert(t);
1119 exp *temp=callee->evaluate(e, t);
1120
1121 // Let the application handle the argument translation.
1122 a->transArgs(e);
1123
1124 // Translate the call.
1125 temp->transCall(e, t);
1126
1127 return t->result;
1128}

1309types::ty *promote(coenv &e, types::ty *x, types::ty *y)
1310{
1311 struct promoter : public collector {
1312 env &e;
1313
1314 promoter(env &e)
1315 : e(e) {}
1316
1317 types::ty *both (types::ty *x, types::ty *y) {
1318 overloaded *o=new overloaded;
1319 o->add(x); o->add(y);
1320 return o;
1321 }
1322
1323 types::ty *base (types::ty *x, types::ty *y) {
1324 if (equivalent(x,y))
1325 return x;
1326 else {
1327 bool castToFirst=e.castable(x, y, symbol::castsym);
1328 bool castToSecond=e.castable(y, x, symbol::castsym);
1329
1330 return (castToFirst && castToSecond) ? both(x,y) :
1331 castToFirst ? x :
1332 castToSecond ? y :
1333 0;
1334 }
1335 }
1336 };
1337
1338 promoter p(e.e);
1339 return p.collect(x,y);
1340}
