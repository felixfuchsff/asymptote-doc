185// Helper function to get action string from button/mods (uses GLFW constants)
186// This is generic code that can be shared between Vulkan and OpenGL renderers
187static std::string getActionString(int button, int mods)
188{
189 size_t Button;
190 size_t nButtons = 5;
191 switch(button) {
192 case GLFW_MOUSE_BUTTON_LEFT:
193 Button = 0;
194 break;
195 case GLFW_MOUSE_BUTTON_MIDDLE:
196 Button = 1;
197 break;
198 case GLFW_MOUSE_BUTTON_RIGHT:
199 Button = 2;
200 break;
201 default:
202 Button = nButtons;
203 }
204
205 size_t Mod;
206 size_t nMods = 4;
207
208 if (mods == 0)
209 Mod = 0;
210 else if(mods & GLFW_MOD_SHIFT)
211 Mod = 1;
212 else if(mods & GLFW_MOD_CONTROL)
213 Mod = 2;
214 else if(mods & GLFW_MOD_ALT)
215 Mod = 3;
216 else
217 Mod = nMods;
218
219 if(Button < nButtons) {
220 auto left = getSetting<vm::array *>("leftbutton");
221 auto middle = getSetting<vm::array *>("middlebutton");
222 auto right = getSetting<vm::array *>("rightbutton");
223 auto wheelup = getSetting<vm::array *>("wheelup");
224 auto wheeldown = getSetting<vm::array *>("wheeldown");
225 vm::array* buttonArray[] = {left, middle, right, wheelup, wheeldown};
226 auto a = buttonArray[Button];
227 size_t size = checkArray(a);
228
229 if(Mod < size)
230 return read<std::string>(a, Mod);
231 }
232
233 return "";
234}

269void glfwRunLoop(GLFWwindow* window,
270 std::function<bool()> shouldContinue,
271 std::function<bool()> shouldDisplay,
272 std::function<void()> doDisplay,
273 std::function<void()> processMessages,
274 std::function<std::function<void()>()> getIdleFunc,
275 std::function<bool()> shouldWait)
276{
277 while (shouldContinue()) {
278 if (shouldDisplay()) {
279 doDisplay();
280 }
281
282 if (processMessages) {
283 processMessages();
284 }
285
286 if (getIdleFunc) {
287 auto idleFunc = getIdleFunc();
288 if (idleFunc != nullptr) {
289 idleFunc();
290 glfwPollEvents();
291 } else {
292 // Use shouldWait to decide between wait and poll
293 if (shouldWait && shouldWait()) {
294 glfwWaitEvents();
295 } else {
296 glfwPollEvents();
297 }
298 }
299 } else {
300 glfwPollEvents();
301 }
302 }
