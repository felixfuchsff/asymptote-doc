177void assessClosure(lambda *body) {
178 // If we have already determined if it needs closure, just return.
179 if (body->closureReq != lambda::MAYBE_NEEDS_CLOSURE)
180 return;
181
182 for (program::label l = body->code->begin(); l != body->code->end(); ++l)
183 if (l->op == inst::pushclosure ||
184 l->op == inst::pushframe) {
185 body->closureReq = lambda::NEEDS_CLOSURE;
186 return;
187 }
188
189 body->closureReq = lambda::DOESNT_NEED_CLOSURE;
190}

227void stack::debug()
228{

230 if(indebugger) {em.clear(); return;}
231
232 switch(debugOp) {
233 case 'i': // inst
234 breakpoint();
235 break;
236 case 's': // step
237 if((!curPos.match(lastPos.filename()) || !curPos.match(lastPos.Line())))
238 breakpoint();
239 break;
240 case 'n': // next
241 if(curPos.match(lastPos.filename()) && !curPos.match(lastPos.Line()))
242 breakpoint();
243 break;
244 case 'f': // file
245 if(!curPos.match(lastPos.filename()))
246 breakpoint();
247 break;
248 case 'r': // return
249 if(curPos.match(breakPos.filename()))
250 breakpoint();
251 break;
252 case 'c': // continue
253 default:
254 for(auto p=bplist.begin(); p != bplist.end(); ++p) {
255 if(curPos.match(p->f.name()) && curPos.match(p->f.line()) &&
256 (newline || !curPos.match(breakPos.filename()) ||
257 !curPos.match(breakPos.Line()))) {
258 breakPos=curPos;
259 breakpoint(p->r);
260 newline=false;
261 break;
262 }
263 if(!newline &&
264 (curPos.match(lastPos.filename()) && !curPos.match(lastPos.Line())))
265 newline=true;
266 }
267 break;
268 }

272void stack::runWithOrWithoutClosure(lambda *l, vars_t vars, vars_t parent)
273{
274 // The size of the frame (when running without closure).
275 size_t frameSize = l->parentIndex;
276
277#ifdef SIMPLE_FRAME
278 // Link to the variables, be they in a closure or on the stack.
279 vmFrame *varlink;
280
281# define SET_VARLINK assert(vars); varlink = vars;
282# define VAR(n) ( (varlink)[(n) + frameStart] )
283# define FRAMEVAR(frame,n) (frame[(n)])
284#else
285 // Link to the variables, be they in a closure or on the stack.
286 mem::vector<item> *varlink=NULL;
287
288# define SET_VARLINK assert(vars); varlink = &vars->vars
289# define VAR(n) ( (*varlink)[(n) + frameStart] )
290# define FRAMEVAR(frame,n) ((*frame)[(n)])
291#endif
292
293 size_t frameStart = 0;
294
295 // Set up the closure, if necessary.
296 if (vars == 0)
297 {
298#ifndef SIMPLE_FRAME
299 assessClosure(l);
300 if (l->closureReq == lambda::NEEDS_CLOSURE)
301#endif
302 {
303 /* make new activation record */
304 vars = vm::make_frame(l, parent);
305 assert(vars);
306 }
307#ifndef SIMPLE_FRAME
308 else
309 {
310 assert(l->closureReq == lambda::DOESNT_NEED_CLOSURE);
311
312 // Use the stack to store variables.
313 varlink = &theStack;
314
315 // Record where the parameters start on the stack.
316 frameStart = theStack.size() - frameSize;
317
318 // Add the parent's closure to the frame.
319 push(parent);
320 ++frameSize;
321
322 size_t newFrameSize = (size_t)l->framesize;
323
324 if (newFrameSize > frameSize) {
325 theStack.resize(frameStart + newFrameSize);
326 frameSize = newFrameSize;
327 }
328 }
329#endif
330 }
331
332 if (vars) {
333 marshall(l->parentIndex, vars);
334
335 SET_VARLINK;
336 }
337
338 /* start the new function */
339 program::label ip = l->code->begin();
340 processDataStruct& P=processData();
341 position& topPos=P.topPos;
342 string& fileName=P.fileName;
343 unsigned int offset=P.xmapCount;
344
345 bool traceless=!settings::debug;
346 bool xasy=settings::xasy || offset || settings::keys;
347
348 if(xasy && curPos.filename() == fileName)
349 topPos=curPos.shift(offset);
350
351 try {
352 for (;;) {
353 const inst &i = *ip;
354 curPos = i.pos;
355
356#ifdef PROFILE
357 prof.recordInstruction();
358#endif
359
360#ifdef DEBUG_STACK
361 printInst(cout, ip, l->code->begin());
362 cout << " (";
363 i.pos.printTerse(cout);
364 cout << ")\n";
365#endif
366
367 if(settings::verbose > 4) em.trace(curPos);
368
369 if(!bplist.empty()) debug();
370
371 if(errorstream::interrupt) throw interrupted();
372
373 switch (i.op)
374 {
375 case inst::varpush:
376 push(VAR(get<Int>(i)));
377 break;
378
379 case inst::varsave:
380 VAR(get<Int>(i)) = top();
381 break;
382
383#ifdef COMBO
384 case inst::varpop:
385 VAR(get<Int>(i)) = pop();
386 break;
387#endif
388
389 case inst::ret: {
390 if (vars == 0)
391 // Delete the frame from the stack.
392 // TODO: Optimize for common cases.
393 theStack.erase(theStack.begin() + frameStart,
394 theStack.begin() + frameStart + frameSize);
395 return;
396 }
397
398 case inst::pushframe:
399 {
400 assert(vars);
401 Int size = get<Int>(i);
402 vars=make_pushframe(size, vars);
403
404 SET_VARLINK;
405
406 break;
407 }
408
409 case inst::popframe:
410 {
411 assert(vars);
412 vars=get<vmFrame *>(VAR(0));
413
414 SET_VARLINK;
415
416 break;
417 }
418
419 case inst::pushclosure:
420 assert(vars);
421 push(vars);
422 break;
423
424 case inst::nop:
425 break;
426
427 case inst::pop:
428 pop();
429 break;
430
431 case inst::intpush:
432 case inst::constpush:
433 push(i.ref);
434 break;
435
436 case inst::fieldpush: {
437 vars_t frame = pop<vars_t>();
438 if (!frame)
439 error(dereferenceNullPointer);
440 push(FRAMEVAR(frame, get<Int>(i)));
441 break;
442 }
443
444 case inst::fieldsave: {
445 vars_t frame = pop<vars_t>();
446 if (!frame)
447 error(dereferenceNullPointer);
448 FRAMEVAR(frame, get<Int>(i)) = top();
449 break;
450 }
451
452#if COMBO
453 case inst::fieldpop: {
454#error NOT REIMPLEMENTED
455 vars_t frame = pop<vars_t>();
456 if (!frame)
457 error(dereferenceNullPointer);
458 FRAMEVAR(get<Int>(i)) = pop();
459 break;
460 }
461#endif
462
463
464 case inst::builtin: {
465 bltin func = get<bltin>(i);
466#ifdef PROFILE
467 prof.beginFunction(func);
468#endif
469 func(this);
470#ifdef PROFILE
471 prof.endFunction(func);
472#endif
473 break;
474 }
475
476 case inst::jmp:
477 ip = get<program::label>(i);
478 continue;
479
480 case inst::cjmp:
481 if (pop<bool>()) { ip = get<program::label>(i); continue; }
482 break;
483
484 case inst::njmp:
485 if (!pop<bool>()) { ip = get<program::label>(i); continue; }
486 break;
487
488 case inst::jump_if_not_default:
489 if (!isdefault(pop())) { ip = get<program::label>(i); continue; }
490 break;
491
492#ifdef COMBO
493 case inst::gejmp: {
494 Int y = pop<Int>();
495 Int x = pop<Int>();
496 if (x>=y)
497 { ip = get<program::label>(i); continue; }
498 break;
499 }
500
501#if 0
502 case inst::jump_if_func_eq: {
503 callable * b=pop<callable *>();
504 callable * a=pop<callable *>();
505 if (a->compare(b))
506 { ip = get<program::label>(i); continue; }
507 break;
508 }
509
510 case inst::jump_if_func_neq: {
511 callable * b=pop<callable *>();
512 callable * a=pop<callable *>();
513 if (!a->compare(b))
514 { ip = get<program::label>(i); continue; }
515 break;
516 }
517#endif
518#endif
519
520 case inst::push_default:
521 push(Default);
522 break;
523
524 case inst::popcall: {
525 /* get the function reference off of the stack */
526 if(xasy && curPos.filename() == fileName)
527 topPos=curPos.shift(offset);
528
529 callable* f = pop<callable*>();
530 if(traceless)
531 f->call(this);
532 else {
533 em.traceback.push_back(curPos);
534 f->call(this);
535 if(em.traceback.size())
536 em.traceback.pop_back();
537 }
538 break;
539 }
540
541 case inst::makefunc: {
542 func *f = new func;
543 f->closure = pop<vars_t>();
544 f->body = get<lambda*>(i);
545
546 push((callable*)f);
547 break;
548 }
549
550 default:
551 error("Internal VM error: Bad stack operand");
552 }
553
554#ifdef DEBUG_STACK
555 draw(cerr);
556 vm::draw(cerr,vars);
557 cerr << "\n";
558#endif
559
560 ++ip;
561 }
562 } catch (bad_item_value&) {
563 error("Trying to use uninitialized value.");
564 }
565
566#undef SET_VARLINK
567#undef VAR
568#undef FRAMEVAR
569}
