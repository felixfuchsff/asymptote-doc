45 // Used for determining the type when the context does not establish
46 // the name as a variable or a type. First, the function looks for a
47 // non-function variable fitting the description. If one fits, the
48 // type of the variable is returned. Failing that, the function looks
49 // for a fitting type and returns that. If nothing is found, an
50 // appropriate error is reported and ty_error is returned.
51 // Because this is used only on qualifiers (ie. names to the left of a
52 // dot), it does not look at function variables.
53 virtual types::ty *getType(coenv &e);
54
55 // Pushes the highest level frame possible onto the stack. Returning
56 // the frame pushed. If no frame can be pushed, returns 0.
57 // NOTE: This duplicates some functionality with getVarEntry.
58 virtual trans::frame *frameTrans(coenv &e);
59 // Helper function for the case where the name is known to be a type.
60 virtual trans::frame *tyFrameTrans(coenv &e) = 0;
61
62 // Constructs the varEntry part of the tyEntry for the name. Like
63 // getType, this is called on the qualifier, instead of the full name.
64 // This reports no errors, and returns 0 if there is no varEntry to
65 // use.
66 virtual trans::varEntry *getVarEntry(coenv &e) = 0;
67
68 // As a variable:
69 // Translates the name (much like an expression).
70 virtual void varTrans(action act, coenv &e, types::ty *target) = 0;
71 // Returns the possible variable types. Unlike exp, returns 0 if none
72 // match.
73 virtual types::ty *varGetType(coenv &e) = 0;
74 virtual trans::varEntry *getCallee(coenv &e, types::signature *sig) = 0;
75
76 // As a type:
77 // Determines the type, as used in a variable declaration.
78 virtual types::ty *typeTrans(coenv &e) = 0;
79 // Constructs the tyEntry of the name, needed so that we know the
80 // parent frame for allocating new objects of that type. Reports
81 // errors unless em.isSuppressed() is true.
82 virtual trans::tyEntry *tyEntryTrans(coenv &e) = 0;
83
84 virtual void prettyprint(ostream &out, Int indent) = 0;
85 virtual void print(ostream& out) const {
86 out << "<base name>";
87 }
88 virtual void printPath(ostream& out) const { print(out); }
89 virtual symbol asPath() const {
90 ostringstream out;
91 printPath(out);
