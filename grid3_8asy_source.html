8struct grid3 {
9 path3 axea,axeb;
10 bounds bds;
11 triple dir;
12 valuetime vt;
13 ticklocate locate;
14 void create(picture pic, path3 axea, path3 axeb, path3 axelevel,
15 real min, real max, position pos, autoscaleT t) {
16 real position=pos.position.x;
17 triple level;
18 if(pos.relative) {
19 position=reltime(axelevel,position);
20 level=point(axelevel,position)-point(axelevel,0);
21 } else {
22 triple v=unit(point(axelevel,1)-point(axelevel,0));
23 triple zerolevel=dot(-point(axelevel,0),v)*v;
24 level=zerolevel+position*v;
25 }
26 this.axea=shift(level)*axea;
27 this.axeb=shift(level)*axeb;
28 bds=autoscale(min,max,t.scale);
29 locate=ticklocate(min,max,t,bds.min,bds.max,
30 Dir(point(axeb,0)-point(axea,0)));
31 }
32};

185void grid3(picture pic=currentpicture,
186 grid3routinetype gridroutine=XYZgrid,
187 int N=0, int n=0, real Step=0, real step=0,
188 bool begin=true, bool end=true,
189 pen pGrid=grey, pen pgrid=lightgrey,
190 bool above=false)
191{
192 for(int j=0; j < gridroutine.length; ++j) {
193 grid3routines gridroutinej=gridroutine[j];
194 for(int i=0; i < gridroutinej.length; ++i) {
195 grid3 gt=gridroutinej[i](pic);
196 pic.add(new void(picture f, transform3 t, transform3 T,
197 projection P, triple, triple) {
198 picture d;
199 ticks3 ticks=Ticks3(1,F="%",ticklabel=null,
200 beginlabel=false,endlabel=false,
201 N=N,n=n,Step=Step,step=step,
202 begin=begin,end=end,
203 Size=0,size=0,extend=true,
204 pTick=pGrid,ptick=pgrid);
205 ticks(d,t,"",gt.axea,gt.axeb,nullpen,None,NoMargin3,gt.locate,
206 gt.bds.divisor,opposite=true,primary=false,P);
207 add(f,t*T*inverse(t)*d);
208 },above=above);
209 addPath(pic,gt.axea,pGrid);
210 addPath(pic,gt.axeb,pGrid);
211 }
212 }
213}
