190template<class T>
191T pow(T x, Int y)
192{
193 if(y == 0) return 1.0;
194 if(x == 0.0 && y > 0) return 0.0;
195 if(y < 0) {y=-y; x=1/x;}
196
197 T r=1.0;
198 for(;;) {
199 if(y & 1) r *= x;
200 if((y >>= 1) == 0) return r;
201 x *= x;
202 }
203}

210template <>
211struct power<Int> {
212 Int operator() (Int x, Int p, size_t i=0) {
213 if(p == 0) return 1;
214 Int sign=1;
215 if(x < 0) {
216 if(p % 2) sign=-1;
217 x=-x;
218 }
219 if(p > 0) {
220 if(x == 0) return 0;
221 Int r = 1;
222 for(;;) {
223 if(p & 1) {
224 if(r > Int_MAX/x) integeroverflow(i);
225 r *= x;
226 }
227 if((p >>= 1) == 0)
228 return sign*r;
229 if(x > Int_MAX/x) integeroverflow(i);
230 x *= x;
231 }
232 } else {
233 if(x == 1) return sign;
234 ostringstream buf;
235 if(i > 0) buf << "array element " << i << ": ";
236 buf << "Only 1 and -1 can be raised to negative exponents as integers.";
237 vm::error(buf);
238 return 0;
239 }
