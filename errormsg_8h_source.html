41 // The filename without the directory and without the '.asy' suffix.
42 // Note that this assumes name are separated by a forward slash.
43 string moduleName() const {
44 size_t start = filename.rfind('/');
45 if (start == filename.npos)
46 start = 0;
47 else
48 // Step over slash.
49 ++start;
50
51 size_t end = filename.rfind(".asy");
52 if (end != filename.size() - 4)
53 end = filename.size();
54
55 return filename.substr(start, end-start);
56 }
