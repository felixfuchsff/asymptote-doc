1/*****
2 * rendererloader.cc
3 * On Unix: Probe for Vulkan and OpenGL availability at runtime via dlopen.
4 * On Windows: Vulkan is linked against vulkan-1.dll; the renderer is
5 * directly instantiated. If no hardware GPU is detected (e.g., pre-2012
6 * hardware, VirtualBox VM), a llvmpipe fallback is activated by writing
7 * an ICD manifest (lvp_icd.json) and loading vulkan_lvp.dll (Lavapipe).
8 * On macOS (Intel x86_64): If Metal is unavailable (e.g., headless VM),
9 * fall back to a shipped llvmpipe driver by locating libvulkan_lvp.dylib
10 * via the Asymptote search path and writing an ICD manifest.
11 * Apple Silicon (arm64) is not supported by llvmpipe; use SwiftShader instead.
12 *
13 * The main asy binary has zero link-time dependencies on Vulkan or OpenGL
14 * on Unix. All Vulkan-specific code lives in libasyvulkan.so, loaded at runtime.
15 * All OpenGL-specific code lives in libasyopengl.so, loaded at runtime.
16 *
17 * On Windows, Vulkan is linked directly into the asy binary.
18 *
19 * For WebGL (html) and v3d output, AsyWebGLRender is used instead, which
20 * requires no GPU libraries - it only sets up state for client-side rendering.
21 *****/

105/**
106 * Attempt to load libasyvulkan.so and create a Vulkan renderer.
107 * Returns true on success (gl is set to the new AsyVkRender, vulkan=true).
108 * Returns false on failure (gl remains nullptr, vulkan=false).
109 */
110static bool tryLoadVulkanLib()
111{
112#ifdef _WIN32
113 // On Windows, try LoadLibrary.
114 vulkanLibHandle = LoadLibraryA("libasyvulkan.dll");
115 if (!vulkanLibHandle) {
116 if (settings::verbose > 1)
117 std::cout << "Failed to load libasyvulkan.dll; falling back to OpenGL"
118 << std::endl;
119 return false;
120 }
121
122 // Get the factory function.
123 typedef void *(*CreateAsyVkRenderFn)();
124 CreateAsyVkRenderFn fn =
125 reinterpret_cast<CreateAsyVkRenderFn>(GetProcAddress(vulkanLibHandle, "createAsyVkRender"));
126#else
127 mem::string locPath = settings::locateFile("libasyvulkan.so", true, "");
128 std::string pathStr = mem::stdString(locPath);
129
130 vulkanLibHandle = dlopen(pathStr.c_str(), RTLD_NOW | RTLD_LOCAL);
131
132 if (!vulkanLibHandle) {
133 if (settings::verbose > 1)
134 std::cout << "Failed to load libasyvulkan.so ("
135 << dlerror() << "); falling back to OpenGL" << std::endl;
136 return false;
137 }
138
139 if (settings::verbose > 2)
140 std::cout << "Loaded libasyvulkan.so from: " << pathStr << std::endl;
141
142 // Get the factory function.
143 typedef void *(*CreateAsyVkRenderFn)();
144 CreateAsyVkRenderFn fn =
145 reinterpret_cast<CreateAsyVkRenderFn>(dlsym(vulkanLibHandle, "createAsyVkRender"));
146#endif
147
148 if (!fn) {
149 if (settings::verbose > 1)
150 std::cout << "Failed to find createAsyVkRender in libasyvulkan.so"
151 << std::endl;
152 return false;
153 }
154
155 // Create the Vulkan renderer via the factory function.
156 void *vkRenderer = fn();
157 if (!vkRenderer) {
158 if (settings::verbose > 1)
159 std::cout << "createAsyVkRender() returned NULL" << std::endl;
160 return false;
161 }
162
163 gl = static_cast<camp::AsyRender*>(vkRenderer);
164
165#ifdef HAVE_PTHREAD
166 if (gl)
167 gl->threadMgr.mainthread = pthread_self();
168#endif
169
170 signalRendererReady();
171 vulkan = true;
172 return true;
173}

318 * This does NOT require Vulkan or OpenGL libraries - it only sets up state

350/**
351 * Create the renderer object by probing for Vulkan/OpenGL availability.
352 * Called lazily from initRenderer() when shipout3() first needs GPU rendering.
353 * On Unix this loads the appropriate shared library (libasyvulkan.so or
354 * libasyopengl.so) via dlopen. The render thread's glrenderWrapper() is
355 * safe with a null gl pointer, so there is no need to pre-initialise the
356 * renderer in main().
357 *
358 * On Windows: Vulkan is statically linked; directly instantiate AsyVkRender.
359 * On Unix: Try to load the requested renderer via dlopen/LoadLibrary.
360 * If the user requested Vulkan (-vulkan, which is the default), attempt
361 * to load libasyvulkan.so first. If that fails or the user requested
362 * OpenGL (-novulkan), load libasyopengl.so as a fallback.
363 */
364void createRenderer()
365{
366 if (gl != nullptr)
367 return; // Already created
368
369 bool useVulkan = settings::getSetting<bool>("vulkan");
370
371#ifdef _WIN32
372 // On Windows, Vulkan is linked against vulkan-1.dll. Initialize the
373 // Vulkan-Hpp dynamic dispatcher before creating the renderer.
374 VULKAN_HPP_DEFAULT_DISPATCHER.init(vkGetInstanceProcAddr);
375
376 // Probe for a hardware GPU. If none is available (e.g., pre-2012
377 // hardware, VirtualBox VM), fall back to llvmpipe via Lavapipe.
378 bool hasHardwareGPU = false;
379 try {
380 VkInstanceCreateInfo info = {};
381 info.sType = VK_STRUCTURE_TYPE_INSTANCE_CREATE_INFO;
382
383 VkInstance tmpInst;
384 if (vkCreateInstance(&info, nullptr, &tmpInst) == VK_SUCCESS) {
385 uint32_t devCount = 0;
386 vkEnumeratePhysicalDevices(tmpInst, &devCount, nullptr);
387 if (devCount > 0) {
388 std::vector<VkPhysicalDevice> devices(devCount);
389 vkEnumeratePhysicalDevices(tmpInst, &devCount, devices.data());
390 for (uint32_t i = 0; i < devCount; ++i) {
391 VkPhysicalDeviceProperties props;
392 vkGetPhysicalDeviceProperties(devices[i], &props);
393 if (props.deviceType == VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU ||
394 props.deviceType == VK_PHYSICAL_DEVICE_TYPE_INTEGRATED_GPU ||
395 props.deviceType == VK_PHYSICAL_DEVICE_TYPE_VIRTUAL_GPU) {
396 hasHardwareGPU = true;
397 break;
398 }
399 }
400 }
401 vkDestroyInstance(tmpInst, nullptr);
402 }
403 } catch (...) {
404 // If probing fails for any reason, proceed without the check.
405 }
406
407 if (!hasHardwareGPU) {
408 // No hardware GPU found -- set up llvmpipe (Lavapipe) fallback.
409 // Strategy: write lvp_icd.json next to the executable and set
410 // VK_ICD_FILENAMES so the Vulkan loader picks it up on the next
411 // instance creation.
412
413 // 1) Determine the directory of our own executable.
414 std::string exeDir;
415 {
416 char buf[MAX_PATH];
417 DWORD len = GetModuleFileNameA(nullptr, buf, MAX_PATH);
418 if (len > 0 && len < MAX_PATH) {
419 std::string exePath(buf);
420 size_t slash = exePath.find_last_of("\\/");
421 if (slash != std::string::npos)
422 exeDir = exePath.substr(0, slash + 1);
423 }
424 }
425
426 // 2) Write lvp_icd.json next to the executable.
427 std::string icdName = "lvp_icd.json";
428 std::string icdPath;
429 if (!exeDir.empty())
430 icdPath = exeDir + icdName;
431 else
432 icdPath = icdName;
433
434 {
435 std::ofstream ofs(icdPath.c_str(), std::ios::trunc);
436 if (ofs.is_open()) {
437 ofs << "{\n"
438 << " \"file_format_version\": \"1.0.0\",\n"
439 << " \"ICD\": {\n"
440 << " \"library_path\": \".\\\\vulkan_lvp.dll\",\n"
441 << " \"api_version\": \"1.3.0\"\n"
442 << " }\n"
443 << "}\n";
444 if (settings::verbose > 1)
445 std::cout << "Wrote Lavapipe ICD manifest: " << icdPath
446 << std::endl;
447 }
448 }
449
450 // 3) Set VK_ICD_FILENAMES so the loader finds our manifest.
451 _putenv_s("VK_ICD_FILENAMES", icdPath.c_str());
452
453 // 4) Load vulkan_lvp.dll (the Lavapipe driver).
454 std::string lvpDllPath;
455 if (!exeDir.empty())
456 lvpDllPath = exeDir + "vulkan_lvp.dll";
457 else
458 lvpDllPath = "vulkan_lvp.dll";
459
460 lvpLibHandle = LoadLibraryA(lvpDllPath.c_str());
461 if (lvpLibHandle) {
462 if (settings::verbose > 1)
463 std::cout << "Loaded llvmpipe fallback: " << lvpDllPath
464 << std::endl;
465 } else {
466 if (settings::verbose > 1)
467 std::cout << "Warning: failed to load " << lvpDllPath
468 << "; proceeding without llvmpipe" << std::endl;
469 }
470
471 // 5) Re-initialize the Vulkan dispatcher so it picks up the new ICD.
472 VULKAN_HPP_DEFAULT_DISPATCHER.init(vkGetInstanceProcAddr);
473 }
474
475 // Directly instantiate the Vulkan renderer.
476 try {
477 gl = new camp::AsyVkRender();
478#ifdef HAVE_PTHREAD
479 if (gl)
480 gl->threadMgr.mainthread = pthread_self();
481#endif
482 signalRendererReady();
483 vulkan = true;
484 if (settings::verbose > 1)
485 std::cout << "Using Vulkan renderer" << std::endl;
486 } catch (const std::exception &e) {
487 // Vulkan renderer failed to initialize (e.g., no display, no GPU).
488 // Leave gl as nullptr so initRenderer() reports the error.
489 std::cerr << "Vulkan renderer initialization failed: " << e.what()
490 << std::endl;
491 } catch (...) {
492 std::cerr << "Vulkan renderer initialization failed (unknown error)"
493 << std::endl;
494 }
495#else
496#if defined(__APPLE__) && defined(__x86_64__)
497 // On Intel Macs, detect Metal availability BEFORE any Vulkan calls.
498 // MoltenVK's ICD is found by the Vulkan loader, but if Metal is not
499 // available (e.g., headless VM), vkCreateInstance fails with
500 // ErrorIncompatibleDriver before we can enumerate devices.
501 // Solution: probe Metal directly via dlopen/dlsym, and if unavailable,
502 // use a shipped llvmpipe driver found via the Asymptote search path.
503 {
504 // Step 1: Open Metal.framework
505 void *metalHandle = dlopen("/System/Library/Frameworks/Metal.framework/Metal",
506 RTLD_NOW | RTLD_LOCAL);
507
508 bool metalAvailable = false;
509 if (metalHandle) {
510 // Step 2: Resolve MTLCreateSystemDefaultDevice
511 typedef void *(*PFN_MTLCreateSystemDefaultDevice)();
512 PFN_MTLCreateSystemDefaultDevice createDevice =
513 reinterpret_cast<PFN_MTLCreateSystemDefaultDevice>(
514 dlsym(metalHandle, "MTLCreateSystemDefaultDevice"));
515
516 if (createDevice) {
517 // Step 3: Call it - nil means Metal is not available
518 void *device = createDevice();
519 if (device != nullptr) {
520 metalAvailable = true;
521 }
522 }
523 dlclose(metalHandle);
524 }
525
526 // Step 4: If Metal is NOT available and Vulkan was requested,
527 // use the shipped llvmpipe driver from the Asymptote search path.
528 if (!metalAvailable && useVulkan) {
529 // Locate the shipped llvmpipe ICD JSON via the Asymptote path.
530 mem::string icdJsonPath = settings::locateFile("lvp_icd.x86_64.json", true, "");
531 std::string icdJsonStr = mem::stdString(icdJsonPath);
532
533 if (!icdJsonStr.empty()) {
534 // Convert to absolute path (the Vulkan loader needs it).
535 char *absPath = realpath(icdJsonStr.c_str(), nullptr);
536 if (absPath != nullptr) {
537 icdJsonStr = absPath;
538 free(absPath);
539 }
540
541 setenv("VK_ICD_FILENAMES", icdJsonStr.c_str(), true);
542 headlessRenderer = true;
543
544 if (settings::verbose > 1)
545 std::cout << "Metal unavailable on macOS Intel; "
546 << "using llvmpipe fallback (" << icdJsonStr << ")" << std::endl;
547 } else {
548 if (settings::verbose > 1)
549 std::cout << "Metal unavailable and llvmpipe ICD not found" << std::endl;
550 }
551 }
552 }
553#endif // __APPLE__ && __x86_64__
554
555 // If user wants Vulkan, try to load the shared library first.
556 if (useVulkan) {
557 if (tryLoadVulkanLib()) {
558#ifdef __APPLE__
559 // GLFW lazily calls dlopen("libvulkan.1.dylib") inside
560 // glfwCreateWindowSurface to get vkGetInstanceProcAddr.
561 // In a portable/staged build the bundled Vulkan loader's
562 // install name is "@executable_path/lib/libvulkan.1.dylib",
563 // so that bare-name dlopen fails on a machine without the SDK.
564 // Fix: hand GLFW the function pointer from our already-loaded
565 // Vulkan loader so it never needs to dlopen it itself.
566 {
567 PFN_vkGetInstanceProcAddr pfn =
568 (PFN_vkGetInstanceProcAddr)dlsym(
569 vulkanLibHandle, "vkGetInstanceProcAddr");
570 if (pfn)
571 glfwInitVulkanLoader(pfn);
572 }
573#endif // __APPLE__
574 return;
575 }
576 // Vulkan failed to load; fall through to try OpenGL.
577 if (settings::verbose > 1)
578 std::cout << "Vulkan unavailable, falling back to OpenGL" << std::endl;
579 }
580
581 // Try to load the OpenGL renderer.
582 if (tryLoadOpenGLLib()) {
583 return;
584 }
585#endif // _WIN32
586
587 // On Unix: both renderers failed to load. Leave gl as nullptr; the error will be
588 // reported lazily in initRenderer() when 3D rendering is actually requested.
589#ifndef _WIN32
590 vulkan = false;
591#endif
592}

594/**
595 * Initialise the renderer, optionally selecting based on output format.
596 *
597 * For WebGL (html) and v3d formats, creates AsyWebGLRender which requires
598 * no GPU libraries - it only sets up state for client-side rendering.
599 *
600 * For all other formats, lazily calls createRenderer() to probe for Vulkan/
601 * OpenGL and instantiate the appropriate renderer. This defers all GPU
602 * library loading until a shipout3() call actually requires rendering.
603 *
604 * @param format Output format string (e.g., "html", "v3d", or empty/NULL for default)
605 */
606void initRenderer(const char* format)
607{
608 // For WebGL and v3d output, use the lightweight AsyWebGLRender
609 // which doesn't require Vulkan or OpenGL libraries
610 bool isFormat3D = (format != nullptr &&
611 (strcmp(format, "html") == 0 || strcmp(format, "v3d") == 0));
612
613 // If we have a WebGL renderer but now need GPU rendering (or vice versa),
614 // reset and re-initialize with the appropriate renderer
615 if (initializedRenderer) {
616 bool currentIsWebGL = dynamic_cast<AsyWebGLRender*>(gl) != nullptr;
617 if (currentIsWebGL != isFormat3D) {
618 initializedRenderer = false;
619 // Clear the old renderer so a new one gets created below.
620 // We don't delete it to avoid triggering cleanup code
621 // (e.g., glslang::FinalizeProcess()) that can cause issues.
622 if (currentIsWebGL)
623 gl = nullptr;
624 }
625 }
626
627 if (initializedRenderer)
628 return; // Already fully initialised for this format type
629
630 if (isFormat3D) {
631 createWebGLRenderer();
632 } else if (gl == nullptr) {
633 // Lazy initialisation: probe for Vulkan/OpenGL only when GPU
634 // rendering is actually needed.
635#ifdef HAVE_PTHREAD
636 // In threaded mode, delegate renderer creation to the glrenderWrapper
637 // thread (the OS main thread) to avoid a race condition caused by
638 // dlopen from a secondary thread.
639 // since gl is still nullptr and we can't use gl->threadMgr yet.
640 if(AsyRender::threads) {
641 pthread_mutex_lock(&main_wait_mutex);
642 pthread_cond_signal(&main_wait_cond); // Wake glrenderWrapper
643 pthread_cond_wait(&main_wait_cond, &main_wait_mutex); // Wait for done
644 pthread_mutex_unlock(&main_wait_mutex);
645 } else
646#endif
647 createRenderer();
648 }
649
650 if (gl == nullptr) {
651 camp::reportError("No 3D rendering available");
652 }
653
654 initializedRenderer = true;
655
656 if (settings::verbose > 2) {
657 if (isFormat3D && format)
658 std::cout << "Using WebGL renderer for " << format << " output" << std::endl;
659 }
660}
