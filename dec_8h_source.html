673// Abstract base class for
674// from _ access _; (fromaccessdec)
675// and
676// from _ unravel _; (unraveldec)
677class fromdec : public dec {
678protected:
679 struct qualifier {
680 // The varEntry holds the location and the type of the highest framed
681 // structure that can be put on the stack. The record holds the actual
682 // type of the qualifier.
683 // For example:
684 // struct A {
685 // struct B {
686 // static int x;
687 // }
688 // }
689 // A a=new A;
690 // from a.B unravel x;
691 //
692 // Here, v->getType() will yield A and v->getLocation() will yield the
693 // location of the variable a, but the record type t will be B.
694 record *t;
695 varEntry *v;
696
697 qualifier(record *t, varEntry *v)
698 : t(t), v(v) {}
699 };
700
701 // Return the qualifier from which the fields are taken. If t==0, it is
702 // assumed that an error occurred and was reported.
703 virtual qualifier getQualifier(coenv &e, record *r) = 0;
704 idpairlist *fields;
705
706public:
707 fromdec(position pos, idpairlist *fields)
708 : dec(pos), fields(fields) {}
709
710 void prettyprint(ostream &out, Int indent) override;
711
712 void transAsField(coenv &e, record *r) override;
713};
