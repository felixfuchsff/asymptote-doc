217 if (flushResult != 0)

222#if defined(_WIN32)

226 // DANGER: There's a severe but rare issue with certain systems
227 // involving a potential memory leak.
228
229 // See https://github.com/Kreijstal/fmem/issues/6
230
231 // Right now, we have no reasonable way to determine if a tmpfile
232 // implementation is being used, so we cannot have a way to
233 // conditionally free the memory.
234
235 // On most systems, we have the open_memstream and Windows tmpfile API,
236 // where the allocation/mapping is handled by the system; hence
237 // there is no need to free the pointer ourselves.
238 // But the tmpfile implementation uses malloc that doesn't
239 // get freed, so it is our job to manually free it.
240 fmem_mem(&fmInstance, &streamPtr, &retSize);
241
242 if (streamPtr == nullptr)
243 {
244 return {};
245 }
246
247 auto* bytePtr = static_cast<uint8_t*>(streamPtr);
248 std::vector ret(bytePtr, bytePtr + retSize);
249 return ret;
250#else
251 // for sanity check, always ensure that we have a vector of bytes
252 static_assert(sizeof(char) == sizeof(uint8_t));
253
254 if (buffer == nullptr)
255 {
256 return {};
257 }

259 auto* retPtr = reinterpret_cast<uint8_t*>(buffer);
260 return {retPtr, retPtr + size};
