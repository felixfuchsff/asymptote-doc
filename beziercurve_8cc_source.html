59// Use a uniform partition to draw a Bezier curve.
60// p is an array of 4 triples representing the control points.
61// Ii are the vertex indices.
62void BezierCurve::render(const triple *p, uint32_t I0, uint32_t I1)
63{
64 triple p0=p[0];
65 triple p1=p[1];
66 triple p2=p[2];
67 triple p3=p[3];
68 if(Straightness(p0,p1,p2,p3) < res2) { // Segment is flat
69 triple P[]={p0,p3};
70 if(!offscreen(2,P)) {
71 data.indices.push_back(I0);
72 data.indices.push_back(I1);
73 }
74 } else { // Segment is not flat
75 if(offscreen(4,p)) return;
76 triple m0=0.5*(p0+p1);
77 triple m1=0.5*(p1+p2);
78 triple m2=0.5*(p2+p3);
79 triple m3=0.5*(m0+m1);
80 triple m4=0.5*(m1+m2);
81 triple m5=0.5*(m3+m4);
82
83 triple s0[]={p0,m0,m3,m5};
84 triple s1[]={m5,m4,m2,p3};
85
86 triple n0=normal(bezierPh(p0,p1,p2,p3),bezierPPh(p0,p1,p2,p3));
87 size_t i0=data.addVertex(MaterialVertex{m5,n0,MaterialIndex});
88
89 render(s0,I0,i0);
90 render(s1,i0,I1);
91 }
92}
