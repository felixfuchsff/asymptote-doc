268 void operator init(pen p) {
269 real[] c=colors(rgb(p));
270 real r=c[0];
271 real g=c[1];
272 real b=c[2];
273 real M=max(r,g,b);
274 real m=min(r,g,b);
275 if(M == m) this.h=0;
276 else {
277 real denom=1/(M-m);
278 if(M == r) {
279 this.h=60*(g-b)*denom;
280 if(g < b) h += 360;
281 } else if(M == g) {
282 this.h=60*(b-r)*denom+120;
283 } else
284 this.h=60*(r-g)*denom+240;
285 }
286 this.s=M == 0 ? 0 : 1-m/M;
287 this.v=M;
288 }
289 // return an rgb pen corresponding to h in [0,360) and s and v in [0,1].
290 pen rgb() {

353// opacity.
354pen mean(pen[] p, real opacity(real[])=min)
355{
356 if(p.length == 0) return nullpen;
357 real[] a=rgba(p[0]);
358 real[] t=new real[p.length];
359 t[0]=a[3];
360 for(int i=1; i < p.length; ++i) {
361 real[] b=rgba(p[i]);
362 a += b;
363 t[i]=b[3];
364 }
365 a /= p.length;
366 return rgb(a[0],a[1],a[2])+opacity(opacity(t));
367}
