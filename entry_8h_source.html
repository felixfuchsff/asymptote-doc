132// As looked-up types can be allocated in a new expression, we need to know
133// what frame they should be allocated on. Type entries store this extra
134// information along with the type.
135class tyEntry : public entry {
136public:
137 ty *t;
138 varEntry *v; // NOTE: Name isn't very descriptive.
139
140 tyEntry(ty *t, varEntry *v, record *where, position pos)
141 : entry(where, pos), t(t), v(v) {}
142
143 tyEntry(tyEntry *base, permission perm, record *r)
144 : entry(*base, perm, r), t(base->t), v(base->v) {}
145
146 // Records need a varEntry that refers back to the qualifier qv; i.e. in
147 // the last new of the code
148 // struct A {
149 // struct B {}
150 // }
151 // A a=new A;
152 // unravel a;
153 // new B;
154 // we need to put a's frame on the stack before allocating an instance of B.
155 // NOTE: A possible optimization could be to only qualify the varEntry if
156 // the type is a record, as other types don't use the varEntry.
157private:

159 : entry(*base, *qv), t(base->t), v(qualifyVarEntry(qv, base->v)) {}

161public:

307 // Features for iterating over the entire table.
308 class const_iterator {
309 const core_venv& core;
310 size_t index;
311
312 public:
313 const_iterator(const core_venv& core, size_t index)
314 : core(core), index(index) {}
315
316 const cell& operator * () const {
317 return core.cellByIndex(index);
318 }
319 const cell* operator -> () const {
320 return &core.cellByIndex(index);
321 }
322
323 const_iterator& operator ++ () {
324 // Advance to the next filled cell, or stop at the end of the array.
325 do {
326 ++index;
327 } while (!(*this)->filled() && index < core.capacity);
328
329 DEBUG_CACHE_ASSERT((*this)->filled() || (*this) == core.end());
330
331 return *this;
332 }
333
334 friend bool operator == (const const_iterator& a, const const_iterator& b)
335 {
336 // For speed, we don't compare the hashtables.
337 return a.index == b.index;
338 }
339 friend bool operator != (const const_iterator& a, const const_iterator& b)
340 {
341 // For speed, we don't compare the hashtables.
342 return a.index != b.index;
343 }
344 };
345
346 const_iterator begin() const {
347 size_t index = 0;
348 while (index < capacity && !cellByIndex(index).filled())
349 ++index;
350 return const_iterator(*this, index);
351 }
352
353 const_iterator end() const {
354 return const_iterator(*this, capacity);
355 }
