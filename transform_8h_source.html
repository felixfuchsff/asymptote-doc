5 * The transform datatype stores an affine transformation on the plane
6 * The datamembers are x, y, xx, xy, yx, and yy. A pair (x,y) is
7 * transformed as
8 * x' = t.x + t.xx * x + t.xy * y
9 * y' = t.y + t.yx * x + t.yy * y

128 friend transform inverse(const transform& t)
129 {
130 double d = det(t);
131 if (d == 0.0)
132 reportError("inverting singular transform");
133
134 d=1.0/d;
135 return transform((t.xy * t.y - t.yy * t.x)*d,
136 (t.yx * t.x - t.xx * t.y)*d,
137 t.yy*d, -t.xy*d, -t.yx*d, t.xx*d);
138 }

197// return rotate(angle(v)) if z != (0,0); otherwise return identity.
198inline transform rotate(pair z)
199{
200 double d=z.length();
201 if(d == 0.0) return identity;
202 d=1.0/d;
203 return transform (0.0, 0.0, d*z.getx(), -d*z.gety(), d*z.gety(), d*z.getx());
204}

206inline transform rotatearound(pair z, double theta)
207{
208 // Notice the operators are applied from right to left.
209 // Could be optimized.
210 return shift(z) * rotate(theta) * shift(-z);
211}

213inline transform reflectabout(pair z, pair w)
214{
215 if (z == w)
216 reportError("points determining line to reflect about must be distinct");
217
218 // Also could be optimized.
219 transform basis = shift(z) * scale(w-z);
220 transform flip = yscale(-1.0);
221
222 return basis * flip * inverse(basis);
223}

225// Return the rotational part of t.
226inline transform rotation(transform t)
227{
228 pair z(2.0*t.getxx()*t.getyy(),t.getyx()*t.getyy()-t.getxx()*t.getxy());
229 if(t.getxx() < 0) z=-z;
230 return rotate(atan2(z.gety(),z.getx()));
231}
