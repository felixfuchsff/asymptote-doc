113 def build_args_for_gcc(
114 self,
115 src_file: str,
116 out_file: Optional[str],
117 addr_flags: Optional[List[str]] = None,
118 ):
119 base_args = (
120 [f"-I{include_dir}" for include_dir in self._include_dirs]
121 + [f"-D{macro}" for macro in self._macros]
122 + [f"-std=c++{self._standard}"]
123 + self._extra_flags
124 )
125
126 if addr_flags:
127 base_args.extend(addr_flags)
128
129 if out_file:
130 base_args.extend(["-o", out_file])

132 base_args.append(src_file)

190def escape_windows_path(raw_path: str) -> str:
191 escape_chars = {" ", "$", "#"}
192 with io.StringIO() as ret_str_io:
193 for char in raw_path:
194 if char in escape_chars:
195 ret_str_io.write("\\")
196 ret_str_io.write(char)
197 return ret_str_io.getvalue()

200def compile_for_preproc_and_depfile_msvc(
201 compile_opt: CompileOptions, src_in: str, preproc_out: str, depfile_out: str
202):
203 with tempfile.TemporaryDirectory() as td:
204 args = [compile_opt.compiler] + compile_opt.build_args_for_msvc(
205 src_in,
206 None,
207 [
208 "/DNOSYM",
209 "/DDEPEND",
210 "/P",
211 f"/Fi{preproc_out}",
212 "/sourceDependencies",
213 f"{td}/srcdep.json",
214 ],
215 )
216 execute_and_report_err(args, "MSVC Error")
217 with open(f"{td}/srcdep.json", "r", encoding="utf-8") as fread:
218 dep_data = json.load(fread)
219
220 include_fil_str = " ".join(
221 escape_windows_path(include_fil)
222 for include_fil in dep_data["Data"].get("Includes", [])
223 )
224
225 with open(depfile_out, "w", encoding="utf-8") as depfile_writer:
226 depfile_writer.write(escape_windows_path(preproc_out))
227 depfile_writer.write(": ")
228 depfile_writer.write(include_fil_str)
