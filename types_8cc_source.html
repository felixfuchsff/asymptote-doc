78#define FIELD(Type, name, func) \
79 if (sig == 0 && id == name) { \
80 static trans::virtualFieldAccess a(run::func); \
81 static trans::varEntry v(Type(), &a, 0, nullPos); \
82 return &v; \

85#define RWFIELD(Type, name, getter, setter) \
86 if (sig == 0 && id == name) { \
87 static trans::virtualFieldAccess a(run::getter, run::setter); \
88 static trans::varEntry v(Type(), &a, 0, nullPos); \
89 return &v; \
90 }

92#define SIGFIELD(Type, name, func) \
93 if (id == name && \
94 equivalent(sig, Type()->getSignature())) \
95 { \
96 static trans::virtualFieldAccess a(run::func, 0, run::func##Helper); \
97 static trans::varEntry v(Type(), &a, 0, nullPos); \
98 return &v; \
99 }
100
101#define DSIGFIELD(name, sym, func) \
102 if (id == sym && \
103 equivalent(sig, name##Type()->getSignature())) \
104 { \
105 static trans::virtualFieldAccess a(run::func, 0, run::func##Helper); \
106 /* for some fields, v needs to be dynamic */ \
107 /* e.g. when the function type depends on an array type. */ \
108 trans::varEntry *v = \
109 new trans::varEntry(name##Type(), &a, 0, nullPos); \

137trans::varEntry *primitiveTy::virtualField(symbol id, signature *sig)
138{
139 switch (kind) {
140 case ty_string:
141 SIGFIELD(hashMethodType,SYM(hash),stringHash);
142 break;
143 case ty_Int:
144 SIGFIELD(hashMethodType,SYM(hash),intHash);
145 break;
146 case ty_real:
147 SIGFIELD(hashMethodType,SYM(hash),realHash);
148 break;
149 case ty_pair:
150 FIELD(primReal,SYM(x),pairXPart);
151 FIELD(primReal,SYM(y),pairYPart);
152 break;
153 case ty_triple:
154 FIELD(primReal,SYM(x),tripleXPart);
155 FIELD(primReal,SYM(y),tripleYPart);
156 FIELD(primReal,SYM(z),tripleZPart);
157 break;
158 case ty_transform:
159 FIELD(primReal,SYM(x),transformXPart);
160 FIELD(primReal,SYM(y),transformYPart);
161 FIELD(primReal,SYM(xx),transformXXPart);
162 FIELD(primReal,SYM(xy),transformXYPart);
163 FIELD(primReal,SYM(yx),transformYXPart);
164 FIELD(primReal,SYM(yy),transformYYPart);
165 break;
166 case ty_tensionSpecifier:
167 FIELD(primReal,SYM(out),tensionSpecifierOutPart);
168 FIELD(primReal,SYM(in),tensionSpecifierInPart);
169 FIELD(primBoolean,SYM(atLeast),tensionSpecifierAtleastPart);
170 break;
171 case ty_curlSpecifier:
172 FIELD(primReal,SYM(value),curlSpecifierValuePart);
173 FIELD(primInt,SYM(side),curlSpecifierSidePart);
174 break;
175 case ty_file:
176 FIELD(primString,SYM(name),namePart);
177 FIELD(primString,SYM(mode),modePart);
178 FILEFIELD(IntArray,dimensionType,dimension,SYM(dimension));
179 FILEFIELD(primBoolean,modeType,line,SYM(line));
180 FILEFIELD(primBoolean,modeType,csv,SYM(csv));
181 FILEFIELD(primBoolean,modeType,word,SYM(word));
182 FILEFIELD(primBoolean,modeType,singlereal,SYM(singlereal));
183 FILEFIELD(primBoolean,modeType,singleint,SYM(singleint));
184 FILEFIELD(primBoolean,modeType,signedint,SYM(signedint));
185 SIGFIELD(readType,SYM(read),readSet);
186 break;
187 default:
188 break;
189 }
190 return 0;

218ty *primitiveTy::virtualFieldGetType(symbol id)
219{
220 if (id == SYM(hash)) {
221 if (kind == ty_string || kind == ty_Int || kind == ty_real) {
222 return hashMethodType();
223 }
224 }
225
226 if(kind == ty_file) {
227 if (id == SYM(dimension))
228 return overloadedDimensionType();
229
230 if (id == SYM(line) || id == SYM(csv) ||
231 id == SYM(word) || id == SYM(singlereal) ||
232 id == SYM(singleint) || id == SYM(signedint))
233 return overloadedModeType();
234
235 if (id == SYM(read))
236 return readType();
237 }
238
239 trans::varEntry *v = virtualField(id, 0);
240
241 return v ? v->getType() : 0;
242}

449// Equivalence by design does not look at the presence of default values.
450bool equivalent(const signature *s1, const signature *s2)
451{
452 if (s1 == s2)
453 return true;
454
455 // Handle null signature
456 if (s1 == 0 || s2 == 0)
457 return false;
458
459 // Two open signatures are always equivalent, as the formals are ignored.
460 if (s1->isOpen)
461 return s2->isOpen;
462 else if (s2->isOpen)
463 return false;
464
465 if (s1->formals.size() != s2->formals.size()) {
466 return false;
467 }
468
469 if (!std::equal(s1->formals.begin(),s1->formals.end(),s2->formals.begin(),
470 (bool (*)(const formal&,const formal&)) equivalent))
471 return false;
472
473 if (s1->rest.t)
474 return s2->rest.t && equivalent(s1->rest, s2->rest);
475 else
476 return s2->rest.t == 0;
477}

479bool argumentEquivalent(const signature *s1, const signature *s2)
480{
481 // Handle null signature
482 if (s1 == 0)
483 return s2 == 0;
484 else if (s2 == 0)
485 return false;
486
487 if (s1->formals.size() != s2->formals.size())
488 return false;
489
490 return std::equal(s1->formals.begin(),s1->formals.end(),s2->formals.begin(),
491 (bool (*)(const formal&,const formal&))
492 argumentEquivalent) &&
493 argumentEquivalent(s1->rest, s2->rest);
494}

542void overloaded::addDistinct(ty *t, bool special)
543{
544 if (t->kind == ty_overloaded) {
545 overloaded *ot = (overloaded *)t;
546 for (ty_vector::iterator st = ot->sub.begin();
547 st != ot->sub.end();

557 {
558 if (equivalent(t, *st, special))

561
562 // Nonequivalent in signature - add it.
563 this->add(t);
564 }

585bool equivalent(const ty *t1, const ty *t2)
586{
587 // The same pointer must point to the same type.
588 if (t1 == t2)
589 return true;
590
591 // Ensure if an overloaded type is compared to a non-overloaded one, that the
592 // overloaded type's method is called.
593 if (t2->kind == ty_overloaded)
594 return t2->equiv(t1);
595 if (t1->kind == ty_overloaded)
596 return t1->equiv(t2);
597
598 // Outside of overloaded types, different kinds mean different types.
599 if (t1->kind != t2->kind)
600 return false;
601
602 return t1->equiv(t2);
603}
