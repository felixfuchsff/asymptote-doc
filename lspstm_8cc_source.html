83DEC_CREATE_SYM_MAP_FUNCTION_STM(whileStm)
84{
85
86 // while (<xyz>) { <body> }
87 // the <xyz> part belongs in the main context as the while statement,
88 // as it cannot declare new variables and only knows the symbols from that
89 // context.
90
91 test->createSymMap(symContext);
92
93 // for the body part, { <body> } are encapsulated in
94 // the blockStm, while <body> are direct statements.
95 // If the while block does not use { <body> }, then the body
96 // can be considered the same context as it cannot declare new variables and
97 // again, can only uses the variable already known before this while
98 // statement.
99
100 body->createSymMap(symContext);
