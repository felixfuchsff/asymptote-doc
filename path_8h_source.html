202 inline double norm(const pair& z0, const pair& c0, const pair& c1,
203 const pair& z1) const {
204 return Fuzz2*camp::max((c0-z0).abs2(),
205 camp::max((c1-z0).abs2(),(z1-z0).abs2()));

208 pair predir(Int t, bool normalize=true) const {
209 if(!cycles && t <= 0) return pair(0,0);
210 pair z1=point(t);
211 pair c1=precontrol(t);
212 pair dir=3.0*(z1-c1);
213 if(!normalize) return dir;
214 pair z0=point(t-1);
215 pair c0=postcontrol(t-1);
216 double epsilon=norm(z0,c0,c1,z1);
217 if(dir.abs2() > epsilon) return unit(dir);
218 dir=2.0*c1-c0-z1;
219 if(dir.abs2() > epsilon) return unit(dir);
220 return unit(z1-z0+3.0*(c0-c1));
221 }
222
223 pair postdir(Int t, bool normalize=true) const {
224 if(!cycles && t >= n-1) return pair(0,0);
225 pair c0=postcontrol(t);
226 pair z0=point(t);
227 pair dir=3.0*(c0-z0);
228 if(!normalize) return dir;
229 pair z1=point(t+1);
230 pair c1=precontrol(t+1);
231 double epsilon=norm(z0,c0,c1,z1);
232 if(dir.abs2() > epsilon) return unit(dir);
233 dir=z0-2.0*c0+c1;
234 if(dir.abs2() > epsilon) return unit(dir);
235 return unit(z1-z0+3.0*(c0-c1));
236 }
237
238 pair dir(Int t, Int sign, bool normalize=true) const {
239 if(sign == 0) {
240 pair v=predir(t,normalize)+postdir(t,normalize);
241 return normalize ? unit(v) : 0.5*v;
242 }
243 if(sign > 0) return postdir(t,normalize);
244 return predir(t,normalize);
245 }
246
247 pair dir(double t, bool normalize=true) const {
248 if(!cycles) {
249 if(t <= 0) return postdir((Int) 0,normalize);
250 if(t >= n-1) return predir(n-1,normalize);
251 }
252 Int i=Floor(t);
253 t -= i;
254 if(t == 0) return dir(i,0,normalize);
255 pair z0=point(i);
256 pair c0=postcontrol(i);
257 pair c1=precontrol(i+1);
258 pair z1=point(i+1);
259 pair a=3.0*(z1-z0)+9.0*(c0-c1);
260 pair b=6.0*(z0+c1)-12.0*c0;
261 pair c=3.0*(c0-z0);
262 pair dir=a*t*t+b*t+c;
263 if(!normalize) return dir;
264 double epsilon=norm(z0,c0,c1,z1);
265 if(dir.abs2() > epsilon) return unit(dir);
266 dir=2.0*a*t+b;
267 if(dir.abs2() > epsilon) return unit(dir);
268 return unit(a);
269 }
270
271 pair postaccel(Int t) const {
272 if(!cycles && t >= n-1) return pair(0,0);
273 pair z0=point(t);
274 pair c0=postcontrol(t);
275 pair c1=precontrol(t+1);
276 return 6.0*(z0+c1)-12.0*c0;
277 }
278
279 pair preaccel(Int t) const {
280 if(!cycles && t <= 0) return pair(0,0);
281 pair c0=postcontrol(t-1);
282 pair c1=precontrol(t);
283 pair z1=point(t);
284 return 6.0*(z1+c0)-12.0*c1;
285 }
286
287 pair accel(Int t, Int sign) const {
288 if(sign == 0) return 0.5*(preaccel(t)+postaccel(t));
289 if(sign > 0) return postaccel(t);
290 return preaccel(t);
291 }
292
293 pair accel(double t) const {
294 if(!cycles) {
295 if(t <= 0) return postaccel((Int) 0);
296 if(t >= n-1) return preaccel(n-1);
297 }
298 Int i=Floor(t);
299 t -= i;
300 if(t == 0) return 0.5*(postaccel(i)+preaccel(i));
301 pair z0=point(i);
302 pair c0=postcontrol(i);
303 pair c1=precontrol(i+1);
304 pair z1=point(i+1);
305 return 6.0*t*(z1-z0+3.0*(c0-c1))+6.0*(z0+c1)-12.0*c0;
306 }
