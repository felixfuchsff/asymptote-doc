92inline bool operator == (const vm::array& a, const vm::array& b)
93{
94 size_t asize=a.size();
95 size_t bsize=b.size();
96 if(asize != bsize) return false;
97 for(size_t i=0; i < asize; ++i)
98 if(vm::read<double>(a,i) != vm::read<double>(b,i)) return false;
99 return true;
100}

104 return a.pattern == b.pattern && a.offset == b.offset && a.scale == b.scale

602 void greytocmyk() {

608 void rgbtocmyk() {
609 double sat=rgbsaturation();
610 grey=1.0-sat;
611 if(sat) {
612 double scale=1.0/sat;
613 r=1.0-r*scale;
614 g=1.0-g*scale;
615 b=1.0-b*scale;
616 }
617 color=CMYK;
618 }

622 r=(1.0-r)*sat;
623 g=(1.0-g)*sat;

671 // Try to upgrade to the specified colorspace.
672 bool promote(ColorSpace c) {
673 if(color == c) return true;
674
675 switch(color) {
676 case PATTERN:
677 case INVISIBLE:
678 break;
679 case DEFCOLOR:
680 {
681 return true;
682 break;
683 }
684 break;
685 case GRAYSCALE:
686 {
687 if(c == RGB) {greytorgb(); return true;}
688 else if(c == CMYK) {greytocmyk(); return true;}
689 break;
690 }
691 case RGB:
692 {
693 if(c == CMYK) {rgbtocmyk(); return true;}
694 break;
695 }
696 case CMYK:
697 {
698 break;
699 }
700 }
701 return false;
702 }

930 // The bounds of the circle or ellipse the pen describes.
931 bbox bounds() const
932 {
933 double maxx, maxy;
934 pair shift;
935
936 if(!P.empty()) return P.bounds();
937
938 transform t=getTransform();
939
940 if(t.isIdentity()) {
941 maxx = 1;
942 maxy = 1;
943 shift = pair(0,0);
944 } else {
945 double xx = t.getxx(), xy = t.getxy(),
946 yx = t.getyx(), yy = t.getyy();
947
948 // These are the maximum x and y values that a linear transform can map
949 // a point in the unit circle. This can be proven by the Lagrange
950 // Multiplier theorem or by properties of the dot product.
951 maxx = length(pair(xx,xy));
952 maxy = length(pair(yx,yy));
953
954 shift = t*pair(0,0);
955 }
956
957 bbox b;
958 pair z=0.5*width()*pair(maxx,maxy);
959 b += z + shift;
960 b += -z + shift;
961
962 return b;
963 }
