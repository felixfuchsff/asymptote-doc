58 real atime(pair m, path g, real fuzz=trembleFuzz())
59 {// Return the time of the point on path g nearest to m, within fuzz.
60 if(length(g) == 0) return 0.0;
61 real[] t=intersect(m,g,fuzz);
62 if(t.length > 0) return t[1];
63 real ot;
64 static real eps=sqrt(realEpsilon);
65 real abmax=abs(max(g)-m), abmin=abs(min(g)-m);
66 real initr=abs(m-midpoint(g));
67 real maxR=2*max(abmax,abmin), step=eps, r=initr;
68 real shx=1e-4;
69 transform T=shift(m);
70 path ig;
71 if(t.length > 0) ot=t[1];
72 real rm=0, rM=r;
73 while(rM-rm > eps) {
74 r=(rm+rM)/2;
75 t=intersect(T*scale(r)*unitcircle,g,fuzz);
76 if(t.length <= 0) {
77 rm=r;
78 } else {
79 rM=r;
80 ot=t[1];
81 }
82 }
83 return ot;
84 }

86 path addnode(path g, real t)
87 {// Add a node to 'g' at point(g,t).
88 real l=length(g);
89 real rt=t % 1;
90 if(l == 0 || (t > l && !cyclic(g)) || rt == 0) return g;
91 if(cyclic(g)) t=t % l;
92 int t0=floor(t);
93 int t1=t0+1;
94 pair z0=point(g,t0), z1=point(g,t1),
95 c0=postcontrol(g,t0), c1=precontrol(g,t1),
96 m0=(1-rt)*z0+rt*c0, m1=(1-rt)*c0+rt*c1,
97 m2=(1-rt)*c1+rt*z1, m3=(1-rt)*m0+rt*m1,
98 m4=(1-rt)*m1+rt*m2;
99 guide og=subpath(g,0,t0)..controls m0 and m3..point(g,t);
100 if(cyclic(g)) {
101 if(t1 < l)
102 og=og..controls m4 and m2..subpath(g,t1,l)&cycle;
103 else og=og..controls m4 and m2..cycle;
104 } else og=og..controls m4 and m2..subpath(g,t1,l);
105 return og;
106 }

108 path addnodes(path g, real fuzz=trembleFuzz()...pair[] P)
109 {
110 pair[] P=single(P);
111 if(length(g) == 0 || P.length == 0 || magneticRadius <= 0) return g;
112 path og=g;
113 for(pair tp: P) {
114 real t=atime(tp,og,fuzz);
115 real d=abs(tp-point(og,t));
116 if(d < magneticRadius) og=addnode(og,t);
117 }
118 return og;
119 }

121 path addnodes(path g, int n)
122 {// Add 'n' nodes between each node of 'g'.
123 real l=length(g);
124 if(n == 0 || l == 0) return g;
125 path og=g;
126 int np=0;
127 for(int i=0; i < l; ++i) {
128 real step=1/(n+1);
129 for(int j=0; j < n; ++j) {
130 og=addnode(og,i*(n+1)+j+step);
131 step=1/(n-j);
132 }
133 }
134 return og;
135 }

145 path deform(path g...pair[] magneticPoints) {
146 /* Return g as it was handwriting.
147 The postcontrols and precontrols of the nodes of g will be rotated
148 by an angle proportional to 'angle'(in degrees).
149 If frequency < 1, floor(1/frequency) nodes will be added to g to
150 increase the control points.
151 If frequency>= 1, one point for floor(frequency) will be used to deform
152 the path.
153 'random' controls the randomized coefficient which will be multiplied
154 by 'angle'.
155 random is 0 means don't use randomized coefficient;
156 The higher 'random' is, the more the trembling is randomized. */
157 if(length(g) == 0) return g;
158 g=addnodes(g,fuzz*abs(max(g)-min(g))...magneticPoints);
159 path tg=g;
160 frequency=abs(frequency);
161 int f=abs(floor(1/frequency)-1);
162 tg=addnodes(tg,f);
163 int frequency=floor(frequency);
164 int tf=(frequency == 0) ? 1 : frequency;
165 int l=length(tg);
166 guide og=point(tg,0);
167 random=abs(random);
168 int rsgn(real x){
169 int d2=floor(100*x)-10*floor(10*x);
170 if(d2 == 0) return 1;
171 return 2 % d2 == 0 ? 1 : -1;
172 }
173 real randf()
174 {
175 real or;
176 if(random != 0) {
177 if(1 % tf != 0) or=0;
178 else {
179 real ur=unitrand();
180 or=rsgn(ur)*angle*(1+ur^(1/random));
181 }
182 } else or=rsgn(unitrand())*1.5*angle;
183 return or;
184 }
185
186 real first=randf();
187 for(int i=1; i <= l; ++i) {
188 pair P=point(tg,i);
189 real a=randf();
190 pair post=rotate(a,point(tg,i-1))*postcontrol(tg,i-1);
191 pair pre=rotate((a+randf())/2,P)*precontrol(tg,i);
192 if(i == l && (cyclic(tg)))
193 og=og..controls post and pre..cycle;
194 else
195 og=og..controls post and pre..P;
196 }
197 return og;
198 }
