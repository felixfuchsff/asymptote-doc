606// This is used to encapsulate iteration over the subtypes of an overloaded
607// type. The base method need only be implemented to handle non-overloaded
608// types.
609class collector {
610public:
611 virtual ~collector() {}
612 virtual ty *base(ty *target, ty *source) = 0;
613
614 virtual ty *collect(ty *target, ty *source) {
615 if (overloaded *o=dynamic_cast<overloaded *>(target)) {
616 ty_vector &sub=o->sub;
617
618 overloaded *oo=new overloaded;
619 for(ty_vector::iterator x = sub.begin(); x != sub.end(); ++x) {
620 types::ty *t=collect(*x, source);
621 if (t)
622 oo->add(t);
623 }
624
625 return oo->simplify();
626 }
627 else if (overloaded *o=dynamic_cast<overloaded *>(source)) {
628 ty_vector &sub=o->sub;
629
630 overloaded *oo=new overloaded;
631 for(ty_vector::iterator y = sub.begin(); y != sub.end(); ++y) {
632 // NOTE: A possible speed optimization would be to replace this with a
633 // call to base(), but this is only correct if we can guarantee that an
634 // overloaded type has no overloaded sub-types.
635 types::ty *t=collect(target, *y);
636 if (t)
637 oo->add(t);
638 }
639
640 return oo->simplify();
641 }
642 else
643 return base(target, source);
644 }
