22// Return a frame containing n copies of the path g shifted by space
23// drawn with pen p.
24frame duplicate(path g, int n=1, pair space=0, pen p=currentpen)
25{
26 if(space == 0) space=dotsize(p);
27 frame f;
28 int pos=0;
29 int sign=1;
30 int m=(n+1) % 2;
31 for(int i=1; i <= n; ++i) {
32 draw(f,shift(space*(pos-0.5*m))*g,p);
33 pos += i*sign;
34 sign *= -1;
35 }
36 return f;
37}

134// Mark the oriented angle AOB counterclockwise with optional Label, arrows, and markers.
135// With radius < 0, AOB-2pi is marked clockwise.
136void markangle(picture pic=currentpicture, Label L="",
137 int n=1, real radius=0, real space=0,
138 pair A, pair O, pair B, arrowbar arrow=None,
139 pen p=currentpen, filltype filltype=NoFill,
140 margin margin=NoMargin, marker marker=nomarker)
141{
142 if(space == 0) space=markanglespace(p);
143 if(radius == 0) radius=markangleradius(p);
144 picture lpic,phantom;
145 frame ff;
146 path lpth;
147 p=squarecap+p;
148 pair OB=unit(B-O), OA=unit(A-O);
149 real xoa=degrees(OA,false);
150 real gle=degrees(acos(dot(OA,OB)));
151 if((conj(OA)*OB).y < 0) gle *= -1;
152 bool ccw=radius > 0;
153 if(!ccw) radius=-radius;
154 bool drawarrow = !arrow(phantom,arc((0,0),radius,xoa,xoa+gle,ccw),p,margin);
155 if(drawarrow && margin == NoMargin) margin=TrueMargin(0,0.5linewidth(p));
156 if(filltype != NoFill) {
157 lpth=margin(arc((0,0),radius+(n-1)*space,xoa,xoa+gle,ccw),p).g;
158 pair p0=relpoint(lpth,0), p1=relpoint(lpth,1);
159 pair ac=p0-p0-A+O, bd=p1-p1-B+O, det=(conj(ac)*bd).y;
160 pair op=(det == 0) ? O : p0+(conj(p1-p0)*bd).y*ac/det;
161 filltype.fill(ff,op--lpth--relpoint(lpth,1)--cycle,p);
162 add(lpic,ff);
163 }
164 for(int i=0; i < n; ++i) {
165 lpth=margin(arc((0,0),radius+i*space,xoa,xoa+gle,ccw),p).g;
166 draw(lpic,lpth,p=p,arrow=arrow,margin=NoMargin,marker=marker);
167 }
168 Label lL=L.copy();
169 real position=lL.position.position.x;
170 if(lL.defaultposition) {lL.position.relative=true; position=0.5;}
171 if(lL.position.relative) position=reltime(lpth,position);
172 if(lL.align.default) {
173 lL.align.relative=true;
174 lL.align.dir=unit(point(lpth,position));
175 }
176 label(lpic,lL,point(lpth,position),align=NoAlign, p=p);
177 add(pic,lpic,O);
178}
